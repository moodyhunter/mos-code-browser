102/**
103 * mutex_acquire
104 * Try to acquire a mutex and block the calling thread.
105 * \param mutex The mutex to lock.
106 * \return true if the mutex was locked, false if it was already locked.
107 *
108 * This syscall blocks the calling thread until the target mutex has been released
109 * by another thread. Once the mutex is acquired, it returns true and the calling
110 * thread will be scheduled to run again.
111 */
112should_inline bool syscall_mutex_acquire(bool *mutex)
113{
114 return platform_syscall1(18, (long) mutex);
115}
116
117/**
118 * mutex_release
119 * Release a mutex.
120 * \param mutex The mutex to release.
121 * \return true if the mutex was released, false if it was already released.
122 *
123 * This syscall releases a mutex that was previously acquired by the some thread.
124 */
125should_inline bool syscall_mutex_release(bool *mutex)
126{
127 return platform_syscall1(19, (long) mutex);
128}
