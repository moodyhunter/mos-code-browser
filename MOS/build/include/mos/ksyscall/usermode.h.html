10always_inline void invoke_ksyscall_panic(void)
11{
12 platform_syscall0(0);
13}
14
15always_inline fd_t invoke_ksyscall_file_open(const char *file_path, file_open_flags flags)
16{
17 return platform_syscall2(1, (long) file_path, (long) flags);
18}
19
20always_inline bool invoke_ksyscall_file_stat(const char *file_path, file_stat_t *stat)
21{
22 return platform_syscall2(2, (long) file_path, (long) stat);
23}
24
25always_inline size_t invoke_ksyscall_io_read(fd_t fd, void *buffer, size_t size, size_t offset)
26{
27 return platform_syscall4(3, (long) fd, (long) buffer, (long) size, (long) offset);
28}
29
30always_inline size_t invoke_ksyscall_io_write(fd_t fd, const void *buffer, size_t size, size_t offset)
31{
32 return platform_syscall4(4, (long) fd, (long) buffer, (long) size, (long) offset);
33}
34
35always_inline bool invoke_ksyscall_io_close(fd_t fd)
36{
37 return platform_syscall1(5, (long) fd);
38}
39
40always_inline void invoke_ksyscall_exit(u32 exit_code)
41{
42 platform_syscall1(6, (long) exit_code);
43}

45always_inline void invoke_ksyscall_yield_cpu(void)
