9should_inline long dispatch_ksyscall(const long number, long arg1, long arg2, long arg3, long arg4, long arg5, long arg6, long arg7, long arg8)
10{
11 (void) arg1;
12 (void) arg2;
13 (void) arg3;
14 (void) arg4;
15 (void) arg5;
16 (void) arg6;
17 (void) arg7;
18 (void) arg8;
19
20 long ret = 0;
21 extern noreturn void ksyscall_panic(void);
22 if (number == 0)
23 ksyscall_panic();
24
25 extern fd_t ksyscall_file_open(const char *file_path, file_open_flags flags);
26 if (number == 1)
27 ret = (long) ksyscall_file_open((const char *) arg1, (file_open_flags) arg2);
28
29 extern bool ksyscall_file_stat(const char *file_path, file_stat_t *stat);
30 if (number == 2)
31 ret = (long) ksyscall_file_stat((const char *) arg1, (file_stat_t *) arg2);
32
33 extern size_t ksyscall_io_read(fd_t fd, void *buffer, size_t size, size_t offset);
34 if (number == 3)
35 ret = (long) ksyscall_io_read((fd_t) arg1, (void *) arg2, (size_t) arg3, (size_t) arg4);
36
37 extern size_t ksyscall_io_write(fd_t fd, const void *buffer, size_t size, size_t offset);
38 if (number == 4)
39 ret = (long) ksyscall_io_write((fd_t) arg1, (const void *) arg2, (size_t) arg3, (size_t) arg4);
40
41 extern bool ksyscall_io_close(fd_t fd);
42 if (number == 5)
43 ret = (long) ksyscall_io_close((fd_t) arg1);
44
45 extern noreturn void ksyscall_exit(u32 exit_code);
46 if (number == 6)
47 ksyscall_exit((u32) arg1);
48
49 extern void ksyscall_yield_cpu(void);
50 if (number == 7)
51 ksyscall_yield_cpu();
52
53 return ret;
54}
