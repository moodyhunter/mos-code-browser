5#include "lib/containers.h"
6#include "mos/types.h"
7
8#define STD_COLOR_LIGHT 0x8
9
10typedef enum
11{
12 Black = 0x0,
13 Blue = 0x1,
14 Green = 0x2,
15 Cyan = 0x3,
16 Red = 0x4,
17 Magenta = 0x5,
18 Brown = 0x6,
19 Gray = 0x7,
20 DarkGray = Black | STD_COLOR_LIGHT,
21 LightBlue = Blue | STD_COLOR_LIGHT,
22 LightGreen = Green | STD_COLOR_LIGHT,
23 LightCyan = Cyan | STD_COLOR_LIGHT,
24 LightRed = Red | STD_COLOR_LIGHT,
25 LightMagenta = Magenta | STD_COLOR_LIGHT,
26 Yellow = Brown | STD_COLOR_LIGHT,
27 White = Gray | STD_COLOR_LIGHT,
28} standard_color_t;
29
30typedef enum
31{
32 CONSOLE_CAP_NONE = 0,
33 CONSOLE_CAP_COLOR = 1 << 0,
34 CONSOLE_CAP_CLEAR = 1 << 1,
35 CONSOLE_CAP_READ = 1 << 2,
36 CONSOLE_CAP_SETUP = 1 << 3,
37 CONSOLE_CAP_GET_SIZE = 1 << 4,
38 CONSOLE_CAP_CURSOR_HIDE = 1 << 5,
39 CONSOLE_CAP_CURSOR_MOVE = 1 << 6,
40} console_caps_t;
41
42typedef struct console_t console_t;
43
44struct console_t
45{
46 as_linked_list;
47
48 char name[64];
49 console_caps_t caps;
50
51 bool (*setup)(console_t *con);
52 bool (*get_size)(console_t *con, u32 *width, u32 *height);
53
54 bool (*set_cursor)(console_t *con, bool show);
55 bool (*move_cursor)(console_t *con, u32 x, u32 y);
56 bool (*get_cursor)(console_t *con, u32 *x, u32 *y);
57
58 // VGA standard color codes
59 bool (*get_color)(console_t *con, standard_color_t *fg, standard_color_t *bg);
60 bool (*set_color)(console_t *con, standard_color_t fg, standard_color_t bg);
61
62 int (*read)(console_t *con, char *dest, size_t size);
63 int (*write)(console_t *con, const char *data, size_t size);
64
65 bool (*clear)(console_t *con);
66 bool (*close)(console_t *con);
67
68 void *data;
69};

72void mos_install_console(console_t *con);
