10#define PER_CPU_DECLARE(type, name) \
11 struct \
12 { \
13 type percpu_value[MOS_MAX_CPU_COUNT]; \
14 } name

21typedef struct _thread thread_t;

23typedef enum
24{
25 VM_NONE = 1 << 1,
26 VM_READ = 1 << 2,
27 VM_WRITE = 1 << 3,
28 VM_USER = 1 << 4,
29 VM_WRITE_THROUGH = 1 << 5,
30 VM_CACHE_DISABLED = 1 << 6,
31 VM_GLOBAL = 1 << 7,
32 VM_EXEC = 1 << 8,
33} vm_flags;
34
35typedef enum
36{
37 PGALLOC_HINT_KHEAP,
38 PGALLOC_HINT_USERSPACE,
39} pgalloc_hints;
40
41typedef enum
42{
43 VMTYPE_APPCODE,
44 VMTYPE_APPDATA,
45 VMTYPE_STACK,
46 VMTYPE_KSTACK,
47 VMTYPE_FILE,
48} vm_type;
49
50typedef struct
51{
52 u32 id;
53 thread_t *thread;
54 void *platform_context;
55 uintptr_t scheduler_stack;
56 paging_handle_t pagetable;
57} cpu_t;
58
59typedef struct
60{
61 uintptr_t vaddr;
62 uintptr_t paddr;
63 size_t size_bytes;
64 bool available;
65} memblock_t;
66
67typedef struct
68{
69 uintptr_t vaddr;
70 size_t npages;
71 vm_flags flags; // the expected flags for the region, regardless of the copy-on-write state
72} vmblock_t;
73
74typedef enum
75{
76 MMAP_DEFAULT = 0 << 0,
77 MMAP_COW = 1 << 0,
78 MMAP_PRIVATE = 1 << 1,
79} mmap_flags;
80
81typedef struct
82{
83 vmblock_t vm;
84 vm_type type;
85
86 // if MMAP_COW is set, then the flags in vm contains 'original' flags
87 // of this block. Which means if there're no VM_WRITE flag, then the block
88 // should not be writable.
89 mmap_flags map_flags;
90} proc_vmblock_t;
91
92typedef struct
93{
94 struct
95 {
96 const uintptr_t code_start;
97 const uintptr_t code_end;
98 const uintptr_t rodata_start;
99 const uintptr_t rodata_end;
100 const uintptr_t rw_start;
101 const uintptr_t rw_end;
102 } regions;
103
104 u32 num_cpus;
105 u32 boot_cpu_id;

124 void (*const mm_destroy_user_pgd)(paging_handle_t table);

127 vmblock_t (*const mm_alloc_pages_at)(paging_handle_t table, uintptr_t vaddr, size_t npages, vm_flags vflags);

138 void (*const switch_to_thread)(uintptr_t *old_stack, thread_t *new_thread);
