56paging_handle_t platform_mm_create_user_pgd(void)
57{
58 vmblock_t block = pg_page_alloc(x86_kpg_infra, ALIGN_UP_TO_PAGE(sizeof(x86_pg_infra_t)) / MOS_PAGE_SIZE, PGALLOC_HINT_KHEAP, VM_READ | VM_WRITE);
59 x86_pg_infra_t *infra = (x86_pg_infra_t *) block.vaddr;
60 memset(infra, 0, sizeof(x86_pg_infra_t));
61 paging_handle_t handle;
62 handle.ptr = (uintptr_t) infra;
63
64 // pg_do_map_pages(infra, 0, 0, 1, VM_NONE); // ! the zero page is not writable, nor readable by user
65 pg_do_map_pages(infra, MOS_PAGE_SIZE, MOS_PAGE_SIZE, 1 MB / MOS_PAGE_SIZE - 1, VM_GLOBAL | VM_WRITE);
66
67 // physical address of kernel page table
68 const uintptr_t kpgtable_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) x86_kpg_infra->pgtable);
69
70 // this is a bit of a hack, but it's the easiest way that I can think of ...
71 const int kernel_pagedir_id_start = MOS_KERNEL_START_VADDR / MOS_PAGE_SIZE / 1024; // addr / (size of page) / (# pages of a page directory)
72 for (int i = kernel_pagedir_id_start; i < 1024; i++)
73 {
74 x86_pgdir_entry *pgd = &infra->pgdir[i];
75 pgd->present = true;
76 pgd->writable = true;
77 pgd->usermode = false;
78 // redirect it to the kernel page table
79 // use pre-allocated (pre-calculated) physical address, otherwise some newly mapped pgdirs won't be applied correctly
80 pgd->page_table_paddr = (kpgtable_paddr + i * 1024 * sizeof(x86_pgtable_entry)) >> 12;
81 }
82 return handle;
83}

108vmblock_t platform_mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
109{
110 x86_pg_infra_t *from_infra = x86_get_pg_infra(from);
111 x86_pg_infra_t *to_infra = x86_get_pg_infra(to);
112
113 // uintptr_t start_paddr = pg_page_get_mapped_paddr(from_infra, fvaddr);
114
115 for (size_t i = 0; i < npages; i++)
116 {
117 uintptr_t from_vaddr = fvaddr + i * MOS_PAGE_SIZE;
118 uintptr_t to_vaddr = tvaddr + i * MOS_PAGE_SIZE;
119 // uintptr_t expected_paddr = start_paddr + i * MOS_PAGE_SIZE;
120 uintptr_t paddr = pg_page_get_mapped_paddr(from_infra, from_vaddr);
121 vm_flags flags = pg_page_get_flags(from_infra, from_vaddr);
122 pg_do_map_page(to_infra, to_vaddr, paddr, flags);
123 }
124
125 vmblock_t block = {
126 .vaddr = tvaddr,
127 .npages = npages,
128 .flags = pg_page_get_flags(from_infra, fvaddr),
129 };
130
131 return block;
132}
