67paging_handle_t platform_mm_create_user_pgd(void)
68{
69 const size_t npages = ALIGN_UP_TO_PAGE(sizeof(x86_pg_infra_t)) / MOS_PAGE_SIZE;
70 vmblock_t block = mm_alloc_pages(x86_platform.kernel_pgd, npages, PGALLOC_HINT_KHEAP, VM_RW);
71 x86_pg_infra_t *infra = (x86_pg_infra_t *) block.vaddr;
72 memzero(infra, sizeof(x86_pg_infra_t));
73 paging_handle_t handle;
74 handle.pgd = (uintptr_t) infra;
75
76 // physical address of kernel page table
77 const uintptr_t kpgtable_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) x86_kpg_infra->pgtable);
78
79 // this is a bit of a hack, but it's the easiest way that I can think of ...
80 const int kernel_pagedir_id_start = MOS_KERNEL_START_VADDR / MOS_PAGE_SIZE / 1024; // addr / (size of page) / (# pages of a page directory)
81 for (int i = kernel_pagedir_id_start; i < 1024; i++)
82 {
83 x86_pgdir_entry *pgd = &infra->pgdir[i];
84 pgd->present = true;
85 pgd->writable = true;
86 pgd->usermode = false;
87 // redirect it to the kernel page table
88 // use pre-allocated (pre-calculated) physical address, otherwise some newly mapped pgdirs won't be applied correctly
89 pgd->page_table_paddr = (kpgtable_paddr + i * 1024 * sizeof(x86_pgtable_entry)) >> 12;
90 }
91 return handle;
92}

119void platform_mm_map_pages(paging_handle_t table, vmblock_t block)
120{
121 mos_debug(x86_paging, "mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table " PTR_FMT, block.npages, block.vaddr, block.paddr, table.pgd);
122
123 x86_pg_infra_t *infra = x86_get_pg_infra(table);
124 for (size_t i = 0; i < block.npages; i++)
125 pg_do_map_page(infra, block.vaddr + i * MOS_PAGE_SIZE, block.paddr + i * MOS_PAGE_SIZE, block.flags);
126}

128void platform_mm_unmap_pages(paging_handle_t table, uintptr_t vaddr_start, size_t n_pages)
129{
130 mos_debug(x86_paging, "unmapping %zu pages starting at " PTR_FMT " @ table " PTR_FMT, n_pages, vaddr_start, table.pgd);
131
132 x86_pg_infra_t *infra = x86_get_pg_infra(table);
133 for (size_t i = 0; i < n_pages; i++)
134 pg_do_unmap_page(infra, vaddr_start + i * MOS_PAGE_SIZE);
135}

137vmblock_t platform_mm_get_block_info(paging_handle_t table, uintptr_t vaddr, size_t npages)
138{
139 x86_pg_infra_t *infra = x86_get_pg_infra(table);
140 vmblock_t block;
141 block.vaddr = vaddr;
142 block.paddr = pg_page_get_mapped_paddr(infra, vaddr);
143 block.npages = npages;
144 block.flags = pg_page_get_flags(infra, vaddr);
145 return block;
146}

148vmblock_t platform_mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
149{
150 x86_pg_infra_t *from_infra = x86_get_pg_infra(from);
151 x86_pg_infra_t *to_infra = x86_get_pg_infra(to);
152
153 for (size_t i = 0; i < npages; i++)
154 {
155 uintptr_t from_vaddr = fvaddr + i * MOS_PAGE_SIZE;
156 uintptr_t to_vaddr = tvaddr + i * MOS_PAGE_SIZE;
157 uintptr_t paddr = pg_page_get_mapped_paddr(from_infra, from_vaddr);
158 vm_flags flags = pg_page_get_flags(from_infra, from_vaddr);
159 pg_do_map_page(to_infra, to_vaddr, paddr, flags);
160 }
161
162 vmblock_t block = {
163 .vaddr = tvaddr,
164 .npages = npages,
165 .flags = pg_page_get_flags(from_infra, fvaddr),
166 };
167
168 return block;
169}
