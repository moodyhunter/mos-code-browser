74paging_handle_t platform_mm_create_user_pgd(void)
75{
76 vmblock_t block = pg_page_alloc(x86_kpg_infra, ALIGN_UP_TO_PAGE(sizeof(x86_pg_infra_t)) / MOS_PAGE_SIZE, PGALLOC_HINT_KHEAP, VM_RW);
77 x86_pg_infra_t *infra = (x86_pg_infra_t *) block.vaddr;
78 memset(infra, 0, sizeof(x86_pg_infra_t));
79 paging_handle_t handle;
80 handle.ptr = (uintptr_t) infra;
81
82 // pg_do_map_pages(infra, 0, 0, 1, VM_NONE); // ! the zero page is not writable, nor readable by user
83 pg_do_map_pages(infra, MOS_PAGE_SIZE, MOS_PAGE_SIZE, 1 MB / MOS_PAGE_SIZE - 1, VM_GLOBAL | VM_WRITE);
84
85 // physical address of kernel page table
86 const uintptr_t kpgtable_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) x86_kpg_infra->pgtable);
87
88 // this is a bit of a hack, but it's the easiest way that I can think of ...
89 const int kernel_pagedir_id_start = MOS_KERNEL_START_VADDR / MOS_PAGE_SIZE / 1024; // addr / (size of page) / (# pages of a page directory)
90 for (int i = kernel_pagedir_id_start; i < 1024; i++)
91 {
92 x86_pgdir_entry *pgd = &infra->pgdir[i];
93 pgd->present = true;
94 pgd->writable = true;
95 pgd->usermode = false;
96 // redirect it to the kernel page table
97 // use pre-allocated (pre-calculated) physical address, otherwise some newly mapped pgdirs won't be applied correctly
98 pgd->page_table_paddr = (kpgtable_paddr + i * 1024 * sizeof(x86_pgtable_entry)) >> 12;
99 }
100 return handle;
101}

126vmblock_t platform_mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
127{
128 x86_pg_infra_t *from_infra = x86_get_pg_infra(from);
129 x86_pg_infra_t *to_infra = x86_get_pg_infra(to);
130
131 // uintptr_t start_paddr = pg_page_get_mapped_paddr(from_infra, fvaddr);
132
133 for (size_t i = 0; i < npages; i++)
134 {
135 uintptr_t from_vaddr = fvaddr + i * MOS_PAGE_SIZE;
136 uintptr_t to_vaddr = tvaddr + i * MOS_PAGE_SIZE;
137 // uintptr_t expected_paddr = start_paddr + i * MOS_PAGE_SIZE;
138 uintptr_t paddr = pg_page_get_mapped_paddr(from_infra, from_vaddr);
139 vm_flags flags = pg_page_get_flags(from_infra, from_vaddr);
140 pg_do_map_page(to_infra, to_vaddr, paddr, flags);
141 }
142
143 vmblock_t block = {
144 .vaddr = tvaddr,
145 .npages = npages,
146 .flags = pg_page_get_flags(from_infra, fvaddr),
147 };
148
149 return block;
150}
