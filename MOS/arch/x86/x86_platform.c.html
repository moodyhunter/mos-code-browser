43void do_backtrace(u32 max)
44{
45 static bool is_tracing = false;
46 if (is_tracing)
47 return;
48 pr_info("Stack trace:");
49 struct frame_t
50 {
51 struct frame_t *ebp;
52 uintptr_t eip;
53 } *frame = NULL;
54
55 __asm__("movl %%ebp,%1" : "=r"(frame) : "r"(frame));
56 for (u32 i = 0; frame && i < max; i++)
57 {
58 pr_warn(" " PTR_FMT, frame->eip);
59 frame = frame->ebp;
60 }
61}

63void x86_kpanic_hook()
64{
65 pmem_freelist_dump();
66 pr_emph("Current task: %s", current_thread ? current_thread->owner->name : "<none>");
67 if (current_thread)
68 {
69 pr_emph("Task Page Table:");
70 x86_mm_dump_page_table(x86_get_pg_infra(current_process->pagetable));
71 }
72 pr_emph("Current Page Table:");
73 x86_mm_dump_page_table(x86_get_pg_infra(current_cpu->pagetable));
74 pr_emph("Kernel Page Table:");
75 x86_mm_dump_page_table(x86_kpg_infra);
76 do_backtrace(20);
77}

79void x86_start_kernel(x86_startup_info *info)
80{
81 multiboot_info_t *mb_info = info->mb_info;
82 uintptr_t initrd_size = info->initrd_size;
83 x86_disable_interrupts();
84 mos_install_console(&com1_console.console);
85
86 x86_gdt_init();
87 x86_idt_init();
88 x86_tss_init();
89 x86_irq_handler_init();
90
91 pic_unmask_irq(IRQ_TIMER);
92 pic_unmask_irq(IRQ_KEYBOARD);
93 pic_unmask_irq(IRQ_COM1);
94 pic_unmask_irq(IRQ_COM2);
95
96 if (mb_info->flags & MULTIBOOT_INFO_CMDLINE)
97 strncpy(mos_cmdline, mb_info->cmdline, sizeof(mos_cmdline));
98
99 u32 count = mb_info->mmap_length / sizeof(multiboot_memory_map_t);
100 x86_mem_init(mb_info->mmap_addr, count);
101
102 x86_mm_prepare_paging();
103 x86_platform.kernel_pg.ptr = (uintptr_t) x86_kpg_infra;
104
105 if (initrd_size)
106 {
107 reg_t cr3 = x86_get_cr3();
108 const uintptr_t initrd_vaddr = MOS_X86_INITRD_VADDR;
109 x86_pgtable_entry *table = (x86_pgtable_entry *) (((x86_pgdir_entry *) cr3)[initrd_vaddr >> 22].page_table_paddr << 12);
110 const uintptr_t initrd_paddr = table->phys_addr << 12;
111 const size_t initrd_n_pages = ALIGN_UP_TO_PAGE(initrd_size) / MOS_PAGE_SIZE;
112 pg_map_pages(x86_kpg_infra, initrd_vaddr, initrd_paddr, initrd_n_pages, VM_GLOBAL);
113 }
114
115 x86_acpi_init();
116 x86_smp_init();
117
118 // ! map the bios memory area, should it be done like this?
119 pr_info("mapping bios memory area...");
120 for (u32 i = 0; i < x86_mem_regions_count; i++)
121 if (x86_mem_regions[i].paddr == info->bios_region_start)
122 x86_bios_region = &x86_mem_regions[i];
123
124 memblock_t *bios_block = x86_bios_region;
125 pg_do_map_pages(x86_kpg_infra, bios_block->paddr, bios_block->paddr, bios_block->size_bytes / MOS_PAGE_SIZE, VM_GLOBAL);
126
127 x86_mm_enable_paging();
128
129 mos_kernel_mm_init(); // since then, we can use the kernel heap (kmalloc)
130
131 // the stack memory to be used if we enter the kernelmode by a trap / interrupt
132 extern tss32_t tss_entry;
133#define KSTACK_PAGES 1
134 const vmblock_t esp0_block = pg_page_alloc(x86_kpg_infra, KSTACK_PAGES, PGALLOC_HINT_KHEAP, VM_READ | VM_WRITE);
135 tss_entry.esp0 = esp0_block.vaddr + KSTACK_PAGES * MOS_PAGE_SIZE;
136
137 pr_emph("kernel stack at " PTR_FMT, (uintptr_t) tss_entry.esp0);
138 tss32_flush(GDT_SEGMENT_TSS);
139
140 // mos_install_kpanic_hook(x86_kpanic_hook);
141 mos_install_console(&vga_text_mode_console);
142 x86_install_interrupt_handler(IRQ_COM1, serial_irq_handler);
143
144 if (initrd_size)
145 {
146 initrd_blockdev_t *initrd_blockdev = kmalloc(sizeof(initrd_blockdev_t));
147 initrd_blockdev->memblock = (memblock_t){ .available = true, .vaddr = MOS_X86_INITRD_VADDR, .size_bytes = initrd_size };
148
149 initrd_blockdev_preinstall(initrd_blockdev);
150 blockdev_register(&initrd_blockdev->blockdev);
151 }
152
153 mos_start_kernel(mos_cmdline);
154}
