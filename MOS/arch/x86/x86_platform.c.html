42void x86_keyboard_handler(u32 irq)
43{
44 MOS_ASSERT(irq == IRQ_KEYBOARD);
45 int scancode = port_inb(0x60);
46 char buf[9] = { 0 };
47 for (int i = 7; i >= 0; i--)
48 {
49 buf[i] = '0' + (scancode & 1);
50 scancode >>= 1;
51 }
52
53 pr_info("scancode: 0b%s", buf);
54}

56void do_backtrace(u32 max)
57{
58 static bool is_tracing = false;
59 if (is_tracing)
60 return;
61 pr_info("Stack trace:");
62 struct frame_t
63 {
64 struct frame_t *ebp;
65 uintptr_t eip;
66 } *frame = NULL;
67
68 __asm__("movl %%ebp,%1" : "=r"(frame) : "r"(frame));
69 for (u32 i = 0; frame && i < max; i++)
70 {
71 pr_warn(" " PTR_FMT, frame->eip);
72 frame = frame->ebp;
73 }
74}

76void x86_kpanic_hook()
77{
78 pmem_freelist_dump();
79 const char *cpu_pagetable_source = current_cpu->pagetable.ptr == (uintptr_t) x86_kpg_infra ? "Kernel" : NULL;
80
81 if (current_thread)
82 {
83 pr_emph("Current task: %s (tid: %d, pid: %d)", current_process->name, current_thread->tid, current_process->pid);
84 pr_emph("Task Page Table:");
85 x86_mm_dump_page_table(x86_get_pg_infra(current_process->pagetable));
86 if (current_cpu->pagetable.ptr == current_process->pagetable.ptr)
87 cpu_pagetable_source = "Current Process";
88 }
89 else
90 {
91 pr_emph("Kernel Page Table:");
92 x86_mm_dump_page_table(x86_kpg_infra);
93 }
94
95 if (cpu_pagetable_source)
96 {
97 pr_emph("CPU Page Table: %s", cpu_pagetable_source);
98 }
99 else
100 {
101 pr_emph("CPU Page Table:");
102 x86_mm_dump_page_table(x86_get_pg_infra(current_cpu->pagetable));
103 }
104 // do_backtrace(20);
105}

107void x86_start_kernel(x86_startup_info *info)
108{
109 multiboot_info_t *mb_info = info->mb_info;
110 uintptr_t initrd_size = info->initrd_size;
111 x86_disable_interrupts();
112 console_register(&com1_console.console);
113
114 x86_gdt_init();
115 x86_idt_init();
116 x86_tss_init();
117 x86_irq_handler_init();
118
119 if (mb_info->flags & MULTIBOOT_INFO_CMDLINE)
120 strncpy(mos_cmdline, mb_info->cmdline, sizeof(mos_cmdline));
121
122 u32 count = mb_info->mmap_length / sizeof(multiboot_memory_map_t);
123 x86_mem_init(mb_info->mmap_addr, count);
124
125 x86_mm_prepare_paging();
126 current_cpu->pagetable.ptr = (uintptr_t) x86_kpg_infra;
127
128 if (initrd_size)
129 {
130 reg_t cr3 = x86_get_cr3();
131 const uintptr_t initrd_vaddr = MOS_X86_INITRD_VADDR;
132 x86_pgtable_entry *table = (x86_pgtable_entry *) (((x86_pgdir_entry *) cr3)[initrd_vaddr >> 22].page_table_paddr << 12);
133 const uintptr_t initrd_paddr = table->phys_addr << 12;
134 const size_t initrd_n_pages = ALIGN_UP_TO_PAGE(initrd_size) / MOS_PAGE_SIZE;
135 pg_map_pages(x86_kpg_infra, initrd_vaddr, initrd_paddr, initrd_n_pages, VM_GLOBAL);
136 }
137
138 // ! map the bios memory area, should it be done like this?
139 pr_info("mapping bios memory area...");
140 for (u32 i = 0; i < x86_mem_regions_count; i++)
141 if (x86_mem_regions[i].address == info->bios_region_start)
142 x86_bios_region = &x86_mem_regions[i];
143
144 x86_pmblock_t *bios_block = x86_bios_region;
145 pg_do_map_pages(x86_kpg_infra, BIOS_VADDR(bios_block->address), bios_block->address, bios_block->size_bytes / MOS_PAGE_SIZE, VM_GLOBAL);
146 bios_block->address = BIOS_VADDR(bios_block->address);
147
148 pr_info("Parsing ACPI tables...");
149 x86_acpi_init();
150 acpi_parse_madt();
151
152 pr_info("Initializing APICs...");
153 lapic_memory_setup();
154 lapic_enable();
155 ioapic_init();
156
157 pr_info("Starting APs...");
158 cpuid_print_cpu_info();
159 x86_platform.boot_cpu_id = lapic_get_id();
160 per_cpu(x86_platform.cpu)->id = lapic_get_id();
161 x86_smp_init();
162
163 x86_mm_enable_paging();
164
165 mos_kernel_mm_init(); // since then, we can use the kernel heap (kmalloc)
166
167 mos_install_kpanic_hook(x86_kpanic_hook);
168 console_register(&vga_text_mode_console);
169
170 x86_install_interrupt_handler(IRQ_TIMER, x86_timer_handler);
171 x86_install_interrupt_handler(IRQ_KEYBOARD, x86_keyboard_handler);
172 x86_install_interrupt_handler(IRQ_COM1, serial_irq_handler);
173
174 ioapic_enable_interrupt(IRQ_TIMER, 0);
175 ioapic_enable_interrupt(IRQ_KEYBOARD, 0);
176 ioapic_enable_interrupt(IRQ_COM1, 0);
177
178 if (initrd_size)
179 {
180 initrd_blockdev_t *initrd_blockdev = kmalloc(sizeof(initrd_blockdev_t));
181 initrd_blockdev->memblock = (x86_pmblock_t){ .available = true, .address = MOS_X86_INITRD_VADDR, .size_bytes = initrd_size };
182
183 initrd_blockdev_preinstall(initrd_blockdev);
184 blockdev_register(&initrd_blockdev->blockdev);
185 }
186
187 mos_start_kernel(mos_cmdline);
188}
