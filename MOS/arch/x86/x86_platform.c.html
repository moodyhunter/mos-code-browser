56void x86_keyboard_handler(u32 irq)
57{
58 MOS_ASSERT(irq == IRQ_KEYBOARD);
59 int scancode = port_inb(0x60);
60 char buf[9] = { 0 };
61 for (int i = 7; i >= 0; i--)
62 {
63 buf[i] = '0' + (scancode & 1);
64 scancode >>= 1;
65 }
66
67 pr_info("scancode: 0b%s", buf);
68}

70void do_backtrace(u32 max)
71{
72 static bool is_tracing = false;
73 if (is_tracing)
74 return;
75 pr_info("Stack trace:");
76 struct frame_t
77 {
78 struct frame_t *ebp;
79 uintptr_t eip;
80 } *frame = NULL;
81
82 __asm__("movl %%ebp,%1" : "=r"(frame) : "r"(frame));
83 for (u32 i = 0; frame && i < max; i++)
84 {
85 pr_warn(" " PTR_FMT, frame->eip);
86 frame = frame->ebp;
87 }
88}

90void x86_kpanic_hook()
91{
92 const char *cpu_pagetable_source = current_cpu->pagetable.pgd == (uintptr_t) x86_kpg_infra ? "Kernel" : NULL;
93
94 if (current_thread)
95 {
96 pr_emph("Current task: %s (tid: %d, pid: %d)", current_process->name, current_thread->tid, current_process->pid);
97 pr_emph("Task Page Table:");
98 x86_mm_dump_page_table(x86_get_pg_infra(current_process->pagetable));
99 if (current_cpu->pagetable.pgd == current_process->pagetable.pgd)
100 cpu_pagetable_source = "Current Process";
101 }
102 else
103 {
104 pr_emph("Kernel Page Table:");
105 x86_mm_dump_page_table(x86_kpg_infra);
106 }
107
108 if (cpu_pagetable_source)
109 {
110 pr_emph("CPU Page Table: %s", cpu_pagetable_source);
111 }
112 else
113 {
114 pr_emph("CPU Page Table:");
115 x86_mm_dump_page_table(x86_get_pg_infra(current_cpu->pagetable));
116 }
117 // do_backtrace(20);
118}

120void x86_start_kernel(x86_startup_info *info)
121{
122 x86_platform.k_code.vaddr = (uintptr_t) &__MOS_KERNEL_CODE_START;
123 x86_platform.k_code.npages = (ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_END) - ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_START)) / MOS_PAGE_SIZE;
124 x86_platform.k_code.paddr = (uintptr_t) &__MOS_KERNEL_CODE_START - MOS_KERNEL_START_VADDR;
125 x86_platform.k_code.flags = VM_GLOBAL | VM_READ | VM_EXEC;
126
127 x86_platform.k_rwdata.vaddr = (uintptr_t) &__MOS_KERNEL_RW_START;
128 x86_platform.k_rwdata.npages = (ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_END) - ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_START)) / MOS_PAGE_SIZE;
129 x86_platform.k_rwdata.paddr = (uintptr_t) &__MOS_KERNEL_RW_START - MOS_KERNEL_START_VADDR;
130 x86_platform.k_rwdata.flags = VM_GLOBAL | VM_READ | VM_WRITE;
131
132 x86_platform.k_rodata.vaddr = (uintptr_t) &__MOS_KERNEL_RODATA_START;
133 x86_platform.k_rodata.npages = (ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_END) - ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_START)) / MOS_PAGE_SIZE;
134 x86_platform.k_rodata.paddr = (uintptr_t) &__MOS_KERNEL_RODATA_START - MOS_KERNEL_START_VADDR;
135 x86_platform.k_rodata.flags = VM_GLOBAL | VM_READ;
136
137 console_register(&com1_console.console);
138 pr_info("mos_startup_info: initrd %zu bytes, mbinfo at: " PTR_FMT ", magic " PTR_FMT, info->initrd_size, (uintptr_t) info->mb_info, (uintptr_t) info->mb_magic);
139
140 const multiboot_info_t *mb_info = info->mb_info;
141 const uintptr_t initrd_size = info->initrd_size;
142 const uintptr_t initrd_paddr = ((x86_pgtable_entry *) (((x86_pgdir_entry *) x86_get_cr3())[MOS_X86_INITRD_VADDR >> 22].page_table_paddr << 12))->phys_addr << 12;
143
144 x86_gdt_init();
145 x86_idt_init();
146 x86_tss_init();
147 x86_irq_handler_init();
148
149 if (mb_info->flags & MULTIBOOT_INFO_CMDLINE)
150 strncpy(mos_cmdline, mb_info->cmdline, sizeof(mos_cmdline));
151
152 current_cpu->pagetable.pgd = x86_platform.kernel_pgd.pgd = (uintptr_t) x86_kpg_infra;
153 current_cpu->pagetable.um_page_map = x86_platform.kernel_pgd.um_page_map = NULL; // a kernel page table does not have a user-mode page map
154
155 const u32 memregion_count = mb_info->mmap_length / sizeof(multiboot_memory_map_t);
156 x86_mem_init(mb_info->mmap_addr, memregion_count);
157 MOS_ASSERT_X(x86_platform.mem_regions.count != memregion_count, "x86_mem_init() failed to initialize all memory regions");
158 mos_pmalloc_setup();
159
160 x86_mm_prepare_paging();
161
162 pr_info("mapping kernel space...");
163 mm_map_pages(x86_platform.kernel_pgd, x86_platform.k_code);
164 mm_map_pages(x86_platform.kernel_pgd, x86_platform.k_rodata);
165 mm_map_pages(x86_platform.kernel_pgd, x86_platform.k_rwdata);
166
167 pr_info("mapping bios memory area...");
168 mm_map_allocated_pages(x86_platform.kernel_pgd, x86_bios_block);
169 mm_map_allocated_pages(x86_platform.kernel_pgd, x86_ebda_block);
170
171 pr_info("reserving memory for AP boot...");
172 x86_smp_copy_trampoline();
173
174 x86_mm_enable_paging();
175 mos_kernel_mm_init(); // since then, we can use the kernel heap (kmalloc)
176
177 if (initrd_size)
178 {
179 vmblock_t initrd_block = (vmblock_t){
180 .vaddr = MOS_X86_INITRD_VADDR,
181 .flags = VM_READ | VM_GLOBAL,
182 .paddr = initrd_paddr,
183 .npages = ALIGN_UP_TO_PAGE(initrd_size) / MOS_PAGE_SIZE,
184 };
185 mm_map_pages(x86_platform.kernel_pgd, initrd_block);
186
187 initrd_blockdev_t *initrd_blockdev = kzalloc(sizeof(initrd_blockdev_t));
188 initrd_blockdev->blockdev = (blockdev_t){ .name = "initrd", .read = initrd_read };
189 initrd_blockdev->memblock = (memregion_t){ .available = true, .address = MOS_X86_INITRD_VADDR, .size_bytes = initrd_size };
190 blockdev_register(&initrd_blockdev->blockdev);
191 }
192
193 pr_info("Parsing ACPI tables...");
194 acpi_rsdp_t *rsdp = acpi_find_rsdp(BIOS_VADDR(X86_EBDA_MEMREGION_PADDR), EBDA_MEMREGION_SIZE);
195 if (!rsdp)
196 {
197 rsdp = acpi_find_rsdp(BIOS_VADDR(X86_BIOS_MEMREGION_PADDR), BIOS_MEMREGION_SIZE);
198 if (!rsdp)
199 mos_panic("RSDP not found");
200 }
201
202 // find the RSDT and map the whole region it's in
203 for (u32 i = 0; i < x86_platform.mem_regions.count; i++)
204 {
205 memregion_t *this_region = &x86_platform.mem_regions.regions[i];
206 if (rsdp->xsdt_addr)
207 mos_panic("XSDT not supported");
208 if (rsdp->v1.rsdt_addr >= this_region->address && rsdp->v1.rsdt_addr < this_region->address + this_region->size_bytes)
209 {
210 vmblock_t bios_vmblock = (vmblock_t){
211 .npages = this_region->size_bytes / MOS_PAGE_SIZE,
212 .vaddr = BIOS_VADDR(this_region->address),
213 .flags = VM_READ | VM_GLOBAL,
214 .paddr = this_region->address,
215 };
216 mm_map_allocated_pages(x86_platform.kernel_pgd, bios_vmblock);
217 this_region->address = BIOS_VADDR(this_region->address);
218 break;
219 }
220 }
221
222 acpi_parse_rsdt(rsdp);
223
224 pr_info("Initializing APICs...");
225 madt_parse_table();
226 lapic_memory_setup();
227 lapic_enable();
228 pic_remap_irq();
229 ioapic_init();
230
231 pr_info("Starting APs...");
232 current_cpu->id = x86_platform.boot_cpu_id = lapic_get_id();
233 x86_smp_start_all();
234
235 mos_install_kpanic_hook(x86_kpanic_hook);
236
237 x86_install_interrupt_handler(IRQ_TIMER, x86_timer_handler);
238 x86_install_interrupt_handler(IRQ_KEYBOARD, x86_keyboard_handler);
239 x86_install_interrupt_handler(IRQ_COM1, serial_irq_handler);
240
241 ioapic_enable_interrupt(IRQ_TIMER, 0);
242 ioapic_enable_interrupt(IRQ_KEYBOARD, 0);
243 ioapic_enable_interrupt(IRQ_COM1, 0);
244
245 // TODO: move this to a userspace program
246 pr_info("paging: mapping video memory...");
247 vmblock_t video_block = (vmblock_t){
248 .npages = 1,
249 .vaddr = BIOS_VADDR(X86_VIDEO_DEVICE_PADDR),
250 .flags = VM_RW | VM_GLOBAL,
251 .paddr = X86_VIDEO_DEVICE_PADDR,
252 };
253 mm_map_allocated_pages(current_cpu->pagetable, video_block);
254 console_register(&vga_text_mode_console);
255
256 mos_start_kernel(mos_cmdline);
257}
