20should_inline bool verify_sdt_checksum(acpi_sdt_header_t *tableHeader)
21{
22 u8 sum = 0;
23 for (u32 i = 0; i < tableHeader->length; i++)
24 sum += ((char *) tableHeader)[i];
25 return sum == 0;
26}

28void acpi_parse_rsdt(acpi_rsdp_t *rsdp)
29{
30 pr_info("Initializing ACPI with RSDP at " PTR_FMT, (uintptr_t) rsdp);
31 // !! "MUST" USE XSDT IF FOUND !!
32 if (rsdp->xsdt_addr)
33 mos_panic("XSDT not supported");
34
35 x86_acpi_rsdt = container_of(BIOS_VADDR(rsdp->v1.rsdt_addr), acpi_rsdt_t, sdt_header);
36 if (!verify_sdt_checksum(&x86_acpi_rsdt->sdt_header))
37 mos_panic("RSDT checksum error");
38
39 if (strncmp(x86_acpi_rsdt->sdt_header.signature, "RSDT", 4) != 0)
40 mos_panic("RSDT signature mismatch");
41
42 const size_t num_headers = (x86_acpi_rsdt->sdt_header.length - sizeof(acpi_sdt_header_t)) / sizeof(u32);
43 for (size_t i = 0; i < num_headers; i++)
44 {
45 acpi_sdt_header_t *addr = (acpi_sdt_header_t *) BIOS_VADDR((uintptr_t) x86_acpi_rsdt->sdts[i]);
46
47 if (strncmp(addr->signature, ACPI_SIGNATURE_FADT, 4) == 0)
48 {
49 x86_acpi_fadt = container_of(addr, acpi_fadt_t, sdt_header);
50 if (!verify_sdt_checksum(&x86_acpi_fadt->sdt_header))
51 mos_panic("FADT checksum error");
52 pr_info2("acpi: FADT at %p", (void *) x86_acpi_fadt);
53 }
54 else if (strncmp(addr->signature, ACPI_SIGNATURE_MADT, 4) == 0)
55 {
56 x86_acpi_madt = container_of(addr, acpi_madt_t, sdt_header);
57 if (!verify_sdt_checksum(&x86_acpi_madt->sdt_header))
58 mos_panic("MADT checksum error");
59 pr_info2("acpi: MADT at %p", (void *) x86_acpi_madt);
60 }
61 else if (strncmp(addr->signature, ACPI_SIGNATURE_HPET, 4) == 0)
62 {
63 x86_acpi_hpet = container_of(addr, acpi_hpet_t, header);
64 MOS_WARNING_PUSH
65 MOS_WARNING_DISABLE("-Waddress-of-packed-member")
66 if (!verify_sdt_checksum(&x86_acpi_hpet->header))
67 mos_panic("HPET checksum error");
68 MOS_WARNING_POP
69 pr_info2("acpi: HPET at %p", (void *) x86_acpi_hpet);
70 }
71 else
72 {
73 pr_info2("acpi: unknown entry %.4s", addr->signature);
74 }
75 }
76}

78acpi_rsdp_t *acpi_find_rsdp(uintptr_t start, size_t size)
79{
80 for (uintptr_t addr = start; addr < start + size; addr += 0x10)
81 {
82 if (strncmp((const char *) addr, ACPI_SIGNATURE_RSDP, 8) == 0)
83 {
84 pr_info2("ACPI: RSDP magic at %p", (void *) addr);
85 acpi_rsdp_t *rsdp = (acpi_rsdp_t *) addr;
86
87 // check the checksum
88 u8 sum = 0;
89 for (u32 i = 0; i < sizeof(acpi_rsdp_v1_t); i++)
90 sum += ((u8 *) rsdp)[i];
91
92 if (sum != 0)
93 {
94 pr_info2("ACPI: RSDP checksum failed");
95 continue;
96 }
97 pr_info2("ACPI: RSDP checksum ok");
98 pr_info("ACPI: oem: '%s', revision: %d", rsdp->v1.oem_id, rsdp->v1.revision);
99
100 if (rsdp->v1.revision != 0)
101 mos_panic("ACPI: RSDP revision %d not supported", rsdp->v1.revision);
102
103 return rsdp;
104 }
105 }
106 return NULL;
