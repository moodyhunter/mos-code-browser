22should_inline bool verify_sdt_checksum(acpi_sdt_header_t *tableHeader)
23{
24 u8 sum = 0;
25 for (u32 i = 0; i < tableHeader->length; i++)
26 sum += ((char *) tableHeader)[i];
27 return sum == 0;
28}

30void x86_acpi_init()
31{
32 acpi_rsdp_t *rsdp = find_acpi_rsdp(X86_EBDA_MEMREGION_PADDR | BIOS_VADDR_MASK, EBDA_MEMREGION_SIZE);
33 if (!rsdp)
34 {
35 rsdp = find_acpi_rsdp(X86_BIOS_MEMREGION_PADDR | BIOS_VADDR_MASK, BIOS_MEMREGION_SIZE);
36 if (!rsdp)
37 mos_panic("RSDP not found");
38 }
39
40 // !! "MUST" USE XSDT IF FOUND !!
41 x86_acpi_rsdt = container_of(BIOS_VADDR(rsdp->v1.rsdt_addr), acpi_rsdt_t, sdt_header);
42 if (!verify_sdt_checksum(&x86_acpi_rsdt->sdt_header))
43 mos_panic("RSDT checksum error");
44
45 MOS_ASSERT(strncmp(x86_acpi_rsdt->sdt_header.signature, "RSDT", 4) == 0);
46
47 const size_t count = (x86_acpi_rsdt->sdt_header.length - sizeof(acpi_sdt_header_t)) / sizeof(u32);
48 for (size_t i = 0; i < count; i++)
49 {
50 acpi_sdt_header_t *addr = (acpi_sdt_header_t *) BIOS_VADDR((uintptr_t) x86_acpi_rsdt->sdts[i]);
51 pr_info2("acpi: RSDT entry %zu: %.4s", i, addr->signature);
52
53 if (strncmp(addr->signature, ACPI_SIGNATURE_FADT, 4) == 0)
54 {
55 x86_acpi_fadt = container_of(addr, acpi_fadt_t, sdt_header);
56 if (!verify_sdt_checksum(&x86_acpi_fadt->sdt_header))
57 mos_panic("FADT checksum error");
58 }
59 else if (strncmp(addr->signature, ACPI_SIGNATURE_MADT, 4) == 0)
60 {
61 x86_acpi_madt = container_of(addr, acpi_madt_t, sdt_header);
62 if (!verify_sdt_checksum(&x86_acpi_madt->sdt_header))
63 mos_panic("MADT checksum error");
64 }
65 else if (strncmp(addr->signature, ACPI_SIGNATURE_HPET, 4) == 0)
66 {
67 x86_acpi_hpet = container_of(addr, acpi_hpet_t, sdt_header);
68 MOS_WARNING_PUSH
69 MOS_WARNING_DISABLE("-Waddress-of-packed-member")
70 if (!verify_sdt_checksum(&x86_acpi_hpet->sdt_header))
71 mos_panic("HPET checksum error");
72 MOS_WARNING_POP
73 }
74 else
75 {
76 pr_info2("acpi: unknown entry %.4s", addr->signature);
77 }
78 }
79
80 if (!x86_acpi_madt)
81 mos_panic("MADT not found");
82
83 madt_entry_foreach(entry, x86_acpi_madt)
84 {
85 switch (entry->type)
86 {
87 case 0:
88 {
89 acpi_madt_et0_lapic_t *lapic = container_of(entry, acpi_madt_et0_lapic_t, madt_entry_header);
90 pr_info2("acpi: MADT entry LAPIC [%p]", (void *) lapic);
91 break;
92 }
93 case 1:
94 {
95 acpi_madt_et1_ioapic_t *ioapic = container_of(entry, acpi_madt_et1_ioapic_t, madt_entry_header);
96 pr_info2("acpi: MADT entry IOAPIC [%p]", (void *) ioapic);
97 break;
98 }
99 case 2:
100 {
101 acpi_madt_et2_ioapic_override_t *int_override = container_of(entry, acpi_madt_et2_ioapic_override_t, madt_entry_header);
102 pr_info2("acpi: MADT entry IOAPIC override [%p]", (void *) int_override);
103 break;
104 }
105 case 3:
106 {
107 acpi_madt_et3_ioapic_nmi_t *int_override = container_of(entry, acpi_madt_et3_ioapic_nmi_t, madt_entry_header);
108 pr_info2("acpi: MADT entry IOAPIC NMI [%p]", (void *) int_override);
109 break;
110 }
111 case 4:
112 {
113 acpi_madt_et4_lapic_nmi_t *nmi = container_of(entry, acpi_madt_et4_lapic_nmi_t, madt_entry_header);
114 pr_info2("acpi: MADT entry LAPIC NMI [%p]", (void *) nmi);
115 break;
116 }
117 case 5:
118 {
119 acpi_madt_et5_lapic_addr_t *local_apic_nmi = container_of(entry, acpi_madt_et5_lapic_addr_t, madt_entry_header);
120 pr_info2("acpi: MADT entry LAPIC address override [%p]", (void *) local_apic_nmi);
121 break;
122 }
123 case 9:
124 {
125 acpi_madt_et9_lx2apic_t *local_sapic_override = container_of(entry, acpi_madt_et9_lx2apic_t, madt_entry_header);
126 pr_info2("acpi: MADT entry local x2 SAPIC override [%p]", (void *) local_sapic_override);
127 break;
128 }
129 }
130 }
131}

133acpi_rsdp_t *find_acpi_rsdp(uintptr_t start, size_t size)
134{
135 for (uintptr_t addr = start; addr < start + size; addr += 0x10)
136 {
137 if (strncmp((const char *) addr, ACPI_SIGNATURE_RSDP, 8) == 0)
138 {
139 pr_info2("ACPI: RSDP magic at %p", (void *) addr);
140 acpi_rsdp_t *rsdp = (acpi_rsdp_t *) addr;
141
142 // check the checksum
143 u8 sum = 0;
144 for (u32 i = 0; i < sizeof(acpi_rsdp_v1_t); i++)
145 sum += ((u8 *) rsdp)[i];
146
147 if (sum != 0)
148 {
149 pr_info2("ACPI: RSDP checksum failed");
150 continue;
151 }
152 pr_info2("ACPI: RSDP checksum ok");
153 pr_info("ACPI: oem: '%s', revision: %d", rsdp->v1.oem_id, rsdp->v1.revision);
154
155 if (rsdp->v1.revision != 0)
156 mos_panic("ACPI: RSDP revision %d not supported", rsdp->v1.revision);
157
158 return rsdp;
159 }
160 }
161 return NULL;
162}
