10acpi_madt_t *x86_acpi_madt = NULL;
11u32 x86_cpu_lapic[MOS_MAX_CPU_COUNT] = { 0 };
12uintptr_t x86_ioapic_address = 0;
13
14#define IOAPIC_IRQ_OVERRIDE_MAX 255 // u8 can hold up to 255
15static u32 ioapic_irq_override[IOAPIC_IRQ_OVERRIDE_MAX] = { 0 };
16
17u32 x86_ioapic_get_irq_override(u32 irq)
18{
19 if (irq >= IOAPIC_IRQ_OVERRIDE_MAX)
20 return irq;
21 return ioapic_irq_override[irq];
22}
23
24void acpi_parse_madt()
25{
26 if (!x86_acpi_madt)
27 mos_panic("MADT not found");
28
29 x86_platform.num_cpus = 0;
30
31 for (u8 i = 0; i < IOAPIC_IRQ_OVERRIDE_MAX; i++)
32 ioapic_irq_override[i] = i; // Default: no override
33
34 madt_entry_foreach(entry, x86_acpi_madt)
35 {
36 switch (entry->type)
37 {
38 case 0:
39 {
40 acpi_madt_et0_lapic_t *lapic = container_of(entry, acpi_madt_et0_lapic_t, header);
41 pr_info2("acpi: MADT entry LAPIC [%p], id=%u, processor=%u, flags=0x%x", (void *) lapic, lapic->apic_id, lapic->processor_id, lapic->flags);
42
43 if (unlikely(lapic->processor_id >= MOS_MAX_CPU_COUNT))
44 mos_panic("Too many CPUs");
45
46 if (unlikely(x86_cpu_lapic[lapic->processor_id]))
47 mos_panic("Multiple LAPICs for the same processor not supported");
48
49 x86_cpu_lapic[lapic->processor_id] = lapic->apic_id;
50 x86_platform.num_cpus++;
51 break;
52 }
53 case 1:
54 {
55 acpi_madt_et1_ioapic_t *ioapic = container_of(entry, acpi_madt_et1_ioapic_t, header);
56 pr_info2("acpi: MADT entry IOAPIC [%p], id=%u, address=%x, global_irq_base=%u", (void *) ioapic, ioapic->id, ioapic->address, ioapic->global_intr_base);
57 if (unlikely(x86_ioapic_address))
58 mos_panic("Multiple IOAPICs not supported");
59 x86_ioapic_address = ioapic->address;
60 break;
61 }
62 case 2:
63 {
64 acpi_madt_et2_ioapic_override_t *int_override = container_of(entry, acpi_madt_et2_ioapic_override_t, header);
65 pr_info2("acpi: MADT entry IOAPIC override [%p], bus=%u, source=%u, global_irq=%u, flags=0x%x", (void *) int_override, int_override->bus_source,
66 int_override->irq_source, int_override->global_intr, int_override->flags);
67
68 if (unlikely(int_override->bus_source != 0))
69 mos_panic("IOAPIC override for non-ISA bus not supported");
70
71 if (unlikely(int_override->irq_source >= IOAPIC_IRQ_OVERRIDE_MAX))
72 mos_panic("IOAPIC override for IRQ >= 16 not supported");
73
74 if (unlikely(ioapic_irq_override[int_override->irq_source] != int_override->irq_source))
75 mos_panic("Multiple IOAPIC overrides for the same IRQ not supported");
76
77 ioapic_irq_override[int_override->irq_source] = int_override->global_intr;
78 break;
79 }
80 case 3:
81 {
82 acpi_madt_et3_ioapic_nmi_t *int_override = container_of(entry, acpi_madt_et3_ioapic_nmi_t, header);
83 pr_info2("acpi: MADT entry IOAPIC NMI [%p], nmi_source=%u, global_irq=%u, flags=0x%x", (void *) int_override, int_override->nmi_source,
84 int_override->global_irq, int_override->flags);
85 // TODO
86 break;
87 }
88 case 4:
89 {
90 acpi_madt_et4_lapic_nmi_t *nmi = container_of(entry, acpi_madt_et4_lapic_nmi_t, header);
91 pr_info2("acpi: MADT entry LAPIC NMI [%p], processor=%u, flags=0x%x, lint=%u", (void *) nmi, nmi->processor_id, nmi->flags, nmi->lint_number);
92 // TODO
93 break;
94 }
95 case 5:
96 {
97 acpi_madt_et5_lapic_addr_t *local_apic_nmi = container_of(entry, acpi_madt_et5_lapic_addr_t, header);
98 pr_info2("acpi: MADT entry LAPIC address override [%p], address=%p", (void *) local_apic_nmi, (void *) local_apic_nmi->lapic_paddr);
99 // TODO
100 break;
101 }
102 case 9:
103 {
104 acpi_madt_et9_lx2apic_t *local_sapic_override = container_of(entry, acpi_madt_et9_lx2apic_t, header);
105 pr_info2("acpi: MADT entry local x2 SAPIC override [%p], x2apic_id=%u, flags=0x%x, acpi_id=%u", (void *) local_sapic_override,
106 local_sapic_override->processor_lx2apic_id, local_sapic_override->flags, local_sapic_override->acpi_id);
107 // TODO
108 break;
109 }
110 }
111 }
112
113 pr_info("acpi: platform has %u cpu(s)", x86_platform.num_cpus);
114 return;
115}
