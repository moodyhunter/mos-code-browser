14#define PAGEMAP_MAP(map, index) map[index / PAGEMAP_WIDTH] |= 1 << (index % PAGEMAP_WIDTH)
15#define PAGEMAP_UNMAP(map, index) map[index / PAGEMAP_WIDTH] &= ~(1 << (index % PAGEMAP_WIDTH))
16#define BIT_IS_SET(byte, bit) ((byte) & (1 << (bit)))
17
18always_inline void pg_flush_tlb(uintptr_t vaddr)
19{
20 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
21}

23void *pg_page_alloc(x86_pg_infra_t *pg, size_t n_page, pagealloc_flags flags)
24{
25 vm_flags pflags = VM_WRITE;
26 // always allocate after the end of the kernel pages
27 uintptr_t vaddr_begin;
28
29 if (flags & PGALLOC_KHEAP)
30 {
31 // ...so that we can use the kernel heap everywhere?
32 pflags |= VM_GLOBAL;
33 vaddr_begin = MOS_X86_HEAP_BASE_VADDR;
34 }
35 else
36 {
37 vaddr_begin = mos_kernel_end;
38 }
39
40 // simply rename the variable, we are dealing with bitmaps
41 size_t n_bits = n_page;
42 size_t n_zero_bits = 0;
43
44 u8 target_bit = 0;
45 uintptr_t vaddr_map_bit_begin = vaddr_begin / MOS_PAGE_SIZE / PAGEMAP_WIDTH + 1;
46 for (size_t i = vaddr_map_bit_begin; n_zero_bits < n_bits; i++)
47 {
48 if (i >= MM_PAGE_MAP_SIZE)
49 {
50 mos_warn("failed to allocate %zu pages", n_page);
51 return NULL;
52 }
53 pagemap_line_t current_byte = pg->page_map[i];
54
55 if (current_byte == 0)
56 {
57 n_zero_bits += PAGEMAP_WIDTH;
58 continue;
59 }
60 if (current_byte == (pagemap_line_t) ~0)
61 {
62 vaddr_map_bit_begin = i + 1;
63 continue;
64 }
65
66 for (size_t bit = 0; bit < PAGEMAP_WIDTH; bit++)
67 {
68 if (!BIT_IS_SET(current_byte, bit))
69 n_zero_bits++;
70 else
71 n_zero_bits = 0, target_bit = bit + 1, vaddr_map_bit_begin = i;
72 }
73 }
74
75 size_t page_i = vaddr_map_bit_begin * PAGEMAP_WIDTH + target_bit;
76 uintptr_t vaddr = page_i * MOS_PAGE_SIZE;
77 mos_debug("paging: allocating page %zu to %zu (aka starting at " PTR_FMT ")", page_i, page_i + n_page, vaddr);
78
79 uintptr_t paddr = pmem_freelist_find_free(n_page);
80
81 if (paddr == 0)
82 {
83 mos_panic("OOM");
84 return NULL;
85 }
86
87 pg_map_pages(pg, vaddr, paddr, n_page, pflags);
88 return (void *) vaddr;
89}

91bool pg_page_free(x86_pg_infra_t *pg, uintptr_t vptr, size_t n_page)
92{
93 size_t page_index = vptr / MOS_PAGE_SIZE;
94 mos_debug("paging: freeing %zu to %zu", page_index, page_index + n_page);
95 pg_unmap_pages(pg, vptr, n_page);
96 return true;
97}

99void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
100{
101 mos_debug("paging: setting flags [%x] to [" PTR_FMT "] +%zu pages", flags, vaddr, n);
102 size_t start_page = vaddr / MOS_PAGE_SIZE;
103 for (size_t i = 0; i < n; i++)
104 {
105 size_t page_i = start_page + i;
106 size_t pgd_i = page_i / 1024;
107
108 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
109 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
110
111 pg->pgdir[pgd_i].writable = flags & VM_WRITE;
112 pg->pgtable[page_i].writable = flags & VM_WRITE;
113
114 pg->pgdir[pgd_i].usermode = flags & VM_USERMODE;
115 pg->pgtable[page_i].usermode = flags & VM_USERMODE;
116
117 pg->pgdir[pgd_i].cache_disabled = flags & VM_CACHE_DISABLED;
118 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
119
120 pg->pgtable[page_i].global = flags & VM_GLOBAL;
121 pg_flush_tlb(vaddr);
122 }
123}

125void pg_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
126{
127 pmem_freelist_remove_region(paddr_start, n_page * MOS_PAGE_SIZE);
128 pg_do_map_pages(pg, vaddr_start, paddr_start, n_page, flags);
129}

131void pg_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
132{
133 uintptr_t paddr = pg_page_get_mapped_paddr(pg, vaddr_start);
134 pg_do_unmap_pages(pg, vaddr_start, n_page);
135 pmem_freelist_add_region(paddr, n_page * MOS_PAGE_SIZE);
136}

138void pg_copy_page(x86_pg_infra_t *from_pg, x86_pg_infra_t *to_pg, uintptr_t start_vaddr, size_t n_page)
139{
140 for (size_t i = 0; i < n_page; i++)
141 {
142 uintptr_t vaddr = start_vaddr + i * MOS_PAGE_SIZE;
143 uintptr_t paddr = pg_page_get_mapped_paddr(from_pg, vaddr);
144 pg_map_pages(to_pg, vaddr, paddr, 1, VM_WRITE); // !! TODO: copy flags
145 }
146}

148void pg_do_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
149{
150 mos_debug("paging: mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table %lu", n_page, vaddr_start, paddr_start, vaddr_start / MOS_PAGE_SIZE);
151 for (size_t i = 0; i < n_page; i++)
152 pg_do_map_page(pg, vaddr_start + i * MOS_PAGE_SIZE, paddr_start + i * MOS_PAGE_SIZE, flags);
153}

155void pg_do_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
156{
157 mos_debug("paging: unmapping %zu pages starting at " PTR_FMT " @ table %lu", n_page, vaddr_start, vaddr_start / MOS_PAGE_SIZE);
158 for (size_t i = 0; i < n_page; i++)
159 pg_do_unmap_page(pg, vaddr_start + i * MOS_PAGE_SIZE);
160}

162void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
163{
164 // ensure the page is aligned to 4096
165 MOS_ASSERT_X(paddr < X86_MAX_MEM_SIZE, "physical address out of bounds");
166 MOS_ASSERT_X(flags < 0x100, "invalid flags");
167 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
168
169 // ! todo: ensure the offsets are correct for both paddr and vaddr
170 int page_dir_index = vaddr >> 22;
171 int page_table_index = vaddr >> 12 & 0x3ff; // mask out the lower 12 bits
172
173 u32 pte_index = page_dir_index * 1024 + page_table_index;
174
175 x86_pgdir_entry *this_dir = &pg->pgdir[page_dir_index];
176 x86_pgtable_entry *this_table = &pg->pgtable[pte_index];
177 MOS_ASSERT_X(this_table->present == false, "page is already mapped");
178
179 // TODO: dynamically allocate a page table if it doesn't exist
180
181 if (unlikely(!this_dir->present))
182 {
183 this_dir->present = true;
184
185 // kernel page tables are identity mapped
186 uintptr_t table_paddr;
187 if (pg == x86_kpg_infra)
188 table_paddr = (uintptr_t) this_table - MOS_KERNEL_START_VADDR;
189 else
190 table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table);
191 this_dir->page_table_paddr = table_paddr >> 12;
192 }
193
194 this_table->present = true;
195 this_table->phys_addr = (uintptr_t) paddr >> 12;
196
197 this_dir->writable = flags & VM_WRITE;
198 this_table->writable = flags & VM_WRITE;
199
200 this_dir->usermode = flags & VM_USERMODE;
201 this_table->usermode = flags & VM_USERMODE;
202
203 this_dir->cache_disabled = flags & VM_CACHE_DISABLED;
204 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
205
206 this_table->global = flags & VM_GLOBAL;
207
208 // update the mm_page_map
209 PAGEMAP_MAP(pg->page_map, pte_index);
210 pg_flush_tlb(vaddr);
211}

213void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
214{
215 int page_dir_index = vaddr >> 22;
216 int page_table_index = vaddr >> 12 & 0x3ff;
217
218 x86_pgdir_entry *page_dir = &pg->pgdir[page_dir_index];
219 if (unlikely(!page_dir->present))
220 {
221 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
222 return;
223 }
224
225 x86_pgtable_entry *page_table = &pg->pgtable[page_dir_index * 1024 + page_table_index];
226 page_table->present = false;
227
228 // update the mm page map
229 u32 pte_index = page_dir_index * 1024 + page_table_index;
230 PAGEMAP_UNMAP(pg->page_map, pte_index);
231 pg_flush_tlb(vaddr);
232}

234// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
235uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
236{
237 int page_dir_index = vaddr >> 22;
238 int page_table_index = vaddr >> 12 & 0x3ff;
239 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
240
241 if (unlikely(!page_dir->present))
242 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
243
244 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
245 if (unlikely(!page_table->present))
246 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
247
248 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
249}
