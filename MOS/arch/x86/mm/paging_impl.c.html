14#define PAGEMAP_MAP(map, index) map[index / PAGEMAP_WIDTH] |= 1 << (index % PAGEMAP_WIDTH)
15#define PAGEMAP_UNMAP(map, index) map[index / PAGEMAP_WIDTH] &= ~(1 << (index % PAGEMAP_WIDTH))
16#define BIT_IS_SET(byte, bit) ((byte) & (1 << (bit)))
17
18always_inline void pg_flush_tlb(uintptr_t vaddr)
19{
20 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
21}

23vmblock_t pg_page_alloc(x86_pg_infra_t *pg, size_t n_page, pgalloc_flags flags)
24{
25 vm_flags pflags = VM_WRITE;
26 // always allocate after the end of the kernel pages
27 uintptr_t vaddr_begin;
28
29 if (flags & PGALLOC_KHEAP)
30 {
31 // ...so that we can use the kernel heap everywhere?
32 pflags |= VM_GLOBAL;
33 vaddr_begin = MOS_X86_HEAP_BASE_VADDR;
34 }
35 else
36 {
37#pragma message "TODO: Why?"
38 vaddr_begin = mos_kernel_end;
39 }
40
41 // simply rename the variable, we are dealing with bitmaps
42 size_t n_bits = n_page;
43 size_t n_zero_bits = 0;
44
45 u8 target_bit = 0;
46 uintptr_t vaddr_map_bit_begin = vaddr_begin / MOS_PAGE_SIZE / PAGEMAP_WIDTH + 1;
47 for (size_t i = vaddr_map_bit_begin; n_zero_bits < n_bits; i++)
48 {
49 if (i >= MM_PAGE_MAP_SIZE)
50 {
51 mos_warn("failed to allocate %zu pages", n_page);
52 return (vmblock_t){ .mem.available = false };
53 }
54 pagemap_line_t current_byte = pg->page_map[i];
55
56 if (current_byte == 0)
57 {
58 n_zero_bits += PAGEMAP_WIDTH;
59 continue;
60 }
61 if (current_byte == (pagemap_line_t) ~0)
62 {
63 vaddr_map_bit_begin = i + 1;
64 continue;
65 }
66
67 for (size_t bit = 0; bit < PAGEMAP_WIDTH; bit++)
68 {
69 if (!BIT_IS_SET(current_byte, bit))
70 n_zero_bits++;
71 else
72 n_zero_bits = 0, target_bit = bit + 1, vaddr_map_bit_begin = i;
73 }
74 }
75
76 size_t page_i = vaddr_map_bit_begin * PAGEMAP_WIDTH + target_bit;
77 uintptr_t vaddr = page_i * MOS_PAGE_SIZE;
78 mos_debug("paging: allocating page %zu to %zu (aka starting at " PTR_FMT ")", page_i, page_i + n_page, vaddr);
79
80 uintptr_t paddr = pmem_freelist_find_free(n_page);
81
82 if (paddr == 0)
83 {
84 mos_panic("OOM");
85 return (vmblock_t){ .mem.available = false };
86 }
87
88 pg_map_pages(pg, vaddr, paddr, n_page, pflags);
89 vmblock_t block = {
90 .mem.available = true,
91 .mem.vaddr = vaddr,
92 .mem.paddr = paddr,
93 .mem.size_bytes = n_page * MOS_PAGE_SIZE,
94 .flags = pflags,
95 };
96 return block;
97}

99bool pg_page_free(x86_pg_infra_t *pg, uintptr_t vptr, size_t n_page)
100{
101 size_t page_index = vptr / MOS_PAGE_SIZE;
102 mos_debug("paging: freeing %zu to %zu", page_index, page_index + n_page);
103 pg_unmap_pages(pg, vptr, n_page);
104 return true;
105}

107void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
108{
109 mos_debug("paging: setting flags [%x] to [" PTR_FMT "] +%zu pages", flags, vaddr, n);
110 size_t start_page = vaddr / MOS_PAGE_SIZE;
111 for (size_t i = 0; i < n; i++)
112 {
113 size_t page_i = start_page + i;
114 size_t pgd_i = page_i / 1024;
115
116 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
117 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
118
119 pg->pgdir[pgd_i].writable = flags & VM_WRITE;
120 pg->pgtable[page_i].writable = flags & VM_WRITE;
121
122 pg->pgdir[pgd_i].usermode = flags & VM_USERMODE;
123 pg->pgtable[page_i].usermode = flags & VM_USERMODE;
124
125 pg->pgdir[pgd_i].cache_disabled = flags & VM_CACHE_DISABLED;
126 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
127
128 pg->pgtable[page_i].global = flags & VM_GLOBAL;
129 pg_flush_tlb(vaddr);
130 }
131}

133void pg_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
134{
135 pmem_freelist_remove_region(paddr_start, n_page * MOS_PAGE_SIZE);
136 pg_do_map_pages(pg, vaddr_start, paddr_start, n_page, flags);
137}

139void pg_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
140{
141 uintptr_t paddr = pg_page_get_mapped_paddr(pg, vaddr_start);
142 pg_do_unmap_pages(pg, vaddr_start, n_page);
143 pmem_freelist_add_region(paddr, n_page * MOS_PAGE_SIZE);
144}

146void pg_copy_page(x86_pg_infra_t *from_pg, x86_pg_infra_t *to_pg, uintptr_t start_vaddr, size_t n_page)
147{
148 for (size_t i = 0; i < n_page; i++)
149 {
150 uintptr_t vaddr = start_vaddr + i * MOS_PAGE_SIZE;
151 uintptr_t paddr = pg_page_get_mapped_paddr(from_pg, vaddr);
152 pg_map_pages(to_pg, vaddr, paddr, 1, VM_WRITE); // !! TODO: copy flags
153 }
154}

156void pg_do_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
157{
158 mos_debug("paging: mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table %lu", n_page, vaddr_start, paddr_start, vaddr_start / MOS_PAGE_SIZE);
159 for (size_t i = 0; i < n_page; i++)
160 pg_do_map_page(pg, vaddr_start + i * MOS_PAGE_SIZE, paddr_start + i * MOS_PAGE_SIZE, flags);
161}

163void pg_do_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
164{
165 mos_debug("paging: unmapping %zu pages starting at " PTR_FMT " @ table %lu", n_page, vaddr_start, vaddr_start / MOS_PAGE_SIZE);
166 for (size_t i = 0; i < n_page; i++)
167 pg_do_unmap_page(pg, vaddr_start + i * MOS_PAGE_SIZE);

170void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
171{
172 // ensure the page is aligned to 4096
173 MOS_ASSERT_X(paddr < X86_MAX_MEM_SIZE, "physical address out of bounds");
174 MOS_ASSERT_X(flags < 0x100, "invalid flags");
175 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
176
177 // ! todo: ensure the offsets are correct for both paddr and vaddr
178 int page_dir_index = vaddr >> 22;
179 int page_table_index = vaddr >> 12 & 0x3ff; // mask out the lower 12 bits
180
181 u32 pte_index = page_dir_index * 1024 + page_table_index;
182
183 x86_pgdir_entry *this_dir = &pg->pgdir[page_dir_index];
184 x86_pgtable_entry *this_table = &pg->pgtable[pte_index];
185 MOS_ASSERT_X(this_table->present == false, "page is already mapped");
186
187 // TODO: dynamically allocate a page table if it doesn't exist
188
189 if (unlikely(!this_dir->present))
190 {
191 this_dir->present = true;
192
193 // kernel page tables are identity mapped
194 uintptr_t table_paddr;
195 if (pg == x86_kpg_infra)
196 table_paddr = (uintptr_t) this_table - MOS_KERNEL_START_VADDR;
197 else
198 table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table);
199 this_dir->page_table_paddr = table_paddr >> 12;
200 }
201
202 this_table->present = true;
203 this_table->phys_addr = (uintptr_t) paddr >> 12;
204
205 this_dir->writable = flags & VM_WRITE;
206 this_table->writable = flags & VM_WRITE;
207
208 this_dir->usermode = flags & VM_USERMODE;
209 this_table->usermode = flags & VM_USERMODE;
210
211 this_dir->cache_disabled = flags & VM_CACHE_DISABLED;
212 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
213
214 this_table->global = flags & VM_GLOBAL;
215
216 // update the mm_page_map
217 PAGEMAP_MAP(pg->page_map, pte_index);
218 pg_flush_tlb(vaddr);
219}

221void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
222{
223 int page_dir_index = vaddr >> 22;
224 int page_table_index = vaddr >> 12 & 0x3ff;
225
226 x86_pgdir_entry *page_dir = &pg->pgdir[page_dir_index];
227 if (unlikely(!page_dir->present))
228 {
229 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
230 return;
231 }
232
233 x86_pgtable_entry *page_table = &pg->pgtable[page_dir_index * 1024 + page_table_index];
234 page_table->present = false;
235
236 // update the mm page map
237 u32 pte_index = page_dir_index * 1024 + page_table_index;
238 PAGEMAP_UNMAP(pg->page_map, pte_index);
239 pg_flush_tlb(vaddr);
240}

242// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
243uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
244{
245 int page_dir_index = vaddr >> 22;
246 int page_table_index = vaddr >> 12 & 0x3ff;
247 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
248 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
249
250 if (page_dir_index >= 768)
251 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
252
253 if (unlikely(!page_dir->present))
254 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
255
256 if (unlikely(!page_table->present))
257 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
258
259 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
260}
