14#define PAGEMAP_MAP(map, index) map[index / PAGEMAP_WIDTH] |= 1 << (index % PAGEMAP_WIDTH)
15#define PAGEMAP_UNMAP(map, index) map[index / PAGEMAP_WIDTH] &= ~(1 << (index % PAGEMAP_WIDTH))
16#define PAGEMAP_IS_SET(map, index) (map[index / PAGEMAP_WIDTH] & (1 << (index % PAGEMAP_WIDTH)))
17#define BIT_IS_SET(byte, bit) ((byte) & (1 << (bit)))

19always_inline void pg_flush_tlb(uintptr_t vaddr)
20{
21 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
22}

24vmblock_t pg_page_alloc(x86_pg_infra_t *pg, size_t n_page, pgalloc_hints hints, vm_flags vm_flag)
25{
26 vmblock_t block = pg_page_get_free(pg, n_page, hints);
27 return pg_page_alloc_at(pg, block.vaddr, n_page, vm_flag);
28}

30vmblock_t pg_page_alloc_at(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n_page, vm_flags vm_flag)
31{
32 uintptr_t paddr = pmem_freelist_find_free(n_page);
33
34 if (paddr == 0)
35 {
36 mos_panic("OOM");
37 return (vmblock_t){ 0, .npages = 0 };
38 }
39
40 pg_map_pages(pg, vaddr, paddr, n_page, vm_flag);
41 vmblock_t block = {
42 .vaddr = vaddr,
43 .npages = n_page,
44 .flags = vm_flag,
45 };
46 return block;
47}

49vmblock_t pg_page_get_free(x86_pg_infra_t *pg, size_t n_pages, pgalloc_hints hints)
50{
51 uintptr_t vaddr_begin;
52 pagemap_line_t *pagemap = pg->page_map; // paging trick
53
54 switch (hints)
55 {
56 case PGALLOC_HINT_KHEAP:
57 vaddr_begin = MOS_ADDR_KERNEL_HEAP;
58 pagemap = x86_kpg_infra->page_map;
59 break;
60 case PGALLOC_HINT_UHEAP: vaddr_begin = MOS_ADDR_USER_HEAP; break;
61 case PGALLOC_HINT_STACK: vaddr_begin = MOS_ADDR_USER_STACK; break;
62 case PGALLOC_HINT_MMAP: vaddr_begin = MOS_ADDR_USER_MMAP; break;
63 default: mos_panic("invalid pgalloc_hints"); break;
64 }
65
66 size_t n_free_pages = 0;
67 u8 target_starting_bit = 0;
68 uintptr_t map_line_i = vaddr_begin / MOS_PAGE_SIZE / PAGEMAP_WIDTH;
69 for (size_t i = map_line_i; n_free_pages < n_pages; i++)
70 {
71 // TODO: check if we are going beyond the end of user's address space (i.e. larger than 0xC0000000)
72 if (i >= MM_PAGE_MAP_SIZE)
73 {
74 mos_warn("failed to allocate %zu pages", n_pages);
75 return (vmblock_t){ 0 }; // possibly out of virtual memory
76 }
77
78 pagemap_line_t line = pagemap[i];
79
80 if (line == 0)
81 {
82 n_free_pages += PAGEMAP_WIDTH;
83 continue;
84 }
85 else if (line == (pagemap_line_t) ~0)
86 {
87 // a full line of used pages
88 map_line_i = i + 1;
89 n_free_pages = 0;
90 target_starting_bit = 0;
91 continue;
92 }
93
94 for (size_t bit = 0; bit < PAGEMAP_WIDTH; bit++)
95 {
96 if (!BIT_IS_SET(line, bit))
97 {
98 // we have found a free page
99 n_free_pages++;
100 }
101 else
102 {
103 // this page is used
104 if (n_free_pages >= n_pages)
105 // but we have enough free pages
106 break;
107 n_free_pages = 0;
108 target_starting_bit = bit + 1;
109 map_line_i = i;
110 }
111 }
112 }
113
114 size_t page_i = map_line_i * PAGEMAP_WIDTH + target_starting_bit;
115 uintptr_t vaddr = page_i * MOS_PAGE_SIZE;
116
117 MOS_ASSERT_X(!PAGEMAP_IS_SET(pagemap, page_i), "page %zu is already allocated", page_i);
118
119 // sanity check: if we are not asked to allocate in kernel space, we should not be doing so (?)
120 if (hints != PGALLOC_HINT_KHEAP)
121 MOS_ASSERT_X(vaddr < MOS_KERNEL_START_VADDR, "make up your mind to allocate in kernel space!");
122
123 return (vmblock_t){ .vaddr = vaddr, .npages = n_pages };
124}

126void pg_page_free(x86_pg_infra_t *pg, uintptr_t vptr, size_t n_page)
127{
128 size_t page_index = vptr / MOS_PAGE_SIZE;
129 mos_debug("freeing %zu to %zu", page_index, page_index + n_page);
130 pg_unmap_pages(pg, vptr, n_page);
131}

133void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
134{
135 mos_debug("setting flags [0x%x] to [" PTR_FMT "] +%zu pages (pg " PTR_FMT ")", flags, vaddr, n, (uintptr_t) pg);
136 size_t start_page = vaddr / MOS_PAGE_SIZE;
137 for (size_t i = 0; i < n; i++)
138 {
139 size_t page_i = start_page + i;
140 size_t pgd_i = page_i / 1024;
141
142 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
143 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
144
145 pg->pgdir[pgd_i].writable |= flags & VM_WRITE;
146 pg->pgtable[page_i].writable = flags & VM_WRITE;
147
148 pg->pgdir[pgd_i].usermode |= flags & VM_USER;
149 pg->pgtable[page_i].usermode = flags & VM_USER;
150
151 pg->pgdir[pgd_i].cache_disabled |= flags & VM_CACHE_DISABLED;
152 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
153
154 pg->pgdir[pgd_i].write_through |= flags & VM_WRITE_THROUGH;
155 pg->pgtable[page_i].write_through = flags & VM_WRITE_THROUGH;
156
157 pg->pgtable[page_i].global = flags & VM_GLOBAL;
158 pg_flush_tlb(vaddr);
159 }
160}

162void pg_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
163{
164 pmem_freelist_remove_region(paddr_start, n_page * MOS_PAGE_SIZE);
165 pg_do_map_pages(pg, vaddr_start, paddr_start, n_page, flags);
166}

168void pg_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
169{
170 uintptr_t paddr = pg_page_get_mapped_paddr(pg, vaddr_start);
171 pg_do_unmap_pages(pg, vaddr_start, n_page);
172 pmem_freelist_add_region(paddr, n_page * MOS_PAGE_SIZE);
173}

175void pg_do_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
176{
177 mos_debug("paging: mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table %lu", n_page, vaddr_start, paddr_start, vaddr_start / MOS_PAGE_SIZE);
178 for (size_t i = 0; i < n_page; i++)
179 pg_do_map_page(pg, vaddr_start + i * MOS_PAGE_SIZE, paddr_start + i * MOS_PAGE_SIZE, flags);
180}

182void pg_do_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
183{
184 mos_debug("paging: unmapping %zu pages starting at " PTR_FMT " @ table %lu", n_page, vaddr_start, vaddr_start / MOS_PAGE_SIZE);
185 for (size_t i = 0; i < n_page; i++)
186 pg_do_unmap_page(pg, vaddr_start + i * MOS_PAGE_SIZE);
187}

189void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
190{
191 // ensure the page is aligned to 4096
192 MOS_ASSERT_X(paddr < X86_MAX_MEM_SIZE, "physical address out of bounds");
193 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
194
195 if (pg == x86_kpg_infra)
196 MOS_ASSERT_X(vaddr >= MOS_KERNEL_START_VADDR, "make up your mind to allocate userspace address in kernel page table");
197
198 // ! todo: ensure the offsets are correct for both paddr and vaddr
199 u32 pd_index = vaddr >> 22;
200 u32 pt_index = pd_index * 1024 + (vaddr >> 12 & 0x3ff);
201
202 x86_pgdir_entry *this_dir = &pg->pgdir[pd_index];
203 x86_pgtable_entry *this_table = pd_index >= 768 ? &x86_kpg_infra->pgtable[pt_index] : &pg->pgtable[pt_index];
204
205 if (unlikely(!this_dir->present))
206 {
207 this_dir->present = true;
208
209 // kernel page tables are identity mapped
210 uintptr_t table_paddr;
211 if (pg == x86_kpg_infra)
212 table_paddr = (uintptr_t) this_table - MOS_KERNEL_START_VADDR; // kvaddr = kpaddr + MOS_KERNEL_START_VADDR
213 else
214 table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table);
215 this_dir->page_table_paddr = table_paddr >> 12;
216 }
217
218 if (pg != x86_kpg_infra && pd_index >= 768)
219 MOS_ASSERT_X(this_dir->present, "page directory not present for kernel addresses, when mapping user pages");
220
221 MOS_ASSERT_X(this_table->present == false, "page " PTR_FMT " already mapped", vaddr);
222
223 this_table->present = true;
224 this_table->phys_addr = (uintptr_t) paddr >> 12;
225
226 this_dir->writable |= flags & VM_WRITE;
227 this_table->writable = flags & VM_WRITE;
228
229 this_dir->usermode |= flags & VM_USER;
230 this_table->usermode = flags & VM_USER;
231
232 this_dir->cache_disabled |= flags & VM_CACHE_DISABLED;
233 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
234
235 this_table->global = flags & VM_GLOBAL;
236
237 pagemap_line_t *page_map = pd_index < 768 ? pg->page_map : x86_kpg_infra->page_map;
238
239 if (unlikely(PAGEMAP_IS_SET(page_map, pt_index)))
240 mos_panic("vmem " PTR_FMT " already mapped", vaddr);
241 PAGEMAP_MAP(page_map, pt_index);
242
243 MOS_ASSERT_X(PAGEMAP_IS_SET(page_map, pt_index), "page map not set for " PTR_FMT, vaddr);
244
245 pg_flush_tlb(vaddr);
246}

248void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
249{
250 u32 pd_index = vaddr >> 22;
251 u32 pt_index = pd_index * 1024 + (vaddr >> 12 & 0x3ff);
252
253 x86_pgdir_entry *page_dir = &pg->pgdir[pd_index];
254 if (unlikely(!page_dir->present))
255 {
256 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
257 return;
258 }
259
260 x86_pgtable_entry *this_table = pd_index >= 768 ? &x86_kpg_infra->pgtable[pt_index] : &pg->pgtable[pt_index];
261 this_table->present = false;
262
263 // update the mm page map
264 pagemap_line_t *page_map = pd_index < 768 ? pg->page_map : x86_kpg_infra->page_map;
265 if (unlikely(!PAGEMAP_IS_SET(page_map, pt_index)))
266 {
267 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
268 return;
269 }
270
271 PAGEMAP_UNMAP(page_map, pt_index);
272 pg_flush_tlb(vaddr);
273}

275// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
276uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
277{
278 int page_dir_index = vaddr >> 22;
279 int page_table_index = vaddr >> 12 & 0x3ff;
280 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
281 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
282
283 if (page_dir_index >= 768)
284 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
285
286 if (unlikely(!page_dir->present))
287 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
288
289 if (unlikely(!page_table->present))
290 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
291
292 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
293}

295vm_flags pg_page_get_flags(x86_pg_infra_t *pg, uintptr_t vaddr)
296{
297 int page_dir_index = vaddr >> 22;
298 int page_table_index = vaddr >> 12 & 0x3ff;
299 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
300 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
301
302 if (page_dir_index >= 768)
303 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
304
305 if (unlikely(!page_dir->present))
306 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
307
308 if (unlikely(!page_table->present))
309 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
310
311 vm_flags flags = VM_READ;
312 flags |= (page_dir->writable && page_table->writable) ? VM_WRITE : 0;
313 flags |= (page_dir->usermode && page_table->usermode) ? VM_USER : 0;
314 flags |= (page_dir->cache_disabled && page_table->cache_disabled) ? VM_CACHE_DISABLED : 0;
315 flags |= page_table->global ? VM_GLOBAL : 0;
316 return flags;
317}

319bool pg_page_get_is_mapped(x86_pg_infra_t *pg, uintptr_t vaddr)
320{
321 int page_dir_index = vaddr >> 22;
322 int page_table_index = vaddr >> 12 & 0x3ff;
323 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
324 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
325
326 if (page_dir_index >= 768)
327 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
328
329 if (unlikely(!page_dir->present))
330 return false;
331
332 if (unlikely(!page_table->present))
333 return false;
334
335 return true;
336}
