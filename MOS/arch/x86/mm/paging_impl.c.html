9#define PAGING_CORRECT_PGTABLE_SANITY_CHECKS(_pg, _vaddr) \
10 do \
11 { \
12 if (_pg == x86_kpg_infra) \
13 MOS_ASSERT_X(_vaddr >= MOS_KERNEL_START_VADDR, "operating with [userspace address] in the [kernel page] table"); \
14 else \
15 MOS_ASSERT_X(_vaddr < MOS_KERNEL_START_VADDR, "operating with [kernel address] in the [userspace page] table"); \
16 } while (0)
17
18always_inline void pg_flush_tlb(uintptr_t vaddr)
19{
20 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
21}

23void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
24{
25 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
26 PAGING_CORRECT_PGTABLE_SANITY_CHECKS(pg, vaddr);
27
28 mos_debug(x86_paging, "setting flags [0x%x] to [" PTR_FMT "] +%zu pages (pg " PTR_FMT ")", flags, vaddr, n, (uintptr_t) pg);
29 size_t start_page = vaddr / MOS_PAGE_SIZE;
30 for (size_t i = 0; i < n; i++)
31 {
32 size_t page_i = start_page + i;
33 size_t pgd_i = page_i / 1024;
34
35 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
36 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
37
38 pg->pgdir[pgd_i].writable |= flags & VM_WRITE;
39 pg->pgtable[page_i].writable = flags & VM_WRITE;
40
41 pg->pgdir[pgd_i].usermode |= flags & VM_USER;
42 pg->pgtable[page_i].usermode = flags & VM_USER;
43
44 pg->pgdir[pgd_i].cache_disabled |= flags & VM_CACHE_DISABLED;
45 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
46
47 pg->pgdir[pgd_i].write_through |= flags & VM_WRITE_THROUGH;
48 pg->pgtable[page_i].write_through = flags & VM_WRITE_THROUGH;
49
50 pg->pgtable[page_i].global = flags & VM_GLOBAL;
51 pg_flush_tlb(vaddr);
52 }
53}

55void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
56{
57 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
58 MOS_ASSERT_X(paddr % MOS_PAGE_SIZE == 0, "paddr is not aligned to 4096");
59 PAGING_CORRECT_PGTABLE_SANITY_CHECKS(pg, vaddr);
60
61 const u32 pd_index = vaddr >> 22;
62 const u32 pt_index = pd_index * 1024 + (vaddr >> 12 & 0x3ff);
63
64 x86_pgdir_entry *this_dir = &pg->pgdir[pd_index];
65 x86_pgtable_entry *this_table = &pg->pgtable[pt_index];
66
67 if (unlikely(!this_dir->present))
68 {
69 this_dir->present = true;
70
71 if (vaddr >= MOS_KERNEL_START_VADDR)
72 {
73 // kernel page tables are identity mapped, directly subtract the kernel start address
74 // chicken and egg problem: get the physical address of the very first page table
75 this_dir->page_table_paddr = ((uintptr_t) &pg->pgtable[pd_index * 1024] - MOS_KERNEL_START_VADDR) >> 12;
76 }
77 else
78 {
79 this_dir->page_table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table) >> 12;
80 }
81 }
82
83 MOS_ASSERT_X(this_table->present == false, "page " PTR_FMT " already mapped", vaddr);
84
85 this_table->present = true;
86 this_table->phys_addr = (uintptr_t) paddr >> 12;
87
88 this_dir->writable |= flags & VM_WRITE;
89 this_table->writable = flags & VM_WRITE;
90
91 this_dir->usermode |= flags & VM_USER;
92 this_table->usermode = flags & VM_USER;
93
94 this_dir->cache_disabled |= flags & VM_CACHE_DISABLED;
95 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
96
97 this_table->global = flags & VM_GLOBAL;
98
99 pg_flush_tlb(vaddr);
100}

102void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
103{
104 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
105 PAGING_CORRECT_PGTABLE_SANITY_CHECKS(pg, vaddr);
106
107 const u32 pd_index = vaddr >> 22;
108 const u32 pt_index = pd_index * 1024 + (vaddr >> 12 & 0x3ff);
109
110 x86_pgdir_entry *page_dir = &pg->pgdir[pd_index];
111 if (unlikely(!page_dir->present))
112 {
113 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
114 return;
115 }
116
117 pg->pgtable[pt_index].present = false;
118 pg_flush_tlb(vaddr);
119}

121// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
122uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
123{
124 PAGING_CORRECT_PGTABLE_SANITY_CHECKS(pg, vaddr);
125
126 int page_dir_index = vaddr >> 22;
127 int page_table_index = vaddr >> 12 & 0x3ff;
128 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
129 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
130
131 if (vaddr >= MOS_KERNEL_START_VADDR)
132 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
133
134 if (unlikely(!page_dir->present))
135 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
136
137 if (unlikely(!page_table->present))
138 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
139
140 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
141}

143vm_flags pg_page_get_flags(x86_pg_infra_t *pg, uintptr_t vaddr)
144{
145 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
146 PAGING_CORRECT_PGTABLE_SANITY_CHECKS(pg, vaddr);
147
148 int page_dir_index = vaddr >> 22;
149 int page_table_index = vaddr >> 12 & 0x3ff;
150 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
151 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
152
153 if (vaddr >= MOS_KERNEL_START_VADDR)
154 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
155
156 if (unlikely(!page_dir->present))
157 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
158
159 if (unlikely(!page_table->present))
160 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
161
162 vm_flags flags = VM_READ;
163 flags |= (page_dir->writable && page_table->writable) ? VM_WRITE : 0;
164 flags |= (page_dir->usermode && page_table->usermode) ? VM_USER : 0;
165 flags |= (page_dir->cache_disabled && page_table->cache_disabled) ? VM_CACHE_DISABLED : 0;
166 flags |= page_table->global ? VM_GLOBAL : 0;
167 return flags;
168}
