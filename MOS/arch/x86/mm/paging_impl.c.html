14#define PAGEMAP_MAP(map, index) map[index / PAGEMAP_WIDTH] |= 1 << (index % PAGEMAP_WIDTH)
15#define PAGEMAP_UNMAP(map, index) map[index / PAGEMAP_WIDTH] &= ~(1 << (index % PAGEMAP_WIDTH))
16#define BIT_IS_SET(byte, bit) ((byte) & (1 << (bit)))
17
18always_inline void pg_flush_tlb(uintptr_t vaddr)
19{
20 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
21}

23vmblock_t pg_page_alloc(x86_pg_infra_t *pg, size_t n_page, pgalloc_hints flags, vm_flags vm_flag)
24{
25 uintptr_t vaddr_begin;
26
27 if (flags & PGALLOC_HINT_KHEAP)
28 vaddr_begin = MOS_X86_HEAP_BASE_VADDR;
29 else if (flags & PGALLOC_HINT_USERSPACE)
30 vaddr_begin = MOS_USERSPACE_PGALLOC_START;
31 else
32 vaddr_begin = mos_kernel_end;
33
34 // simply rename the variable, we are dealing with bitmaps
35 size_t n_bits = n_page;
36 size_t n_zero_bits = 0;
37
38 u8 target_bit = 0;
39 uintptr_t vaddr_map_bit_begin = vaddr_begin / MOS_PAGE_SIZE / PAGEMAP_WIDTH + 1;
40 for (size_t i = vaddr_map_bit_begin; n_zero_bits < n_bits; i++)
41 {
42 if (i >= MM_PAGE_MAP_SIZE)
43 {
44 mos_warn("failed to allocate %zu pages", n_page);
45 return (vmblock_t){ 0, .npages = 0 };
46 }
47 pagemap_line_t current_byte = pg->page_map[i];
48
49 if (current_byte == 0)
50 {
51 n_zero_bits += PAGEMAP_WIDTH;
52 continue;
53 }
54 if (current_byte == (pagemap_line_t) ~0)
55 {
56 vaddr_map_bit_begin = i + 1;
57 continue;
58 }
59
60 for (size_t bit = 0; bit < PAGEMAP_WIDTH; bit++)
61 {
62 if (!BIT_IS_SET(current_byte, bit))
63 n_zero_bits++;
64 else
65 n_zero_bits = 0, target_bit = bit + 1, vaddr_map_bit_begin = i;
66 }
67 }
68
69 size_t page_i = vaddr_map_bit_begin * PAGEMAP_WIDTH + target_bit;
70 uintptr_t vaddr = page_i * MOS_PAGE_SIZE;
71 return pg_page_alloc_at(pg, vaddr, n_page, vm_flag);
72}

74vmblock_t pg_page_alloc_at(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n_page, vm_flags vm_flag)
75{
76 uintptr_t paddr = pmem_freelist_find_free(n_page);
77
78 if (paddr == 0)
79 {
80 mos_panic("OOM");
81 return (vmblock_t){ 0, .npages = 0 };
82 }
83
84 pg_map_pages(pg, vaddr, paddr, n_page, vm_flag);
85 vmblock_t block = {
86 .vaddr = vaddr,
87 .npages = n_page,
88 .flags = vm_flag,
89 };
90 return block;
91}

93void pg_page_free(x86_pg_infra_t *pg, uintptr_t vptr, size_t n_page)
94{
95 size_t page_index = vptr / MOS_PAGE_SIZE;
96 mos_debug("paging: freeing %zu to %zu", page_index, page_index + n_page);
97 pg_unmap_pages(pg, vptr, n_page);
98}

100void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
101{
102 mos_debug("paging: setting flags [0x%x] to [" PTR_FMT "] +%zu pages", flags, vaddr, n);
103 size_t start_page = vaddr / MOS_PAGE_SIZE;
104 for (size_t i = 0; i < n; i++)
105 {
106 size_t page_i = start_page + i;
107 size_t pgd_i = page_i / 1024;
108
109 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
110 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
111
112 pg->pgdir[pgd_i].writable = flags & VM_WRITE;
113 pg->pgtable[page_i].writable = flags & VM_WRITE;
114
115 pg->pgdir[pgd_i].usermode = flags & VM_USER;
116 pg->pgtable[page_i].usermode = flags & VM_USER;
117
118 pg->pgdir[pgd_i].cache_disabled = flags & VM_CACHE_DISABLED;
119 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
120
121 pg->pgdir[pgd_i].write_through = flags & VM_WRITE_THROUGH;
122 pg->pgtable[page_i].write_through = flags & VM_WRITE_THROUGH;
123
124 pg->pgtable[page_i].global = flags & VM_GLOBAL;
125 pg_flush_tlb(vaddr);
126 }
127}

129void pg_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
130{
131 pmem_freelist_remove_region(paddr_start, n_page * MOS_PAGE_SIZE);
132 pg_do_map_pages(pg, vaddr_start, paddr_start, n_page, flags);
133}

135void pg_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
136{
137 uintptr_t paddr = pg_page_get_mapped_paddr(pg, vaddr_start);
138 pg_do_unmap_pages(pg, vaddr_start, n_page);
139 pmem_freelist_add_region(paddr, n_page * MOS_PAGE_SIZE);
140}

142void pg_copy_page(x86_pg_infra_t *from_pg, x86_pg_infra_t *to_pg, uintptr_t start_vaddr, size_t n_page)
143{
144 for (size_t i = 0; i < n_page; i++)
145 {
146 uintptr_t vaddr = start_vaddr + i * MOS_PAGE_SIZE;
147 uintptr_t paddr = pg_page_get_mapped_paddr(from_pg, vaddr);
148 pg_map_pages(to_pg, vaddr, paddr, 1, VM_WRITE); // !! TODO: copy flags
149 }
150}

152void pg_do_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
153{
154 mos_debug("paging: mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table %lu", n_page, vaddr_start, paddr_start, vaddr_start / MOS_PAGE_SIZE);
155 for (size_t i = 0; i < n_page; i++)
156 pg_do_map_page(pg, vaddr_start + i * MOS_PAGE_SIZE, paddr_start + i * MOS_PAGE_SIZE, flags);
157}

159void pg_do_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
160{
161 mos_debug("paging: unmapping %zu pages starting at " PTR_FMT " @ table %lu", n_page, vaddr_start, vaddr_start / MOS_PAGE_SIZE);
162 for (size_t i = 0; i < n_page; i++)
163 pg_do_unmap_page(pg, vaddr_start + i * MOS_PAGE_SIZE);

166void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
167{
168 // ensure the page is aligned to 4096
169 MOS_ASSERT_X(paddr < X86_MAX_MEM_SIZE, "physical address out of bounds");
170 MOS_ASSERT_X(flags < 0x100, "invalid flags");
171 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
172
173 // ! todo: ensure the offsets are correct for both paddr and vaddr
174 int page_dir_index = vaddr >> 22;
175 int page_table_index = vaddr >> 12 & 0x3ff; // mask out the lower 12 bits
176
177 u32 pte_index = page_dir_index * 1024 + page_table_index;
178
179 x86_pgdir_entry *this_dir = &pg->pgdir[page_dir_index];
180
181 x86_pgtable_entry *this_table = &pg->pgtable[pte_index];
182 if (unlikely(!this_dir->present))
183 {
184 this_dir->present = true;
185
186 // kernel page tables are identity mapped
187 uintptr_t table_paddr;
188 if (pg == x86_kpg_infra)
189 table_paddr = (uintptr_t) this_table - MOS_KERNEL_START_VADDR;
190 else
191 table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table);
192 this_dir->page_table_paddr = table_paddr >> 12;
193 }
194
195 if (pg != x86_kpg_infra && page_dir_index >= 768)
196 MOS_ASSERT_X(this_dir->present, "page directory not present for kernel addresses, when mapping user pages");
197
198 MOS_ASSERT_X(this_table->present == false, "page is already mapped");
199
200 this_table->present = true;
201 this_table->phys_addr = (uintptr_t) paddr >> 12;
202
203 this_dir->writable = flags & VM_WRITE;
204 this_table->writable = flags & VM_WRITE;
205
206 this_dir->usermode = flags & VM_USER;
207 this_table->usermode = flags & VM_USER;
208
209 this_dir->cache_disabled = flags & VM_CACHE_DISABLED;
210 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
211
212 this_table->global = flags & VM_GLOBAL;
213
214 // update the mm_page_map
215 PAGEMAP_MAP(pg->page_map, pte_index);
216 pg_flush_tlb(vaddr);
217}

219void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
220{
221 int page_dir_index = vaddr >> 22;
222 int page_table_index = vaddr >> 12 & 0x3ff;
223
224 x86_pgdir_entry *page_dir = &pg->pgdir[page_dir_index];
225 if (unlikely(!page_dir->present))
226 {
227 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
228 return;
229 }
230
231 x86_pgtable_entry *page_table = &pg->pgtable[page_dir_index * 1024 + page_table_index];
232 page_table->present = false;
233
234 // update the mm page map
235 u32 pte_index = page_dir_index * 1024 + page_table_index;
236 PAGEMAP_UNMAP(pg->page_map, pte_index);
237 pg_flush_tlb(vaddr);
238}

240// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
241uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
242{
243 int page_dir_index = vaddr >> 22;
244 int page_table_index = vaddr >> 12 & 0x3ff;
245 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
246 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
247
248 if (page_dir_index >= 768)
249 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
250
251 if (unlikely(!page_dir->present))
252 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
253
254 if (unlikely(!page_table->present))
255 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
256
257 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
258}

260vm_flags pg_page_get_flags(x86_pg_infra_t *pg, uintptr_t vaddr)
261{
262 int page_dir_index = vaddr >> 22;
263 int page_table_index = vaddr >> 12 & 0x3ff;
264 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
265 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
266
267 if (page_dir_index >= 768)
268 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
269
270 if (unlikely(!page_dir->present))
271 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
272
273 if (unlikely(!page_table->present))
274 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
275
276 vm_flags flags = VM_READ;
277 flags |= page_table->writable ? VM_WRITE : 0;
278 flags |= page_table->usermode ? VM_USER : 0;
279 flags |= page_table->cache_disabled ? VM_CACHE_DISABLED : 0;
280 flags |= page_table->global ? VM_GLOBAL : 0;
281 return flags;
282}
