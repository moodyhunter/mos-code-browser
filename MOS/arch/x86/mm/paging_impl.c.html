14#define PAGEMAP_MAP(map, index) map[index / PAGEMAP_WIDTH] |= 1 << (index % PAGEMAP_WIDTH)
15#define PAGEMAP_UNMAP(map, index) map[index / PAGEMAP_WIDTH] &= ~(1 << (index % PAGEMAP_WIDTH))
16#define BIT_IS_SET(byte, bit) ((byte) & (1 << (bit)))

18always_inline void pg_flush_tlb(uintptr_t vaddr)
19{
20 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
21}

23vmblock_t pg_page_alloc(x86_pg_infra_t *pg, size_t n_page, pgalloc_hints hints, vm_flags vm_flag)
24{
25 vmblock_t block = pg_page_get_free(pg, n_page, hints);
26 return pg_page_alloc_at(pg, block.vaddr, n_page, vm_flag);
27}

29vmblock_t pg_page_alloc_at(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n_page, vm_flags vm_flag)
30{
31 uintptr_t paddr = pmem_freelist_find_free(n_page);
32
33 if (paddr == 0)
34 {
35 mos_panic("OOM");
36 return (vmblock_t){ 0, .npages = 0 };
37 }
38
39 pg_map_pages(pg, vaddr, paddr, n_page, vm_flag);
40 vmblock_t block = {
41 .vaddr = vaddr,
42 .npages = n_page,
43 .flags = vm_flag,
44 };
45 return block;
46}

48vmblock_t pg_page_get_free(x86_pg_infra_t *pg, size_t n_pages, pgalloc_hints flags)
49{
50 uintptr_t vaddr_begin;
51 pagemap_line_t *pagemap = pg->page_map; // paging trick
52
53 switch (flags)
54 {
55 case PGALLOC_HINT_KHEAP:
56 vaddr_begin = MOS_ADDR_KERNEL_HEAP;
57 pagemap = x86_kpg_infra->page_map;
58 break;
59 case PGALLOC_HINT_UHEAP: vaddr_begin = MOS_ADDR_USER_HEAP; break;
60 case PGALLOC_HINT_STACK: vaddr_begin = MOS_ADDR_USER_STACK; break;
61 case PGALLOC_HINT_MMAP: vaddr_begin = MOS_ADDR_USER_MMAP; break;
62 default: mos_panic("invalid pgalloc_hints"); break;
63 }
64
65 // simply rename the variable, we are dealing with bitmaps
66 size_t n_bits = n_pages;
67 size_t n_zero_bits = 0;
68
69 u8 target_bit = 0;
70 uintptr_t vaddr_map_bit_begin = vaddr_begin / MOS_PAGE_SIZE / PAGEMAP_WIDTH + 1;
71 for (size_t i = vaddr_map_bit_begin; n_zero_bits < n_bits; i++)
72 {
73 if (i >= MM_PAGE_MAP_SIZE)
74 {
75 mos_warn("failed to allocate %zu pages", n_pages);
76 return (vmblock_t){ 0, .npages = 0 };
77 }
78 pagemap_line_t current_byte = pagemap[i];
79
80 if (current_byte == 0)
81 {
82 n_zero_bits += PAGEMAP_WIDTH;
83 continue;
84 }
85 if (current_byte == (pagemap_line_t) ~0)
86 {
87 vaddr_map_bit_begin = i + 1;
88 continue;
89 }
90
91 for (size_t bit = 0; bit < PAGEMAP_WIDTH; bit++)
92 {
93 if (!BIT_IS_SET(current_byte, bit))
94 n_zero_bits++;
95 else
96 n_zero_bits = 0, target_bit = bit + 1, vaddr_map_bit_begin = i;
97 }
98 }
99
100 size_t page_i = vaddr_map_bit_begin * PAGEMAP_WIDTH + target_bit;
101 uintptr_t vaddr = page_i * MOS_PAGE_SIZE;
102
103 // sanity check: if we are not asked to allocate in kernel space, we should not be doing so (?)
104 if (flags != PGALLOC_HINT_KHEAP)
105 MOS_ASSERT_X(vaddr < MOS_KERNEL_START_VADDR, "make up your mind to allocate in kernel space!");
106
107 return (vmblock_t){ .vaddr = vaddr, .npages = n_pages };
108}

110void pg_page_free(x86_pg_infra_t *pg, uintptr_t vptr, size_t n_page)
111{
112 size_t page_index = vptr / MOS_PAGE_SIZE;
113 mos_debug("paging: freeing %zu to %zu", page_index, page_index + n_page);
114 pg_unmap_pages(pg, vptr, n_page);
115}

117void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
118{
119 mos_debug("paging: setting flags [0x%x] to [" PTR_FMT "] +%zu pages", flags, vaddr, n);
120 size_t start_page = vaddr / MOS_PAGE_SIZE;
121 for (size_t i = 0; i < n; i++)
122 {
123 size_t page_i = start_page + i;
124 size_t pgd_i = page_i / 1024;
125
126 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
127 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
128
129 pg->pgdir[pgd_i].writable |= flags & VM_WRITE;
130 pg->pgtable[page_i].writable = flags & VM_WRITE;
131
132 pg->pgdir[pgd_i].usermode |= flags & VM_USER;
133 pg->pgtable[page_i].usermode = flags & VM_USER;
134
135 pg->pgdir[pgd_i].cache_disabled |= flags & VM_CACHE_DISABLED;
136 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
137
138 pg->pgdir[pgd_i].write_through |= flags & VM_WRITE_THROUGH;
139 pg->pgtable[page_i].write_through = flags & VM_WRITE_THROUGH;
140
141 pg->pgtable[page_i].global = flags & VM_GLOBAL;
142 pg_flush_tlb(vaddr);
143 }
144}

146void pg_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
147{
148 pmem_freelist_remove_region(paddr_start, n_page * MOS_PAGE_SIZE);
149 pg_do_map_pages(pg, vaddr_start, paddr_start, n_page, flags);
150}

152void pg_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
153{
154 uintptr_t paddr = pg_page_get_mapped_paddr(pg, vaddr_start);
155 pg_do_unmap_pages(pg, vaddr_start, n_page);
156 pmem_freelist_add_region(paddr, n_page * MOS_PAGE_SIZE);
157}

159void pg_do_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
160{
161 mos_debug("paging: mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table %lu", n_page, vaddr_start, paddr_start, vaddr_start / MOS_PAGE_SIZE);
162 for (size_t i = 0; i < n_page; i++)
163 pg_do_map_page(pg, vaddr_start + i * MOS_PAGE_SIZE, paddr_start + i * MOS_PAGE_SIZE, flags);
164}

166void pg_do_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
167{
168 mos_debug("paging: unmapping %zu pages starting at " PTR_FMT " @ table %lu", n_page, vaddr_start, vaddr_start / MOS_PAGE_SIZE);
169 for (size_t i = 0; i < n_page; i++)
170 pg_do_unmap_page(pg, vaddr_start + i * MOS_PAGE_SIZE);
171}

173void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
174{
175 // ensure the page is aligned to 4096
176 MOS_ASSERT_X(paddr < X86_MAX_MEM_SIZE, "physical address out of bounds");
177 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
178
179 // ! todo: ensure the offsets are correct for both paddr and vaddr
180 u32 pd_index = vaddr >> 22;
181 u32 pt_index = pd_index * 1024 + (vaddr >> 12 & 0x3ff);
182
183 x86_pgdir_entry *this_dir = &pg->pgdir[pd_index];
184 x86_pgtable_entry *this_table = pd_index >= 768 ? &x86_kpg_infra->pgtable[pt_index] : &pg->pgtable[pt_index];
185
186 if (unlikely(!this_dir->present))
187 {
188 this_dir->present = true;
189
190 // kernel page tables are identity mapped
191 uintptr_t table_paddr;
192 if (pg == x86_kpg_infra)
193 table_paddr = (uintptr_t) this_table - MOS_KERNEL_START_VADDR; // kvaddr = kpaddr + MOS_KERNEL_START_VADDR
194 else
195 table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table);
196 this_dir->page_table_paddr = table_paddr >> 12;
197 }
198
199 if (pg != x86_kpg_infra && pd_index >= 768)
200 MOS_ASSERT_X(this_dir->present, "page directory not present for kernel addresses, when mapping user pages");
201
202 MOS_ASSERT_X(this_table->present == false, "page is already mapped");
203
204 this_table->present = true;
205 this_table->phys_addr = (uintptr_t) paddr >> 12;
206
207 this_dir->writable |= flags & VM_WRITE;
208 this_table->writable = flags & VM_WRITE;
209
210 this_dir->usermode |= flags & VM_USER;
211 this_table->usermode = flags & VM_USER;
212
213 this_dir->cache_disabled |= flags & VM_CACHE_DISABLED;
214 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
215
216 this_table->global = flags & VM_GLOBAL;
217
218 // update the mm_page_map
219 if (pt_index < MOS_KERNEL_START_VADDR / MOS_PAGE_SIZE)
220 PAGEMAP_MAP(pg->page_map, pt_index);
221 else
222 PAGEMAP_MAP(x86_kpg_infra->page_map, pt_index);
223 pg_flush_tlb(vaddr);
224}

226void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
227{
228 u32 pd_index = vaddr >> 22;
229 u32 pt_index = pd_index * 1024 + (vaddr >> 12 & 0x3ff);
230
231 x86_pgdir_entry *page_dir = &pg->pgdir[pd_index];
232 if (unlikely(!page_dir->present))
233 {
234 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
235 return;
236 }
237
238 x86_pgtable_entry *this_table = pd_index >= 768 ? &x86_kpg_infra->pgtable[pt_index] : &pg->pgtable[pt_index];
239 this_table->present = false;
240
241 // update the mm page map
242 if (pd_index >= 768)
243 PAGEMAP_UNMAP(x86_kpg_infra->page_map, pt_index);
244 else
245 PAGEMAP_UNMAP(pg->page_map, pt_index);
246 pg_flush_tlb(vaddr);
247}

249// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
250uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
251{
252 int page_dir_index = vaddr >> 22;
253 int page_table_index = vaddr >> 12 & 0x3ff;
254 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
255 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
256
257 if (page_dir_index >= 768)
258 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
259
260 if (unlikely(!page_dir->present))
261 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
262
263 if (unlikely(!page_table->present))
264 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
265
266 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
267}

269vm_flags pg_page_get_flags(x86_pg_infra_t *pg, uintptr_t vaddr)
270{
271 int page_dir_index = vaddr >> 22;
272 int page_table_index = vaddr >> 12 & 0x3ff;
273 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
274 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
275
276 if (page_dir_index >= 768)
277 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
278
279 if (unlikely(!page_dir->present))
280 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
281
282 if (unlikely(!page_table->present))
283 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
284
285 vm_flags flags = VM_READ;
286 flags |= (page_dir->writable && page_table->writable) ? VM_WRITE : 0;
287 flags |= (page_dir->usermode && page_table->usermode) ? VM_USER : 0;
288 flags |= (page_dir->cache_disabled && page_table->cache_disabled) ? VM_CACHE_DISABLED : 0;
289 flags |= page_table->global ? VM_GLOBAL : 0;
290 return flags;
291}

293bool pg_page_get_is_mapped(x86_pg_infra_t *pg, uintptr_t vaddr)
294{
295 int page_dir_index = vaddr >> 22;
296 int page_table_index = vaddr >> 12 & 0x3ff;
297 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
298 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
299
300 if (page_dir_index >= 768)
301 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
302
303 if (unlikely(!page_dir->present))
304 return false;
305
306 if (unlikely(!page_table->present))
307 return false;
308
309 return true;
310}
