14#define PAGEMAP_MAP(map, index) map[index / PAGEMAP_WIDTH] |= 1 << (index % PAGEMAP_WIDTH)
15#define PAGEMAP_UNMAP(map, index) map[index / PAGEMAP_WIDTH] &= ~(1 << (index % PAGEMAP_WIDTH))
16#define BIT_IS_SET(byte, bit) ((byte) & (1 << (bit)))

18always_inline void pg_flush_tlb(uintptr_t vaddr)
19{
20 __asm__ volatile("invlpg (%0)" ::"r"(vaddr));
21}

23vmblock_t pg_page_alloc(x86_pg_infra_t *pg, size_t n_page, pgalloc_hints hints, vm_flags vm_flag)
24{
25 vmblock_t block = pg_page_get_free(pg, n_page, hints);
26 return pg_page_alloc_at(pg, block.vaddr, n_page, vm_flag);
27}

29vmblock_t pg_page_alloc_at(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n_page, vm_flags vm_flag)
30{
31 uintptr_t paddr = pmem_freelist_find_free(n_page);
32
33 if (paddr == 0)
34 {
35 mos_panic("OOM");
36 return (vmblock_t){ 0, .npages = 0 };
37 }
38
39 pg_map_pages(pg, vaddr, paddr, n_page, vm_flag);
40 vmblock_t block = {
41 .vaddr = vaddr,
42 .npages = n_page,
43 .flags = vm_flag,
44 };
45 return block;
46}

48vmblock_t pg_page_get_free(x86_pg_infra_t *pg, size_t n_pages, pgalloc_hints flags)
49{
50 uintptr_t vaddr_begin;
51
52 if (flags == PGALLOC_HINT_KHEAP)
53 vaddr_begin = MOS_X86_HEAP_BASE_VADDR;
54 if (flags == PGALLOC_HINT_USERSPACE)
55 vaddr_begin = MOS_USERSPACE_PGALLOC_START;
56
57 // simply rename the variable, we are dealing with bitmaps
58 size_t n_bits = n_pages;
59 size_t n_zero_bits = 0;
60
61 u8 target_bit = 0;
62 uintptr_t vaddr_map_bit_begin = vaddr_begin / MOS_PAGE_SIZE / PAGEMAP_WIDTH + 1;
63 for (size_t i = vaddr_map_bit_begin; n_zero_bits < n_bits; i++)
64 {
65 if (i >= MM_PAGE_MAP_SIZE)
66 {
67 mos_warn("failed to allocate %zu pages", n_pages);
68 return (vmblock_t){ 0, .npages = 0 };
69 }
70 pagemap_line_t current_byte = pg->page_map[i];
71
72 if (current_byte == 0)
73 {
74 n_zero_bits += PAGEMAP_WIDTH;
75 continue;
76 }
77 if (current_byte == (pagemap_line_t) ~0)
78 {
79 vaddr_map_bit_begin = i + 1;
80 continue;
81 }
82
83 for (size_t bit = 0; bit < PAGEMAP_WIDTH; bit++)
84 {
85 if (!BIT_IS_SET(current_byte, bit))
86 n_zero_bits++;
87 else
88 n_zero_bits = 0, target_bit = bit + 1, vaddr_map_bit_begin = i;
89 }
90 }
91
92 size_t page_i = vaddr_map_bit_begin * PAGEMAP_WIDTH + target_bit;
93 uintptr_t vaddr = page_i * MOS_PAGE_SIZE;
94
95 return (vmblock_t){ .vaddr = vaddr, .npages = n_pages };
96}

98void pg_page_free(x86_pg_infra_t *pg, uintptr_t vptr, size_t n_page)
99{
100 size_t page_index = vptr / MOS_PAGE_SIZE;
101 mos_debug("paging: freeing %zu to %zu", page_index, page_index + n_page);
102 pg_unmap_pages(pg, vptr, n_page);
103}

105void pg_page_flag(x86_pg_infra_t *pg, uintptr_t vaddr, size_t n, vm_flags flags)
106{
107 mos_debug("paging: setting flags [0x%x] to [" PTR_FMT "] +%zu pages", flags, vaddr, n);
108 size_t start_page = vaddr / MOS_PAGE_SIZE;
109 for (size_t i = 0; i < n; i++)
110 {
111 size_t page_i = start_page + i;
112 size_t pgd_i = page_i / 1024;
113
114 MOS_ASSERT_X(pg->pgdir[pgd_i].present, "page directory not present");
115 MOS_ASSERT_X(pg->pgtable[page_i].present, "page table not present");
116
117 pg->pgdir[pgd_i].writable |= flags & VM_WRITE;
118 pg->pgtable[page_i].writable = flags & VM_WRITE;
119
120 pg->pgdir[pgd_i].usermode |= flags & VM_USER;
121 pg->pgtable[page_i].usermode = flags & VM_USER;
122
123 pg->pgdir[pgd_i].cache_disabled |= flags & VM_CACHE_DISABLED;
124 pg->pgtable[page_i].cache_disabled = flags & VM_CACHE_DISABLED;
125
126 pg->pgdir[pgd_i].write_through |= flags & VM_WRITE_THROUGH;
127 pg->pgtable[page_i].write_through = flags & VM_WRITE_THROUGH;
128
129 pg->pgtable[page_i].global = flags & VM_GLOBAL;
130 pg_flush_tlb(vaddr);
131 }
132}

134void pg_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
135{
136 pmem_freelist_remove_region(paddr_start, n_page * MOS_PAGE_SIZE);
137 pg_do_map_pages(pg, vaddr_start, paddr_start, n_page, flags);
138}

140void pg_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
141{
142 uintptr_t paddr = pg_page_get_mapped_paddr(pg, vaddr_start);
143 pg_do_unmap_pages(pg, vaddr_start, n_page);
144 pmem_freelist_add_region(paddr, n_page * MOS_PAGE_SIZE);
145}

147void pg_do_map_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, uintptr_t paddr_start, size_t n_page, vm_flags flags)
148{
149 mos_debug("paging: mapping %zu pages (" PTR_FMT "->" PTR_FMT ") @ table %lu", n_page, vaddr_start, paddr_start, vaddr_start / MOS_PAGE_SIZE);
150 for (size_t i = 0; i < n_page; i++)
151 pg_do_map_page(pg, vaddr_start + i * MOS_PAGE_SIZE, paddr_start + i * MOS_PAGE_SIZE, flags);
152}

154void pg_do_unmap_pages(x86_pg_infra_t *pg, uintptr_t vaddr_start, size_t n_page)
155{
156 mos_debug("paging: unmapping %zu pages starting at " PTR_FMT " @ table %lu", n_page, vaddr_start, vaddr_start / MOS_PAGE_SIZE);
157 for (size_t i = 0; i < n_page; i++)
158 pg_do_unmap_page(pg, vaddr_start + i * MOS_PAGE_SIZE);
159}

161void pg_do_map_page(x86_pg_infra_t *pg, uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
162{
163 // ensure the page is aligned to 4096
164 MOS_ASSERT_X(paddr < X86_MAX_MEM_SIZE, "physical address out of bounds");
165 MOS_ASSERT_X(vaddr % MOS_PAGE_SIZE == 0, "vaddr is not aligned to 4096");
166
167 // ! todo: ensure the offsets are correct for both paddr and vaddr
168 int page_dir_index = vaddr >> 22;
169 int page_table_index = vaddr >> 12 & 0x3ff; // mask out the lower 12 bits
170
171 u32 pte_index = page_dir_index * 1024 + page_table_index;
172
173 x86_pgdir_entry *this_dir = &pg->pgdir[page_dir_index];
174
175 x86_pgtable_entry *this_table = &pg->pgtable[pte_index];
176 if (unlikely(!this_dir->present))
177 {
178 this_dir->present = true;
179
180 // kernel page tables are identity mapped
181 uintptr_t table_paddr;
182 if (pg == x86_kpg_infra)
183 table_paddr = (uintptr_t) this_table - MOS_KERNEL_START_VADDR;
184 else
185 table_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) this_table);
186 this_dir->page_table_paddr = table_paddr >> 12;
187 }
188
189 if (pg != x86_kpg_infra && page_dir_index >= 768)
190 MOS_ASSERT_X(this_dir->present, "page directory not present for kernel addresses, when mapping user pages");
191
192 MOS_ASSERT_X(this_table->present == false, "page is already mapped");
193
194 this_table->present = true;
195 this_table->phys_addr = (uintptr_t) paddr >> 12;
196
197 this_dir->writable |= flags & VM_WRITE;
198 this_table->writable = flags & VM_WRITE;
199
200 this_dir->usermode |= flags & VM_USER;
201 this_table->usermode = flags & VM_USER;
202
203 this_dir->cache_disabled |= flags & VM_CACHE_DISABLED;
204 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
205
206 this_table->global = flags & VM_GLOBAL;
207
208 // update the mm_page_map
209 PAGEMAP_MAP(pg->page_map, pte_index);
210 pg_flush_tlb(vaddr);
211}

213void pg_do_unmap_page(x86_pg_infra_t *pg, uintptr_t vaddr)
214{
215 int page_dir_index = vaddr >> 22;
216 int page_table_index = vaddr >> 12 & 0x3ff;
217
218 x86_pgdir_entry *page_dir = &pg->pgdir[page_dir_index];
219 if (unlikely(!page_dir->present))
220 {
221 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
222 return;
223 }
224
225 x86_pgtable_entry *page_table = &pg->pgtable[page_dir_index * 1024 + page_table_index];
226 page_table->present = false;
227
228 // update the mm page map
229 u32 pte_index = page_dir_index * 1024 + page_table_index;
230 PAGEMAP_UNMAP(pg->page_map, pte_index);
231 pg_flush_tlb(vaddr);
232}

234// !! TODO: read real address instead of assuming memory layout = x86_pg_infra_t
235uintptr_t pg_page_get_mapped_paddr(x86_pg_infra_t *pg, uintptr_t vaddr)
236{
237 int page_dir_index = vaddr >> 22;
238 int page_table_index = vaddr >> 12 & 0x3ff;
239 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
240 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
241
242 if (page_dir_index >= 768)
243 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
244
245 if (unlikely(!page_dir->present))
246 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
247
248 if (unlikely(!page_table->present))
249 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
250
251 return (page_table->phys_addr << 12) + (vaddr & 0xfff);
252}

254vm_flags pg_page_get_flags(x86_pg_infra_t *pg, uintptr_t vaddr)
255{
256 int page_dir_index = vaddr >> 22;
257 int page_table_index = vaddr >> 12 & 0x3ff;
258 x86_pgdir_entry *page_dir = pg->pgdir + page_dir_index;
259 x86_pgtable_entry *page_table = pg->pgtable + page_dir_index * 1024 + page_table_index;
260
261 if (page_dir_index >= 768)
262 page_dir = x86_kpg_infra->pgdir + page_dir_index, page_table = x86_kpg_infra->pgtable + page_dir_index * 1024 + page_table_index;
263
264 if (unlikely(!page_dir->present))
265 mos_panic("page directory for address " PTR_FMT " not mapped", vaddr);
266
267 if (unlikely(!page_table->present))
268 mos_panic("vmem " PTR_FMT " not mapped", vaddr);
269
270 vm_flags flags = VM_READ;
271 flags |= (page_dir->writable && page_table->writable) ? VM_WRITE : 0;
272 flags |= (page_dir->usermode && page_table->usermode) ? VM_USER : 0;
273 flags |= (page_dir->cache_disabled && page_table->cache_disabled) ? VM_CACHE_DISABLED : 0;
274 flags |= page_table->global ? VM_GLOBAL : 0;
275 return flags;
276}
