9static void mem_add_region(u64 phys_addr, size_t size, bool available)
10{
11 if (x86_platform.mem_regions.count >= MOS_MAX_SUPPORTED_MEMREGION)
12 mos_panic("too many memory regions added.");
13
14 memregion_t *block = &x86_platform.mem_regions.regions[x86_platform.mem_regions.count++];
15 block->address = phys_addr;
16 block->size_bytes = size;
17 block->available = available;
18
19 if (available)
20 x86_platform.mem.available += size;
21 x86_platform.mem.available += size;
22}

24void x86_mem_init(const multiboot_memory_map_t *map_entry, u32 count)
25{
26 x86_platform.mem_regions.count = 0;
27 x86_platform.mem.total = 0;
28 x86_platform.mem.available = 0;
29
30 pr_info("Multiboot memory map:");
31 for (u32 i = 0; i < count; i++)
32 {
33 char size_buf[32];
34 const multiboot_memory_map_t *entry = map_entry + i;
35
36 u64 region_length = entry->len;
37 u64 region_base = entry->phys_addr;
38
39 format_size(size_buf, sizeof(size_buf), region_length);
40
41 if (region_base > MOS_MAX_VADDR)
42 {
43 pr_warn("ignoring a %s long memory region starting at 0x%llx", size_buf, region_base);
44 continue;
45 }
46
47 if (region_base + region_length > (u64) MOS_MAX_VADDR + 1)
48 {
49 region_length = MOS_MAX_VADDR - region_base;
50 format_size(size_buf, sizeof(size_buf), region_length);
51 pr_warn("truncating memory region at 0x%llx, it extends beyond the maximum address " PTR_FMT, region_base, MOS_MAX_VADDR);
52 }
53
54 mem_add_region(region_base, region_length, entry->type == MULTIBOOT_MEMORY_AVAILABLE);
55
56 char *type_str = "";
57 switch (entry->type)
58 {
59 case MULTIBOOT_MEMORY_AVAILABLE: type_str = "Available"; break;
60 case MULTIBOOT_MEMORY_RESERVED: type_str = "Reserved"; break;
61 case MULTIBOOT_MEMORY_ACPI_RECLAIMABLE: type_str = "ACPI Reclaimable"; break;
62 case MULTIBOOT_MEMORY_NVS: type_str = "NVS"; break;
63 case MULTIBOOT_MEMORY_BADRAM: type_str = "Bad RAM"; break;
64 default: mos_panic("unsupported memory map type: %x", entry->type);
65 }
66
67 pr_info2(" %d: 0x%.8llx - 0x%.8llx: %-10s (%s)", i, region_base, region_base + region_length - 1, type_str, size_buf);
68 }
69
70#define SIZE_BUF_LEN 32
71 char buf[SIZE_BUF_LEN];
72 char buf_available[SIZE_BUF_LEN];
73 char buf_unavailable[SIZE_BUF_LEN];
74 format_size(buf, sizeof(buf), x86_platform.mem.total);
75 format_size(buf_available, sizeof(buf_available), x86_platform.mem.available);
76 format_size(buf_unavailable, sizeof(buf_unavailable), x86_platform.mem.total - x86_platform.mem.available);
77 pr_info("Total Memory: %s (%s available, %s unavailable)", buf, buf_available, buf_unavailable);
78}
