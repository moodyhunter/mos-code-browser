18static void mem_add_region(u64 phys_addr, size_t size, bool available)
19{
20 if (x86_mem_regions_count >= X86_MEM_MAX_N_REGIONS)
21 mos_panic("too many memory regions added.");
22
23 x86_pmblock_t *block = &x86_mem_regions[x86_mem_regions_count];
24 block->address = phys_addr;
25 block->size_bytes = size;
26 block->available = available;
27 x86_mem_regions_count++;
28}

30void x86_mem_init(const multiboot_memory_map_t *map_entry, u32 count)
31{
32 pr_info("Multiboot memory map:");
33 for (u32 i = 0; i < count; i++)
34 {
35 char size_buf[32];
36 const multiboot_memory_map_t *entry = map_entry + i;
37
38 u64 region_length = entry->len;
39 u64 region_base = entry->phys_addr;
40
41 format_size(size_buf, sizeof(size_buf), region_length);
42
43 if (region_base > X86_MAX_MEM_SIZE)
44 {
45 pr_warn("ignoring a %s long memory region starting at 0x%llx", size_buf, region_base);
46 continue;
47 }
48
49 if (region_base + region_length > (u64) X86_MAX_MEM_SIZE + 1)
50 {
51 region_length = X86_MAX_MEM_SIZE - region_base;
52 format_size(size_buf, sizeof(size_buf), region_length);
53 pr_warn("truncating memory region at 0x%llx, it extends beyond the maximum address 0x%x", region_base, X86_MAX_MEM_SIZE);
54 }
55
56 mem_add_region(region_base, region_length, entry->type == MULTIBOOT_MEMORY_AVAILABLE);
57
58 x86_mem_size_total += region_length;
59
60 char *type_str = "";
61 switch (entry->type)
62 {
63 case MULTIBOOT_MEMORY_AVAILABLE: type_str = "Available", x86_mem_available += region_length; break;
64 case MULTIBOOT_MEMORY_RESERVED: type_str = "Reserved"; break;
65 case MULTIBOOT_MEMORY_ACPI_RECLAIMABLE: type_str = "ACPI Reclaimable"; break;
66 case MULTIBOOT_MEMORY_NVS: type_str = "NVS"; break;
67 case MULTIBOOT_MEMORY_BADRAM: type_str = "Bad RAM"; break;
68 default: mos_panic("unsupported memory map type: %x", entry->type);
69 }
70
71 pr_info2(" %d: 0x%.8llx - 0x%.8llx: %-10s (%s)", i, region_base, region_base + region_length - 1, type_str, size_buf);
72 }
73
74#define SIZE_BUF_LEN 32
75 char buf[SIZE_BUF_LEN];
76 char buf_available[SIZE_BUF_LEN];
77 char buf_unavailable[SIZE_BUF_LEN];
78 format_size(buf, sizeof(buf), x86_mem_size_total);
79 format_size(buf_available, sizeof(buf_available), x86_mem_available);
80 format_size(buf_unavailable, sizeof(buf_unavailable), x86_mem_size_total - x86_mem_available);
81 pr_info("Total Memory: %s (%s available, %s unavailable)", buf, buf_available, buf_unavailable);
82}
