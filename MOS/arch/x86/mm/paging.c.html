18void x86_mm_enable_paging(void)
19{
20 mos_debug(x86_paging, "page directory vaddr at: %p", (void *) x86_kpg_infra->pgdir);
21 x86_enable_paging_impl(((uintptr_t) x86_kpg_infra->pgdir) - MOS_KERNEL_START_VADDR);
22 pr_info("paging: enabled");
23
24#if MOS_DEBUG_FEATURE(x86_paging)
25 x86_mm_dump_page_table(x86_kpg_infra);
26#endif
27}

29void page_table_dump_range(x86_pg_infra_t *kpg_infra, size_t start_page, size_t end_page)
30{
31 pr_info(" VM Group " PTR_FMT " (+%zu pages):", (uintptr_t) start_page * MOS_PAGE_SIZE, end_page - start_page);
32
33 size_t i = start_page;
34 while (true)
35 {
36 if (i >= end_page)
37 return;
38
39 x86_pgtable_entry *pgtable = kpg_infra->pgtable + i;
40 MOS_ASSERT(pgtable->present);
41
42 uintptr_t p_range_start = pgtable->phys_addr << 12;
43 uintptr_t p_range_last = p_range_start;
44
45 uintptr_t v_range_start = i * MOS_PAGE_SIZE;
46
47 bool range_begin_usermode = pgtable->usermode;
48 bool range_begin_writable = pgtable->writable;
49
50 // ! Special case: end_page - start_page == 1
51 if (end_page - start_page == 1)
52 {
53 uintptr_t v_range_end = v_range_start + MOS_PAGE_SIZE;
54 uintptr_t p_range_end = p_range_start + MOS_PAGE_SIZE;
55 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
56 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
57 return;
58 }
59
60 while (true)
61 {
62 i++;
63 if (i >= end_page)
64 return;
65
66 x86_pgtable_entry *this_t = kpg_infra->pgtable + i;
67 MOS_ASSERT(this_t->present);
68
69 uintptr_t p_range_current = this_t->phys_addr << 12;
70
71 bool is_last_page = i == end_page - 1;
72 bool is_continuous = p_range_current == p_range_last + MOS_PAGE_SIZE && this_t->usermode == range_begin_usermode && this_t->writable == range_begin_writable;
73
74 if (is_continuous && !is_last_page)
75 {
76 p_range_last = p_range_current;
77 continue;
78 }
79
80 uintptr_t v_range_end = i * MOS_PAGE_SIZE;
81 uintptr_t p_range_end = p_range_current + MOS_PAGE_SIZE;
82 // '!is_continuous' case : print the previous range, but add one page
83 if (!is_continuous)
84 p_range_end = p_range_last + MOS_PAGE_SIZE;
85 else
86 v_range_end += MOS_PAGE_SIZE;
87 MOS_ASSERT(p_range_end - p_range_start == v_range_end - v_range_start);
88 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
89 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
90 break;
91 }
92 }

95void x86_mm_dump_page_table(x86_pg_infra_t *pg)
96{
97 pr_info("paging: dumping page table");
98 enum
99 {
100 PRESENT,
101 ABSENT
102 } current_state = pg->pgtable[0].present ? PRESENT : ABSENT;
103
104 size_t current_state_begin_pg = 0;
105
106 for (int pgd_i = 0; pgd_i < 1024; pgd_i++)
107 {
108 x86_pgdir_entry *pgd = pg->pgdir + pgd_i;
109 if (!pgd->present)
110 {
111 if (current_state == PRESENT)
112 {
113 size_t current_state_end_pg = pgd_i * 1024 + 0;
114 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
115 current_state = ABSENT;
116 current_state_begin_pg = current_state_end_pg;
117 }
118 continue;
119 }
120
121 x86_pgtable_entry *pgtable = NULL;
122 x86_pg_infra_t *dump_pg = NULL;
123 if (pgd_i < 768) // user pages
124 pgtable = pg->pgtable, dump_pg = pg;
125 else // HACK: kernel pages
126 pgtable = x86_kpg_infra->pgtable, dump_pg = x86_kpg_infra;
127
128 for (int pte_i = 0; pte_i < 1024; pte_i++)
129 {
130 x86_pgtable_entry *pte = &pgtable[pgd_i * 1024 + pte_i];
131
132 if (!pte->present)
133 {
134 if (current_state == PRESENT)
135 {
136 size_t current_state_end_pg = pgd_i * 1024 + pte_i;
137 page_table_dump_range(dump_pg, current_state_begin_pg, current_state_end_pg);
138 current_state = ABSENT;
139 current_state_begin_pg = current_state_end_pg;
140 }
141 }
142 else
143 {
144 if (current_state == ABSENT)
145 {
146 current_state = PRESENT;
147 current_state_begin_pg = pgd_i * 1024 + pte_i;
148 }
149 }
150 }
151 }
