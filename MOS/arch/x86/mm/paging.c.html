21static const uintptr_t x86_paging_area_start = (uintptr_t) &__MOS_X86_PAGING_AREA_START;
22static const uintptr_t x86_paging_area_end = (uintptr_t) &__MOS_X86_PAGING_AREA_END;
23x86_pg_infra_t *x86_kpg_infra = (void *) &__MOS_X86_PAGING_AREA_START;

25void x86_mm_prepare_paging()
26{
27 // validate if the memory region calculated from the linker script is correct.
28 size_t paging_area_size = x86_paging_area_end - x86_paging_area_start;
29 MOS_ASSERT_X(paging_area_size >= sizeof(x86_pg_infra_t), "allocated paging area size is too small");
30 mos_debug("paging: provided size: 0x%zu, minimum required size: 0x%zu", paging_area_size, sizeof(x86_pg_infra_t));
31
32 // initialize the page directory
33 memset(x86_kpg_infra, 0, sizeof(x86_pg_infra_t));
34
35 pr_info("paging: setting up physical memory freelist...");
36 pmem_freelist_setup(x86_kpg_infra);
37
38 // map video memory
39 pr_info("paging: mapping video memory...");
40 pg_do_map_pages(x86_kpg_infra, BIOS_VADDR(X86_VIDEO_DEVICE_PADDR), X86_VIDEO_DEVICE_PADDR, 1, VM_GLOBAL | VM_WRITE);
41
42 pr_info("paging: mapping kernel space...");
43 mos_debug("mapping kernel code...");
44 const uintptr_t k_v_code_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_START);
45 const uintptr_t k_v_code_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_END);
46 const uintptr_t k_p_code_start = ALIGN_DOWN_TO_PAGE(k_v_code_start - MOS_KERNEL_START_VADDR);
47 pg_map_pages(x86_kpg_infra, k_v_code_start, k_p_code_start, (k_v_code_end - k_v_code_start) / MOS_PAGE_SIZE, VM_GLOBAL | VM_EXEC);
48
49 mos_debug("mapping kernel rodata...");
50 const uintptr_t k_v_rodata_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_START);
51 const uintptr_t k_v_rodata_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_END);
52 const uintptr_t k_p_rodata_start = ALIGN_DOWN_TO_PAGE(k_v_rodata_start - MOS_KERNEL_START_VADDR);
53 pg_map_pages(x86_kpg_infra, k_v_rodata_start, k_p_rodata_start, (k_v_rodata_end - k_v_rodata_start) / MOS_PAGE_SIZE, VM_GLOBAL);
54
55 mos_debug("mapping kernel writable data...");
56 const uintptr_t k_v_data_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_START);
57 const uintptr_t k_v_data_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_END);
58 const uintptr_t k_p_data_start = ALIGN_DOWN_TO_PAGE(k_v_data_start - MOS_KERNEL_START_VADDR);
59 pg_map_pages(x86_kpg_infra, k_v_data_start, k_p_data_start, (k_v_data_end - k_v_data_start) / MOS_PAGE_SIZE, VM_GLOBAL | VM_WRITE);
60}

62void x86_mm_enable_paging(void)
63{
64 mos_debug("paging: page directory vaddr at: %p", (void *) x86_kpg_infra->pgdir);
65 x86_enable_paging_impl(((uintptr_t) x86_kpg_infra->pgdir) - MOS_KERNEL_START_VADDR);
66 pr_info("paging: enabled");
67
68#if MOS_DEBUG
69 x86_mm_dump_page_table(x86_kpg_infra);
70 pmem_freelist_dump();
71#endif
72}

74void page_table_dump_range(x86_pg_infra_t *kpg_infra, size_t start_page, size_t end_page)
75{
76 pr_info(" VM Group " PTR_FMT " (+%zu pages):", (uintptr_t) start_page * MOS_PAGE_SIZE, end_page - start_page);
77
78 size_t i = start_page;
79 while (true)
80 {
81 if (i >= end_page)
82 return;
83
84 x86_pgtable_entry *pgtable = kpg_infra->pgtable + i;
85 MOS_ASSERT(pgtable->present);
86
87 uintptr_t p_range_start = pgtable->phys_addr << 12;
88 uintptr_t p_range_last = p_range_start;
89
90 uintptr_t v_range_start = i * MOS_PAGE_SIZE;
91
92 bool range_begin_usermode = pgtable->usermode;
93 bool range_begin_writable = pgtable->writable;
94
95 // ! Special case: end_page - start_page == 1
96 if (end_page - start_page == 1)
97 {
98 uintptr_t v_range_end = v_range_start + MOS_PAGE_SIZE;
99 uintptr_t p_range_end = p_range_start + MOS_PAGE_SIZE;
100 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
101 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
102 return;
103 }
104
105 while (true)
106 {
107 i++;
108 if (i >= end_page)
109 return;
110
111 x86_pgtable_entry *this_t = kpg_infra->pgtable + i;
112 MOS_ASSERT(this_t->present);
113
114 uintptr_t p_range_current = this_t->phys_addr << 12;
115
116 bool is_last_page = i == end_page - 1;
117 bool is_continuous = p_range_current == p_range_last + MOS_PAGE_SIZE && this_t->usermode == range_begin_usermode && this_t->writable == range_begin_writable;
118
119 if (is_continuous && !is_last_page)
120 {
121 p_range_last = p_range_current;
122 continue;
123 }
124
125 uintptr_t v_range_end = i * MOS_PAGE_SIZE;
126 uintptr_t p_range_end = p_range_current + MOS_PAGE_SIZE;
127 // '!is_continuous' case : print the previous range, but add one page
128 if (!is_continuous)
129 p_range_end = p_range_last + MOS_PAGE_SIZE;
130 else
131 v_range_end += MOS_PAGE_SIZE;
132 MOS_ASSERT(p_range_end - p_range_start == v_range_end - v_range_start);
133 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
134 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
135 break;
136 }
137 }

140void x86_mm_dump_page_table(x86_pg_infra_t *pg)
141{
142 pr_info("paging: dumping page table");
143 enum
144 {
145 PRESENT,
146 ABSENT
147 } current_state = pg->pgtable[0].present ? PRESENT : ABSENT;
148
149 size_t current_state_begin_pg = 0;
150
151 for (int pgd_i = 0; pgd_i < 1024; pgd_i++)
152 {
153 x86_pgdir_entry *pgd = pg->pgdir + pgd_i;
154 if (!pgd->present)
155 {
156 if (current_state == PRESENT)
157 {
158 size_t current_state_end_pg = pgd_i * 1024 + 0;
159 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
160 current_state = ABSENT;
161 current_state_begin_pg = current_state_end_pg;
162 }
163 continue;
164 }
165
166 x86_pgtable_entry *pgtable = NULL;
167 x86_pg_infra_t *dump_pg = NULL;
168 if (pgd_i < 768) // user pages
169 pgtable = pg->pgtable, dump_pg = pg;
170 else // HACK: kernel pages
171 pgtable = x86_kpg_infra->pgtable, dump_pg = x86_kpg_infra;
172
173 for (int pte_i = 0; pte_i < 1024; pte_i++)
174 {
175 x86_pgtable_entry *pte = &pgtable[pgd_i * 1024 + pte_i];
176
177 if (!pte->present)
178 {
179 if (current_state == PRESENT)
180 {
181 size_t current_state_end_pg = pgd_i * 1024 + pte_i;
182 page_table_dump_range(dump_pg, current_state_begin_pg, current_state_end_pg);
183 current_state = ABSENT;
184 current_state_begin_pg = current_state_end_pg;
185 }
186 }
187 else
188 {
189 if (current_state == ABSENT)
190 {
191 current_state = PRESENT;
192 current_state_begin_pg = pgd_i * 1024 + pte_i;
193 }
194 }
195 }
