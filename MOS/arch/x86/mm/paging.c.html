21static const uintptr_t x86_paging_area_start = (uintptr_t) &__MOS_X86_PAGING_AREA_START;
22static const uintptr_t x86_paging_area_end = (uintptr_t) &__MOS_X86_PAGING_AREA_END;
23x86_pg_infra_t *x86_kpg_infra = (void *) &__MOS_X86_PAGING_AREA_START;

25void x86_mm_prepare_paging()
26{
27 // validate if the memory region calculated from the linker script is correct.
28 size_t paging_area_size = x86_paging_area_end - x86_paging_area_start;
29 MOS_ASSERT_X(paging_area_size >= sizeof(x86_pg_infra_t), "allocated paging area size is too small");
30 mos_debug("paging: provided size: 0x%zu, minimum required size: 0x%zu", paging_area_size, sizeof(x86_pg_infra_t));
31
32 // initialize the page directory
33 memset(x86_kpg_infra, 0, sizeof(x86_pg_infra_t));
34
35 pr_info("paging: setting up physical memory freelist...");
36 pmem_freelist_setup(x86_kpg_infra);
37
38 pr_info("paging: setting up low 1MB identity mapping... (except for the NULL page)");
39 // skip the free list setup, use do_ version
40 pg_do_map_page(x86_kpg_infra, 0, 0, VM_NONE); // ! the zero page is not writable
41 pg_do_map_pages(x86_kpg_infra, MOS_PAGE_SIZE, MOS_PAGE_SIZE, 1 MB / MOS_PAGE_SIZE - 1, VM_GLOBAL | VM_WRITE);
42
43 // map video memory
44 pr_info("paging: mapping video memory...");
45 pg_do_map_pages(x86_kpg_infra, BIOS_VADDR(X86_VIDEO_DEVICE_PADDR), X86_VIDEO_DEVICE_PADDR, 1, VM_GLOBAL | VM_WRITE);
46
47 pr_info("paging: mapping kernel space...");
48 mos_debug("mapping kernel code...");
49 const uintptr_t k_v_code_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_START);
50 const uintptr_t k_v_code_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_END);
51 const uintptr_t k_p_code_start = ALIGN_DOWN_TO_PAGE(k_v_code_start - MOS_KERNEL_START_VADDR);
52 pg_map_pages(x86_kpg_infra, k_v_code_start, k_p_code_start, (k_v_code_end - k_v_code_start) / MOS_PAGE_SIZE, VM_GLOBAL | VM_EXECUTE);
53
54 mos_debug("mapping kernel rodata...");
55 const uintptr_t k_v_rodata_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_START);
56 const uintptr_t k_v_rodata_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_END);
57 const uintptr_t k_p_rodata_start = ALIGN_DOWN_TO_PAGE(k_v_rodata_start - MOS_KERNEL_START_VADDR);
58 pg_map_pages(x86_kpg_infra, k_v_rodata_start, k_p_rodata_start, (k_v_rodata_end - k_v_rodata_start) / MOS_PAGE_SIZE, VM_GLOBAL);
59
60 mos_debug("mapping kernel writable data...");
61 const uintptr_t k_v_data_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_START);
62 const uintptr_t k_v_data_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_END);
63 const uintptr_t k_p_data_start = ALIGN_DOWN_TO_PAGE(k_v_data_start - MOS_KERNEL_START_VADDR);
64 pg_map_pages(x86_kpg_infra, k_v_data_start, k_p_data_start, (k_v_data_end - k_v_data_start) / MOS_PAGE_SIZE, VM_GLOBAL | VM_WRITE);
65}

67void x86_mm_enable_paging(x86_pg_infra_t *infra)
68{
69 mos_debug("paging: page directory vaddr at: %p", (void *) infra->pgdir);
70 x86_enable_paging_impl(((uintptr_t) infra->pgdir) - MOS_KERNEL_START_VADDR);
71 pr_info("paging: enabled");
72
73#if MOS_DEBUG
74 x86_mm_dump_page_table(infra);
75 pmem_freelist_dump();
76#endif
77}

79void page_table_dump_range(x86_pg_infra_t *kpg_infra, size_t start_page, size_t end_page)
80{
81 pr_info(" VM Group " PTR_FMT " (+%zu pages):", (uintptr_t) start_page * MOS_PAGE_SIZE, end_page - start_page);
82
83 size_t i = start_page;
84 while (true)
85 {
86 if (i >= end_page)
87 return;
88
89 x86_pgtable_entry *pgtable = kpg_infra->pgtable + i;
90 MOS_ASSERT(pgtable->present);
91
92 uintptr_t p_range_start = pgtable->phys_addr << 12;
93 uintptr_t p_range_last = p_range_start;
94
95 uintptr_t v_range_start = i * MOS_PAGE_SIZE;
96
97 // ! Special case: end_page - start_page == 1
98 if (end_page - start_page == 1)
99 {
100 uintptr_t v_range_end = v_range_start + MOS_PAGE_SIZE;
101 uintptr_t p_range_end = p_range_start + MOS_PAGE_SIZE;
102 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT, v_range_start, v_range_end, p_range_start, p_range_end);
103 return;
104 }
105
106 while (true)
107 {
108 i++;
109 if (i >= end_page)
110 return;
111
112 x86_pgtable_entry *this_entry = kpg_infra->pgtable + i;
113 MOS_ASSERT(this_entry->present);
114
115 uintptr_t p_range_current = this_entry->phys_addr << 12;
116
117 bool is_last_page = i == end_page - 1;
118 bool is_continuous = p_range_current == p_range_last + MOS_PAGE_SIZE;
119
120 if (is_continuous && !is_last_page)
121 {
122 p_range_last = p_range_current;
123 continue;
124 }
125
126 uintptr_t v_range_end = i * MOS_PAGE_SIZE;
127 uintptr_t p_range_end = p_range_current + MOS_PAGE_SIZE;
128 // '!is_continuous' case : print the previous range, but add one page
129 if (!is_continuous)
130 p_range_end = p_range_last + MOS_PAGE_SIZE;
131 else
132 v_range_end += MOS_PAGE_SIZE;
133 MOS_ASSERT(p_range_end - p_range_start == v_range_end - v_range_start);
134 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT, v_range_start, v_range_end, p_range_start, p_range_end);
135 break;
136 }
137 }

140void x86_mm_dump_page_table(x86_pg_infra_t *pg)
141{
142 pr_info("paging: dumping page table");
143 enum
144 {
145 PRESENT,
146 ABSENT
147 } current_state = pg->pgtable[0].present ? PRESENT : ABSENT;
148
149 size_t current_state_begin_pg = 0;
150
151 for (int pgd_i = 0; pgd_i < 1024; pgd_i++)
152 {
153 x86_pgdir_entry *pgd = pg->pgdir + pgd_i;
154 if (!pgd->present)
155 {
156 if (current_state == PRESENT)
157 {
158 size_t current_state_end_pg = pgd_i * 1024 + 0;
159 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
160 current_state = ABSENT;
161 current_state_begin_pg = current_state_end_pg;
162 }
163 continue;
164 }
165
166 for (int pte_i = 0; pte_i < 1024; pte_i++)
167 {
168 x86_pgtable_entry *pte = &pg->pgtable[pgd_i * 1024 + pte_i];
169 if (!pte->present)
170 {
171 if (current_state == PRESENT)
172 {
173 size_t current_state_end_pg = pgd_i * 1024 + pte_i;
174 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
175 current_state = ABSENT;
176 current_state_begin_pg = current_state_end_pg;
177 }
178 }
179 else
180 {
181 if (current_state == ABSENT)
182 {
183 current_state = PRESENT;
184 current_state_begin_pg = pgd_i * 1024 + pte_i;
185 }
186 }
187 }
188 }

191paging_handle_t x86_um_pgd_create()
192{
193 x86_pg_infra_t *infra = pg_page_alloc(x86_kpg_infra, ALIGN_UP_TO_PAGE(sizeof(x86_pg_infra_t)) / MOS_PAGE_SIZE, PGALLOC_NONE);
194 memset(infra, 0, sizeof(x86_pg_infra_t));
195 paging_handle_t handle;
196 handle.ptr = (uintptr_t) infra;
197
198 // pg_do_map_pages(infra, 0, 0, 1, VM_NONE); // ! the zero page is not writable, nor readable by user
199 pg_do_map_pages(infra, MOS_PAGE_SIZE, MOS_PAGE_SIZE, 1 MB / MOS_PAGE_SIZE - 1, VM_GLOBAL | VM_WRITE);
200
201 // this is a bit of a hack, but it's the easiest way that I can think of ...
202 const int kernel_page_start = MOS_KERNEL_START_VADDR / (1024 * MOS_PAGE_SIZE);
203 for (int i = kernel_page_start; i < 1024; i++)
204 {
205 x86_pgdir_entry *pgd = &infra->pgdir[i];
206 pgd->present = true;
207 pgd->writable = true;
208 pgd->usermode = false;
209 // redirect it to the kernel page table
210 pgd->page_table_paddr = (uintptr_t) x86_kpg_infra->pgdir[i].page_table_paddr;
211 }
212 return handle;
213}
