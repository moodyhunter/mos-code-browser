25void x86_mm_prepare_paging()
26{
27 // validate if the memory region calculated from the linker script is correct.
28 size_t paging_area_size = x86_paging_area_end - x86_paging_area_start;
29 MOS_ASSERT_X(paging_area_size >= sizeof(x86_pg_infra_t), "allocated paging area size is too small");
30 mos_debug("paging: provided size: 0x%zu, minimum required size: 0x%zu", paging_area_size, sizeof(x86_pg_infra_t));
31
32 // initialize the page directory
33 memset(x86_kpg_infra, 0, sizeof(x86_pg_infra_t));
34
35 pr_info("paging: setting up physical memory freelist...");
36 pmem_freelist_setup(x86_kpg_infra);
37
38 pr_info("paging: setting up low 1MB identity mapping... (except for the NULL page)");
39 // skip the free list setup, use do_ version
40 pg_do_map_page(x86_kpg_infra, 0, 0, VM_NONE); // ! the zero page is not writable
41 pg_do_map_pages(x86_kpg_infra, MOS_PAGE_SIZE, MOS_PAGE_SIZE, 1 MB / MOS_PAGE_SIZE - 1, VM_GLOBAL | VM_WRITE);
42
43 // map video memory
44 pr_info("paging: mapping video memory...");
45 pg_do_map_pages(x86_kpg_infra, BIOS_VADDR(X86_VIDEO_DEVICE_PADDR), X86_VIDEO_DEVICE_PADDR, 1, VM_GLOBAL | VM_WRITE);
46
47 pr_info("paging: mapping kernel space...");
48 mos_debug("mapping kernel code...");
49 const uintptr_t k_v_code_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_START);
50 const uintptr_t k_v_code_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_CODE_END);
51 const uintptr_t k_p_code_start = ALIGN_DOWN_TO_PAGE(k_v_code_start - MOS_KERNEL_START_VADDR);
52 pg_map_pages(x86_kpg_infra, k_v_code_start, k_p_code_start, (k_v_code_end - k_v_code_start) / MOS_PAGE_SIZE, VM_GLOBAL | VM_EXEC);
53
54 mos_debug("mapping kernel rodata...");
55 const uintptr_t k_v_rodata_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_START);
56 const uintptr_t k_v_rodata_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RODATA_END);
57 const uintptr_t k_p_rodata_start = ALIGN_DOWN_TO_PAGE(k_v_rodata_start - MOS_KERNEL_START_VADDR);
58 pg_map_pages(x86_kpg_infra, k_v_rodata_start, k_p_rodata_start, (k_v_rodata_end - k_v_rodata_start) / MOS_PAGE_SIZE, VM_GLOBAL);
59
60 mos_debug("mapping kernel writable data...");
61 const uintptr_t k_v_data_start = ALIGN_DOWN_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_START);
62 const uintptr_t k_v_data_end = ALIGN_UP_TO_PAGE((uintptr_t) &__MOS_KERNEL_RW_END);
63 const uintptr_t k_p_data_start = ALIGN_DOWN_TO_PAGE(k_v_data_start - MOS_KERNEL_START_VADDR);
64 pg_map_pages(x86_kpg_infra, k_v_data_start, k_p_data_start, (k_v_data_end - k_v_data_start) / MOS_PAGE_SIZE, VM_GLOBAL | VM_WRITE);
65}

67void x86_mm_enable_paging(void)
68{
69 mos_debug("paging: page directory vaddr at: %p", (void *) x86_kpg_infra->pgdir);
70 x86_enable_paging_impl(((uintptr_t) x86_kpg_infra->pgdir) - MOS_KERNEL_START_VADDR);
71 pr_info("paging: enabled");
72
73#if MOS_DEBUG
74 x86_mm_dump_page_table(x86_kpg_infra);
75 pmem_freelist_dump();
76#endif
77}

79void page_table_dump_range(x86_pg_infra_t *kpg_infra, size_t start_page, size_t end_page)
80{
81 pr_info(" VM Group " PTR_FMT " (+%zu pages):", (uintptr_t) start_page * MOS_PAGE_SIZE, end_page - start_page);
82
83 size_t i = start_page;
84 while (true)
85 {
86 if (i >= end_page)
87 return;
88
89 x86_pgtable_entry *pgtable = kpg_infra->pgtable + i;
90 MOS_ASSERT(pgtable->present);
91
92 uintptr_t p_range_start = pgtable->phys_addr << 12;
93 uintptr_t p_range_last = p_range_start;
94
95 uintptr_t v_range_start = i * MOS_PAGE_SIZE;
96
97 bool range_begin_usermode = pgtable->usermode;
98 bool range_begin_writable = pgtable->writable;
99
100 // ! Special case: end_page - start_page == 1
101 if (end_page - start_page == 1)
102 {
103 uintptr_t v_range_end = v_range_start + MOS_PAGE_SIZE;
104 uintptr_t p_range_end = p_range_start + MOS_PAGE_SIZE;
105 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
106 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
107 return;
108 }
109
110 while (true)
111 {
112 i++;
113 if (i >= end_page)
114 return;
115
116 x86_pgtable_entry *this_t = kpg_infra->pgtable + i;
117 MOS_ASSERT(this_t->present);
118
119 uintptr_t p_range_current = this_t->phys_addr << 12;
120
121 bool is_last_page = i == end_page - 1;
122 bool is_continuous = p_range_current == p_range_last + MOS_PAGE_SIZE && this_t->usermode == range_begin_usermode && this_t->writable == range_begin_writable;
123
124 if (is_continuous && !is_last_page)
125 {
126 p_range_last = p_range_current;
127 continue;
128 }
129
130 uintptr_t v_range_end = i * MOS_PAGE_SIZE;
131 uintptr_t p_range_end = p_range_current + MOS_PAGE_SIZE;
132 // '!is_continuous' case : print the previous range, but add one page
133 if (!is_continuous)
134 p_range_end = p_range_last + MOS_PAGE_SIZE;
135 else
136 v_range_end += MOS_PAGE_SIZE;
137 MOS_ASSERT(p_range_end - p_range_start == v_range_end - v_range_start);
138 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
139 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
140 break;
141 }
142 }

145void x86_mm_dump_page_table(x86_pg_infra_t *pg)
146{
147 pr_info("paging: dumping page table");
148 enum
149 {
150 PRESENT,
151 ABSENT
152 } current_state = pg->pgtable[0].present ? PRESENT : ABSENT;
153
154 size_t current_state_begin_pg = 0;
155
156 for (int pgd_i = 0; pgd_i < 1024; pgd_i++)
157 {
158 x86_pgdir_entry *pgd = pg->pgdir + pgd_i;
159 if (!pgd->present)
160 {
161 if (current_state == PRESENT)
162 {
163 size_t current_state_end_pg = pgd_i * 1024 + 0;
164 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
165 current_state = ABSENT;
166 current_state_begin_pg = current_state_end_pg;
167 }
168 continue;
169 }
170
171 x86_pgtable_entry *pgtable = NULL;
172 x86_pg_infra_t *dump_pg = NULL;
173 if (pgd_i < 768) // user pages
174 pgtable = pg->pgtable, dump_pg = pg;
175 else // HACK: kernel pages
176 pgtable = x86_kpg_infra->pgtable, dump_pg = x86_kpg_infra;
177
178 for (int pte_i = 0; pte_i < 1024; pte_i++)
179 {
180 x86_pgtable_entry *pte = &pgtable[pgd_i * 1024 + pte_i];
181
182 if (!pte->present)
183 {
184 if (current_state == PRESENT)
185 {
186 size_t current_state_end_pg = pgd_i * 1024 + pte_i;
187 page_table_dump_range(dump_pg, current_state_begin_pg, current_state_end_pg);
188 current_state = ABSENT;
189 current_state_begin_pg = current_state_end_pg;
190 }
191 }
192 else
193 {
194 if (current_state == ABSENT)
195 {
196 current_state = PRESENT;
197 current_state_begin_pg = pgd_i * 1024 + pte_i;
198 }
199 }
200 }
201 }

204paging_handle_t x86_um_pgd_create()
205{
206 vmblock_t block = pg_page_alloc(x86_kpg_infra, ALIGN_UP_TO_PAGE(sizeof(x86_pg_infra_t)) / MOS_PAGE_SIZE, PGALLOC_HINT_KHEAP, VM_READ | VM_WRITE);
207 x86_pg_infra_t *infra = (x86_pg_infra_t *) block.vaddr;
208 memset(infra, 0, sizeof(x86_pg_infra_t));
209 paging_handle_t handle;
210 handle.ptr = (uintptr_t) infra;
211
212 // pg_do_map_pages(infra, 0, 0, 1, VM_NONE); // ! the zero page is not writable, nor readable by user
213 pg_do_map_pages(infra, MOS_PAGE_SIZE, MOS_PAGE_SIZE, 1 MB / MOS_PAGE_SIZE - 1, VM_GLOBAL | VM_WRITE);
214
215 // physical address of kernel page table
216 const uintptr_t kpgtable_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, (uintptr_t) x86_kpg_infra->pgtable);
217
218 // this is a bit of a hack, but it's the easiest way that I can think of ...
219 const int kernel_page_start = MOS_KERNEL_START_VADDR / (1024 * MOS_PAGE_SIZE);
220 for (int i = kernel_page_start; i < 1024; i++)
221 {
222 x86_pgdir_entry *pgd = &infra->pgdir[i];
223 pgd->present = true;
224 pgd->writable = true;
225 pgd->usermode = false;
226 // redirect it to the kernel page table
227 // use pre-allocated (pre-calculated) physical address, otherwise some newly mapped pgdirs won't be applied correctly
228 pgd->page_table_paddr = (kpgtable_paddr + i * 1024 * sizeof(x86_pgtable_entry)) >> 12;
229 }
230 return handle;
231}

274vmblock_t x86_mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
275{
276 x86_pg_infra_t *from_infra = x86_get_pg_infra(from);
277 x86_pg_infra_t *to_infra = x86_get_pg_infra(to);
278
279 // uintptr_t start_paddr = pg_page_get_mapped_paddr(from_infra, fvaddr);
280
281 for (size_t i = 0; i < npages; i++)
282 {
283 uintptr_t from_vaddr = fvaddr + i * MOS_PAGE_SIZE;
284 uintptr_t to_vaddr = tvaddr + i * MOS_PAGE_SIZE;
285 // uintptr_t expected_paddr = start_paddr + i * MOS_PAGE_SIZE;
286 uintptr_t paddr = pg_page_get_mapped_paddr(from_infra, from_vaddr);
287 vm_flags flags = pg_page_get_flags(from_infra, from_vaddr);
288 pg_do_map_page(to_infra, to_vaddr, paddr, flags);
289 }
290
291 vmblock_t block = {
292 .vaddr = tvaddr,
293 .npages = npages,
294 .flags = pg_page_get_flags(from_infra, fvaddr),
295 };
296
297 return block;
298}
