21void x86_mm_prepare_paging()
22{
23 // validate if the memory region calculated from the linker script is correct.
24 const size_t paging_area_size = (uintptr_t) &__MOS_X86_PAGING_AREA_END - (uintptr_t) &__MOS_X86_PAGING_AREA_START;
25 MOS_ASSERT_X(paging_area_size >= sizeof(x86_pg_infra_t), "allocated paging area size is too small");
26 mos_debug("paging: provided size: 0x%zu, minimum required size: 0x%zu", paging_area_size, sizeof(x86_pg_infra_t));
27
28 // initialize the page directory
29 memzero(x86_kpg_infra, sizeof(x86_pg_infra_t));
30
31 pr_info("paging: mapping kernel space...");
32 mm_map_pages(current_cpu->pagetable, x86_platform.k_code);
33 mm_map_pages(current_cpu->pagetable, x86_platform.k_rodata);
34 mm_map_pages(current_cpu->pagetable, x86_platform.k_rwdata);
35}
36
37void x86_mm_enable_paging(void)
38{
39 mos_debug("paging: page directory vaddr at: %p", (void *) x86_kpg_infra->pgdir);
40 x86_enable_paging_impl(((uintptr_t) x86_kpg_infra->pgdir) - MOS_KERNEL_START_VADDR);
41 pr_info("paging: enabled");
42
43#if MOS_DEBUG
44 x86_mm_dump_page_table(x86_kpg_infra);
45 pmem_freelist_dump();
46#endif
47}

49void page_table_dump_range(x86_pg_infra_t *kpg_infra, size_t start_page, size_t end_page)
50{
51 pr_info(" VM Group " PTR_FMT " (+%zu pages):", (uintptr_t) start_page * MOS_PAGE_SIZE, end_page - start_page);
52
53 size_t i = start_page;
54 while (true)
55 {
56 if (i >= end_page)
57 return;
58
59 x86_pgtable_entry *pgtable = kpg_infra->pgtable + i;
60 MOS_ASSERT(pgtable->present);
61
62 uintptr_t p_range_start = pgtable->phys_addr << 12;
63 uintptr_t p_range_last = p_range_start;
64
65 uintptr_t v_range_start = i * MOS_PAGE_SIZE;
66
67 bool range_begin_usermode = pgtable->usermode;
68 bool range_begin_writable = pgtable->writable;
69
70 // ! Special case: end_page - start_page == 1
71 if (end_page - start_page == 1)
72 {
73 uintptr_t v_range_end = v_range_start + MOS_PAGE_SIZE;
74 uintptr_t p_range_end = p_range_start + MOS_PAGE_SIZE;
75 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
76 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
77 return;
78 }
79
80 while (true)
81 {
82 i++;
83 if (i >= end_page)
84 return;
85
86 x86_pgtable_entry *this_t = kpg_infra->pgtable + i;
87 MOS_ASSERT(this_t->present);
88
89 uintptr_t p_range_current = this_t->phys_addr << 12;
90
91 bool is_last_page = i == end_page - 1;
92 bool is_continuous = p_range_current == p_range_last + MOS_PAGE_SIZE && this_t->usermode == range_begin_usermode && this_t->writable == range_begin_writable;
93
94 if (is_continuous && !is_last_page)
95 {
96 p_range_last = p_range_current;
97 continue;
98 }
99
100 uintptr_t v_range_end = i * MOS_PAGE_SIZE;
101 uintptr_t p_range_end = p_range_current + MOS_PAGE_SIZE;
102 // '!is_continuous' case : print the previous range, but add one page
103 if (!is_continuous)
104 p_range_end = p_range_last + MOS_PAGE_SIZE;
105 else
106 v_range_end += MOS_PAGE_SIZE;
107 MOS_ASSERT(p_range_end - p_range_start == v_range_end - v_range_start);
108 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
109 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
110 break;
111 }
112 }
113}

115void x86_mm_dump_page_table(x86_pg_infra_t *pg)
116{
117 pr_info("paging: dumping page table");
118 enum
119 {
120 PRESENT,
121 ABSENT
122 } current_state = pg->pgtable[0].present ? PRESENT : ABSENT;
123
124 size_t current_state_begin_pg = 0;
125
126 for (int pgd_i = 0; pgd_i < 1024; pgd_i++)
127 {
128 x86_pgdir_entry *pgd = pg->pgdir + pgd_i;
129 if (!pgd->present)
130 {
131 if (current_state == PRESENT)
132 {
133 size_t current_state_end_pg = pgd_i * 1024 + 0;
134 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
135 current_state = ABSENT;
136 current_state_begin_pg = current_state_end_pg;
137 }
138 continue;
139 }
140
141 x86_pgtable_entry *pgtable = NULL;
142 x86_pg_infra_t *dump_pg = NULL;
143 if (pgd_i < 768) // user pages
144 pgtable = pg->pgtable, dump_pg = pg;
145 else // HACK: kernel pages
146 pgtable = x86_kpg_infra->pgtable, dump_pg = x86_kpg_infra;
147
148 for (int pte_i = 0; pte_i < 1024; pte_i++)
149 {
150 x86_pgtable_entry *pte = &pgtable[pgd_i * 1024 + pte_i];
151
152 if (!pte->present)
153 {
154 if (current_state == PRESENT)
155 {
156 size_t current_state_end_pg = pgd_i * 1024 + pte_i;
157 page_table_dump_range(dump_pg, current_state_begin_pg, current_state_end_pg);
158 current_state = ABSENT;
159 current_state_begin_pg = current_state_end_pg;
160 }
161 }
162 else
163 {
164 if (current_state == ABSENT)
165 {
166 current_state = PRESENT;
167 current_state_begin_pg = pgd_i * 1024 + pte_i;
168 }
169 }
170 }
171 }
