16void x86_mm_prepare_paging()
17{
18 // validate if the memory region calculated from the linker script is correct.
19 const size_t paging_area_size = (uintptr_t) &__MOS_X86_PAGING_AREA_END - (uintptr_t) &__MOS_X86_PAGING_AREA_START;
20 MOS_ASSERT_X(paging_area_size >= sizeof(x86_pg_infra_t), "allocated paging area size is too small");
21 mos_debug(x86_paging, "provided size: 0x%zu, minimum required size: 0x%zu", paging_area_size, sizeof(x86_pg_infra_t));
22
23 // initialize the page directory
24 memzero(x86_kpg_infra, sizeof(x86_pg_infra_t));
25}
26
27void x86_mm_enable_paging(void)
28{
29 mos_debug(x86_paging, "page directory vaddr at: %p", (void *) x86_kpg_infra->pgdir);
30 x86_enable_paging_impl(((uintptr_t) x86_kpg_infra->pgdir) - MOS_KERNEL_START_VADDR);
31 pr_info("paging: enabled");
32
33#if MOS_DEBUG_FEATURE(x86_paging)
34 x86_mm_dump_page_table(x86_kpg_infra);
35#endif
36}

38void page_table_dump_range(x86_pg_infra_t *kpg_infra, size_t start_page, size_t end_page)
39{
40 pr_info(" VM Group " PTR_FMT " (+%zu pages):", (uintptr_t) start_page * MOS_PAGE_SIZE, end_page - start_page);
41
42 size_t i = start_page;
43 while (true)
44 {
45 if (i >= end_page)
46 return;
47
48 x86_pgtable_entry *pgtable = kpg_infra->pgtable + i;
49 MOS_ASSERT(pgtable->present);
50
51 uintptr_t p_range_start = pgtable->phys_addr << 12;
52 uintptr_t p_range_last = p_range_start;
53
54 uintptr_t v_range_start = i * MOS_PAGE_SIZE;
55
56 bool range_begin_usermode = pgtable->usermode;
57 bool range_begin_writable = pgtable->writable;
58
59 // ! Special case: end_page - start_page == 1
60 if (end_page - start_page == 1)
61 {
62 uintptr_t v_range_end = v_range_start + MOS_PAGE_SIZE;
63 uintptr_t p_range_end = p_range_start + MOS_PAGE_SIZE;
64 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
65 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
66 return;
67 }
68
69 while (true)
70 {
71 i++;
72 if (i >= end_page)
73 return;
74
75 x86_pgtable_entry *this_t = kpg_infra->pgtable + i;
76 MOS_ASSERT(this_t->present);
77
78 uintptr_t p_range_current = this_t->phys_addr << 12;
79
80 bool is_last_page = i == end_page - 1;
81 bool is_continuous = p_range_current == p_range_last + MOS_PAGE_SIZE && this_t->usermode == range_begin_usermode && this_t->writable == range_begin_writable;
82
83 if (is_continuous && !is_last_page)
84 {
85 p_range_last = p_range_current;
86 continue;
87 }
88
89 uintptr_t v_range_end = i * MOS_PAGE_SIZE;
90 uintptr_t p_range_end = p_range_current + MOS_PAGE_SIZE;
91 // '!is_continuous' case : print the previous range, but add one page
92 if (!is_continuous)
93 p_range_end = p_range_last + MOS_PAGE_SIZE;
94 else
95 v_range_end += MOS_PAGE_SIZE;
96 MOS_ASSERT(p_range_end - p_range_start == v_range_end - v_range_start);
97 pr_info(" " PTR_FMT "-" PTR_FMT " -> " PTR_FMT "-" PTR_FMT " (%s, %s)", v_range_start, v_range_end, p_range_start, p_range_end,
98 range_begin_writable ? "rw" : "ro", range_begin_usermode ? "user" : "kernel");
99 break;
100 }
101 }

104void x86_mm_dump_page_table(x86_pg_infra_t *pg)
105{
106 pr_info("paging: dumping page table");
107 enum
108 {
109 PRESENT,
110 ABSENT
111 } current_state = pg->pgtable[0].present ? PRESENT : ABSENT;
112
113 size_t current_state_begin_pg = 0;
114
115 for (int pgd_i = 0; pgd_i < 1024; pgd_i++)
116 {
117 x86_pgdir_entry *pgd = pg->pgdir + pgd_i;
118 if (!pgd->present)
119 {
120 if (current_state == PRESENT)
121 {
122 size_t current_state_end_pg = pgd_i * 1024 + 0;
123 page_table_dump_range(pg, current_state_begin_pg, current_state_end_pg);
124 current_state = ABSENT;
125 current_state_begin_pg = current_state_end_pg;
126 }
127 continue;
128 }
129
130 x86_pgtable_entry *pgtable = NULL;
131 x86_pg_infra_t *dump_pg = NULL;
132 if (pgd_i < 768) // user pages
133 pgtable = pg->pgtable, dump_pg = pg;
134 else // HACK: kernel pages
135 pgtable = x86_kpg_infra->pgtable, dump_pg = x86_kpg_infra;
136
137 for (int pte_i = 0; pte_i < 1024; pte_i++)
138 {
139 x86_pgtable_entry *pte = &pgtable[pgd_i * 1024 + pte_i];
140
141 if (!pte->present)
142 {
143 if (current_state == PRESENT)
144 {
145 size_t current_state_end_pg = pgd_i * 1024 + pte_i;
146 page_table_dump_range(dump_pg, current_state_begin_pg, current_state_end_pg);
147 current_state = ABSENT;
148 current_state_begin_pg = current_state_end_pg;
149 }
150 }
151 else
152 {
153 if (current_state == ABSENT)
154 {
155 current_state = PRESENT;
156 current_state_begin_pg = pgd_i * 1024 + pte_i;
157 }
158 }
159 }
