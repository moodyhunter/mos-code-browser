12#define RESERVED_LOMEM 1 MB
13
14#define PMEM_FREELIST_SIZE_FOR(mem_size) (((mem_size) / 2 / MOS_PAGE_SIZE) * sizeof(pmem_range_t))
15
16typedef struct pmem_range_t
17{
18 struct pmem_range_t *next;
19 uintptr_t paddr;
20 size_t n_pages;
21} pmem_range_t;
22
23extern const void __MOS_X86_PMEM_FREE_LIST;
24static pmem_range_t *const pmem_freelist_storage = (pmem_range_t *) &__MOS_X86_PMEM_FREE_LIST;
25static pmem_range_t *pmem_freelist = (pmem_range_t *) &__MOS_X86_PMEM_FREE_LIST;
26static size_t pmem_freelist_count = 0;
27
28size_t pmem_freelist_size()
29{
30 return ALIGN_UP_TO_PAGE(PMEM_FREELIST_SIZE_FOR(x86_mem_available));
31}

33void pmem_freelist_dump()
34{
35 pr_info("paging: pmem freelist has %zu entries", pmem_freelist_count);
36 pmem_range_t *this = pmem_freelist;
37 while (this)
38 {
39 const uintptr_t pstart = this->paddr;
40 const uintptr_t pend = pstart + this->n_pages * MOS_PAGE_SIZE;
41 char sbuf[32];
42 format_size(sbuf, sizeof(sbuf), this->n_pages * MOS_PAGE_SIZE);
43 pr_info(" [" PTR_FMT "] entry: " PTR_FMT "-" PTR_FMT " (%zu page(s), %s)", (uintptr_t) this, pstart, pend, this->n_pages, sbuf);
44 this = this->next;
45 }
46}

48void pmem_freelist_setup(void)
49{
50 size_t list_size = pmem_freelist_size();
51 pr_info2("paging: %zu bytes (aligned) required for physical memory freelist", list_size);
52
53 memset(pmem_freelist, 0, list_size);
54
55 // add current physical memory region to the freelist
56 for (size_t i = 0; i < x86_mem_regions_count; i++)
57 {
58 memblock_t *r = &x86_mem_regions[i];
59 if (!r->available)
60 continue;
61
62 if (r->paddr + r->size_bytes < RESERVED_LOMEM)
63 {
64 pr_emph("paging: ignored low memory: " PTR_FMT "-" PTR_FMT " (%zu bytes)", r->paddr, r->paddr + r->size_bytes, r->size_bytes);
65 continue;
66 }
67
68 size_t alignment_loss = pmem_freelist_add_region(r->paddr, r->size_bytes);
69 if (alignment_loss)
70 pr_emph("paging: %zu bytes of memory loss due to alignment", alignment_loss);
71 }
72}

74#define this_start (this->paddr)
75#define this_end (this_start + this->n_pages * MOS_PAGE_SIZE)
76
77size_t pmem_freelist_add_region(uintptr_t start_addr, size_t size_bytes)
78{
79 const uintptr_t aligned_start = ALIGN_UP_TO_PAGE(start_addr);
80 const uintptr_t aligned_end = ALIGN_DOWN_TO_PAGE(start_addr + size_bytes);
81
82 const size_t pages_in_region = (aligned_end - aligned_start) / MOS_PAGE_SIZE;
83
84 mos_debug("paging: adding physical memory region " PTR_FMT "-" PTR_FMT " to freelist.", aligned_start, aligned_end);
85
86 pmem_range_t *this = pmem_freelist;
87 pmem_range_t *prev = NULL;
88
89 while (this)
90 {
91 // * if the aligned start/end address is within the current region, it overlaps
92 if ((this_start <= aligned_start && aligned_start < this_end) || (this_start < aligned_end && aligned_end <= this_end))
93 {
94 mos_panic("new pmem " PTR_FMT "-" PTR_FMT " overlaps with " PTR_FMT "-" PTR_FMT, aligned_start, aligned_end, this_start, this_end);
95 return 0;
96 }
97
98 // prepend to 'this' region
99 if (aligned_end == this_start)
100 {
101 uintptr_t new_end = this_start + pages_in_region * MOS_PAGE_SIZE;
102 mos_debug("paging: enlarge range [" PTR_FMT "-" PTR_FMT "]: starts at " PTR_FMT, this_start, this_end, new_end);
103 this->paddr = aligned_start;
104 this->n_pages += pages_in_region;
105 goto end;
106 }
107
108 // append to the previous region (if possible)
109 if (prev)
110 {
111 // * the new region "may" overlap with next region
112 // * if we 'append' to the current region, we will not be able to check for such overlap
113 // * (because we would have 'goto end'-ed)
114 const uintptr_t prev_start = prev->paddr;
115 const uintptr_t prev_end = prev_start + prev->n_pages * MOS_PAGE_SIZE;
116
117 if (aligned_start == prev_end)
118 {
119 prev->n_pages += pages_in_region;
120 const uintptr_t new_ends = prev_end + pages_in_region * MOS_PAGE_SIZE;
121 mos_debug("paging: enlarged " PTR_FMT "-" PTR_FMT ": ends at " PTR_FMT, prev_start, prev_end, new_ends);
122 goto end;
123 }
124 }
125
126 // same as above, we only 'prepend' the region before 'this'
127 if (aligned_end < this_start)
128 {
129 pmem_range_t *new_range = &pmem_freelist_storage[pmem_freelist_count++];
130 new_range->paddr = aligned_start;
131 new_range->n_pages = pages_in_region;
132 new_range->next = this;
133 if (prev)
134 prev->next = new_range;
135 else
136 pmem_freelist = new_range;
137 goto end;
138 }
139
140 prev = this;
141 this = this->next;
142 }
143
144 MOS_ASSERT(this == NULL && prev != NULL);
145 pmem_range_t *new_range = &pmem_freelist_storage[pmem_freelist_count++];
146 prev->next = new_range;
147
148 // the freelist is in the correct order, so we can insert the new range here
149 new_range->paddr = aligned_start;
150 new_range->n_pages = pages_in_region;
151 new_range->next = NULL;
152
153end:
154 // return the amount of memory that was lost due to alignment
155 return (aligned_start - start_addr) + ((start_addr + size_bytes) - aligned_end);
156}

158void pmem_freelist_remove_region(uintptr_t start_addr, size_t size_bytes)
159{
160 const uintptr_t end_addr = start_addr + size_bytes;
161
162 // the addresses must be aligned to page size
163 MOS_ASSERT(start_addr % MOS_PAGE_SIZE == 0);
164 MOS_ASSERT(end_addr % MOS_PAGE_SIZE == 0);
165
166 const size_t pages_in_region = (end_addr - start_addr) / MOS_PAGE_SIZE;
167
168 mos_debug("paging: removing physical memory region " PTR_FMT "-" PTR_FMT " from freelist.", start_addr, end_addr);
169
170 bool needs_cleanup = false;
171 bool freed = false;
172 void *cleanup_target_memptr = NULL;
173
174 pmem_range_t *this = pmem_freelist;
175 pmem_range_t *prev = NULL;
176 while (this)
177 {
178 pmem_range_t *next = this->next;
179 if (this_start <= start_addr && this_end >= end_addr)
180 {
181 // starts before this region
182 if (start_addr < this_start)
183 continue;
184
185 // or ends after this region
186 if (end_addr > this_end)
187 continue;
188
189 const size_t part_1_size = start_addr - this_start;
190 const size_t part_2_size = this_end - end_addr;
191
192 if (part_1_size == 0 && part_2_size != 0)
193 {
194 // part 1 is empty, which means we are removing from the front of the region
195 this->paddr = end_addr;
196 this->n_pages = part_2_size / MOS_PAGE_SIZE;
197 mos_debug("paging: this pmem block now starts at " PTR_FMT ", with %zu pages", this_start, this->n_pages);
198 }
199 else if (part_1_size != 0 && part_2_size == 0)
200 {
201 // part 2 is empty, which means we are removing the tail of part 1
202 mos_debug("paging: shrunk " PTR_FMT "-" PTR_FMT ": ends at " PTR_FMT, this_start, this_end,
203 this_start + (this->n_pages - pages_in_region) * MOS_PAGE_SIZE);
204 this->n_pages -= pages_in_region;
205 }
206 else if (part_1_size == 0 && part_2_size == 0)
207 {
208 // part 1 and part 2 are empty, so remove this entry
209 if (prev)
210 prev->next = next;
211 else
212 pmem_freelist = next;
213
214 // have to do some cleanup
215 needs_cleanup = true;
216 cleanup_target_memptr = this;
217 mos_debug("paging: removed " PTR_FMT "-" PTR_FMT " from freelist.", this_start, this_end);
218 }
219 else
220 {
221 // neither part 1 nor part 2 is empty, so we have to allocate a new entry for part 2
222 mos_debug("paging: split " PTR_FMT "-" PTR_FMT " into " PTR_FMT "-" PTR_FMT " and " PTR_FMT "-" PTR_FMT, this_start, this_end,
223 this_start, start_addr, end_addr, end_addr + part_2_size);
224 pmem_range_t *new = (pmem_range_t *) pmem_freelist_storage + pmem_freelist_count++;
225 memset(new, 0, sizeof(pmem_range_t));
226 new->next = next;
227 this->next = new;
228 this->n_pages = part_1_size / MOS_PAGE_SIZE;
229 new->paddr = end_addr;
230 new->n_pages = part_2_size / MOS_PAGE_SIZE;
231 }
232 freed = true;
233 break;
234 }
235 prev = this;
236 this = next;
237 }
238
239 if (!freed)
240 mos_panic("paging: " PTR_FMT "-" PTR_FMT " is not in the freelist.", start_addr, end_addr);
241
242 if (needs_cleanup)
243 {
244 // firstly find who is at 'pmem_freelist_count' that should be copied to 'cleanup_target_memptr'
245 // minus 1 because it's a count, make it an index
246 pmem_range_t *copy_source = &pmem_freelist_storage[pmem_freelist_count - 1];
247 pmem_freelist_count--;
248
249 pmem_range_t *r_prev = NULL;
250 for (pmem_range_t *r_this = pmem_freelist; r_this != copy_source && r_this->next; r_this = r_this->next)
251 r_prev = r_this;
252
253 if (cleanup_target_memptr != copy_source)
254 {
255 memmove(cleanup_target_memptr, copy_source, sizeof(pmem_range_t));
256 memset(copy_source, 0, sizeof(pmem_range_t)); // prevent from being used again
257
258 if (copy_source != r_prev->next)
259 {
260 pmem_freelist = cleanup_target_memptr;
261 }
262 else
263 {
264 r_prev->next = cleanup_target_memptr;
265 }
266 }
267 }
268}

273uintptr_t pmem_freelist_find_free(size_t pages)
274{
275 pmem_range_t *this = pmem_freelist;
276 while (this)
277 {
278 if (this->n_pages >= pages)
279 {
280 // ! do not remove the range from the freelist
281 uintptr_t addr = this->paddr;
282 mos_debug("paging: allocated %zu pages from freelist, starting at " PTR_FMT, pages, addr);
283 return addr;
284 }
285 this = this->next;
286 }
287
288 // !! OOM
289 mos_warn("paging: out of memory?");
290 MOS_UNREACHABLE();
291 return 0;
292}
