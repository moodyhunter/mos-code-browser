49#define STARTUP_ASSERT(cond, type) \
50 do \
51 { \
52 if (!unlikely(cond)) \
53 { \
54 print_debug_info('E', type); \
55 while (1) \
56 __asm__("hlt"); \
57 } \
58 } while (0)

62__startup_code should_inline void print_debug_info(char a, char b)
63{
64 *((volatile char *) video_device_address + 0) = a;
65 *((volatile char *) video_device_address + 1) = LightMagenta;
66 *((volatile char *) video_device_address + 2) = b;
67 *((volatile char *) video_device_address + 3) = Gray;
68 *((volatile char *) video_device_address + 4) = '\0';
69 *((volatile char *) video_device_address + 5) = White;
70}

72__startup_code should_inline void startup_setup_pgd(int pgdid, x86_pgtable_entry *pgtable)
73{
74 STARTUP_ASSERT(pgdid < 1024, 'r'); // pgdid must be less than 1024
75 STARTUP_ASSERT(pgtable != NULL, 't'); // pgtable must not be NULL
76 STARTUP_ASSERT((uintptr_t) pgtable % 4096 == 0, 'a'); // pgtable must be aligned to 4096
77 STARTUP_ASSERT(!startup_pgd[pgdid].present, 'p'); // pgdir entry already present
78
79 mos_startup_memzero((void *) (startup_pgd + pgdid), sizeof(x86_pgdir_entry));
80 startup_pgd[pgdid].present = true;
81 startup_pgd[pgdid].page_table_paddr = (uintptr_t) pgtable >> 12;
82}

84__startup_code void mos_startup_map_single_page(uintptr_t vaddr, uintptr_t paddr, vm_flags flags)
85{
86 const size_t dir_index = vaddr >> 22;
87 const size_t table_index = (vaddr >> 12) & 0x3FF;
88
89 __startup_rwdata static int used_pgd = 0;
90
91 x86_pgdir_entry *this_dir = &startup_pgd[dir_index];
92 if (!this_dir->present)
93 {
94 size_t pagedir_entry_table_index = 0;
95
96 if (vaddr - (dir_index << 22) > 0)
97 pagedir_entry_table_index = used_pgd * 1024;
98 else if (vaddr - (dir_index << 22) == 0)
99 pagedir_entry_table_index = ALIGN_UP(used_pgd * 1024, 1024);
100 else
101 STARTUP_ASSERT(false, 'v');
102
103 startup_setup_pgd(dir_index, &pages[pagedir_entry_table_index]);
104 used_pgd++;
105 }
106
107 // this_dir should be present now
108 STARTUP_ASSERT(this_dir->present, 'm');
109 this_dir->writable = flags & VM_WRITE;
110
111 x86_pgtable_entry *this_table = (x86_pgtable_entry *) (this_dir->page_table_paddr << 12) + table_index;
112 if (this_table->present)
113 {
114 STARTUP_ASSERT(this_table->phys_addr == (paddr >> 12), 'd'); // page already mapped to different physical address
115 return;
116 }
117 mos_startup_memzero((void *) this_table, sizeof(x86_pgtable_entry));
118 this_table->present = true;
119 this_table->phys_addr = (uintptr_t) paddr >> 12;
120 this_table->writable = flags & VM_WRITE;
121 this_table->global = flags & VM_GLOBAL;
122 this_table->cache_disabled = flags & VM_CACHE_DISABLED;
123 this_table->write_through = flags & VM_WRITE_THROUGH;
124
125 __asm__ volatile("invlpg (%0)" ::"r"(vaddr) : "memory");
126}

128// x86_startup does the following:
129// 1. Identity map the VGA buffer to kvirt address space
130// 2. Identity map the code section '.mos.startup*'
131// 3. Map the kernel code, rodata, data, bss and kpage tables
132// 4. Find the initrd and map it
133// 5. Find a possible location for the kernel stack and map it
134// 4. Enable paging
135__startup_code asmlinkage void x86_startup(x86_startup_info *startup)
136{
137 char step = 'a';
138
139 STARTUP_ASSERT(startup->mb_magic == MULTIBOOT_BOOTLOADER_MAGIC, '1');
140 STARTUP_ASSERT(startup->mb_info->flags & MULTIBOOT_INFO_MEM_MAP, '2');
141
142 mos_startup_memzero((void *) startup_pgd, sizeof(x86_pgdir_entry) * 1024);
143 mos_startup_memzero((void *) pages, 512 KB);
144
145 debug_print_step();
146 mos_startup_map_identity((uintptr_t) startup->mb_info, sizeof(multiboot_info_t), VM_NONE);
147
148 // multiboot stuff
149 if (startup->mb_info->flags & MULTIBOOT_INFO_CMDLINE)
150 mos_startup_map_identity((uintptr_t) startup->mb_info->cmdline, mos_startup_strlen(startup->mb_info->cmdline), VM_NONE);
151
152 STARTUP_ASSERT(startup->mb_info->mmap_addr, 'm');
153 mos_startup_map_identity((uintptr_t) startup->mb_info->mmap_addr, startup->mb_info->mmap_length * sizeof(multiboot_memory_map_t), VM_NONE);
154
155 // map the VGA buffer, from 0xB8000
156 mos_startup_map_bios(X86_VIDEO_DEVICE, VIDEO_WIDTH * VIDEO_HEIGHT * 2, VM_WRITE);
157
158 // map the bios memory regions
159 mos_startup_map_bios(X86_BIOS_MEMREGION_PADDR, BIOS_MEMREGION_SIZE, VM_READ);
160 mos_startup_map_bios(X86_EBDA_MEMREGION_PADDR, EBDA_MEMREGION_SIZE, VM_READ);
161
162 // ! we do not separate the startup code and data to simplify the setup.
163 // ! this page directory will be removed as soon as the kernel is loaded, it shouldn't be a problem.
164 mos_startup_map_identity(startup_start, startup_end - startup_start, VM_READ | VM_WRITE | VM_EXEC);
165
166 debug_print_step();
167 const size_t kernel_code_pgsize = ALIGN_UP_TO_PAGE(kernel_code_vend - kernel_code_vstart) / MOS_PAGE_SIZE;
168 mos_startup_map_pages(kernel_code_vstart, kernel_code_vstart - MOS_KERNEL_START_VADDR, kernel_code_pgsize, VM_EXEC);
169
170 const size_t kernel_ro_pgsize = ALIGN_UP_TO_PAGE(kernel_ro_vend - kernel_ro_vstart) / MOS_PAGE_SIZE;
171 mos_startup_map_pages(kernel_ro_vstart, kernel_ro_vstart - MOS_KERNEL_START_VADDR, kernel_ro_pgsize, VM_NONE);
172
173 const size_t kernel_rw_pgsize = ALIGN_UP_TO_PAGE(kernel_rw_vend - kernel_rw_vstart) / MOS_PAGE_SIZE;
174 mos_startup_map_pages(kernel_rw_vstart, kernel_rw_vstart - MOS_KERNEL_START_VADDR, kernel_rw_pgsize, VM_WRITE);
175
176 if (startup->mb_info->flags & MULTIBOOT_INFO_MODS && startup->mb_info->mods_count != 0)
177 {
178 multiboot_module_t *mod = (multiboot_module_t *) startup->mb_info->mods_addr;
179 const size_t initrd_pgsize = ALIGN_UP_TO_PAGE(mod->mod_end - mod->mod_start) / MOS_PAGE_SIZE;
180 startup->initrd_size = mod->mod_end - mod->mod_start;
181 mos_startup_map_pages(MOS_X86_INITRD_VADDR, mod->mod_start, initrd_pgsize, VM_NONE);
182 debug_print_step();
183 }
184
185 __asm__ volatile("mov %0, %%cr3" ::"r"(startup_pgd));
186 debug_print_step();
187
188 __asm__ volatile("mov %%cr0, %%eax; or $0x80000000, %%eax; mov %%eax, %%cr0" ::: "eax");
189 video_device_address = BIOS_VADDR(X86_VIDEO_DEVICE);
190 debug_print_step();
191
192 acpi_rsdp_t *rsdp = acpi_find_rsdp(BIOS_VADDR(X86_EBDA_MEMREGION_PADDR), EBDA_MEMREGION_SIZE);
193 if (!rsdp)
194 rsdp = acpi_find_rsdp(BIOS_VADDR(X86_BIOS_MEMREGION_PADDR), BIOS_MEMREGION_SIZE);
195 STARTUP_ASSERT(rsdp, 'R');
196
197 const multiboot_memory_map_t *map_entries = (multiboot_memory_map_t *) startup->mb_info->mmap_addr;
198 for (u32 i = 0; i < startup->mb_info->mmap_length / sizeof(multiboot_memory_map_t); i++)
199 {
200 u64 region_length = map_entries[i].len;
201 u64 region_base = map_entries[i].phys_addr;
202 if (rsdp->v1.rsdt_addr >= region_base && rsdp->v1.rsdt_addr < region_base + region_length)
203 {
204 mos_startup_map_bios(region_base, region_length, VM_NONE);
205 startup->bios_region_start = region_base;
206 break;
207 }
208 }
209
210 debug_print_step();
211}
