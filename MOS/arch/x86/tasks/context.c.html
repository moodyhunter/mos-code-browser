22void x86_setup_thread_context(thread_t *thread, downwards_stack_t *proxy_stack, thread_entry_t entry, void *arg)
23{
24 x86_thread_context_t *context = kmalloc(sizeof(x86_thread_context_t));
25 context->inner.instruction = (uintptr_t) entry;
26 context->ebp = 0;
27
28 thread->context = &context->inner;
29 // TODO: remove this hack
30 stack_push(proxy_stack, &arg, sizeof(uintptr_t));
31 uintptr_t zero = 0;
32 stack_push(proxy_stack, &zero, sizeof(uintptr_t)); // a fake return address
33}

35void x86_copy_thread_context(platform_context_t *from, platform_context_t **to)
36{
37 x86_thread_context_t *from_arg = container_of(from, x86_thread_context_t, inner);
38 x86_thread_context_t *to_arg = kmalloc(sizeof(x86_thread_context_t));
39 memcpy(to_arg, from_arg, sizeof(x86_thread_context_t));
40 *to = &to_arg->inner;
41}

43void x86_switch_to_thread(uintptr_t *old_stack, thread_t *to)
44{
45 bool need_pgd_switch = current_cpu->pagetable.ptr != to->owner->pagetable.ptr;
46 uintptr_t pgd_paddr = need_pgd_switch ? pg_page_get_mapped_paddr(x86_kpg_infra, to->owner->pagetable.ptr) : 0;
47 tss_entry.esp0 = to->kernel_stack.top;
48 x86_thread_context_t *context = container_of(to->context, x86_thread_context_t, inner);
49
50 bool need_iret_switching = to->status == THREAD_STATUS_CREATED;
51
52 mos_update_current(to); // this updates to->status to THREAD_STATUS_RUNNING
53
54 if (unlikely(need_iret_switching))
55 // eax is set to 0, which is exactly what a child process should get (from fork)
56 x86_context_switch_impl(old_stack, to->stack.head, pgd_paddr, SWITCH_MODE_IRET, context);
57 else
58 x86_context_switch_impl(old_stack, to->stack.head, pgd_paddr, SWITCH_MODE_NORMAL, 0);
59}
60
61void x86_switch_to_scheduler(uintptr_t *old_stack_ptr, uintptr_t new_stack)
62{
63 // pgd = 0 so that we don't switch to a different page table
64 // pass 0 as switch_mode so that we don't switch to IRET
65 x86_context_switch_impl(old_stack_ptr, new_stack, 0, SWITCH_MODE_NORMAL, 0);
66}

68void x86_timer_handler()
