11void x86_setup_thread_context(thread_t *thread, thread_entry_t entry, void *arg)
12{
13 // x86_um_thread_startup needs [arg, entry_point] on the stack5
14 uintptr_t entry_addr = (uintptr_t) entry;
15 stack_push(&thread->stack, &arg, sizeof(uintptr_t));
16 stack_push(&thread->stack, &entry_addr, sizeof(uintptr_t));
17 stack_grow(&thread->stack, sizeof(reg_t) * 4); // space for esi, edi, ebx, ebp
18}
19
20void x86_context_switch(uintptr_t *old_stack, thread_t *to)
21{
22 // TODO: update TSS->esp0 on each context switch
23 uintptr_t x = pg_page_get_mapped_paddr(x86_kpg_infra, to->owner->pagetable.ptr);
24 x86_context_switch_impl(old_stack, (uintptr_t) to->stack.head, x);
25}
26
27void x86_context_switch_to_scheduler(uintptr_t *old_stack, uintptr_t new_stack)
28{
29 x86_context_switch_impl(old_stack, new_stack, 0);
30}
