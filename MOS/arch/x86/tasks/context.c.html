26void x86_setup_thread_context(thread_t *thread, downwards_stack_t *proxy_stack, thread_entry_t entry, void *arg)
27{
28 MOS_UNUSED(entry);
29 MOS_UNUSED(thread);
30 stack_push(proxy_stack, &arg, sizeof(uintptr_t));
31}

33void x86_switch_to_thread(uintptr_t *old_stack, thread_t *to)
34{
35 uintptr_t pgd_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, to->owner->pagetable.ptr);
36 tss_entry.esp0 = to->kernel_stack.top;
37 switch_stack_arg arg = { .ebp = (reg32_t) ((x86_stack_frame *) current_cpu->platform_context)->ebp, .eip = (reg32_t) to->current_instruction };
38
39 bool need_iret_switching = to->status == THREAD_STATUS_FORKED || to->status == THREAD_STATUS_CREATED;
40
41 mos_update_current(to); // this updates to->status to THREAD_STATUS_RUNNING
42
43 if (unlikely(need_iret_switching))
44 x86_context_switch_impl(old_stack, to->stack.head, pgd_paddr, SWITCH_MODE_IRET, (void *) &arg);
45 else
46 x86_context_switch_impl(old_stack, to->stack.head, pgd_paddr, 0, 0);
47}
48
49void x86_switch_to_scheduler(uintptr_t *old_stack_ptr, uintptr_t new_stack)
50{
51 // pgd = 0 so that we don't switch to a different page table
52 // pass 0 as switch_mode so that we don't switch to IRET
53 x86_context_switch_impl(old_stack_ptr, new_stack, 0, 0, 0);
54}
