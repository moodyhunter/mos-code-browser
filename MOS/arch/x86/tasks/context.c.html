25void x86_setup_thread_context(thread_t *thread, thread_entry_t entry, void *arg)
26{
27 x86_thread_context_t *context = kzalloc(sizeof(x86_thread_context_t));
28 context->inner.instruction = (uintptr_t) entry;
29 context->inner.stack = thread->mode == THREAD_MODE_KERNEL ? thread->k_stack.head : thread->u_stack.head;
30 context->arg = arg;
31 thread->context = &context->inner;
32}
33
34void x86_copy_thread_context(platform_context_t *from, platform_context_t **to)
35{
36 x86_thread_context_t *from_arg = container_of(from, x86_thread_context_t, inner);
37 x86_thread_context_t *to_arg = kmalloc(sizeof(x86_thread_context_t));
38 memcpy(to_arg, from_arg, sizeof(x86_thread_context_t));
39 *to = &to_arg->inner;
40}

42void x86_switch_to_thread(uintptr_t *scheduler_stack, thread_t *to)
43{
44 const bool need_pgd_switch = current_cpu->pagetable.pgd != to->owner->pagetable.pgd;
45 const uintptr_t pgd_paddr = need_pgd_switch ? pg_page_get_mapped_paddr(x86_kpg_infra, to->owner->pagetable.pgd) : 0;
46
47 per_cpu(x86_tss.tss)->esp0 = to->k_stack.top;
48
49 const x86_thread_context_t *context = container_of(to->context, x86_thread_context_t, inner);
50
51 switch_func_t switch_func = NULL;
52 if (likely(to->state != THREAD_STATE_CREATED))
53 {
54 switch_func = x86_switch_impl_normal;
55 }
56 else
57 {
58 // eax will be set to 0 (when need_iret_switching), which is exactly what a child process should get (from fork)
59 switch_func = to->mode == THREAD_MODE_KERNEL ? x86_switch_impl_new_kernel_thread : x86_switch_impl_new_user_thread;
60 }
61
62 mos_update_current(to); // this updates to->mode to THREAD_MODE_RUNNING
63 x86_context_switch_impl(scheduler_stack, to->k_stack.head, pgd_paddr, switch_func, context);
64}
65
66void x86_switch_to_scheduler(uintptr_t *old_stack, uintptr_t scheduler_stack)
67{
68 // pgd = 0 so that we don't switch to a different page table
69 x86_context_switch_impl(old_stack, scheduler_stack, 0, x86_switch_impl_normal, NULL);
70}
