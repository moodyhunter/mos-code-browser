25// called from assembly
26asmlinkage void x86_switch_impl_setup_user_thread()
27{
28 thread_t *current = current_thread;
29 x86_thread_context_t *context = container_of(current->context, x86_thread_context_t, inner);
30
31 if (current == current->owner->threads[0])
32 {
33 // this is the main thread of a new process, so we need to set up the arguments
34 pr_info2("setting up main thread (id: %d) of process '%s' (%d)", current->tid, current->owner->name, current->owner->pid);
35
36 MOS_ASSERT_X(!context->arg, "arg is not NULL for main thread of process '%s' (%d)", current->owner->name, current->owner->pid);
37
38 const char *const *const src_argv = current->owner->argv.argv;
39 const size_t argc = current->owner->argv.argc;
40
41 const char *real_argv[argc + 1];
42 for (size_t i = 0; i < argc; i++)
43 {
44 real_argv[i] = stack_grow(&current->u_stack, strlen(src_argv[i]) + 1);
45 strcpy((char *) real_argv[i], src_argv[i]);
46 }
47 real_argv[argc] = NULL;
48
49 // stack layout:
50 // | argc | argv1 | argv2 | ... | argvN | NULL | arg0 | arg1 | ... | argN | NULL |
51
52 stack_push(&current->u_stack, &real_argv, sizeof(char *) * (argc + 1)); // the argv vector
53
54 const uintptr_t argv_ptr = current->u_stack.head; // stack_push changes head, so we need to save it
55 stack_push(&current->u_stack, &argv_ptr, sizeof(uintptr_t));
56 stack_push(&current->u_stack, &argc, sizeof(size_t));
57 }
58 else
59 {
60 stack_push(&current->u_stack, &context->arg, sizeof(void *));
61 }
62
63 context->inner.stack = current->u_stack.head;
64}

66void x86_setup_thread_context(thread_t *thread, thread_entry_t entry, void *arg)
67{
68 x86_thread_context_t *context = kzalloc(sizeof(x86_thread_context_t));
69 context->inner.instruction = (uintptr_t) entry;
70 context->inner.stack = thread->mode == THREAD_MODE_KERNEL ? thread->k_stack.head : thread->u_stack.head;
71 context->arg = arg;
72 thread->context = &context->inner;
73}

75void x86_copy_thread_context(platform_context_t *from, platform_context_t **to)
76{
77 x86_thread_context_t *from_arg = container_of(from, x86_thread_context_t, inner);
78 x86_thread_context_t *to_arg = kmalloc(sizeof(x86_thread_context_t));
79 memcpy(to_arg, from_arg, sizeof(x86_thread_context_t));
80 *to = &to_arg->inner;
81}

83void x86_switch_to_thread(uintptr_t *scheduler_stack, thread_t *to)
84{
85 const bool need_pgd_switch = current_cpu->pagetable.pgd != to->owner->pagetable.pgd;
86 const uintptr_t pgd_paddr = need_pgd_switch ? pg_page_get_mapped_paddr(x86_kpg_infra, to->owner->pagetable.pgd) : 0;
87
88 per_cpu(x86_tss.tss)->esp0 = to->k_stack.top;
89
90 const x86_thread_context_t *context = container_of(to->context, x86_thread_context_t, inner);
91
92 switch_func_t switch_func = NULL;
93 if (likely(to->state != THREAD_STATE_CREATED))
94 {
95 switch_func = x86_switch_impl_normal;
96 }
97 else
98 {
99 // eax will be set to 0 (when need_iret_switching), which is exactly what a child process should get (from fork)
100 switch_func = to->mode == THREAD_MODE_KERNEL ? x86_switch_impl_new_kernel_thread : x86_switch_impl_new_user_thread;
101 }
102
103 mos_update_current(to); // this updates to->mode to THREAD_MODE_RUNNING
104 x86_context_switch_impl(scheduler_stack, to->k_stack.head, pgd_paddr, switch_func, context);
105}
106
107void x86_switch_to_scheduler(uintptr_t *old_stack, uintptr_t scheduler_stack)
108{
109 // pgd = 0 so that we don't switch to a different page table
110 x86_context_switch_impl(old_stack, scheduler_stack, 0, x86_switch_impl_normal, NULL);
111}
