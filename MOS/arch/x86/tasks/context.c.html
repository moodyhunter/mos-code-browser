14void x86_setup_thread_context(thread_t *thread, thread_entry_t entry, void *arg)
15{
16 // x86_um_thread_startup needs [arg, entry_point] on the stack5
17 uintptr_t entry_addr = (uintptr_t) entry;
18 stack_push(&thread->stack, &arg, sizeof(uintptr_t));
19 stack_push(&thread->stack, &entry_addr, sizeof(uintptr_t));
20}

22void x86_switch_to_thread(uintptr_t *old_stack, thread_t *to)
23{
24 uintptr_t pgd_paddr = pg_page_get_mapped_paddr(x86_kpg_infra, to->owner->pagetable.ptr);
25 uintptr_t post_switch_func = 0;
26 uintptr_t post_switch_func_arg = 0;
27
28 if (unlikely(to->status == THREAD_STATUS_CREATED))
29 post_switch_func = (uintptr_t) x86_um_thread_startup;
30
31 if (unlikely(to->status == THREAD_STATUS_FORKED))
32 {
33 post_switch_func = (uintptr_t) x86_um_thread_post_fork;
34 post_switch_func_arg = to->current_instruction;
35 }
36 x86_enable_interrupts();
37 to->status = THREAD_STATUS_RUNNING;
38 x86_context_switch_impl(old_stack, (uintptr_t) to->stack.head, pgd_paddr, post_switch_func, post_switch_func_arg);
39}
40
41void x86_switch_to_scheduler(uintptr_t *old_stack_ptr, uintptr_t new_stack)
42{
43 // pgd = 0 so that we don't switch to a different page table
44 x86_context_switch_impl(old_stack_ptr, new_stack, 0, 0, 0);
45}
