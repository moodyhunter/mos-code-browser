3#pragma once
4
5#include "mos/types.h"
6
7#define GDT_NULL 0x00
8#define GDT_SEGMENT 0x10
9#define GDT_PRESENT 0x80
10
11#define GDT_GRANULARITY_BYTE 0x40
12#define GDT_GRANULARITY_PAGE 0xC0
13
14typedef struct
15{
16 u32 limit_low : 16;
17 u32 base_low : 24;
18 u32 accessed : 1;
19 u32 read_write : 1; // readable for code, writable for data
20 u32 conforming_expand_down : 1; // conforming for code, expand down for data
21 u32 executable : 1; // 1 for code, 0 for data
22 u32 code_data_segment : 1; // should be 1 for everything but TSS and LDT
23 u32 dpl : 2; // privilege level
24 u32 present : 1;
25 u32 limit_high : 4;
26 u32 available : 1; // only used in software; has no effect on hardware
27 u32 long_mode : 1;
28 u32 pm32_segment : 1; // 32-bit opcodes for code, uint32_t stack for data
29 u32 granularity : 1; // 1 to use 4k page addressing, 0 for byte addressing
30 u32 base_high : 8;
31} __packed gdt_entry32_t;
32
33typedef struct
34{
35 u16 limit;
36 gdt_entry32_t *base;
37} __packed gdt_ptr32_t;
38
39static_assert(sizeof(gdt_entry32_t) == 8, "gdt_entry_t is not 8 bytes");
40static_assert(sizeof(gdt_ptr32_t) == 6, "gdt_ptr_t is not 6 bytes");
