7typedef enum
8{
9 COM1 = 0x3F8,
10 COM2 = 0x2F8,
11 COM3 = 0x3E8,
12 COM4 = 0x2E8,
13 COM5 = 0x5F8,
14 COM6 = 0x4F8,
15 COM7 = 0x5E8,
16 COM8 = 0x4E8
17} serial_port_t;
18
19typedef enum
20{
21 OFFSET_INTTERUPT_ENABLE = 1, // Interrupt Enable Register
22 OFFSET_INTERRUPT_ID_FIFO = 2, // Interrupt ID Register and FIFO Control Register
23 OFFSET_LINE_CONTROL = 3, // Line Control Register
24 OFFSET_MODEM_CONTROL = 4, // Modem Control Register
25 OFFSET_LINE_STATUS = 5, // Line Status Register
26 OFFSET_MODEM_STATUS = 6, // Modem Status Register
27 OFFSET_SCRATCH = 7, // Scratch Register
28
29 OFFSET_DLAB_DIVISOR_LSB = 0, // With DLAB set to 1, this is the least significant byte of the divisor value for setting the baud rate.
30 OFFSET_DLAB_DIVISOR_MSB = 1, // With DLAB set to 1, this is the most significant byte of the divisor value.
31} serial_register_offset_t;
32
33typedef enum
34{
35 CHAR_LENGTH_5 = 0x0,
36 CHAR_LENGTH_6 = 0x1,
37 CHAR_LENGTH_7 = 0x2,
38 CHAR_LENGTH_8 = 0x3,
39} serial_char_length_t;
40
41typedef enum
42{
43 STOP_BITS_1, // 1 stop bit
44 STOP_BITS_15_OR_2, // 1.5 or 2 stop bits
45} serial_stop_bits_t;
46
47typedef enum
48{
49 PARITY_NONE = 0, // No parity
50 PARITY_ODD = 1, // Odd parity
51 PARITY_EVEN = 2, // Even parity
52 PARITY_MARK = 3, // Parity bit always 1
53 PARITY_SPACE = 4, // Parity bit always 0
54} serial_port_parity_t;
55
56typedef enum
57{
58 INTERRUPT_DATA_AVAILABLE = 1 << 0, // Data ready interrupt
59 INTERRUPT_TRANSMITTER_EMPTY = 1 << 1, // Transmitter empty interrupt
60 INTERRUPT_BREAK_ERROR = 1 << 2, // Break error interrupt
61 INTERRUPT_STATUS_CHANGE = 1 << 3, // Status change interrupt
62 INTERRUPT_NONE = 0, // No interrupts
63 INTERRUPT_ALL = INTERRUPT_DATA_AVAILABLE | INTERRUPT_TRANSMITTER_EMPTY | INTERRUPT_BREAK_ERROR | INTERRUPT_STATUS_CHANGE,
64} serial_interrupt_t;
65
66typedef enum
67{
68 MODEM_DTR = 1 << 0, // Data Terminal Ready
69 MODEM_RTS = 1 << 1, // Request To Send
70 MODEM_UNUSED_PIN1 = 1 << 2, // Unused
71 MODEM_IRQ = 1 << 3, // Interrupt Request
72 MODEM_LOOP = 1 << 4, // Loopback
73} serial_modem_control_t;
74
75typedef enum
76{
77 LINE_DATA_READY = 1 << 0, // Data ready to be read.
78 LINE_ERR_OVERRUN = 1 << 1, // There has been data lost.
79 LINE_ERR_PARITY = 1 << 2, // Parity error.
80 LINE_ERR_FRAMING = 1 << 3, // Stop bit is missing.
81 LINE_ERR_BREAK = 1 << 4, // Break detected.
82 LINE_TRANSMITR_BUF_EMPTY = 1 << 5, // (transmitter buffer is empty) Data can be sent.
83 LINE_TRANSMITR_EMPTY = 1 << 6, // Transmitter is not doing anything.
84 LINE_ERR_IMPENDING = 1 << 7, // There is an error with a word in the input buffer
85} serial_line_status_t;
86
87typedef enum
88{
89 MODEM_DCTS = 1 << 0, // Clear To Send input has changed since last read.
90 MODEM_DDSR = 1 << 1, // Data Set Ready input has changed since last read.
91 MODEM_TERI = 1 << 2, // Ring Indicator input has changed since last read.
92 MODEM_DDCD = 1 << 3, // Data Carrier Detect input has changed since last read.
93 MODEM_CLEAR_TO_SEND = 1 << 4,
94 MODEM_DATA_SET_READY = 1 << 5,
95 MODEM_RING_INDICATOR = 1 << 6,
96 MODEM_DATA_CARRIER_DETECT = 1 << 7,
97} serial_modem_status_t;
98
99typedef enum
100{
101 BAUD_RATE_115200 = 1,
102 BAUD_RATE_57600 = 2,
103 BAUD_RATE_38400 = 3,
104 BAUD_RATE_19200 = 4,
105 BAUD_RATE_9600 = 5,
106 BAUD_RATE_4800 = 6,
107 BAUD_RATE_2400 = 7,
108 BAUD_RATE_1200 = 8,
109 BAUD_RATE_600 = 9,
110 BAUD_RATE_300 = 10,
111 BAUD_RATE_110 = 11,
112} serial_baud_rate;
