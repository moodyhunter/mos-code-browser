5#include "mos/types.h"
6
7typedef struct
8{
9 u16 isr_low; // The lower 16 bits of the ISR's address
10 u16 segment; // The GDT segment selector that the CPU will load into CS before calling the ISR
11 u32 args : 5;
12 u32 reserved : 3;
13 u32 type : 4; // The type of interrupt
14 u32 s : 1;
15 u32 dpl : 2;
16 u32 present : 1;
17 u32 isr_high : 16; // The upper 16 bits of the ISR's address
18} __packed idt_entry32_t;
19
20typedef struct
21{
22 u16 limit;
23 idt_entry32_t *base;
24} __packed idtr32_t;
25
26static_assert(sizeof(idt_entry32_t) == 8, "idt_entry32_t is not 8 bytes");
27static_assert(sizeof(idtr32_t) == 6, "idtr32_t is not 6 bytes");
