7#define APIC_REG_LAPIC_ID 0x20
8typedef enum
9{
10 APIC_DELIVER_MODE_FIXED = 0,
11 APIC_DELIVER_MODE_LOWEST_PRIORITY = 1,
12 APIC_DELIVER_MODE_SMI = 2,
13 APIC_DELIVER_MODE_NMI = 4,
14 APIC_DELIVER_MODE_INIT = 5,
15 APIC_DELIVER_MODE_INIT_DEASSERT = APIC_DELIVER_MODE_INIT,
16 APIC_DELIVER_MODE_STARTUP = 6,
17} lapic_delivery_mode_t;
18
19typedef enum
20{
21 APIC_DEST_MODE_PHYSICAL = 0,
22 APIC_DEST_MODE_LOGICAL = 1,
23} lapic_dest_mode_t;
24
25typedef enum
26{
27 APIC_SHORTHAND_NONE = 0,
28 APIC_SHORTHAND_SELF = 1,
29 APIC_SHORTHAND_ALL = 2,
30 APIC_SHORTHAND_ALL_EXCLUDING_SELF = 3,
31} lapic_shorthand_t;

45should_inline u8 lapic_get_id()
46{
47 // https://stackoverflow.com/a/71756491
48 // https://github.com/rust-osdev/apic/blob/master/src/registers.rs
49 // shift 24 because the ID is in the upper 8 bits
50 return lapic_read32(APIC_REG_LAPIC_ID) >> 24;
51}
