23typedef struct

26 {
27 struct
28 {

40
41 union
42 {
43 struct
44 {
45 u8 brand_id : 8;
46 u8 cpuflush_size : 8;
47 /** Maximum number of addressable IDs for logical processors in this physical package;
48 * The nearest power-of-2 integer that is not smaller than this value is the number of
49 * unique initial APIC IDs reserved for addressing different logical processors in a
50 * physical package.
51 * Former use: Number of logical processors per physical processor; two for the Pentium
52 * 4 processor with Hyper-Threading Technology.[10] */
53 u8 logical_processors_per_package : 8;
54 u8 local_apic_id : 8;
55 } __packed;
56 reg32_t raw;
57 } ebx;
