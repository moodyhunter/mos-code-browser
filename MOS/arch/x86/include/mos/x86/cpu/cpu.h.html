8should_inline void cpu_get_msr(u32 msr, u32 *lo, u32 *hi)
9{
10 __asm__ volatile("rdmsr" : "=a"(*lo), "=d"(*hi) : "c"(msr));
11}
12
13should_inline void cpu_set_msr(u32 msr, u32 lo, u32 hi)
14{
15 __asm__ volatile("wrmsr" : : "a"(lo), "d"(hi), "c"(msr));
16}
17
18should_inline void x86_cpu_halt()
19{
20 __asm__ volatile("hlt");
21}
22
23should_inline reg_t x86_get_cr3()
24{
25 reg_t cr3;
26 __asm__ volatile("mov %%cr3, %0" : "=r"(cr3));
27 return cr3;
28}

30always_inline u32 x86_cpu_get_id(void)
31{
32 int lapic_id = 0;
33 __asm__ volatile("cpuid" : "=b"(lapic_id) : "a"(0x01), "c"(0x00) : "edx");
34 return lapic_id >> 24;
35}
