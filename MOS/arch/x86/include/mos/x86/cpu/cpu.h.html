9should_inline void cpu_get_msr(u32 msr, u32 *lo, u32 *hi)
10{
11 __asm__ volatile("rdmsr" : "=a"(*lo), "=d"(*hi) : "c"(msr));
12}
13
14should_inline void cpu_set_msr(u32 msr, u32 lo, u32 hi)
15{
16 __asm__ volatile("wrmsr" : : "a"(lo), "d"(hi), "c"(msr));
17}

19should_inline void x86_cpu_halt()

24should_inline reg_t x86_get_cr3()

26 reg_t cr3;
