27// How many bytes from the start of the file we search for the header.
28#define MULTIBOOT_SEARCH 8192
29#define MULTIBOOT_HEADER_ALIGN 4
30
31#define MULTIBOOT_HEADER_MAGIC 0x1BADB002 // The magic field should contain this.
32#define MULTIBOOT_BOOTLOADER_MAGIC 0x2BADB002 // This should be in %eax.
33#define MULTIBOOT_MOD_ALIGN 0x00001000 // Alignment of multiboot modules.
34#define MULTIBOOT_INFO_ALIGN 0x00000004 // Alignment of the multiboot info structure.
35
36// Flags set in the 'flags' member of the multiboot header.
37#define MULTIBOOT_PAGE_ALIGN 0x00000001 // Align all boot modules on i386 page (4KB) boundaries.
38#define MULTIBOOT_MEMORY_INFO 0x00000002 // Must pass memory information to OS.
39#define MULTIBOOT_VIDEO_MODE 0x00000004 // Must pass video information to OS.
40#define MULTIBOOT_AOUT_KLUDGE 0x00010000 // This flag indicates the use of the address fields in the header.
41
42// Flags to be set in the 'flags' member of the multiboot info structure.
43typedef enum
44{
45 MULTIBOOT_INFO_MEMORY = 0x00000001, // is there basic lower/upper memory information?
46 MULTIBOOT_INFO_BOOTDEV = 0x00000002,
47 MULTIBOOT_INFO_CMDLINE = 0x00000004,
48 MULTIBOOT_INFO_MODS = 0x00000008,
49 MULTIBOOT_INFO_AOUT_SYMS = 0x00000010, // mutually exclusive
50 MULTIBOOT_INFO_ELF_SHDR = 0X00000020, // mutually exclusive
51 MULTIBOOT_INFO_MEM_MAP = 0x00000040,
52 MULTIBOOT_INFO_DRIVE_INFO = 0x00000080,
53 MULTIBOOT_INFO_CONFIG_TABLE = 0x00000100,
54 MULTIBOOT_INFO_BOOT_LOADER_NAME = 0x00000200,
55 MULTIBOOT_INFO_APM_TABLE = 0x00000400,
56 MULTIBOOT_INFO_VBE_INFO = 0x00000800,
57 MULTIBOOT_INFO_FRAMEBUFFER_INFO = 0x00001000,
58} multiboot_info_flags_t;
59
60struct multiboot_header
61{
62 /* Must be MULTIBOOT_MAGIC - see above. */
63 u32 magic;
64
65 /* Feature flags. */
66 u32 flags;
67
68 /* The above fields plus this one must equal 0 mod 2^32. */
69 u32 checksum;
70
71 /* These are only valid if MULTIBOOT_AOUT_KLUDGE is set. */
72 u32 header_addr;
73 u32 load_addr;
74 u32 load_end_addr;
75 u32 bss_end_addr;
76 u32 entry_addr;
77
78 /* These are only valid if MULTIBOOT_VIDEO_MODE is set. */
79 u32 mode_type;
80 u32 width;
81 u32 height;
82 u32 depth;
83};
