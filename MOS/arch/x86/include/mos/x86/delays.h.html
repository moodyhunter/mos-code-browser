8always_inline u64 rdtsc(void)
9{
10#if MOS_32BITS
11 u64 x;
12 __asm__ volatile("rdtsc" : "=A"(x));
13 return x;
14#elif defined(MOS_64BITS)
15 u64 a, d;
16 __asm__ volatile("rdtsc" : "=a"(a), "=d"(d));
17 return (d << 32) | a;
18#endif
19}
20
21always_inline void mdelay(u64 ms)
22{
23 u64 end = rdtsc() + ms * 2000 * 1000;
24 while (rdtsc() < end)
25 ;
26}
27
28always_inline void udelay(u64 us)
29{
30 u64 end = rdtsc() + us * 2000;
31 while (rdtsc() < end)
32 ;
33}
