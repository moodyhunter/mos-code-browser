7u32 x86_cpu_get_id(void)
8{
9 int lapic_id = 0;
10 __asm__ volatile("cpuid" : "=b"(lapic_id) : "a"(0x01), "c"(0x00) : "edx");
11 return lapic_id >> 24;
12}
