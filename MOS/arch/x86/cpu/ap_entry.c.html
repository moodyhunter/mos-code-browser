3#include "mos/printk.h"
4#include "mos/x86/cpu/cpuid.h"

9void ap_begin_exec()
10{
11 x86_ap_gdt_init();
12 x86_idt_init();
13 x86_mm_enable_paging();
14 lapic_enable();
15
16 processor_version_t info;
17 cpuid_get_processor_info(&info);
18
19 pr_info("smp: AP %u started", info.ebx.local_apic_id);
20 cpuid_print_cpu_info();
21
22 per_cpu(x86_platform.cpu)->id = info.ebx.local_apic_id;
23 u8 lapic_id = lapic_get_id();
24 if (lapic_id != info.ebx.local_apic_id)
25 mos_warn("smp: AP %u: LAPIC ID mismatch: %u != %u", info.ebx.local_apic_id, lapic_id, info.ebx.local_apic_id);
26
27 while (1)
28 __asm__ volatile("cli; hlt;");
29}
