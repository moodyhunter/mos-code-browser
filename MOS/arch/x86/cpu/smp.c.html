14volatile enum
15{
16 AP_STATUS_INVALID = 0,
17 AP_STATUS_BSP_STARTUP_SENT = 1,
18 AP_STATUS_START_REQUEST = 2,
19 AP_STATUS_START = 3,
20} ap_state;
21
22#define X86_AP_TRAMPOLINE_ADDR 0x8000
23
24extern char x86_ap_trampoline[];
25extern x86_pgdir_entry startup_pgd[1024];
26volatile uintptr_t ap_stack_addr = 0;
27volatile const uintptr_t ap_pgd_addr = (uintptr_t) &startup_pgd;

29// clang-format off
30#define wait_for(state) do {} while (ap_state != state)
31// clang-format on
32
33static void x86_cpu_start(int apic_id)
34{
35 ap_state = AP_STATUS_INVALID;
36
37 lapic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT, APIC_DEST_MODE_PHYSICAL, true, true, APIC_SHORTHAND_NONE);
38 mdelay(100);
39 lapic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT_DEASSERT, APIC_DEST_MODE_PHYSICAL, false, true, APIC_SHORTHAND_NONE);
40
41 ap_state = AP_STATUS_BSP_STARTUP_SENT;
42 pr_info2("smp: bsp sent startup to cpu %u", apic_id);
43
44 mdelay(100);
45 lapic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
46 mdelay(100);
47 lapic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
48
49 wait_for(AP_STATUS_START_REQUEST);
50 pr_info2("smp: cpu %u received start request", apic_id);
51
52 ap_state = AP_STATUS_START;
53 pr_info2("smp: started cpu %u", apic_id);
54}
55
56#undef wait_for
57
58void x86_smp_copy_trampoline()
59{
60 mos_startup_map_bytes(X86_AP_TRAMPOLINE_ADDR, X86_AP_TRAMPOLINE_ADDR, 4 KB, VM_READ | VM_WRITE | VM_EXEC);
61 memcpy((void *) X86_AP_TRAMPOLINE_ADDR, x86_ap_trampoline, 4 KB);
62}
63
64void x86_smp_start_all()
65{
66 for (u32 i = 0; i < x86_platform.num_cpus; i++)
67 {
68 u32 apic_id = x86_cpu_lapic[i];
69 if (apic_id == x86_platform.boot_cpu_id)
70 continue;
71
72 extern const void __MOS_KERNEL_HIGHER_STACK_TOP;
73 ap_stack_addr = (uintptr_t) &__MOS_KERNEL_HIGHER_STACK_TOP - (i * 1 MB);
74 pr_info("smp: starting AP %d, LAPIC %d, stack " PTR_FMT, i, apic_id, ap_stack_addr);
75 x86_cpu_start(apic_id);
76 }
