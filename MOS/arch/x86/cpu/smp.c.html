19volatile enum
20{
21 AP_STATUS_INVALID = 0,
22 AP_STATUS_BSP_STARTUP_SENT = 1,
23 AP_STATUS_START_REQUEST = 2,
24 AP_STATUS_START = 3,
25} ap_state;
26
27volatile uintptr_t ap_stack_addr = 0;
28volatile uintptr_t ap_pgd_addr = 0;
29extern void x86_ap_trampoline();
30
31void ap_begin_exec()
32{
33 x86_ap_gdt_init();
34 x86_idt_init();
35 x86_mm_enable_paging();
36 lapic_enable();
37
38 processor_version_t info;
39 cpuid_get_processor_info(&info);
40
41 pr_info("smp: AP %u started", info.ebx.local_apic_id);
42 cpuid_print_cpu_info();
43
44 per_cpu(x86_platform.cpu)->id = info.ebx.local_apic_id;
45 u8 lapic_id = lapic_get_id();
46 if (lapic_id != info.ebx.local_apic_id)
47 {
48 mos_warn("smp: AP %u: LAPIC ID mismatch: %u != %u", info.ebx.local_apic_id, lapic_id, info.ebx.local_apic_id);
49 }
50
51 while (1)
52 __asm__ volatile("cli; hlt;");
53}

55// clang-format off
56#define wait_for(state) do {} while (ap_state != state)
57// clang-format on
58
59void x86_cpu_start(int apic_id, uintptr_t stack_addr)
60{
61 ap_state = AP_STATUS_INVALID;
62 ap_pgd_addr = x86_get_cr3();
63 ap_stack_addr = stack_addr;
64
65 lapic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT, APIC_DEST_MODE_PHYSICAL, true, true, APIC_SHORTHAND_NONE);
66 mdelay(100);
67 lapic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT_DEASSERT, APIC_DEST_MODE_PHYSICAL, false, true, APIC_SHORTHAND_NONE);
68
69 ap_state = AP_STATUS_BSP_STARTUP_SENT;
70 pr_info2("smp: bsp sent startup to cpu %u", apic_id);
71
72 mdelay(100);
73 lapic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
74 mdelay(100);
75 lapic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
76
77 wait_for(AP_STATUS_START_REQUEST);
78 pr_info2("smp: cpu %u received start request", apic_id);
79
80 ap_state = AP_STATUS_START;
81 pr_info2("smp: started cpu %u", apic_id);
82}

84#undef wait_for

86void x86_smp_init()
87{
88 // we are still using the old page tables, use mos_startup_map_bytes
89 mos_startup_map_bytes(X86_AP_TRAMPOLINE_ADDR, X86_AP_TRAMPOLINE_ADDR, 4 KB, VM_WRITE);
90 memcpy((void *) X86_AP_TRAMPOLINE_ADDR, (void *) (uintptr_t) &x86_ap_trampoline, 4 KB);
91
92 for (u32 i = 0; i < x86_platform.num_cpus; i++)
93 {
94 u32 apic_id = x86_cpu_lapic[i];
95 if (apic_id == x86_platform.boot_cpu_id)
96 continue;
97
98 extern const void __MOS_KERNEL_HIGHER_STACK_TOP;
99 const uintptr_t stack = (uintptr_t) &__MOS_KERNEL_HIGHER_STACK_TOP - (i * 1 MB);
100 pr_info("smp: starting AP %d, LAPIC %d, stack " PTR_FMT, i, apic_id, stack);
101 x86_cpu_start(apic_id, stack);
102 }
