34void print_cpu_info()
35{
36 char manufacturer[13];
37 cpuid_get_manufacturer(manufacturer);
38 processor_version_t info;
39 cpuid_get_processor_info(&info);
40 char brand_string[49];
41 cpuid_get_brand_string(brand_string);
42 pr_info2("CPU: %s (%s)", brand_string, manufacturer);
43 pr_info2(" Family %u, Model %u, Stepping %u", info.eax.family, info.eax.model, info.eax.stepping);
44 pr_info2(" Type: %s, Ext family: %u, Ext model: %u", cpuid_type_str[info.eax.type], info.eax.ext_family, info.eax.ext_model);
45}

47void ap_begin_exec()
48{
49 x86_ap_gdt_init();
50 x86_idt_init();
51 x86_mm_enable_paging();
52
53 processor_version_t info;
54 cpuid_get_processor_info(&info);
55
56 pr_info("smp: AP %u started", info.ebx.local_apic_id);
57 print_cpu_info();
58
59 per_cpu(x86_platform.cpu)->id = info.ebx.local_apic_id;
60
61 unsigned int x = apic_reg_read_offset_32(APIC_REG_LAPIC_ID);
62 if (x != info.ebx.local_apic_id)
63 {
64 mos_warn("smp: AP %u: LAPIC ID mismatch: %u != %u", info.ebx.local_apic_id, x, info.ebx.local_apic_id);
65 }
66
67 while (1)
68 __asm__ volatile("cli; hlt");
69}

75void x86_cpu_start(int apic_id, uintptr_t stack_addr)
76{
77 ap_state = AP_STATUS_INVALID;
78 ap_pgd_addr = x86_get_cr3();
79 ap_stack_addr = stack_addr;
80
81 apic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT, APIC_DEST_MODE_PHYSICAL, true, true, APIC_SHORTHAND_NONE);
82 mdelay(100);
83 apic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT_DEASSERT, APIC_DEST_MODE_PHYSICAL, false, true, APIC_SHORTHAND_NONE);
84
85 ap_state = AP_STATUS_BSP_STARTUP_SENT;
86 pr_info2("smp: bsp sent startup to cpu %u", apic_id);
87
88 mdelay(100);
89 apic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
90 mdelay(100);
91 apic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
92
93 wait_for(AP_STATUS_START_REQUEST);
94 pr_info2("smp: cpu %u received start request", apic_id);
95
96 ap_state = AP_STATUS_START;
97 pr_info2("smp: started cpu %u", apic_id);
98}

102void x86_smp_init()
103{
104 pr_info("smp: boot cpu:");
105 print_cpu_info();
106
107 pr_info2("disabling 8259 PIC...");
108 pic_disable();
109
110 pr_info2("enabling APIC...");
111 apic_enable();
112
113 u32 num_cpus = 0;
114 processor_version_t info;
115 cpuid_get_processor_info(&info);
116 x86_platform.boot_cpu_id = info.ebx.local_apic_id;
117 per_cpu(x86_platform.cpu)->id = x86_platform.boot_cpu_id;
118
119 madt_entry_foreach(entry, x86_acpi_madt)
120 {
121 if (entry->type == 0)
122 {
123 acpi_madt_et0_lapic_t *et0 = (acpi_madt_et0_lapic_t *) entry;
124 pr_info2("cpu %d: processor id %d, apic id %d", num_cpus, et0->processor_id, et0->apic_id);
125 lapics[num_cpus] = et0->apic_id;
126 num_cpus++;
127 }
128 }
129
130 pr_info("smp: platform has %u cpu(s)", num_cpus);
131 x86_platform.num_cpus = num_cpus;
132
133 // !! TODO: is this the correct way?
134 mos_startup_map_bytes(X86_AP_TRAMPOLINE_ADDR, X86_AP_TRAMPOLINE_ADDR, 4 KB, VM_WRITE);
135 memcpy((void *) X86_AP_TRAMPOLINE_ADDR, (void *) (uintptr_t) &x86_ap_trampoline, 4 KB);
136
137 for (u32 i = 0; i < num_cpus; i++)
138 {
139 u32 apic_id = lapics[i];
140 if (apic_id == x86_platform.boot_cpu_id)
141 continue;
142
143 extern const void __MOS_KERNEL_HIGHER_STACK_TOP;
144 const uintptr_t stack = (uintptr_t) &__MOS_KERNEL_HIGHER_STACK_TOP - (i * 1 MB);
145 pr_info("smp: starting AP %d, LAPIC %d, stack " PTR_FMT, i, apic_id, stack);
146 x86_cpu_start(apic_id, stack);
147 }
