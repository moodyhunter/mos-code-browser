18#define X86_AP_TRAMPOLINE_ADDR 0x8000
19
20volatile enum
21{
22 AP_STATUS_INVALID = 0,
23 AP_STATUS_BSP_STARTUP_SENT = 1,
24 AP_STATUS_AP_WAIT_FOR_STACK_ALLOC = 2,
25 AP_STATUS_STACK_ALLOCATED = 3,
26 AP_STATUS_STACK_INITIALIZED = 4,
27 AP_STATUS_START = 5,
28 AP_STATUS_STARTED = 6,
29} ap_state;

31volatile uintptr_t ap_stack_addr = 0;
32static u32 lapics[MOS_MAX_CPU_COUNT] = { 0 };

35void print_cpu_info()
36{
37 char manufacturer[13];
38 cpuid_get_manufacturer(manufacturer);
39 processor_version_t info;
40 cpuid_get_processor_info(&info);
41 pr_info2(" %s, Family %u, Model %u, Stepping %u", manufacturer, info.eax.eax.family, info.eax.eax.model, info.eax.eax.stepping);
42 pr_info2(" type: %s, ext family: %u, ext model: %u", cpuid_type_str[info.eax.eax.type], info.eax.eax.ext_family, info.eax.eax.ext_model);
43}

45void ap_begin_exec()
46{
47 x86_ap_gdt_init();
48 x86_idt_init();
49
50 processor_version_t info;
51 cpuid_get_processor_info(&info);
52
53 pr_info("smp: AP %u started", info.ebx.ebx.local_apic_id);
54 print_cpu_info();
55
56 per_cpu(x86_platform.cpu)->id = info.ebx.ebx.local_apic_id;
57
58 // TODO: enable paging
59 while (1)
60 ;
61}

67void x86_cpu_start(int apic_id, uintptr_t stack_addr)
68{
69 ap_state = AP_STATUS_INVALID;
70 apic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT, APIC_DEST_MODE_PHYSICAL, true, true, APIC_SHORTHAND_NONE);
71 mdelay(100);
72 apic_interrupt_full(0, apic_id, APIC_DELIVER_MODE_INIT_DEASSERT, APIC_DEST_MODE_PHYSICAL, false, true, APIC_SHORTHAND_NONE);
73
74 ap_state = AP_STATUS_BSP_STARTUP_SENT;
75 pr_info2("smp: bsp sent startup to cpu %u", apic_id);
76
77 mdelay(100);
78 apic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
79 mdelay(100);
80 apic_interrupt(X86_AP_TRAMPOLINE_ADDR >> 12, apic_id, APIC_DELIVER_MODE_STARTUP, APIC_DEST_MODE_PHYSICAL, APIC_SHORTHAND_NONE);
81
82 wait_for(AP_STATUS_AP_WAIT_FOR_STACK_ALLOC);
83 pr_info2("smp: received startup from cpu %u", apic_id);
84
85 // allocate stack for ap
86 ap_stack_addr = stack_addr;
87 ap_state = AP_STATUS_STACK_ALLOCATED;
88 wait_for(AP_STATUS_STACK_INITIALIZED);
89 pr_info2("smp: initialized stack for cpu %u", apic_id);
90
91 ap_state = AP_STATUS_START;
92 wait_for(AP_STATUS_STARTED);
93 pr_info2("smp: started cpu %u", apic_id);
94}

96#undef wait_for

98void x86_smp_init(x86_pg_infra_t *pg)
99{
100 pr_info("smp: boot cpu:");
101 print_cpu_info();
102
103 pr_info2("disabling 8259 PIC...");
104 pic_disable();
105
106 pr_info2("enabling APIC...");
107 apic_enable();
108
109 u32 num_cpus = 0;
110 processor_version_t info;
111 cpuid_get_processor_info(&info);
112 x86_platform.boot_cpu_id = info.ebx.ebx.local_apic_id;
113 per_cpu(x86_platform.cpu)->id = x86_platform.boot_cpu_id;
114
115 madt_entry_foreach(entry, x86_acpi_madt)
116 {
117 if (entry->type == 0)
118 {
119 acpi_madt_et0_lapic_t *et0 = (acpi_madt_et0_lapic_t *) entry;
120 pr_info2("cpu %d: processor id %d, apic id %d", num_cpus, et0->processor_id, et0->apic_id);
121 lapics[num_cpus] = et0->apic_id;
122 num_cpus++;
123 }
124 }
125
126 pr_info("smp: platform has %u cpu(s)", num_cpus);
127 x86_platform.num_cpus = num_cpus;
128
129 mos_startup_map_pages(X86_AP_TRAMPOLINE_ADDR, X86_AP_TRAMPOLINE_ADDR, 4 KB / MOS_PAGE_SIZE, VM_WRITE);
130 memcpy((void *) X86_AP_TRAMPOLINE_ADDR, (void *) (uintptr_t) &x86_ap_trampoline, 4 KB);
131
132 for (u32 i = 0; i < num_cpus; i++)
133 {
134 u32 apic_id = lapics[i];
135 if (apic_id == x86_platform.boot_cpu_id)
136 continue;
137
138 uintptr_t stack = (uintptr_t) pg_page_alloc(pg, 4, PGALLOC_NONE);
139 pr_info("smp: starting AP %d, LAPIC %d, stack " PTR_FMT, i, apic_id, stack);
140 x86_cpu_start(apic_id, stack);
141 }
142}
