9// Reinitialize the PIC controllers.
10// Giving them specified vector offsets rather than 8h and 70h, as configured by default
11// ICW: Initialization command words
12#define ICW1_ICW4 0x01 /* ICW4 (not) needed */
13#define ICW1_SINGLE 0x02 /* Single (cascade) mode */
14#define ICW1_INTERVAL4 0x04 /* Call address interval 4 (8) */
15#define ICW1_LEVEL 0x08 /* Level triggered (edge) mode */
16#define ICW1_INIT 0x10 /* Initialization - required! */
17
18#define ICW4_8086 0x01 /* 8086/88 (MCS-80/85) mode */
19#define ICW4_AUTO 0x02 /* Auto (normal) EOI */
20#define ICW4_BUF_SLAVE 0x08 /* Buffered mode/slave */
21#define ICW4_BUF_MASTER 0x0C /* Buffered mode/master */
22#define ICW4_SFNM 0x10 /* Special fully nested (not) */
23
24void pic_remap_irq(int offset_master, int offset_slave)
25{
26 u8 a1, a2;
27
28 a1 = port_inb(PIC1_DATA); // save masks
29 a2 = port_inb(PIC2_DATA);
30
31 port_outb(PIC1_COMMAND, ICW1_INIT | ICW1_ICW4); // starts the initialization sequence (in cascade mode)
32 port_outb(PIC2_COMMAND, ICW1_INIT | ICW1_ICW4);
33
34 port_outb(PIC1_DATA, offset_master); // ICW2: Master PIC vector offset
35 port_outb(PIC2_DATA, offset_slave); // ICW2: Slave PIC vector offset
36
37 port_outb(PIC1_DATA, 4); // ICW3: tell Master PIC that there is a slave PIC at IRQ2 (0000 0100)
38 port_outb(PIC2_DATA, 2); // ICW3: tell Slave PIC its cascade identity (0000 0010)
39
40 port_outb(PIC1_DATA, ICW4_8086);
41 port_outb(PIC2_DATA, ICW4_8086);
42
43 port_outb(PIC1_DATA, a1); // restore saved masks.
44 port_outb(PIC2_DATA, a2);
45}

47void pic_mask_irq(x86_irq_enum_t irq)
48{
49 x86_port_t port;
50 u8 value;
51 if (irq < 8)
52 port = PIC1_DATA;
53 else
54 port = PIC2_DATA, irq -= 8;
55 value = port_inb(port) | (1 << irq);
56 port_outb(port, value);
57}
58
59void pic_unmask_irq(x86_irq_enum_t irq)
60{
61 x86_port_t port;
62 u8 value;
63 if (irq < 8)
64 port = PIC1_DATA;
65 else
66 port = PIC2_DATA, irq -= 8;
67 value = port_inb(port) & ~(1 << irq);
68 port_outb(port, value);
69}
70
71void pic_disable(void)
72{
73 port_outb(PIC2_DATA, 0xFF);
74 port_outb(PIC1_DATA, 0xFF);
75}
