47typedef struct
48{
49 as_linked_list;
50 void (*handler)(u32 irq);
51} x86_irq_handler_t;

53static void x86_handle_irq(x86_stack_frame *frame);
54static void x86_handle_exception(x86_stack_frame *frame);
55list_node_t irq_handlers[IRQ_MAX_COUNT];

57void x86_irq_handler_init(void)
58{
59 for (int i = 0; i < IRQ_MAX_COUNT; i++)
60 linked_list_init(&irq_handlers[i]);
61}

63void x86_disable_interrupts()
64{
65 __asm__ volatile("cli");
66}
67
68void x86_enable_interrupts()
69{
70 __asm__ volatile("sti");
71}

73bool x86_install_interrupt_handler(u32 irq, void (*handler)(u32 irq))
74{
75 x86_irq_handler_t *desc = kmalloc(sizeof(x86_irq_handler_t));
76 desc->handler = handler;
77 list_node_append(&irq_handlers[irq], list_node(desc));
78 return true;
79}

81void x86_handle_interrupt(u32 esp)
82{
83 x86_stack_frame *frame = (x86_stack_frame *) esp;
84
85 if (frame->interrupt_number < IRQ_BASE)
86 x86_handle_exception(frame);
87 else if (frame->interrupt_number < MOS_SYSCALL_INTR)
88 x86_handle_irq(frame);
89 else if (frame->interrupt_number == MOS_SYSCALL_INTR)
90 {
91#pragma message "TODO: Implement syscall handling for other arguments"
92 int result = dispatch_ksyscall(frame->eax, frame->ebx, frame->ecx, frame->edx, 0, 0, 0, 0, 0);
93 frame->eax = result;
94 }
95 else
96 pr_warn("Unknown interrupt number: %d", frame->interrupt_number);
97}

99static void x86_handle_exception(x86_stack_frame *stack)
100{
101 x86_disable_interrupts();
102 MOS_ASSERT(stack->interrupt_number < EXCEPTION_COUNT);
103
104 const char *name = x86_exception_names[stack->interrupt_number];
105 const char *intr_type = "";
106
107 // Faults: These can be corrected and the program may continue as if nothing happened.
108 // Traps: Traps are reported immediately after the execution of the trapping instruction.
109 // Aborts: Some severe unrecoverable error.
110 switch ((x86_exception_enum_t) stack->interrupt_number)
111 {
112 case EXCEPTION_DIVIDE_ERROR:
113 case EXCEPTION_DEBUG:
114 case EXCEPTION_NMI:
115 case EXCEPTION_OVERFLOW:
116 case EXCEPTION_BOUND_RANGE_EXCEEDED:
117 case EXCEPTION_INVALID_OPCODE:
118 case EXCEPTION_DEVICE_NOT_AVAILABLE:
119 case EXCEPTION_COPROCESSOR_SEGMENT_OVERRUN:
120 case EXCEPTION_INVALID_TSS:
121 case EXCEPTION_SEGMENT_NOT_PRESENT:
122 case EXCEPTION_STACK_SEGMENT_FAULT:
123 case EXCEPTION_GENERAL_PROTECTION_FAULT:
124 case EXCEPTION_FPU_ERROR:
125 case EXCEPTION_ALIGNMENT_CHECK:
126 case EXCEPTION_SIMD_ERROR:
127 case EXCEPTION_VIRTUALIZATION_EXCEPTION:
128 case EXCEPTION_CONTROL_PROTECTION_EXCEPTION:
129 case EXCEPTION_HYPERVISOR_EXCEPTION:
130 case EXCEPTION_VMM_COMMUNICATION_EXCEPTION:
131 case EXCEPTION_SECURITY_EXCEPTION:
132 {
133 intr_type = "fault";
134 break;
135 }
136
137 case EXCEPTION_BREAKPOINT:
138 {
139 mos_warn("Breakpoint not handled.");
140 return;
141 }
142
143 case EXCEPTION_PAGE_FAULT:
144 {
145 intr_type = "page fault";
146 bool is_write = (stack->error_code & 0x2) != 0;
147
148 uintptr_t fault_address;
149 __asm__ volatile("mov %%cr2, %0" : "=r"(fault_address));
150#if MOS_MEME
151 mos_panic("\n页面错误\n\n\"" PTR_FMT "\" 指令引用的 \"" PTR_FMT "\" 内存。该内存不能为 \"%s\"。\n"
152 "要终止程序，请单击 \"确定\"。\n" //
153 "要调试程序，请单击 \"取消\"。\n\n\n", //
154 (uintptr_t) stack->eip, fault_address, is_write ? "written" : "read");
155#else
156 bool present = (stack->error_code & 0x1) != 0;
157 bool is_user = (stack->error_code & 0x4) != 0;
158 mos_panic("Page Fault: %s code at " PTR_FMT " is trying to %s a %s address " PTR_FMT, //
159 is_user ? "Userspace" : "Kernel", //
160 (uintptr_t) stack->intrrupt.eip, //
161 is_write ? "write into" : "read from", //
162 present ? "present" : "non-present", //
163 fault_address);
164#endif
165 break;
166 }
167
168 case EXCEPTION_DOUBLE_FAULT:
169 case EXCEPTION_MACHINE_CHECK:
170 {
171 intr_type = "abort";
172 break;
173 }
174 default:
175 {
176 name = "unknown";
177 intr_type = "unknown";
178 }
179 }
180
181 mos_panic("x86 %s:\n"
182 "Interrupt #%d ('%s', error code %d)\n"
183 "General Purpose Registers:\n"
184 " EAX: 0x%08x EBX: 0x%08x ECX: 0x%08x EDX: 0x%08x\n"
185 " ESI: 0x%08x EDI: 0x%08x EBP: 0x%08x ESP: 0x%08x\n"
186 " EIP: 0x%08x\n"
187 "Segment Registers:\n"
188 " DS: 0x%08x ES: 0x%08x FS: 0x%08x GS: 0x%08x\n"
189 "Context:\n"
190 " EFLAGS: 0x%08x\n"
191 " Instruction: 0x%x:%08x\n"
192 " Stack: 0x%x:%08x",
193 intr_type, //
194 stack->interrupt_number, //
195 name, //
196 stack->error_code, //
197 stack->eax, stack->ebx, stack->ecx, stack->edx, //
198 stack->esi, stack->edi, stack->ebp, stack->esp, //
199 stack->intrrupt.eip, //
200 stack->ds, stack->es, stack->fs, stack->gs, //
201 stack->intrrupt.eflags, //
202 stack->intrrupt.cs, stack->intrrupt.eip, //
203 stack->intrrupt.ss, stack->intrrupt.esp //
204 );
205}

207static void x86_handle_irq(x86_stack_frame *frame)
208{
209 int irq = frame->interrupt_number - IRQ_BASE;
210
211 if (irq == 7 || irq == 15)
212 {
213 // these irqs may be fake ones, test it
214 u8 pic = (irq < 8) ? PIC1 : PIC2;
215 port_outb(pic + 3, 0x03);
216 if ((port_inb(pic) & 0x80) != 0)
217 goto irq_handled;
218 }
219
220 bool irq_handled = false;
221 list_foreach(x86_irq_handler_t, handler, irq_handlers[irq])
222 {
223 irq_handled = true;
224 handler->handler(irq);
225 }
226
227 if (unlikely(!irq_handled))
228 pr_warn("IRQ %d not handled!", irq);
229
230irq_handled:
231 if (irq >= 8)
232 port_outb(PIC2_COMMAND, PIC_EOI);
233 port_outb(PIC1_COMMAND, PIC_EOI);
234}
