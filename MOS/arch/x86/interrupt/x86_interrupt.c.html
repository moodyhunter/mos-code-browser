52typedef struct
53{
54 as_linked_list;
55 void (*handler)(u32 irq);
56} x86_irq_handler_t;

58static void x86_handle_irq(x86_stack_frame *frame);

62void x86_irq_handler_init(void)
63{
64 for (int i = 0; i < IRQ_MAX_COUNT; i++)
65 linked_list_init(&irq_handlers[i]);
66}

68void x86_disable_interrupts()
69{
70 __asm__ volatile("cli");
71}
72
73void x86_enable_interrupts()
74{
75 __asm__ volatile("sti");

78bool x86_install_interrupt_handler(u32 irq, void (*handler)(u32 irq))
79{
80 x86_irq_handler_t *desc = kmalloc(sizeof(x86_irq_handler_t));
81 desc->handler = handler;
82 list_node_append(&irq_handlers[irq], list_node(desc));
83 return true;
84}

108void x86_handle_interrupt(u32 esp)
109{
110 x86_stack_frame *frame = (x86_stack_frame *) esp;
111
112 thread_t *current = current_thread;
113 MOS_ASSERT_X(current != NULL, "No current thread");
114
115 current->stack.head = frame->iret_params.esp;
116 x86_thread_context_t *context = container_of(current->context, x86_thread_context_t, inner);
117 context->ebp = frame->ebp;
118 context->inner.instruction = frame->iret_params.eip;
119
120 if (frame->interrupt_number < IRQ_BASE)
121 x86_handle_exception(frame);
122 else if (frame->interrupt_number < MOS_SYSCALL_INTR)
123 x86_handle_irq(frame);
124 else if (frame->interrupt_number == MOS_SYSCALL_INTR)
125 {
126#pragma message "TODO: Implement syscall handling for other arguments"
127 frame->eax = (reg32_t) dispatch_syscall(frame->eax, frame->ebx, frame->ecx, frame->edx, frame->esi, frame->edi, 0, 0, 0);
128 }
129 else
130 {
131 pr_warn("Unknown interrupt number: %d", frame->interrupt_number);
132 }
133
134 if (unlikely(current->status != THREAD_STATUS_RUNNING))
135 pr_warn("Thread %d is not in 'running' state", current->tid);
136
137 frame->iret_params.eflags |= 0x200; // enable interrupts
138}

140static void x86_handle_exception(x86_stack_frame *stack)
141{
142 MOS_ASSERT(stack->interrupt_number < EXCEPTION_COUNT);
143
144 const char *name = x86_exception_names[stack->interrupt_number];
145 const char *intr_type = "";
146
147 // Faults: These can be corrected and the program may continue as if nothing happened.
148 // Traps: Traps are reported immediately after the execution of the trapping instruction.
149 // Aborts: Some severe unrecoverable error.
150 switch ((x86_exception_enum_t) stack->interrupt_number)
151 {
152 case EXCEPTION_DIVIDE_ERROR:
153 case EXCEPTION_DEBUG:
154 case EXCEPTION_NMI:
155 case EXCEPTION_OVERFLOW:
156 case EXCEPTION_BOUND_RANGE_EXCEEDED:
157 case EXCEPTION_INVALID_OPCODE:
158 case EXCEPTION_DEVICE_NOT_AVAILABLE:
159 case EXCEPTION_COPROCESSOR_SEGMENT_OVERRUN:
160 case EXCEPTION_INVALID_TSS:
161 case EXCEPTION_SEGMENT_NOT_PRESENT:
162 case EXCEPTION_STACK_SEGMENT_FAULT:
163 case EXCEPTION_GENERAL_PROTECTION_FAULT:
164 case EXCEPTION_FPU_ERROR:
165 case EXCEPTION_ALIGNMENT_CHECK:
166 case EXCEPTION_SIMD_ERROR:
167 case EXCEPTION_VIRTUALIZATION_EXCEPTION:
168 case EXCEPTION_CONTROL_PROTECTION_EXCEPTION:
169 case EXCEPTION_HYPERVISOR_EXCEPTION:
170 case EXCEPTION_VMM_COMMUNICATION_EXCEPTION:
171 case EXCEPTION_SECURITY_EXCEPTION:
172 {
173 intr_type = "fault";
174 break;
175 }
176
177 case EXCEPTION_BREAKPOINT:
178 {
179 mos_warn("Breakpoint not handled.");
180 return;
181 }
182
183 case EXCEPTION_PAGE_FAULT:
184 {
185 intr_type = "page fault";
186
187 uintptr_t fault_address;
188 __asm__ volatile("mov %%cr2, %0" : "=r"(fault_address));
189
190 bool present = (stack->error_code & 0x1) != 0;
191 bool is_write = (stack->error_code & 0x2) != 0;
192 bool is_user = (stack->error_code & 0x4) != 0;
193 bool is_exec = false;
194
195 if (fault_address < 1 KB)
196 {
197 x86_dump_registers(stack);
198 mos_panic("%s NULL pointer dereference at " PTR_FMT " caused by instruction " PTR_FMT ".\n",
199 is_user ? "User" : "Kernel", //
200 fault_address, //
201 (uintptr_t) stack->iret_params.eip //
202 );
203 }
204
205 if (current_thread)
206 {
207 pr_emph("page fault: thread %d, process %s (pid %d) at " PTR_FMT ", instruction " PTR_FMT, //
208 current_thread->tid, //
209 current_process->name, //
210 current_process->pid, //
211 fault_address, //
212 (uintptr_t) stack->iret_params.eip //
213 );
214 }
215
216 bool result = cow_handle_page_fault(fault_address, present, is_write, is_user, is_exec);
217 if (result)
218 return;
219
220#if MOS_MEME
221 MOS_UNUSED(is_user);
222 MOS_UNUSED(present);
223 mos_panic("\n\n\n\"" PTR_FMT "\" \"" PTR_FMT "\" \"%s\"\n"
224 " \"\"\n" //
225 " \"\"\n\n\n", //
226 (uintptr_t) stack->interrupt.eip, fault_address, is_write ? "written" : "read");
227#else
228 if (is_user && !is_write && present)
229 pr_warn("'%s' privilege violation?", current_process->name);
230 x86_dump_registers(stack);
231 mos_panic("Page Fault: %s code at " PTR_FMT " is trying to %s a %s address " PTR_FMT, //
232 is_user ? "Userspace" : "Kernel", //
233 (uintptr_t) stack->iret_params.eip, //
234 is_write ? "write into" : "read from", //
235 present ? "present" : "non-present", //
236 fault_address);
237#endif
238 break;
239 }
240
241 case EXCEPTION_DOUBLE_FAULT:
242 case EXCEPTION_MACHINE_CHECK:
243 {
244 intr_type = "abort";
245 break;
246 }
247 default:
248 {
249 name = "unknown";
250 intr_type = "unknown";
251 }
252 }
253
254 x86_dump_registers(stack);
255 mos_panic("x86 %s:\nInterrupt #%d ('%s', error code %d)\n", intr_type, stack->interrupt_number, name, stack->error_code);
256}

258void irq_send_eoi(u8 irq)
259{
260 if (irq >= 8)
261 port_outb(PIC2_COMMAND, PIC_EOI);
262 port_outb(PIC1_COMMAND, PIC_EOI);
263}

265static void x86_handle_irq(x86_stack_frame *frame)
266{
267 int irq = frame->interrupt_number - IRQ_BASE;
268
269 if (irq == 7 || irq == 15)
270 {
271 // these irqs may be fake ones, test it
272 u8 pic = (irq < 8) ? PIC1 : PIC2;
273 port_outb(pic + 3, 0x03);
274 if ((port_inb(pic) & 0x80) != 0)
275 {
276 irq_send_eoi(irq);
277 return;
278 }
279 }
280
281 irq_send_eoi(irq);
282
283 bool irq_handled = false;
284 list_foreach(x86_irq_handler_t, handler, irq_handlers[irq])
285 {
286 irq_handled = true;
287 handler->handler(irq);
288 }
289
290 if (unlikely(!irq_handled))
291 pr_warn("IRQ %d not handled!", irq);
292}
