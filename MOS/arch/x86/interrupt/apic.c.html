31static uintptr_t lapic_paddr_base = 0;
32
33#define IA32_APIC_BASE_MSR 0x1B
34#define IA32_APIC_BASE_MSR_ENABLE 0x800
35
36void apic_set_base_addr(uintptr_t base_addr)
37{
38 u32 edx = 0;
39 u32 eax = (base_addr & 0xfffff0000) | IA32_APIC_BASE_MSR_ENABLE;
40 cpu_set_msr(IA32_APIC_BASE_MSR, eax, edx);
41}

43u32 apic_reg_read_offset_32(u32 offset)
44{
45 mos_debug("offset: %x", offset);
46 mos_warn("This didn't work properly, need to investigate.");
47 return *(volatile u32 *) (lapic_paddr_base + offset);
48}
49
50void apic_reg_write_offset_32(u32 offset, u32 value)
51{
52 mos_debug("reg: 0x%x, value: 0x%.8x", offset, value);
53 *(volatile u32 *) (lapic_paddr_base + offset) = value;
54}
55
56void apic_reg_write_offset_64(u32 offset, u64 value)
57{
58 mos_debug("reg: 0x%x, value: 0x%.16llx", offset, value);
59 *(volatile u32 *) (lapic_paddr_base + offset + 0x10) = value >> 32;
60 *(volatile u32 *) (lapic_paddr_base + offset) = value & 0xffffffff;
61}

63static void apic_wait_sent()
64{
65 // Wait for the delivery status bit to be set
66 while (apic_reg_read_offset_32(APIC_INTERRUPT_COMMAND_REG_BEGIN) & SET_BITS(12, 1, 1))
67 ;
68}

70void apic_interrupt_full(u8 vec, u8 dest, apic_delivery_mode_t delivery_mode, apic_dest_mode_t dest_mode, bool level, bool trigger, apic_shorthand_t shorthand)
71{
72 u64 value = 0;
73 value |= SET_BITS(0, 8, vec); // Interrupt Vector
74 value |= SET_BITS(8, 3, delivery_mode); // Delivery mode
75 value |= SET_BITS(11, 1, dest_mode); // Logical destination mode
76 value |= SET_BITS(12, 1, 0); // Delivery status (0)
77 value |= SET_BITS(14, 1, level); // Level
78 value |= SET_BITS(15, 1, trigger); // Trigger mode
79 value |= SET_BITS(18, 2, shorthand); // Destination shorthand
80 value |= SET_BITS(56, 8, (u64) dest); // Destination
81
82 apic_reg_write_offset_32(APIC_REG_ERROR_STATUS, 0);
83 apic_reg_write_offset_64(APIC_INTERRUPT_COMMAND_REG_BEGIN, value);
84 apic_wait_sent();
85}
86
87void apic_interrupt(u8 vec, u8 dest, apic_delivery_mode_t delivery_mode, apic_dest_mode_t dest_mode, apic_shorthand_t shorthand)
88{
89 apic_interrupt_full(vec, dest, delivery_mode, dest_mode, true, false, shorthand);
90}

92void apic_enable()
93{
94 // CPUID.01h:EDX[bit 9]
95 processor_version_t info;
96 cpuid_get_processor_info(&info);
97
98 if (!info.edx.onboard_apic)
99 mos_panic("APIC is not supported");
100
101 if (!info.edx.msr)
102 mos_panic("MSR is not present");
103
104 lapic_paddr_base = x86_acpi_madt->lapic_addr;
105 pr_info("apic: mapped address: " PTR_FMT, lapic_paddr_base);
106
107 apic_set_base_addr(lapic_paddr_base);
108
109 // map both the current pagedir and x86_kpg_infra
110 mos_startup_map_bios(lapic_paddr_base, 1 KB, VM_GLOBAL | VM_WRITE | VM_CACHE_DISABLED);
111 pg_do_map_pages(x86_kpg_infra, BIOS_VADDR(lapic_paddr_base), lapic_paddr_base, 1, VM_GLOBAL | VM_WRITE | VM_CACHE_DISABLED);
112 lapic_paddr_base = BIOS_VADDR(lapic_paddr_base);
113
114#define APIC_SOFTWARE_ENABLE (1 << 8)
115 // set the Spurious Interrupt Vector Register
116 // To enable the APIC, set bit 8 (or 0x100) of this register.
117 // All the other bits are currently reserved.
118 apic_reg_write_offset_32(APIC_REG_SPURIOUS_INTR_VEC, apic_reg_read_offset_32(APIC_REG_SPURIOUS_INTR_VEC) | APIC_SOFTWARE_ENABLE);
119
120 u32 version_reg = apic_reg_read_offset_32(APIC_REG_LAPIC_VERSION);
121 u32 max_lvt_entry = (version_reg >> 16) & 0xff;
122 u32 version_id = version_reg & 0xff;
123 pr_info("apic: version: %x, max LVT entry: %x", version_id, max_lvt_entry);
124}
