54#define IA32_APIC_BASE_MSR 0x1B
55#define IA32_APIC_BASE_MSR_ENABLE 0x800
56
57void lapic_set_base_addr(uintptr_t base_addr)
58{
59 u32 edx = 0;
60 u32 eax = (base_addr & 0xfffff0000) | IA32_APIC_BASE_MSR_ENABLE;
61 cpu_set_msr(IA32_APIC_BASE_MSR, eax, edx);
62}

64u32 lapic_read32(u32 offset)
65{
66 mos_debug("reg: %x", offset);
67 return lapic_regs[offset / sizeof(u32)];
68}
69
70u64 lapic_read64(u32 offset)
71{
72 mos_debug("reg: %x", offset);
73 u32 high = lapic_regs[(offset + 0x10) / sizeof(u32)];
74 u32 low = lapic_regs[offset / sizeof(u32)];
75 return ((u64) high << 32) | low;
76}
77
78void lapic_write32(u32 offset, u32 value)
79{
80 mos_debug("reg: %x, value: 0x%.8x", offset, value);
81 lapic_regs[offset / sizeof(u32)] = value;
82#ifdef MOS_DEBUG
83 u32 read_value = lapic_read32(offset);
84 if (read_value != value)
85 mos_warn("INCORRECT: 0x%.8x", read_value);
86#endif
87}
88
89void lapic_write64(u32 offset, u64 value)
90{
91 mos_debug("reg: %x, value: 0x%.16llx", offset, value);
92 lapic_regs[(offset + 0x10) / sizeof(u32)] = value >> 32;
93 lapic_regs[offset / sizeof(u32)] = value & 0xffffffff;
94#ifdef MOS_DEBUG
95 u64 read_value = lapic_read64(offset);
96 if (read_value != value)
97 mos_warn("INCORRECT: 0x%.16llx", read_value);
98#endif
99}

101static void lapic_wait_sent()
102{
103 // Wait for the delivery status bit to be set
104 while (lapic_read32(APIC_INTERRUPT_COMMAND_REG_BEGIN) & SET_BITS(12, 1, 1))
105 ;
106}
107
108void lapic_interrupt_full(u8 vec, u8 dest, lapic_delivery_mode_t delivery_mode, lapic_dest_mode_t dest_mode, bool level, bool trigger, lapic_shorthand_t shorthand)
109{
110 u64 value = 0;
111 value |= SET_BITS(0, 8, vec); // Interrupt Vector
112 value |= SET_BITS(8, 3, delivery_mode); // Delivery mode
113 value |= SET_BITS(11, 1, dest_mode); // Logical destination mode
114 value |= SET_BITS(12, 1, 0); // Delivery status (0)
115 value |= SET_BITS(14, 1, level); // Level
116 value |= SET_BITS(15, 1, trigger); // Trigger mode
117 value |= SET_BITS(18, 2, shorthand); // Destination shorthand
118 value |= SET_BITS(56, 8, (u64) dest); // Destination
119
120 lapic_write32(APIC_REG_ERROR_STATUS, 0);
121 lapic_write64(APIC_INTERRUPT_COMMAND_REG_BEGIN, value);
122 lapic_wait_sent();
123}
124
125void lapic_interrupt(u8 vec, u8 dest, lapic_delivery_mode_t delivery_mode, lapic_dest_mode_t dest_mode, lapic_shorthand_t shorthand)
126{
127 lapic_interrupt_full(vec, dest, delivery_mode, dest_mode, true, false, shorthand);
128}

130void lapic_memory_setup()
131{
132 // CPUID.01h:EDX[bit 9]
133 processor_version_t info;
134 cpuid_get_processor_info(&info);
135
136 if (!info.edx.onboard_apic)
137 mos_panic("APIC is not supported");
138
139 if (!info.edx.msr)
140 mos_panic("MSR is not present");
141
142 uintptr_t base_addr = x86_acpi_madt->lapic_addr;
143 pr_info("LAPIC: base address: " PTR_FMT, base_addr);
144
145 if ((uintptr_t) base_addr != BIOS_VADDR(base_addr))
146 {
147 pr_info("LAPIC: remapping it to " PTR_FMT, BIOS_VADDR(base_addr));
148 base_addr = BIOS_VADDR(base_addr);
149 }
150
151 // map both the current pagedir and x86_kpg_infra (because we are now using the former one)
152 mos_startup_map_bios(base_addr, 1 KB, VM_RW | VM_GLOBAL | VM_CACHE_DISABLED);
153 pg_do_map_page(x86_kpg_infra, base_addr, base_addr, VM_RW | VM_GLOBAL | VM_CACHE_DISABLED);
154
155 lapic_regs = (u32 *) base_addr;
156}

158void lapic_enable()
159{
160 lapic_set_base_addr((uintptr_t) lapic_regs);
161
162 // (https://wiki.osdev.org/APIC#Local_APIC_configuration)
163 // To enable the Local APIC to receive interrupts it is necessary to configure the "Spurious Interrupt Vector Register".
164 // The correct value for this field is
165 // - the IRQ number that you want to map the spurious interrupts to within the lowest 8 bits, and
166 // - the 8th bit set to 1
167 // to actually enable the APIC
168 lapic_write32(APIC_REG_SPURIOUS_INTR_VEC, lapic_read32(APIC_REG_SPURIOUS_INTR_VEC) | (1 << 8));
169
170 u32 current_cpu_id = lapic_read32(APIC_REG_LAPIC_ID);
171 u32 version_reg = lapic_read32(APIC_REG_LAPIC_VERSION);
172 u32 max_lvt_entry = (version_reg >> 16) & 0xff;
173 u32 version_id = version_reg & 0xff;
174 pr_info("LAPIC{%d}: version: %x, max LVT entry: %x", current_cpu_id, version_id, max_lvt_entry);
175}
