27#define IOAPIC_REG_ID 0x00
28#define IOAPIC_REG_VERSION 0x01
29#define IOAPIC_REG_ARB_ID 0x02
30#define IOAPIC_REG_REDIR_TABLE(n) (0x10 + (n) *2)
31
32typedef struct
33{
34 u8 interrupt_vec : 8;
35 u8 delivery_mode : 3; // 0 = Normal, 1 = Low Priority, 2 = SMI, 4 = NMI, 5 = INIT, 7 = External
36 bool destination_mode : 1; // 0 = Physical, 1 = Logical
37 bool pending : 1; // 1 = if this interrupt is going to be sent, but the APIC is busy
38 ioapic_polarity_t polarity : 1; // 0 = Active High, 1 = Active Low
39 bool something : 1; // 0 = a local APIC has sent an EOI, 1 = a local APIC has received the interrupt
40 ioapic_trigger_mode_t trigger_mode : 1; // 0 = Edge, 1 = Level
41 bool mask : 1; // 0 = Enabled, 1 = Disabled
42 u64 reserved : 39;
43 struct
44 {
45 u8 target_apic_id : 4;
46 u8 something_else : 4;
47 } __packed destination;
48} __packed ioapic_redirection_entry_t;
49
50static_assert(sizeof(ioapic_redirection_entry_t) == sizeof(u64), "ioapic_register_1 is not 64 bits");
51
52static u32 volatile *ioapic = NULL;
53
54should_inline u32 ioapic_read(u32 reg)
55{
56 ioapic[0] = reg & 0xff;
57 return ioapic[4];
58}
59
60should_inline void ioapic_write(u32 reg, u32 value)
61{
62 ioapic[0] = reg & 0xff;
63 ioapic[4] = value;
64}
65
66should_inline void ioapic_write_redirection_entry(u32 irq, ioapic_redirection_entry_t entry)
67{
68 union
69 {
70 u64 value;
71 ioapic_redirection_entry_t entry;
72 } __packed u = { .entry = entry };
73
74 ioapic_write(IOAPIC_REG_REDIR_TABLE(irq), u.value & 0xffffffff);
75 ioapic_write(IOAPIC_REG_REDIR_TABLE(irq) + 1, u.value >> 32);
76}
77
78should_inline ioapic_redirection_entry_t ioapic_read_redirection_entry(u32 irq)
79{
80 union
81 {
82 u64 value;
83 ioapic_redirection_entry_t entry;
84 } __packed u = { 0 };
85
86 u.value = ioapic_read(IOAPIC_REG_REDIR_TABLE(irq));
87 u.value |= (u64) ioapic_read(IOAPIC_REG_REDIR_TABLE(irq) + 1) << 32;
88
89 return u.entry;
90}

92void ioapic_init()
93{
94 MOS_ASSERT_X(x86_ioapic_address != 0, "ioapic: no ioapic found in madt");
95 ioapic = (u32 volatile *) x86_ioapic_address;
96 const vmblock_t ioapic_block = (vmblock_t){
97 .vaddr = x86_ioapic_address,
98 .paddr = x86_ioapic_address,
99 .npages = 1,
100 .flags = VM_RW | VM_GLOBAL | VM_CACHE_DISABLED,
101 };
102 mm_map_allocated_pages(x86_platform.kernel_pgd, ioapic_block);
103
104 const u32 ioapic_id = ioapic_read(IOAPIC_REG_ID) >> 24 & 0xf; // get the 24-27 bits
105
106 const union
107 {
108 u32 value;
109 struct
110 {
111 u32 version : 8; // 0 - 7
112 u32 reserved : 8;
113 u32 max_redir_entries : 8; // 16 - 23
114 u32 reserved2 : 8;
115 } __packed;
116 } version = { .value = ioapic_read(IOAPIC_REG_VERSION) };
117
118 const u32 arb_id = ioapic_read(IOAPIC_REG_ARB_ID) >> 24 & 0xf; // get the 24-27 bits
119
120 pr_info2("ioapic: max IRQs: %d, id: %d, version: %d, arb: %d", version.max_redir_entries + 1, ioapic_id, version.version, arb_id);
121
122 for (int i = 0; i < version.max_redir_entries + 1; i++)
123 ioapic_disable(i);
124}

126void ioapic_enable_with_mode(u32 irq, u32 cpu, ioapic_trigger_mode_t trigger_mode, ioapic_polarity_t polarity)
127{
128 pr_info2("ioapic: enable irq %d, cpu %d, trigger_mode %d, polarity %d", irq, cpu, trigger_mode, polarity);
129
130 ioapic_redirection_entry_t entry = { 0 };
131 entry.interrupt_vec = irq + ISR_MAX_COUNT; // the vector number received by the CPU
132 entry.polarity = polarity;
133 entry.trigger_mode = trigger_mode;
134 entry.destination.target_apic_id = cpu;
135
136 u32 irq_overrided = x86_ioapic_get_irq_override(irq); // the irq number received by the ioapic "pin"
137 ioapic_write_redirection_entry(irq_overrided, entry);
138}
139
140void ioapic_disable(u32 irq)
141{
142 ioapic_redirection_entry_t entry = { 0 };
143 entry.interrupt_vec = irq + ISR_MAX_COUNT;
144 entry.mask = true;
145 ioapic_write_redirection_entry(irq, entry);
146}
