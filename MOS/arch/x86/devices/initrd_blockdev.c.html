9void initrd_blockdev_preinstall(initrd_blockdev_t *dev)
10{
11 dev->blockdev.name = "initrd";
12 dev->blockdev.read = initrd_read;
13 dev->blockdev.write = initrd_write;
14}
15
16size_t initrd_read(blockdev_t *dev, void *buf, size_t size, size_t offset)
17{
18 initrd_blockdev_t *initrd = container_of(dev, initrd_blockdev_t, blockdev);
19 size_t read = 0;
20 if (offset >= initrd->memblock.size_bytes)
21 {
22 return 0;
23 }
24
25 size_t bytes_to_read = size;
26
27 if (offset + bytes_to_read > initrd->memblock.size_bytes)
28 bytes_to_read = initrd->memblock.size_bytes - offset;
29
30 memcpy(buf, (void *) (initrd->memblock.vaddr + offset), bytes_to_read);
31 read = bytes_to_read;
32 return read;
33}
34
35size_t initrd_write(blockdev_t *dev, const void *buf, size_t size, size_t offset)
36{
37 MOS_UNUSED(dev);
38 MOS_UNUSED(buf);
39 MOS_UNUSED(size);
40 MOS_UNUSED(offset);
41 mos_warn("initrd device does not support writing");
42 return 0;
43}
