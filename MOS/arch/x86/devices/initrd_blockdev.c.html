7size_t initrd_read(blockdev_t *dev, void *buf, size_t size, size_t offset)
8{
9 initrd_blockdev_t *initrd = container_of(dev, initrd_blockdev_t, blockdev);
10 size_t read = 0;
11 if (offset >= initrd->memblock.size_bytes)
12 {
13 return 0;
14 }
15
16 size_t bytes_to_read = size;
17
18 if (offset + bytes_to_read > initrd->memblock.size_bytes)
19 bytes_to_read = initrd->memblock.size_bytes - offset;
20
21 memcpy(buf, (void *) (initrd->memblock.address + offset), bytes_to_read);
22 read = bytes_to_read;
23 return read;
24}
