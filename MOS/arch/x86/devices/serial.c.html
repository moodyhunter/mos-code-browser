14void set_baudrate_divisor(int com, serial_baud_rate divisor)
15{
16 // Set the most significant bit of the Line Control Register. This is the DLAB bit, and allows access to the divisor registers.
17 byte_t control = to_union(byte_t) port_inb(com + OFFSET_LINE_CONTROL);
18 control.bits.msb = true;
19
20 port_outb(com + OFFSET_LINE_CONTROL, control.byte);
21
22 // Send the least significant byte of the divisor value to [PORT + 0].
23 port_outb(com + OFFSET_DLAB_DIVISOR_LSB, divisor & 0xFF);
24
25 // Send the most significant byte of the divisor value to [PORT + 1].
26 port_outb(com + OFFSET_DLAB_DIVISOR_MSB, divisor >> 8);
27
28 // Clear the most significant bit of the Line Control Register.
29 control.bits.msb = false; // clear DLAB bit
30 port_outb(com + OFFSET_LINE_CONTROL, control.byte);
31}

33void set_data_bits(int com, serial_char_length_t length)
34{
35 char control = port_inb(com + OFFSET_LINE_CONTROL);
36 control &= length;
37 port_outb(com + OFFSET_LINE_CONTROL, control);
38}
39
40void set_stop_bits(int com, serial_stop_bits_t stop_bits)
41{
42 byte_t control = to_union(byte_t) port_inb(com + OFFSET_LINE_CONTROL);
43 control.bits.b1 = stop_bits == STOP_BITS_15_OR_2;
44 port_outb(com + OFFSET_LINE_CONTROL, control.byte);
45}
46
47void set_parity(int com, serial_port_parity_t parity)
48{
49 u8 byte = port_inb(com + OFFSET_LINE_CONTROL);
50 byte |= ((u8) parity) << 3;
51 port_outb(com + OFFSET_LINE_CONTROL, byte);
52}

54void serial_set_interrupts(int com, int interrupts)
55{
56 char control = port_inb(com + OFFSET_INTTERUPT_ENABLE);
57 control = interrupts;
58 port_outb(com + OFFSET_INTTERUPT_ENABLE, control);
59}

61void serial_set_modem_options(int com, serial_modem_control_t control, bool enable)
62{
63 byte_t byte = to_union(byte_t) port_inb(com + OFFSET_MODEM_CONTROL);
64 switch (control)
65 {
66 case MODEM_DTR: byte.bits.b0 = enable; break;
67 case MODEM_RTS: byte.bits.b1 = enable; break;
68 case MODEM_UNUSED_PIN1: byte.bits.b2 = enable; break;
69 case MODEM_IRQ: byte.bits.b3 = enable; break;
70 case MODEM_LOOP: byte.bits.b4 = enable; break;
71 }
72 port_outb(com + OFFSET_MODEM_CONTROL, byte.byte);
73}

85bool serial_device_setup(serial_device_t *device)
86{
87 serial_port_t port = device->port;
88
89 serial_set_interrupts(port, INTERRUPT_NONE);
90 set_baudrate_divisor(port, device->baud_rate);
91 set_data_bits(port, device->char_length);
92 set_stop_bits(port, device->stop_bits);
93 set_parity(port, device->parity);
94
95 serial_set_modem_options(port, MODEM_DTR, true);
96 serial_set_modem_options(port, MODEM_RTS, true);
97
98 // Try send a byte to the serial port.
99 // If it fails, then the serial port is not connected.
100 {
101 const char challenge = 'H';
102 char response = { 0 };
103 serial_set_modem_options(port, MODEM_LOOP, true);
104 serial_device_write(device, &challenge, 1);
105 serial_device_read(device, &response, 1);
106 serial_set_modem_options(port, MODEM_LOOP, false);
107 if (response != 'H')
108 return false;
109 }
110
111 serial_set_modem_options(port, MODEM_IRQ, true);
112 serial_set_interrupts(port, INTERRUPT_DATA_AVAILABLE);
113 return true;
114}

116void serial_dev_wait_ready_to_read(serial_device_t *device)
117{
118 serial_port_t port = device->port;
119 while (!(serial_get_line_status(port) & LINE_DATA_READY))
120 ;
121}
122
123void serial_dev_wait_ready_to_write(serial_device_t *device)
124{
125 serial_port_t port = device->port;
126 while (!(serial_get_line_status(port) & LINE_TRANSMITR_BUF_EMPTY))
127 ;
128}

130int serial_device_write(serial_device_t *device, const char *data, size_t length)
131{
132 serial_port_t port = device->port;
133 size_t i = 0;
134 for (; i < length; i++)
135 {
136 serial_dev_wait_ready_to_write(device);
137 port_outb(port, data[i]);
138 }
139 return i;
140}
141
142int serial_device_read(serial_device_t *device, char *data, size_t length)
143{
144 serial_port_t port = device->port;
145 size_t i = 0;
146 while (i < length)
147 {
148 serial_dev_wait_ready_to_read(device);
149 data[i] = port_inb(port);
150 i++;
151 }
152 return i;
153}
154
155int serial_dev_readline(serial_device_t *device, char *buffer, int max_length)
156{
157 serial_port_t port = device->port;
158 int i = 0;
159 while (i < max_length)
160 {
161 serial_dev_wait_ready_to_read(device);
162 char c = port_inb(port);
163 if (c == '\r')
164 break;
165 if (c == '\n')
166 break;
167 buffer[i] = c;
168 i++;
169 }
170 buffer[i] = '\0';
171 return i;
172}
