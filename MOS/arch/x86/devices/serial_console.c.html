11const char *ansi_reset = ANSI_COLOR_RESET;
12
13bool serial_console_setup(console_t *console)
14{
15 console->write = serial_console_write;
16 console->read = serial_console_read;
17 console->caps |= CONSOLE_CAP_READ;
18 if (console->caps & CONSOLE_CAP_COLOR)
19 {
20 console->set_color = serial_console_set_color;
21 console->get_color = serial_console_get_color;
22 }
23 linked_list_init(list_node(console));
24 serial_console_t *serial_console = container_of(console, serial_console_t, console);
25 return serial_device_setup(&serial_console->device);
26}
27
28int serial_console_write(console_t *console, const char *str, size_t len)
29{
30 serial_console_t *serial_console = container_of(console, serial_console_t, console);
31 int x = serial_device_write(&serial_console->device, str, len);
32 serial_device_write(&serial_console->device, ansi_reset, strlen(ansi_reset));
33 return x;
34}
35
36int serial_console_read(console_t *console, char *str, size_t len)
37{
38 serial_console_t *serial_console = container_of(console, serial_console_t, console);
39 return serial_device_read(&serial_console->device, str, len);
40}

42void get_ansi_color(char *buf, standard_color_t fg, standard_color_t bg)
43{
44 MOS_UNUSED(bg);
45 static const char *g_ansi_colors[] = {
46 [Black] = "" ANSI_COLOR(black),
47 [Blue] = "" ANSI_COLOR(blue),
48 [Green] = "" ANSI_COLOR(green),
49 [Cyan] = "" ANSI_COLOR(cyan),
50 [Red] = "" ANSI_COLOR(red),
51 [Magenta] = "" ANSI_COLOR(magenta),
52 [Brown] = "" ANSI_COLOR(yellow),
53 [Gray] = "" ANSI_COLOR(white, bright),
54 [DarkGray] = "" ANSI_COLOR(white),
55 [LightBlue] = "" ANSI_COLOR(blue, bright),
56 [LightGreen] = "" ANSI_COLOR(green, bright),
57 [LightCyan] = "" ANSI_COLOR(cyan, bright),
58 [LightRed] = "" ANSI_COLOR(red, bright),
59 [LightMagenta] = "" ANSI_COLOR(magenta, bright),
60 [Yellow] = "" ANSI_COLOR(yellow, bright),
61 [White] = "" ANSI_COLOR(white, bright),
62 };
63
64 const char *color = g_ansi_colors[fg];
65
66 // TODO: add support for background colors
67 if (bg == Red)
68 color = ANSI_COLOR(red, blink);
69
70 strcat(buf, color);
71}
72
73bool serial_console_set_color(console_t *device, standard_color_t fg, standard_color_t bg)
74{
75 serial_console_t *serial_console = container_of(device, serial_console_t, console);
76 serial_console->fg = fg;
77 serial_console->bg = bg;
78 char buf[64] = { 0 };
79 get_ansi_color(buf, fg, bg);
80 serial_device_write(&serial_console->device, buf, strlen(buf));
81 return true;
82}
83
84bool serial_console_get_color(console_t *device, standard_color_t *fg, standard_color_t *bg)
85{
86 serial_console_t *serial_console = container_of(device, serial_console_t, console);
87 *fg = serial_console->fg;
88 *bg = serial_console->bg;
89 return true;
90}
