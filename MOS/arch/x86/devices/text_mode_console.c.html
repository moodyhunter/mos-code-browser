11#define VIDEO_WIDTH 80
12#define VIDEO_HEIGHT 25
13
14typedef struct
15{
16 char character;
17 u8 color;
18} __packed video_cell_t;
19
20typedef struct
21{
22 video_cell_t cells[VIDEO_HEIGHT][VIDEO_WIDTH];
23} __packed video_buffer_t;
24
25static const size_t VIDEO_LINE_SIZE = VIDEO_WIDTH * sizeof(video_cell_t);
26
27static video_buffer_t *video_buffer = (video_buffer_t *) BIOS_VADDR(X86_VIDEO_DEVICE_PADDR);
28static u8 cursor_x = 0;
29static u8 cursor_y = 0;
30static standard_color_t foreground_color = White;
31static standard_color_t background_color = Black;
32
33static void screen_scroll(void)
34{
35 memmove(video_buffer, video_buffer->cells[1], (VIDEO_HEIGHT - 1) * VIDEO_LINE_SIZE);
36 memset((void *) (video_buffer->cells[VIDEO_HEIGHT - 1]), 0, VIDEO_LINE_SIZE);
37}
38
39static bool screen_get_size(console_t *console, u32 *width, u32 *height)
40{
41 MOS_UNUSED(console);
42 if (width)
43 *width = VIDEO_WIDTH;
44 if (height)
45 *height = VIDEO_HEIGHT;
46 return width || height;
47}
48
49static bool screen_get_cursor_pos(console_t *console, u32 *x, u32 *y)
50{
51 MOS_UNUSED(console);
52 if (!x && !y)
53 return false;
54 if (x)
55 *x = cursor_x;
56 if (y)
57 *y = cursor_y;
58 return true;
59}
60
61static bool screen_set_cursor_pos(console_t *console, u32 x, u32 y)
62{
63 MOS_UNUSED(console);
64 if (x >= VIDEO_WIDTH || y >= VIDEO_HEIGHT)
65 return false;
66
67 cursor_x = x;
68 cursor_y = y;
69
70 u16 pos = y * VIDEO_WIDTH + x;
71
72 port_outb(0x3D4, 0x0F);
73 port_outb(0x3D5, (u8) (pos & 0xFF));
74 port_outb(0x3D4, 0x0E);
75 port_outb(0x3D5, (u8) ((pos >> 8) & 0xFF));
76
77 return true;
78}

80static void screen_print_char(console_t *console, char c)
81{
82 if (c == '\n')
83 {
84 cursor_x = 0;
85 cursor_y++;
86 }
87 else
88 {
89 video_buffer->cells[cursor_y][cursor_x].character = c;
90 video_buffer->cells[cursor_y][cursor_x].color = background_color << 4 | foreground_color;
91 cursor_x++;
92 }
93 if (cursor_x >= VIDEO_WIDTH)
94 {
95 cursor_x = 0;
96 cursor_y++;
97 }
98 if (cursor_y >= VIDEO_HEIGHT)
99 {
100 screen_scroll();
101 cursor_y--;
102 }
103 screen_set_cursor_pos(console, cursor_x, cursor_y);
104}
105
106static bool screen_enable_cursur(console_t *console, bool enable)
107{
108 MOS_UNUSED(console);
109 if (enable)
110 {
111 u8 start_scanline = 13, end_scanline = 15;
112 port_outb(0x3D4, 0x0A);
113 port_outb(0x3D5, (port_inb(0x3D5) & 0xC0) | start_scanline);
114
115 port_outb(0x3D4, 0x0B);
116 port_outb(0x3D5, (port_inb(0x3D5) & 0xE0) | end_scanline);
117 }
118 else
119 {
120 port_outb(0x3D4, 0x0A);
121 port_outb(0x3D5, 0x20);
122 }
123 return true;
124}

126static bool screen_get_color(console_t *console, standard_color_t *fg, standard_color_t *bg)
127{
128 MOS_UNUSED(console);
129 if (fg)
130 *fg = foreground_color;
131 if (bg)
132 *bg = background_color;
133 return fg || bg;
134}
135
136static bool screen_set_color(console_t *console, standard_color_t fg, standard_color_t bg)
137{
138 MOS_UNUSED(console);
139 foreground_color = fg;
140 background_color = bg;
141 return true;
142}
143
144static int screen_print_string(console_t *console, const char *str, size_t limit)
145{
146 MOS_UNUSED(console);
147 size_t i = 0;
148 for (i = 0; (str[i] != '\0') && (i < limit); i++)
149 screen_print_char(console, str[i]);
150 return i;
151}
152
153static bool screen_clear(console_t *console)
154{
155 MOS_UNUSED(console);
156 int i = 0;
157 for (i = 0; i < VIDEO_WIDTH * VIDEO_HEIGHT; i++)
158 screen_print_char(console, ' ');
159 cursor_x = 0;
160 cursor_y = 0;
161 return true;
162}
