10static gdt_ptr32_t gdt_ptr;
11static gdt_entry32_t gdt[GDT_ENTRY_COUNT] __aligned(8);
12
13extern tss32_t tss_entry;
14typedef enum
15{
16 GDT_ENTRY_CODE,
17 GDT_ENTRY_DATA,
18} gdt_entry_type_t;
19
20typedef enum
21{
22 GDT_RING_KERNEL = 0,
23 GDT_RING_1 = 1,
24 GDT_RING_2 = 2,
25 GDT_RING_USER = 3,
26} gdt_ring_t;
27
28typedef enum
29{
30 GDT_GRAN_BYTE = 0,
31 GDT_GRAN_PAGE = 1,
32} gdt_gran_t;
33
34static gdt_entry32_t *gdt32_set_entry(u32 id, u32 base, u32 limit, gdt_entry_type_t entry_type, gdt_ring_t dpl, gdt_gran_t gran)
35{
36 MOS_ASSERT_X(id < GDT_ENTRY_COUNT, "GDT entry id out of bounds: %u", id);
37 gdt_entry32_t *entry = &gdt[id];
38 entry->base_low = MASK_BITS(base, 24);
39 entry->base_high = MASK_BITS((base >> 24), 8);
40 entry->limit_low = MASK_BITS(limit, 16);
41 entry->limit_high = MASK_BITS((limit >> 16), 4);
42 entry->present = 1;
43 entry->available = 1;
44 entry->read_write = true;
45 entry->long_mode = false;
46 entry->pm32_segment = true;
47 entry->code_data_segment = true;
48 entry->dpl = dpl;
49 entry->executable = (entry_type == GDT_ENTRY_CODE);
50 entry->granularity = gran;
51 entry->accessed = false; // "Best left clear (0),"
52
53 // ! This has to be false forever, for:
54 // * 1) allow system calls to be executed in ring 0,
55 // * otherwise, it means this segment "can" be executed by all rings outer than `dpl`
56 // * 2) always 0 for a TSS segment.
57 entry->conforming_expand_down = false;
58 return entry;
59}
60
61void x86_gdt_init()
62{
63 memset(gdt, 0, sizeof(gdt));
64
65 // {Kernel,User}{Code,Data} Segments
66 // We are using a flat memory model, so the base is 0 and the limit is all the way up to the end of the address space.
67 gdt32_set_entry(1, 0x00000000, 0xFFFFFFFF, GDT_ENTRY_CODE, GDT_RING_KERNEL, GDT_GRAN_PAGE);
68 gdt32_set_entry(2, 0x00000000, 0xFFFFFFFF, GDT_ENTRY_DATA, GDT_RING_KERNEL, GDT_GRAN_PAGE);
69 gdt32_set_entry(3, 0x00000000, 0xFFFFFFFF, GDT_ENTRY_CODE, GDT_RING_USER, GDT_GRAN_PAGE);
70 gdt32_set_entry(4, 0x00000000, 0xFFFFFFFF, GDT_ENTRY_DATA, GDT_RING_USER, GDT_GRAN_PAGE);
71
72 // TSS segment
73 gdt_entry32_t *tss_seg = gdt32_set_entry(5, (uintptr_t) &tss_entry, sizeof(__typeof__(tss_entry)), GDT_ENTRY_CODE, GDT_RING_KERNEL, GDT_GRAN_BYTE);
74
75 // ! Set special attributes for the TSS segment.
76 tss_seg->code_data_segment = 0; // indicates TSS/LDT (see also `accessed`)
77 tss_seg->accessed = 1; // With a system entry (`code_data_segment` = 0), 1 indicates TSS and 0 indicates LDT
78 tss_seg->read_write = 0; // For a TSS, indicates busy (1) or not busy (0).
79 tss_seg->executable = 1; // For a TSS, 1 indicates 32-bit (1) or 16-bit (0).
80 tss_seg->available = 0; // 0 for a TSS
81
82 gdt_ptr.base = gdt;
83 gdt_ptr.limit = sizeof(gdt) - 1;
84 gdt32_flush(&gdt_ptr);
85}

87void x86_ap_gdt_init()
88{
89 gdt32_flush_only(&gdt_ptr);
90}
