35typedef enum
36{
37 FILE_PERM_READ = 1 << 2,
38 FILE_PERM_WRITE = 1 << 1,
39 FILE_PERM_EXEC = 1 << 0,
40} file_perm_t;
41
42typedef struct
43{
44 file_perm_t owner;
45 file_perm_t group;
46 file_perm_t other;
47} __packed file_permissions_t;
48
49always_inline void file_format_perm(file_permissions_t perms, char buf[10])
50{
51 buf[0] = perms.owner & FILE_PERM_READ ? 'r' : '-';
52 buf[1] = perms.owner & FILE_PERM_WRITE ? 'w' : '-';
53 buf[2] = perms.owner & FILE_PERM_EXEC ? 'x' : '-';
54 buf[3] = perms.group & FILE_PERM_READ ? 'r' : '-';
55 buf[4] = perms.group & FILE_PERM_WRITE ? 'w' : '-';
56 buf[5] = perms.group & FILE_PERM_EXEC ? 'x' : '-';
57 buf[6] = perms.other & FILE_PERM_READ ? 'r' : '-';
58 buf[7] = perms.other & FILE_PERM_WRITE ? 'w' : '-';
59 buf[8] = perms.other & FILE_PERM_EXEC ? 'x' : '-';
60 buf[9] = '\0';
61}
