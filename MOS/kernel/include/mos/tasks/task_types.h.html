11typedef struct _terminal terminal_t;
12
13typedef enum
14{
15 THREAD_STATUS_READY, // thread can be scheduled
16 THREAD_STATUS_CREATED, // created or forked, but not ever started
17 THREAD_STATUS_RUNNING, // thread is currently running
18 THREAD_STATUS_BLOCKED, // ?
19 THREAD_STATUS_DEAD, // thread is dead, and will be cleaned up soon by the scheduler
20} thread_status_t;
21
22typedef enum
23{
24 THREAD_FLAG_KERNEL = 0 << 0,
25 THREAD_FLAG_USERMODE = 1 << 0,
26} thread_flags_t;
27
28typedef enum
29{
30 MMAP_DEFAULT = 0 << 0, // default flags
31 MMAP_COW = 1 << 0, // this block is currently copy-on-write-mapped
32 MMAP_PRIVATE = 1 << 1, // this block is private, and should not be shared when forked
33 MMAP_SHAREDMEM = 1 << 2, // this block is shared, and should be shared when forked (aka, two processes can write to the same block)
34} mmap_flags;
35
36typedef struct
37{
38 vmblock_t vm;
39 vm_type type;
40
41 // if MMAP_COW is set, then the flags in vm contains 'original' flags
42 // of this block. Which means if there're no VM_WRITE flag, then the block
43 // should not be writable.
44 mmap_flags map_flags;
45} proc_vmblock_t;
46
47typedef struct _wait_condition wait_condition_t;
48typedef struct _thread thread_t;
49typedef struct _process process_t;
50
51typedef struct _wait_condition
52{
53 void *arg;
54 bool (*verify)(wait_condition_t *condition); // return true if condition is met
55 void (*cleanup)(wait_condition_t *condition);
56} wait_condition_t;
57
58typedef struct _process
59{
60 u32 magic;
61 const char *name;
62 pid_t pid;
63 process_t *parent;
64 terminal_t *terminal;
65 uid_t effective_uid;
66 paging_handle_t pagetable;
67
68 ssize_t files_count;
69 io_t *files[MOS_PROCESS_MAX_OPEN_FILES];
70
71 ssize_t threads_count;
72 thread_t *threads[MOS_PROCESS_MAX_THREADS];
73
74 ssize_t mmaps_count;
75 proc_vmblock_t *mmaps;
76} process_t;
77
78typedef struct _thread
79{
80 u32 magic;
81 tid_t tid;
82 thread_status_t status;
83 process_t *owner;
84 downwards_stack_t stack;
85 downwards_stack_t kernel_stack;
86 platform_context_t *context;
87 thread_flags_t flags;
88 wait_condition_t *waiting_condition;
89} thread_t;
