9#define STD_COLOR_LIGHT 0x8
10
11typedef enum
12{
13 Black = 0x0,
14 Blue = 0x1,
15 Green = 0x2,
16 Cyan = 0x3,
17 Red = 0x4,
18 Magenta = 0x5,
19 Brown = 0x6,
20 Gray = 0x7,
21 DarkGray = Black | STD_COLOR_LIGHT,
22 LightBlue = Blue | STD_COLOR_LIGHT,
23 LightGreen = Green | STD_COLOR_LIGHT,
24 LightCyan = Cyan | STD_COLOR_LIGHT,
25 LightRed = Red | STD_COLOR_LIGHT,
26 LightMagenta = Magenta | STD_COLOR_LIGHT,
27 Yellow = Brown | STD_COLOR_LIGHT,
28 White = Gray | STD_COLOR_LIGHT,
29} standard_color_t;
30
31typedef enum
32{
33 CONSOLE_CAP_NONE = 0,
34 CONSOLE_CAP_COLOR = 1 << 0,
35 CONSOLE_CAP_CLEAR = 1 << 1,
36 CONSOLE_CAP_READ = 1 << 2,
37 CONSOLE_CAP_SETUP = 1 << 3,
38 CONSOLE_CAP_GET_SIZE = 1 << 4,
39 CONSOLE_CAP_CURSOR_HIDE = 1 << 5,
40 CONSOLE_CAP_CURSOR_MOVE = 1 << 6,
41} console_caps_t;
42
43typedef struct console_t console_t;
44
45struct console_t
46{
47 as_linked_list;
48
49 const char *name;
50 console_caps_t caps;
51
52 bool (*setup)(console_t *con);
53 bool (*get_size)(console_t *con, u32 *width, u32 *height);
54
55 bool (*set_cursor)(console_t *con, bool show);
56 bool (*move_cursor)(console_t *con, u32 x, u32 y);
57 bool (*get_cursor)(console_t *con, u32 *x, u32 *y);
58
59 // VGA standard color codes
60 bool (*get_color)(console_t *con, standard_color_t *fg, standard_color_t *bg);
61 bool (*set_color)(console_t *con, standard_color_t fg, standard_color_t bg);
62
63 // not meant to be called directly, use console_read / console_write instead
64 int (*read_impl)(console_t *con, char *dest, size_t size);
65 int (*write_impl)(console_t *con, const char *data, size_t size);
66
67 bool (*clear)(console_t *con);
68 bool (*close)(console_t *con);
69
70 void *data;
71 spinlock_t lock __aligned(8);
72};
