8typedef enum
9{
10 ELF_BITS_INVALID = 0,
11 ELF_BITS_32 = 1,
12 ELF_BITS_64 = 2,
13} elf_bits;
14
15#if MOS_32BITS
16#define ELF_BITS_MOS_DEFAULT ELF_BITS_32
17#else
18#define ELF_BITS_MOS_DEFAULT ELF_BITS_64
19#endif
20
21typedef enum
22{
23 ELF_ENDIANNESS_INVALID = 0,
24 ELF_ENDIANNESS_LSB = 1,
25 ELF_ENDIANNESS_MSB = 2,
26} elf_endianness;
27
28#if MOS_LITTLE_ENDIAN
29#define ELF_ENDIANNESS_MOS_DEFAULT ELF_ENDIANNESS_LSB
30#else
31#define ELF_ENDIANNESS_MOS_DEFAULT ELF_ENDIANNESS_MSB
32#endif

34typedef enum
35{
36 ELF_VERSION_NONE = 0,
37 ELF_VERSION_CURRENT = 1,
38} elf_version_type;
39
40typedef enum
41{
42 ELF_OSABI_NONE = 0,
43 ELF_OSABI_LINUX = 3,
44 ELF_OSABI_HURD = 4,
45 ELF_OSABI_SOLARIS = 6,
46 ELF_OSABI_FREEBSD = 9,
47 ELF_OSABI_ARM_AEABI = 64,
48 ELF_OSABI_ARM = 97,
49 ELF_OSABI_MOS = 254, // ! Long live the MOS!
50 ELF_OSABI_STANDALONE = 255,
51} elf_osabi_type;
52
53typedef struct
54{
55 char magic[4];
56 elf_bits bits : 8;
57 elf_endianness endianness : 8;
58 elf_version_type version : 8;
59 elf_osabi_type osabi : 8;
60 u8 abiversion : 8;
61 u8 __padding[7];
62} elf_identity_t;
63
64static_assert(sizeof(elf_identity_t) == 16, "elf_identity_t has wrong size");
65
66typedef enum
67{
68 ELF_OBJTYPE_NONE = 0,
69 ELF_OBJTYPE_RELOCATABLE = 1,
70 ELF_OBJTYPE_EXECUTABLE = 2,
71 ELF_OBJTYPE_SHARED_OBJECT = 3,
72 ELF_OBJTYPE_CORE = 4,
73 ELF_OBJTYPE_PROCESSOR_SPECIFIC_LO = 0xff00,
74 ELF_OBJTYPE_PROCESSOR_SPECIFIC_HI = 0xffff,
75} elf_object_type;
76
77typedef enum
78{
79 ELF_MACHINE_NONE = 0,
80 ELF_MACHINE_X86 = 0x03,
81 ELF_MACHINE_MIPS = 0x08,
82 ELF_MACHINE_ARM = 0x28,
83 ELF_MACHINE_X86_64 = 0x3e,
84 ELF_MACHINE_AARCH64 = 0xb7,
85 ELF_MACHINE_RISCV = 0xf3,
86} elf_machine_type;
87
88typedef struct
89{
90 elf_identity_t identity;
91 elf_object_type object_type : 16;
92 elf_machine_type machine_type : 16;
93
94 u32 version;
95
96 uintptr_t entry_point;
97 size_t ph_offset;
98 size_t sh_offset;
99
100 u32 flags;
101 u16 header_size;
102
103 struct
104 {
105 u16 entry_size, count;
106 } __packed ph;
107
108 struct
109 {
110 u16 entry_size, count;
111 } __packed sh;
112
113 u16 sh_strtab_index;
114} __packed elf_header_t;
115
116typedef enum
117{
118 ELF_PH_T_NULL = 0, // Unused entry
119 ELF_PH_T_LOAD = 1, // Loadable segment
120 ELF_PH_T_DYNAMIC = 2, // Dynamic linking information
121 ELF_PH_T_INTERP = 3, // Interpreter information
122 ELF_PH_T_NOTE = 4, // Auxiliary information
123 ELF_PH_T_SHLIB = 5, // reserved
124 ELF_PH_T_PHDR = 6, // Segment containing program header table
125 ELF_PH_T_TLS = 7, // Thread-local storage segment
126
127 ELF_PH_T_OS_LOW = 0x60000000, // reserved
128 ELF_PH_T_OS_HIGH = 0x6fffffff, // reserved
129 ELF_PH_T_PROCESSOR_LO = 0x70000000, // reserved
130 ELF_PH_T_PROCESSOR_HI = 0x7fffffff, // reserved
131} elf_program_header_type;
132
133typedef enum
134{
135 ELF_PH_F_X = 1 << 0, // Executable
136 ELF_PH_F_W = 1 << 1, // Writable
137 ELF_PH_F_R = 1 << 2, // Readable
138} elf_program_header_flags;
139
140typedef struct
141{
142 elf_program_header_type header_type;
143#if MOS_64BITS
144 u32 p_flags; // Segment independent flags (64-bit only)
145#endif
146 uintptr_t data_offset; // Offset of the segment in the file
147 uintptr_t vaddr; // Virtual address of the segment
148 uintptr_t _reserved; // reserved
149 uintptr_t segsize_in_file; // Size of the segment in the file (may be 0)
150 uintptr_t segsize_in_mem; // Size of the segment in memory (may be 0)
151#if MOS_32BITS
152 u32 p_flags; // Segment independent flags (32-bit only)
153#endif
154 uintptr_t required_alignment;
155} __packed elf_program_hdr_t;
156
157typedef enum
158{
159 ELF_SH_T_NULL = 0, // Unused entry
160 ELF_SH_T_PROGBITS = 1, // Program data
161 ELF_SH_T_SYMTAB = 2, // Symbol table
162 ELF_SH_T_STRTAB = 3, // String table
163 ELF_SH_T_RELA = 4, // Relocation entries with addends
164 ELF_SH_T_HASH = 5, // Symbol hash table
165 ELF_SH_T_DYNAMIC = 6, // Dynamic linking information
166 ELF_SH_T_NOTE = 7, // Auxiliary information
167 ELF_SH_T_NOBITS = 8, // Data
168 ELF_SH_T_REL = 9, // Relocation entries without addends`
169 ELF_SH_T_SHLIB = 10, // Reserved
170 ELF_SH_T_DYNSYM = 11, // Dynamic linker symbol table
171 ELF_SH_T_INIT_ARRAY = 14, // Array of constructors
172 ELF_SH_T_FINI_ARRAY = 15, // Array of destructors
173 ELF_SH_T_PREINIT_ARRAY = 16, // Array of pre-constructors
174 ELF_SH_T_GROUP = 17, // Section group
175 ELF_SH_T_SYMTAB_SHNDX = 18, // Extended section indeces
176 ELF_SH_T_NUM = 19, // Number of defined types
177 ELF_SH_T_LOOS = 0x60000000, // Start of OS-specific
178} elf_section_header_type;
179
180typedef enum
181{
182 ELF_SH_ATTR_WRITE = 1, // Writable
183 ELF_SH_ATTR_ALLOC = 2, // Occupies memory during execution
184 ELF_SH_ATTR_EXECINSTR = 4, // Executable
185 ELF_SH_ATTR_MERGE = 0x10, // Might be merged
186 ELF_SH_ATTR_STRINGS = 0x20, // Contains nul-terminated strings
187 ELF_SH_ATTR_INFO_LINK = 0x40, // `sh_info' contains SHT index
188 ELF_SH_ATTR_LINK_ORDER = 0x80, // Preserve order after combining
189 ELF_SH_ATTR_OS_NONCONFORMING = 0x100, // Non-standard OS specific
190 ELF_SH_ATTR_GROUP = 0x200, // Section is member of a group
191 ELF_SH_ATTR_TLS = 0x400, // Section hold thread-local data
192} elf_section_attribute;
193
194typedef struct
195{
196 u32 name_index; // Section name (string table (.shstrtab) index)
197 elf_section_header_type header_type;
198#if MOS_64BITS
199 u64 attributes; // sizeof(long)
200#else
201 elf_section_attribute attributes;
202#endif
203 uintptr_t sh_addr; // Virtual address of the section in memory, if loaded
204 uintptr_t sh_offset; // Offset of the section in the file
205 size_t sh_size; // Size of the section in bytes
206 u32 sh_link;
207 u32 sh_info;
208 uintptr_t sh_addralign;
209 size_t sh_entsize;
210} __packed elf_section_hdr_t;
