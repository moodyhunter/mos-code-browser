9#define PER_CPU_DECLARE(type, name) \
10 struct \
11 { \
12 type percpu_value[MOS_MAX_CPU_COUNT]; \
13 } name
14
15#define per_cpu(var) (&(var.percpu_value[platform_current_cpu_id()]))

17#define current_cpu per_cpu(platform_info->cpu)
18#define current_thread (current_cpu->thread)

27typedef enum
28{
29 VM_NONE = 1 << 1,
30 VM_READ = 1 << 2,
31 VM_WRITE = 1 << 3,
32 VM_USER = 1 << 4,
33 VM_WRITE_THROUGH = 1 << 5,
34 VM_CACHE_DISABLED = 1 << 6,
35 VM_GLOBAL = 1 << 7,
36 VM_EXEC = 1 << 8,
37
38 // composite flags (for convenience)
39 VM_RW = VM_READ | VM_WRITE,
40 VM_USER_RW = VM_USER | VM_RW,
41} vm_flags;
42
43typedef enum
44{
45 PGALLOC_HINT_KHEAP,
46 PGALLOC_HINT_UHEAP,
47 PGALLOC_HINT_STACK,
48 PGALLOC_HINT_MMAP,
49} pgalloc_hints;
50
51typedef enum
52{
53 VMTYPE_APPCODE, // '.text' section
54 VMTYPE_APPDATA, // '.data' sections
55 VMTYPE_APPDATA_ZERO, // '.bss' sections (zeroed out)
56 VMTYPE_HEAP, // userspace heap
57 VMTYPE_STACK, // userspace stack
58 VMTYPE_KSTACK, // kernel stack
59 VMTYPE_SHM, // shared memory
60 VMTYPE_FILE, // file mapping (mmap)
61} vm_type;
62
63typedef enum
64{
65 HEAP_GET_BASE,
66 HEAP_GET_TOP,
67 HEAP_SET_TOP,
68 HEAP_GET_SIZE,
69 HEAP_GROW_PAGES,
70} heap_control_op;
71
72typedef struct
73{
74 uintptr_t instruction;
75} __packed platform_context_t;

77typedef struct
78{
79 u32 id;
80 thread_t *thread;
81 uintptr_t scheduler_stack;
82 paging_handle_t pagetable;
83} cpu_t;
84
85typedef struct
86{
87 uintptr_t vaddr;
88 size_t npages;
89 vm_flags flags; // the expected flags for the region, regardless of the copy-on-write state
90} vmblock_t;

92typedef struct
93{
94 struct
95 {
96 const uintptr_t code_start;

102 } regions;
103
104 u32 num_cpus;
105 u32 boot_cpu_id;
