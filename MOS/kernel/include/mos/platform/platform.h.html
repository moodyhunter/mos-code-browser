9#define PER_CPU_DECLARE(type, name) \
10 struct \
11 { \
12 type percpu_value[MOS_MAX_CPU_COUNT]; \
13 } name
14
15#define per_cpu(var) (&(var.percpu_value[platform_current_cpu_id()]))

28typedef enum
29{
30 VM_NONE = 1 << 1,
31 VM_READ = 1 << 2,
32 VM_WRITE = 1 << 3,
33 VM_USER = 1 << 4,
34 VM_WRITE_THROUGH = 1 << 5,
35 VM_CACHE_DISABLED = 1 << 6,
36 VM_GLOBAL = 1 << 7,
37 VM_EXEC = 1 << 8,
38
39 // composite flags (for convenience)
40 VM_RW = VM_READ | VM_WRITE,
41 VM_USER_RW = VM_USER | VM_RW,
42} vm_flags;
43
44typedef enum
45{
46 PGALLOC_HINT_KHEAP,
47 PGALLOC_HINT_UHEAP,
48 PGALLOC_HINT_STACK,
49 PGALLOC_HINT_MMAP,
50} pgalloc_hints;
51
52typedef enum
53{
54 VMTYPE_APPCODE, // '.text' section
55 VMTYPE_APPDATA, // '.data' sections
56 VMTYPE_APPDATA_ZERO, // '.bss' sections (zeroed out)
57 VMTYPE_HEAP, // userspace heap
58 VMTYPE_STACK, // userspace stack
59 VMTYPE_KSTACK, // kernel stack
60 VMTYPE_SHM, // shared memory
61 VMTYPE_FILE, // file mapping (mmap)
62} vm_type;
63
64typedef enum
