8#define __startup_code __section(.mos.startup.text)
9#define __startup_rodata __section(.mos.startup.rodata)
10#define __startup_rwdata __section(.mos.startup.data)
11
12// map a single page of a given physical address to a given virtual address
13// vaddr and paddr must be page-aligned
14__startup_code void mos_startup_map_single_page(uintptr_t vaddr, uintptr_t paddr, vm_flags flags);
15
16__startup_code always_inline void mos_startup_memzero(void *start, size_t size)
17{
18 u8 *ptr = start;
19 for (size_t i = 0; i < size; i++)
20 ptr[i] = 0;
21}
22
23__startup_code always_inline size_t mos_startup_strlen(const char *str)
24{
25 size_t len = 0;
26 while (str[len])
27 len++;
28 return len;
29}
30
31__startup_code always_inline void mos_startup_map_pages(uintptr_t vaddr, uintptr_t paddr, size_t npages, vm_flags flags)
32{
33 for (size_t i = 0; i < npages; i++)
34 mos_startup_map_single_page(vaddr + i * MOS_PAGE_SIZE, paddr + i * MOS_PAGE_SIZE, flags);
35}
36
37__startup_code always_inline void mos_startup_map_bytes(uintptr_t vaddr, uintptr_t paddr, size_t nbytes, vm_flags flags)
38{
39 paddr = ALIGN_DOWN_TO_PAGE(paddr);
40 const uintptr_t aligned_vaddr = ALIGN_DOWN_TO_PAGE(vaddr);
41 const size_t npages = ALIGN_UP_TO_PAGE(nbytes + (vaddr - aligned_vaddr)) / MOS_PAGE_SIZE;
42 mos_startup_map_pages(aligned_vaddr, paddr, npages, flags);
43}
44
45__startup_code always_inline void mos_startup_map_identity(uintptr_t paddr, size_t nbytes, vm_flags flags)
46{
47 paddr = ALIGN_DOWN_TO_PAGE(paddr);
48 mos_startup_map_bytes(paddr, paddr, nbytes, flags);
49}
50
51__startup_code always_inline void mos_startup_map_bios(uintptr_t paddr, size_t nbytes, vm_flags flags)
52{
53 paddr = ALIGN_DOWN_TO_PAGE(paddr);
54 mos_startup_map_bytes(BIOS_VADDR(paddr), paddr, nbytes, flags | VM_CACHE_DISABLED);
55}
