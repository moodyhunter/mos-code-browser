12#define MOS_UNIMPLEMENTED(content) mos_panic("\nUNIMPLEMENTED: %s", content)
13#define MOS_UNREACHABLE() mos_panic("\nUNREACHABLE line %d reached in file: %s", __LINE__, __FILE__)

15#define MOS_ASSERT_ONCE(...) MOS_ASSERT_X(once(), __VA_ARGS__)
16#define MOS_ASSERT(cond) MOS_ASSERT_X(cond, "")
17#define MOS_ASSERT_X(cond, msg, ...) \
18 do \
19 { \
20 if (unlikely(!(cond))) \
21 mos_panic("Assertion failed: %s \n" msg, #cond, ##__VA_ARGS__); \
22 } while (0)

24typedef enum
25{
26 MOS_LOG_FATAL = 6,
27 MOS_LOG_EMERG = 5,
28 MOS_LOG_WARN = 4,
29 MOS_LOG_EMPH = 3,
30 MOS_LOG_INFO = 2,
31 MOS_LOG_INFO2 = 1,
32 MOS_LOG_DEFAULT = MOS_LOG_INFO,
33} mos_log_level_t;

35#define mos_debug(feat, fmt, ...) \
36 do \
37 { \
38 if (MOS_DEBUG_FEATURE(feat)) \
39 pr_info2("%s: " fmt, #feat, ##__VA_ARGS__); \
40 } while (0)

42#if MOS_CONFIG(MOS_PRINTK_WITH_FILENAME)
43#define lprintk_wrapper(level, fmt, ...) lprintk(level, "%-30s | " fmt, MOS_FILE_LOCATION, ##__VA_ARGS__)
44#else
45#define lprintk_wrapper(level, fmt, ...) lprintk(level, "" fmt, ##__VA_ARGS__)
46#endif

48// print a colored message without handler, print unconditionally without a handler
49#define pr_info(fmt, ...) lprintk_wrapper(MOS_LOG_INFO, fmt "\r\n", ##__VA_ARGS__)
50#define pr_info2(fmt, ...) lprintk_wrapper(MOS_LOG_INFO2, fmt "\r\n", ##__VA_ARGS__)
51#define pr_emph(fmt, ...) lprintk_wrapper(MOS_LOG_EMPH, fmt "\r\n", ##__VA_ARGS__)
52#define pr_warn(fmt, ...) lprintk_wrapper(MOS_LOG_WARN, fmt "\r\n", ##__VA_ARGS__)
53#define pr_emerg(fmt, ...) lprintk_wrapper(MOS_LOG_EMERG, fmt "\r\n", ##__VA_ARGS__)
54#define pr_fatal(fmt, ...) lprintk_wrapper(MOS_LOG_FATAL, fmt "\r\n", ##__VA_ARGS__)
55
56// these two also invokes a warning/panic handler
57#define mos_warn(fmt, ...) mos_kwarn(__func__, __LINE__, "WARN: " fmt "\r\n", ##__VA_ARGS__)
58#define mos_panic(fmt, ...) mos_kpanic(__func__, __LINE__, "" fmt, ##__VA_ARGS__)
59
60#define mos_warn_once(...) \
61 do \
62 { \
63 if (once()) \
64 mos_warn(__VA_ARGS__); \
65 } while (0)
