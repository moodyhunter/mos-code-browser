3#pragma once
4
5#include "lib/structures/bitmap.h"
6#include "mos/platform/platform.h"

8/**
9 * @defgroup paging Paging
10 * @ingroup mm
11 * @brief This module provides a platform-independent interface to interact
12 * with the paging subsystem, such as allocating and freeing pages, and
13 * mapping and unmapping them.
14 *
15 * @{
16 */
17
18/// @brief Maximum 'lines' in a page map, see also @ref bitmap_line_t.
19#define MOS_PAGEMAP_MAX_LINES BITMAP_LINE_COUNT((uintptr_t) ~0 / MOS_PAGE_SIZE)
20
21/// @brief Number of lines in the page map for user space.
22#define MOS_PAGEMAP_USER_LINES BITMAP_LINE_COUNT(MOS_KERNEL_START_VADDR / MOS_PAGE_SIZE)
23
24/// @brief Number of lines in the page map for kernel space.
25#define MOS_PAGEMAP_KERNEL_LINES (MOS_PAGEMAP_MAX_LINES - MOS_PAGEMAP_USER_LINES)
26
27typedef struct _page_map
28{
29 bitmap_line_t ummap[MOS_PAGEMAP_USER_LINES];
30} page_map_t;
31
32/**
33 * @brief Gets npages unmapped free pages from a page table.
34 *
35 * @param table The page table to allocate from.
36 * @param npages The number of pages to allocate.
37 * @param hints Allocation hints, see @ref pgalloc_hints.
38 * @return vmblock_t The determined block of virtual memory.
39 *
40 * @note This function neither allocates nor maps the pages, it only
41 * determines the block of virtual memory that can be used to allocate
42 * and map the pages.
43 *
44 * @note The returned @ref vmblock_t only contains the virtual address
45 * and the number of pages, it does not contain any physical addresses,
46 * nor the flags of the pages.
47 */
48
49vmblock_t mm_get_free_pages(paging_handle_t table, size_t npages, pgalloc_hints hints);
50
51/**
52 * @brief Allocate npages pages from a page table.
53 *
54 * @param table The page table to allocate from.
55 * @param npages The number of pages to allocate.
56 * @param hints Allocation hints, see @ref pgalloc_hints.
57 * @param flags Flags to set on the pages, see @ref vm_flags.
58 * @return vmblock_t The allocated block of virtual memory, with the
59 * number of pages, physical addresses and flags set.
60 *
61 * @details This function first finds a block of virtual memory using
62 * @ref mm_get_free_pages, then allocates and maps the pages.
63 */
64vmblock_t mm_alloc_pages(paging_handle_t table, size_t npages, pgalloc_hints hints, vm_flags flags);
65
66/**
67 * @brief Allocate npages pages from a page table at a specific virtual address.
68 *
69 * @param table The page table to allocate from.
70 * @param vaddr The virtual address to allocate at.
71 * @param npages The number of pages to allocate.
72 * @param flags Flags to set on the pages, see @ref vm_flags.
73 * @return vmblock_t The allocated block of virtual memory, with the number of pages, physical addresses and flags set.
74 */
75vmblock_t mm_alloc_pages_at(paging_handle_t table, uintptr_t vaddr, size_t npages, vm_flags flags);
76
77/**
78 * @brief Free a block of virtual memory, and return the corresponding physical memory to the allocator.
79 *
80 * @param table The page table to free from.
81 * @param block The block of memory to free.
82 *
83 * @details This function unmaps and frees the pages in the block, it returns the corresponding physical memory
84 * back to the allocator.
85 */
86void mm_free_pages(paging_handle_t table, vmblock_t block);
87
88/**
89 * @brief Map a block of virtual memory to a given block of physical memory, with the physical memory requested from the physical memory allocator.
90 *
91 * @param table The page table to map in.
92 * @param block The block of virtual memory to map.
93 *
94 * @details This function maps the pages in the block, then try requesting the corresponding physical memory
95 * from the physical memory allocator, if the physical memory is not available, the action will fail.
96 *
97 * @warning This function is rarely used directly, it's only used to map the kernel image.
98 */
99void mm_map_pages(paging_handle_t table, vmblock_t block);
100
101/**
102 * @brief Map a block of virtual memory to a given block of physical memory, without the interaction of the physical memory allocator.
103 *
104 * @param table The page table to map in.
105 * @param block The block of virtual memory to map.
106 *
107 * @details This function maps the pages in the block, unlike the above @ref mm_map_pages, this function does
108 * not request the corresponding physical memory from the physical memory allocator, instead, it assumes that
109 * the physical memory is already allocated.
110 *
111 * @note This function can be used to map the hardware memory to a specific virtual address, as hardware memory is
112 * not managed by the physical memory allocator.
113 */
114void mm_map_allocated_pages(paging_handle_t table, vmblock_t block);
115
116/**
117 * @brief Unmap a block of virtual memory.
118 *
119 * @param table The page table to unmap from.
120 * @param vaddr The virtual address to unmap.
121 * @param npages The number of pages to unmap.
122 *
123 * @details This function unmaps the pages in the block, it does not return the corresponding physical memory
124 * back to the allocator.
125 *
126 * @note Imagine that you have shared memory between address space A and B, then B unmaps the shared memory,
127 * the physical memory is still used by A, so it's not returned to the allocator.
128 */
129void mm_unmap_pages(paging_handle_t table, uintptr_t vaddr, size_t npages);
130
131/**
132 * @brief Copy a block of virtual memory from one page table to another.
133 *
134 * @param from The page table to copy from.
135 * @param fvaddr The virtual address to copy from.
136 * @param to The page table to copy to.
137 * @param tvaddr The virtual address to copy to.
138 * @param npages The number of pages to copy.
139 *
140 * @details This function copies the pages mappings, including the flags and physical addresses, from one page
141 * table to another.
142 *
143 * @note This function does not copy the actual data, it's used intensively by CoW, Zero-on-Demand and shared memory.
144 */
145vmblock_t mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages);
146
147/**
148 * @brief Get if a virtual address is mapped in a page table.
149 *
150 * @param table The page table to get the physical address from.
151 * @param vaddr The virtual address to get the physical address of.
152 * @return bool True if the virtual address is mapped, false otherwise.
153 */
154bool mm_get_is_mapped(paging_handle_t table, uintptr_t vaddr);
155
156/**
157 * @brief Get the block info of a virtual address in a page table.
158 *
159 * @param table The page table to get the physical address from.
160 * @param vaddr The virtual address to get the physical address of.
161 * @return vmblock_t The block info of the virtual address.
162 */
163vmblock_t mm_get_block_info(paging_handle_t table, uintptr_t vaddr, size_t n_pages);
164
165/**
166 * @brief Create a user-mode platform-dependent page table.
167 * @return paging_handle_t The created page table.
168 * @note A platform-independent page-map is also created.
169 */
170paging_handle_t mm_create_user_pgd();
171
172/**
173 * @brief Destroy a user-mode platform-dependent page table.
174 * @param table The page table to destroy.
175 * @note The platform-independent page-map is also destroyed.
176 */
177void mm_destroy_user_pgd(paging_handle_t table);
178
