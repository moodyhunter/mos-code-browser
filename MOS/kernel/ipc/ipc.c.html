37static void ipc_server_close(io_t *io)
38{
39 pr_info("closing ipc server (io refcount: %zu)", io->refcount);
40 ipc_server_t *server = container_of(io, ipc_server_t, io);
41 if (server->magic != IPC_SERVER_MAGIC)
42 {
43 mos_warn("invalid magic");
44 return;
45 }
46
47 for (size_t i = 0; i < server->max_pending; i++)
48 {
49 ipc_connection_t *conn = &server->pending[i];
50 if (conn->state == IPC_CONNECTION_STATE_PENDING)
51 conn->state = IPC_CONNECTION_STATE_CLOSED;
52 }
53
54 hashmap_remove(ipc_servers, server->name);
55 kfree(server->name);
56 kfree(server->pending);
57 kfree(server);
58}

64static ipc_connection_t *ipc_server_get_pending(ipc_server_t *server)
65{
66 for (size_t i = 0; i < server->max_pending; i++)
67 {
68 if (server->pending[i].state == IPC_CONNECTION_STATE_PENDING)
69 return &server->pending[i];
70 }
71 return NULL;
72}

108static size_t ipc_connection_server_read(io_t *io, void *buf, size_t buf_size)
109{
110 ipc_connection_t *conn = container_of(io, ipc_connection_t, server_io);
111 if (conn->state != IPC_CONNECTION_STATE_CONNECTED)
112 {
113 pr_warn("connection is not connected");
114 return 0;
115 }
116
117 mutex_try_acquire_may_reschedule(&conn->mutex);
118
119 if (conn->server_data_size == 0)
120 {
121 mutex_release(&conn->mutex);
122 pr_info2("waiting for client data");
123 wait_condition_t *wc = wc_wait_for(conn, wc_ipc_connection_wait_for_server_data, NULL);
124 reschedule_for_wait_condition(wc);
125 pr_info2("ipc server: client data available");
126 mutex_try_acquire_may_reschedule(&conn->mutex);
127 }
128
129 const size_t read_size = MIN(conn->server_data_size, buf_size);
130
131 // not enough data to read
132 if (read_size == 0)
133 return 0;
134
135 // server reads from the back of the ring buffer (where the client writes)
136 size_t read = ring_buffer_pos_pop_back((u8 *) conn->server_data_vaddr, &conn->buffer_pos, buf, read_size);
137 conn->server_data_size -= read; // for the client
138
139 mutex_release(&conn->mutex);
140 return read;
141}

252void ipc_init(void)
253{
254 pr_info("Initializing IPC subsystem...");
255 ipc_servers = kzalloc(sizeof(hashmap_t));
256 hashmap_init(ipc_servers, IPC_CHANNEL_HASHMAP_SIZE, ipc_server_hash, ipc_server_compare);
257}

259io_t *ipc_create(const char *name, size_t max_pending_connections)
260{
261 ipc_server_t *server = hashmap_get(ipc_servers, name);
262
263 if (unlikely(server))
264 {
265 pr_warn("IPC channel '%s' already exists", name);
266 return NULL;
267 }
268
269 pr_info2("creating new channel %s", name);
270 server = kzalloc(sizeof(ipc_server_t));
271 server->magic = IPC_SERVER_MAGIC;
272 server->name = strdup(name);
273 server->max_pending = max_pending_connections;
274 server->pending = kcalloc(max_pending_connections, sizeof(ipc_connection_t));
275 io_init(&server->io, IO_TYPE_NONE, &ipc_server_ops);
276 hashmap_put(ipc_servers, server->name, server);
277
278 return &server->io;
279}

281io_t *ipc_accept(io_t *server)
282{
283 ipc_server_t *ipc_server = container_of(server, ipc_server_t, io);
284 if (ipc_server->magic != IPC_SERVER_MAGIC)
285 {
286 pr_warn("given io is not an IPC server");
287 return NULL;
288 }
289
290 ipc_connection_t *conn = NULL;
291
292 // find a pending connection
293 for (size_t i = 0; i < ipc_server->max_pending; i++)
294 {
295 if (ipc_server->pending[i].state == IPC_CONNECTION_STATE_PENDING)
296 {
297 conn = &ipc_server->pending[i];
298 break;
299 }
300 }
301
302 if (unlikely(!conn))
303 {
304 // no pending connections, wait for one
305 pr_info2("waiting for a pending connection");
306 wait_condition_t *cond = wc_wait_for(ipc_server, wc_ipc_has_pending, NULL);
307 reschedule_for_wait_condition(cond);
308 pr_info2("resuming after pending connection");
309 conn = ipc_server_get_pending(ipc_server);
310 }
311
312 // accept the connection
313 conn->state = IPC_CONNECTION_STATE_CONNECTED;
314
315 // map the shared memory
316 vmblock_t shm_block = shm_map_shared_block(conn->shm_block, current_process);
317 io_init(&conn->server_io, IO_READABLE | IO_WRITABLE, &ipc_connection_server_ops);
318 ring_buffer_pos_init(&conn->buffer_pos, shm_block.npages * MOS_PAGE_SIZE);
319 conn->server_data_vaddr = shm_block.vaddr;
320 return &conn->server_io;
321}

323io_t *ipc_connect(process_t *owner, const char *name, ipc_connect_flags flags, size_t buffer_size)
324{
325 ipc_server_t *server = hashmap_get(ipc_servers, name);
326
327 if (unlikely(!server))
328 {
329 if (flags & IPC_CONNECT_NONBLOCK)
330 {
331 pr_warn("IPC channel '%s' does not exist", name);
332 return NULL;
333 }
334
335 pr_info2("waiting for channel %s to be created", name);
336
337 wait_condition_t *cond = wc_wait_for((void *) strdup(name), wc_ipc_name_is_ready, wc_ipc_name_cleanup);
338 reschedule_for_wait_condition(cond);
339 pr_info2("resuming after channel %s was created", name);
340
341 server = hashmap_get(ipc_servers, name);
342 }
343
344 pr_info2("connecting to channel %s", name);
345 // find a pending connection slot
346 ipc_connection_t *conn = NULL;
347 for (size_t i = 0; i < server->max_pending; i++)
348 {
349 if (server->pending[i].state == IPC_CONNECTION_STATE_INVALID)
350 {
351 conn = &server->pending[i];
352 break;
353 }
354 }
355
356 if (unlikely(!conn))
357 {
358 pr_warn("no pending connection slots available");
359 return NULL;
360 }
361
362 shm_block_t shm_block = shm_allocate(owner, buffer_size / MOS_PAGE_SIZE, MMAP_PRIVATE, VM_RW); // not user accessible, not child-inheritable
363
364 conn->state = IPC_CONNECTION_STATE_PENDING;
365 conn->shm_block = shm_block;
366 conn->server = server;
367 conn->client_data_vaddr = shm_block.block.vaddr;
368
369 // wait for the server to accept the connection
370 wait_condition_t *cond = wc_wait_for(conn, wc_ipc_connection_is_connected_or_closed, NULL);
371 reschedule_for_wait_condition(cond);
372 if (conn->state == IPC_CONNECTION_STATE_CLOSED)
373 {
374 pr_warn("connection was closed before it was accepted");
375 return NULL;
376 }
377 pr_info2("resuming after connection was accepted");
378 io_init(&conn->client_io, IO_READABLE | IO_WRITABLE, &ipc_connection_client_ops);
379 return &conn->client_io;
380}
