16#define IPC_CHANNEL_HASHMAP_SIZE 64
17
18static hashmap_t *ipc_channels;
19
20static hash_t ipc_channel_hash(const void *key)
21{
22 return hashmap_hash_string(key);
23}
24
25static int ipc_channel_compare(const void *a, const void *b)
26{
27 ipc_server_t *ca = (ipc_server_t *) a;
28 ipc_server_t *cb = (ipc_server_t *) b;
29 return strcmp(ca->name, cb->name) == 0;
30}

40static size_t ipc_server_write(io_t *io, const void *buf, size_t count)
41{
42 MOS_UNUSED(io);
43 MOS_UNUSED(buf);
44 MOS_UNUSED(count);
45 return 0;
46}

48static void ipc_server_close(io_t *io)
49{
50 pr_info("closing ipc channel (io refcount: %zu)", io->refcount);
51 ipc_io_t *ipc_io = container_of(io, ipc_io_t, io);
52 ipc_server_t *channel = ipc_io->server;
53 hashmap_remove(ipc_channels, channel->name);
54 kfree(channel);
55}
56
57static io_op_t ipc_ops = {
58 .read = ipc_server_read,
59 .write = ipc_server_write,
60 .close = ipc_server_close,
61};
62
63void ipc_init(void)
64{
65 pr_info("Initializing IPC subsystem...");
66 ipc_channels = kmalloc(sizeof(hashmap_t));
67 hashmap_init(ipc_channels, IPC_CHANNEL_HASHMAP_SIZE, ipc_channel_hash, ipc_channel_compare);
68}

70io_t *ipc_open(process_t *owner, vmblock_t vmblock, const char *name, ipc_open_flags flags)
71{
72 bool create_only = flags & IPC_OPEN_CREATE_ONLY;
73 bool existing_only = flags & IPC_OPEN_EXISTING_ONLY;
74
75 if (unlikely(create_only && existing_only))
76 {
77 pr_warn("IPC channel open flags are invalid");
78 return NULL;
79 }
80
81 ipc_server_t *channel = hashmap_get(ipc_channels, name);
82
83 if (unlikely(!channel && existing_only))
84 {
85 pr_warn("IPC channel %s does not exist", name);
86 return NULL;
87 }
88
89 if (unlikely(channel && create_only))
90 {
91 pr_warn("IPC channel %s already exists", name);
92 return NULL;
93 }
94
95 if (!channel)
96 {
97 return ipc_create_server(owner, vmblock, name);
98 }
99 else
100 {
101 return ipc_create_client(vmblock, name);
102 }
103}
104
105io_t *ipc_create_server(process_t *owner, vmblock_t vmblock, const char *name)
106{
107 pr_info2("creating new channel %s", name);
108 ipc_server_t *channel = kzalloc(sizeof(ipc_server_t));
109 channel->name = duplicate_string(name, strlen(name));
110 channel->address_space = owner->pagetable;
111 channel->vmblock = vmblock;
112 hashmap_put(ipc_channels, channel->name, channel);
113
114 ipc_io_t *ipcio = kzalloc(sizeof(ipc_io_t));
115 ipcio->server = channel;
116 io_init(&ipcio->io, IO_READABLE | IO_WRITABLE, 0, &ipc_ops);
117 return &ipcio->io;
118}
