26static hash_t ipc_server_hash(const void *key)
27{
28 return hashmap_hash_string(key);
29}

31static int ipc_server_compare(const void *a, const void *b)
32{
33 return strcmp((const char *) a, (const char *) b) == 0;
34}

36static void ipc_server_close(io_t *io)
37{
38 pr_info("closing ipc server (io refcount: %zu)", io->refcount);
39 ipc_server_t *server = container_of(io, ipc_server_t, io);
40 if (server->magic != IPC_SERVER_MAGIC)
41 {
42 mos_warn("invalid magic");
43 return;
44 }
45
46 for (size_t i = 0; i < server->max_pending; i++)
47 {
48 ipc_connection_t *conn = &server->pending[i];
49 if (conn->state == IPC_CONNECTION_STATE_PENDING)
50 conn->state = IPC_CONNECTION_STATE_CLOSED;
51 }
52
53 hashmap_remove(ipc_servers, server->name);
54 kfree(server->name);
55 kfree(server->pending);
56 kfree(server);
57}

59static io_op_t ipc_server_ops = {
60 .close = ipc_server_close,
61};

63static ipc_connection_t *ipc_server_get_pending(ipc_server_t *server)
64{
65 for (size_t i = 0; i < server->max_pending; i++)
66 {
67 if (server->pending[i].state == IPC_CONNECTION_STATE_PENDING)
68 return &server->pending[i];
69 }
70 return NULL;
71}

83static bool wc_ipc_has_pending(wait_condition_t *cond)
84{
85 ipc_server_t *server = cond->arg;

89static bool wc_ipc_connection_is_connected_or_closed(wait_condition_t *cond)
90{
91 ipc_connection_t *conn = cond->arg;

107static size_t ipc_connection_server_read(io_t *io, void *buf, size_t buf_size)
108{
109 ipc_connection_t *conn = container_of(io, ipc_connection_t, server_io);
110 if (conn->state != IPC_CONNECTION_STATE_CONNECTED)
111 {
112 pr_warn("connection is not connected");
113 return 0;
114 }
115
116 if (conn->server_data_size == 0)
117 {
118 pr_info2("waiting for client data");
119 wait_condition_t *wc = wc_wait_for(conn, wc_ipc_connection_wait_for_server_data, NULL);
120 reschedule_for_wait_condition(wc);
121 pr_info2("resumed");
122 }
123
124 const size_t read_size = MIN(conn->server_data_size, buf_size);
125
126 // not enough data to read
127 if (read_size == 0)
128 return 0;
129
130 // server reads from the back of the ring buffer (where the client writes)
131 size_t read = ring_buffer_pos_pop_back((u8 *) conn->server_data_vaddr, &conn->buffer_pos, buf, read_size);
132 conn->server_data_size -= read; // for the client
133 return read;
134}

136static size_t ipc_connection_server_write(io_t *io, const void *buf, size_t size)
137{
138 ipc_connection_t *conn = container_of(io, ipc_connection_t, server_io);
139 if (conn->state != IPC_CONNECTION_STATE_CONNECTED)
140 {
141 pr_warn("connection is not connected");
142 return 0;
143 }
144
145 // server writes to the front of the ring buffer (where the client reads)
146 size_t written = ring_buffer_pos_push_front((u8 *) conn->server_data_vaddr, &conn->buffer_pos, buf, size);
147 conn->client_data_size += written; // for the client
148 return written;
149}

231void ipc_init(void)
232{
233 pr_info("Initializing IPC subsystem...");
234 ipc_servers = kzalloc(sizeof(hashmap_t));
235 hashmap_init(ipc_servers, IPC_CHANNEL_HASHMAP_SIZE, ipc_server_hash, ipc_server_compare);
236}

238io_t *ipc_create(const char *name, size_t max_pending_connections)
239{
240 ipc_server_t *server = hashmap_get(ipc_servers, name);
241
242 if (unlikely(server))
243 {
244 pr_warn("IPC channel '%s' already exists", name);
245 return NULL;
246 }
247
248 pr_info2("creating new channel %s", name);
249 server = kzalloc(sizeof(ipc_server_t));
250 server->magic = IPC_SERVER_MAGIC;
251 server->name = duplicate_string(name, strlen(name));
252 server->max_pending = max_pending_connections;
253 server->pending = kcalloc(max_pending_connections, sizeof(ipc_connection_t));
254 io_init(&server->io, IO_TYPE_NONE, &ipc_server_ops);
255 hashmap_put(ipc_servers, server->name, server);
256
257 return &server->io;
258}

260io_t *ipc_accept(io_t *server)
261{
262 ipc_server_t *ipc_server = container_of(server, ipc_server_t, io);
263 if (ipc_server->magic != IPC_SERVER_MAGIC)
264 {
265 pr_warn("given io is not an IPC server");
266 return NULL;
267 }
268
269 ipc_connection_t *conn = NULL;
270
271 // find a pending connection
272 for (size_t i = 0; i < ipc_server->max_pending; i++)
273 {
274 if (ipc_server->pending[i].state == IPC_CONNECTION_STATE_PENDING)
275 {
276 conn = &ipc_server->pending[i];
277 break;
278 }
279 }
280
281 if (unlikely(!conn))
282 {
283 // no pending connections, wait for one
284 pr_info2("waiting for a pending connection");
285 wait_condition_t *cond = wc_wait_for(ipc_server, wc_ipc_has_pending, NULL);
286 reschedule_for_wait_condition(cond);
287 pr_info2("resuming after pending connection");
288 conn = ipc_server_get_pending(ipc_server);
289 }
290
291 // accept the connection
292 conn->state = IPC_CONNECTION_STATE_CONNECTED;
293
294 // map the shared memory
295 vmblock_t shm_block = shm_map_shared_block(conn->shm_block, current_process);
296 io_init(&conn->server_io, IO_READABLE | IO_WRITABLE, &ipc_connection_server_ops);
297 ring_buffer_pos_init(&conn->buffer_pos, shm_block.npages * MOS_PAGE_SIZE);
298 conn->server_data_vaddr = shm_block.vaddr;
299 return &conn->server_io;
300}

302io_t *ipc_connect(process_t *owner, const char *name, ipc_connect_flags flags, size_t buffer_size)
303{
304 ipc_server_t *server = hashmap_get(ipc_servers, name);
305
306 if (unlikely(!server))
307 {
308 if (flags & IPC_CONNECT_NONBLOCK)
309 {
310 pr_warn("IPC channel '%s' does not exist", name);
311 return NULL;
312 }
313
314 pr_info2("waiting for channel %s to be created", name);
315
316 const char *name_copy = duplicate_string(name, strlen(name));
317 wait_condition_t *cond = wc_wait_for((void *) name_copy, wc_ipc_name_is_ready, wc_ipc_name_cleanup);
318 reschedule_for_wait_condition(cond);
319 pr_info2("resuming after channel %s was created", name);
320
321 server = hashmap_get(ipc_servers, name);
322 }
323
324 pr_info2("connecting to channel %s", name);
325 // find a pending connection slot
326 ipc_connection_t *conn = NULL;
327 for (size_t i = 0; i < server->max_pending; i++)
328 {
329 if (server->pending[i].state == IPC_CONNECTION_STATE_INVALID)
330 {
331 conn = &server->pending[i];
332 break;
333 }
334 }
335
336 if (unlikely(!conn))
337 {
338 pr_warn("no pending connection slots available");
339 return NULL;
340 }
341
342 shm_block_t shm_block = shm_allocate(owner, buffer_size / MOS_PAGE_SIZE, MMAP_PRIVATE, VM_RW); // not user accessible, not child-inheritable
343
344 conn->state = IPC_CONNECTION_STATE_PENDING;
345 conn->shm_block = shm_block;
346 conn->server = server;
347 conn->client_data_vaddr = shm_block.block.vaddr;
348
349 // wait for the server to accept the connection
350 wait_condition_t *cond = wc_wait_for(conn, wc_ipc_connection_is_connected_or_closed, NULL);
351 reschedule_for_wait_condition(cond);
352 if (conn->state == IPC_CONNECTION_STATE_CLOSED)
353 {
354 pr_warn("connection was closed before it was accepted");
355 return NULL;
356 }
357 pr_info2("resuming after connection was accepted");
358 io_init(&conn->client_io, IO_READABLE | IO_WRITABLE, &ipc_connection_client_ops);
359 return &conn->client_io;
360}
