11cmdline_t *cmdline_create(const char *cmdline)
12{
13 cmdline_t *cmd = kzalloc(sizeof(cmdline_t));
14
15 while (*cmdline)
16 {
17 if (*cmdline == ' ')
18 cmdline++;
19
20 cmd->options = krealloc(cmd->options, sizeof(cmdline_option_t *) * (cmd->options_count + 1));
21 cmdline_option_t *opt = cmd->options[cmd->options_count++] = kzalloc(sizeof(cmdline_option_t));
22
23 // an arg name ends with an '=' (has parameters) or a space
24 {
25 const char *const start = cmdline;
26 while (*cmdline && !((*cmdline) == ' ' || (*cmdline) == '='))
27 cmdline++;
28 opt->name = duplicate_string(start, cmdline - start);
29 }
30
31 // the option has more arguments
32 if (*cmdline && *cmdline++ != '=')
33 continue;
34
35 while (*cmdline && *cmdline != ' ')
36 {
37 opt->argv = krealloc(opt->argv, sizeof(const char *) * (opt->argc + 1));
38
39 // an option ends with a space or a comma
40 const char *const start = cmdline;
41 while (*cmdline && *cmdline != ' ' && *cmdline != ',')
42 cmdline++;
43
44 opt->argv[opt->argc++] = duplicate_string(start, cmdline - start);
45
46 MOS_ASSERT(*cmdline == ' ' || *cmdline == ',' || *cmdline == '\0');
47
48 // we have another argument
49 if (*cmdline == ',')
50 cmdline++;
51
52 // we have another option
53 if (*cmdline == ' ')
54 {
55 cmdline++;
56 break;
57 }
58 }
59 }
60
61 return cmd;
62}

64static void cmdline_free_option(cmdline_option_t *opt)
65{
66 kfree(opt->name);
67 for (u32 i = 0; i < opt->argc; i++)
68 kfree(opt->argv[i]);
69 if (opt->argc > 0)
70 kfree(opt->argv);
71 kfree(opt);
72}

74bool cmdline_remove_option(cmdline_t *cmdline, const char *arg)
75{
76 for (u32 i = 0; i < cmdline->options_count; i++)
77 {
78 if (strcmp(cmdline->options[i]->name, arg) == 0)
79 {
80 cmdline_free_option(cmdline->options[i]);
81 cmdline->options_count--;
82
83 for (u32 j = i; j < cmdline->options_count; j++)
84 cmdline->options[j] = cmdline->options[j + 1]; // shift the array to the left by one
85
86 cmdline->options = krealloc(cmdline->options, sizeof(cmdline_option_t *) * cmdline->options_count);
87 return true;
88 }
89 }
90 return false;
91}
92
93void cmdline_destroy(cmdline_t *cmdline)
94{
95 for (u32 i = 0; i < cmdline->options_count; i++)
96 cmdline_free_option(cmdline->options[i]);
97
98 kfree(cmdline->options);
99 kfree(cmdline);
100}

102static bool cmdline_is_truthy(const char *arg)
103{
104 return strcmp(arg, "true") == 0 || strcmp(arg, "1") == 0 || strcmp(arg, "yes") == 0 || strcmp(arg, "on") == 0;
105}
106
107static bool cmdline_is_falsy(const char *arg)
108{
109 return strcmp(arg, "false") == 0 || strcmp(arg, "0") == 0 || strcmp(arg, "no") == 0 || strcmp(arg, "off") == 0;
110}

112bool cmdline_arg_get_bool(const char *s, int argc, const char **argv, bool default_value)
113{
114 s = s ? s : "";
115 if (argc == 0)
116 return default_value;
117 else if (argc > 1)
118 pr_warn("%s: too many arguments (%d), only the first one will be used", s, argc);
119
120 if (cmdline_is_truthy(argv[0]))
121 return true;
122 else if (cmdline_is_falsy(argv[0]))
123 return false;
124 else
125 pr_warn("%s: invalid argument '%s', assuming %s", s, argv[0], default_value ? "true" : "false");
126
127 return default_value;
128}
