11cmdline_t *mos_cmdline_create(const char *cmdline)
12{
13 cmdline_t *cmd = kmalloc(sizeof(cmdline_t));
14 cmd->args_count = 0;
15 cmd->arguments = NULL;
16
17 while (*cmdline)
18 {
19 if (*cmdline == ' ')
20 cmdline++;
21 cmdline_arg_t *arg = kmalloc(sizeof(cmdline_arg_t));
22 arg->params = NULL;
23 arg->params_count = 0;
24 cmd->arguments = krealloc(cmd->arguments, sizeof(cmdline_arg_t *) * (cmd->args_count + 1));
25 cmd->arguments[cmd->args_count] = arg;
26 cmd->args_count++;
27
28 // an arg name ends with an '=' (has parameters) or a space
29 {
30 const char *start = cmdline;
31 while (*cmdline && !((*cmdline) == ' ' || (*cmdline) == '='))
32 cmdline++;
33 arg->arg_name = duplicate_string(start, cmdline - start);
34 }
35
36 // the option has parameters
37 if (*cmdline && *cmdline++ != '=')
38 continue;
39
40 while (*cmdline && *cmdline != ' ')
41 {
42 cmdline_param_t *param = kmalloc(sizeof(cmdline_param_t));
43 arg->params = krealloc(arg->params, sizeof(cmdline_param_t *) * (arg->params_count + 1));
44 arg->params[arg->params_count] = param;
45 arg->params_count++;
46
47 if (strncmp(cmdline, "true", 4) == 0)
48 {
49 param->param_type = CMDLINE_PARAM_TYPE_BOOL;
50 param->val.boolean = true;
51 cmdline += 4;
52 }
53 else if (strncmp(cmdline, "false", 5) == 0)
54 {
55 param->param_type = CMDLINE_PARAM_TYPE_BOOL;
56 param->val.boolean = false;
57 cmdline += 5;
58 }
59 else
60 {
61 param->param_type = CMDLINE_PARAM_TYPE_STRING;
62 // an option ends with a space or a comma
63 const char *param_start = cmdline;
64 while (*cmdline && *cmdline != ' ' && *cmdline != ',')
65 cmdline++;
66
67 param->val.string = duplicate_string(param_start, cmdline - param_start);
68 }
69
70 MOS_ASSERT(*cmdline == ' ' || *cmdline == ',' || *cmdline == '\0');
71
72 // we have another option
73 if (*cmdline == ',')
74 cmdline++;
75
76 // we have no other options, continue with another parameter
77 if (*cmdline == ' ')
78 {
79 cmdline++;
80 break;
81 }
82 }
83 }
84
85 return cmd;
86}

88void mos_cmdline_destroy(cmdline_t *cmdline)
89{
90 for (u32 i = 0; i < cmdline->args_count; i++)
91 {
92 cmdline_arg_t *arg = cmdline->arguments[i];
93 kfree(arg->arg_name);
94 for (u32 j = 0; j < arg->params_count; j++)
95 {
96 cmdline_param_t *param = arg->params[j];
97 if (param->param_type == CMDLINE_PARAM_TYPE_STRING)
98 kfree(param->val.string);
99 kfree(param);
100 }
101 if (arg->params_count)
102 kfree(arg->params);
103 kfree(arg);
104 }
105 kfree(cmdline->arguments);
106 kfree(cmdline);
107}

109cmdline_arg_t *mos_cmdline_get_arg(const char *option_name)
110{
111 MOS_ASSERT(mos_cmdline);
112 for (u32 i = 0; i < mos_cmdline->args_count; i++)
113 {
114 if (strcmp(mos_cmdline->arguments[i]->arg_name, option_name) == 0)
115 return mos_cmdline->arguments[i];
116 }
117 return NULL;
118}
