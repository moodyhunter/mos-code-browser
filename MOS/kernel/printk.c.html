12static void deduce_level_color(int loglevel, standard_color_t *fg, standard_color_t *bg)
13{
14 *bg = Black;
15 switch (loglevel)
16 {
17 case MOS_LOG_INFO2: *fg = DarkGray; break;
18 case MOS_LOG_INFO: *fg = Gray; break;
19 case MOS_LOG_EMPH: *fg = Cyan; break;
20 case MOS_LOG_WARN: *fg = Brown; break;
21 case MOS_LOG_EMERG: *fg = Red; break;
22 case MOS_LOG_FATAL: *fg = White, *bg = Red; break;
23 default: break; // do not change the color
24 }
25}
26
27void lprintk(int loglevel, const char *fmt, ...)
28{
29 standard_color_t fg = White, bg = Black;
30 deduce_level_color(loglevel, &fg, &bg);
31
32 char message[PRINTK_BUFFER_SIZE] = { 0 };
33 va_list args;
34 va_start(args, fmt);
35 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
36 va_end(args);
37 const size_t len = strlen(message);
38
39 standard_color_t prev_fg, prev_bg;
40 list_foreach(console_t, console, consoles)
41 {
42 if (console->caps & CONSOLE_CAP_COLOR)
43 {
44 console->get_color(console, &prev_fg, &prev_bg);
45 console->set_color(console, fg, bg);
46 }
47
48 console->write(console, message, len);
49
50 if (console->caps & CONSOLE_CAP_COLOR)
51 {
52 console->set_color(console, prev_fg, prev_bg);
53 }
54 }
55}
