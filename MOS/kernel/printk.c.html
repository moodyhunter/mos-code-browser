14static console_t *printk_console;
15
16void printk_setup_console()
17{
18 cmdline_arg_t *kcon_arg = mos_cmdline_get_arg("kmsg_console");
19
20 if (!kcon_arg || kcon_arg->params_count <= 0)
21 return;
22
23 if (kcon_arg->params_count > 1)
24 pr_warn("too many parameters for kmsg_console, using first one");
25
26 cmdline_param_t *kcon_param = kcon_arg->params[0];
27 if (unlikely(kcon_param->param_type != CMDLINE_PARAM_TYPE_STRING))
28 {
29 pr_warn("kmsg_console parameter is not a string, ignoring");
30 return;
31 }
32
33 const char *kcon_name = kcon_param->val.string;
34
35 console_t *console = console_get(kcon_name);
36 if (console)
37 {
38 pr_emph("Selected console '%s' for future printk", kcon_name);
39 printk_console = console;
40 return;
41 }
42
43 console = console_get_by_prefix(kcon_name);
44 if (console)
45 {
46 pr_emph("Selected console '%s' for future printk (prefix-based)", console->name);
47 printk_console = console;
48 return;
49 }
50
51 mos_warn("No console found for printk based on given name or prefix '%s'", kcon_name);
52 printk_console = NULL;
53}

55static inline void deduce_level_color(int loglevel, standard_color_t *fg, standard_color_t *bg)
56{
57 *bg = Black;
58 switch (loglevel)
59 {
60 case MOS_LOG_INFO2: *fg = DarkGray; break;
61 case MOS_LOG_INFO: *fg = Gray; break;
62 case MOS_LOG_EMPH: *fg = Cyan; break;
63 case MOS_LOG_WARN: *fg = Brown; break;
64 case MOS_LOG_EMERG: *fg = Red; break;
65 case MOS_LOG_FATAL: *fg = White, *bg = Red; break;
66 default: break; // do not change the color
67 }
68}

70static void print_to_console(console_t *con, int loglevel, const char *message, size_t len)
71{
72 if (!con)
73 return;
74
75 spinlock_acquire(&con->lock);
76
77 standard_color_t prev_fg, prev_bg;
78 standard_color_t fg = White, bg = Black;
79 deduce_level_color(loglevel, &fg, &bg);
80
81 if (con->caps & CONSOLE_CAP_COLOR)
82 {
83 con->get_color(con, &prev_fg, &prev_bg);
84 con->set_color(con, fg, bg);
85 }
86
87 console_write(con, message, len);
88
89 if (con->caps & CONSOLE_CAP_COLOR)
90 con->set_color(con, prev_fg, prev_bg);
91
92 spinlock_release(&con->lock);
93}

95static void lvprintk(int loglevel, const char *fmt, va_list args)
96{
97 char message[PRINTK_BUFFER_SIZE] = { 0 };
98 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
99 const size_t len = strlen(message);
100
101 if (likely(printk_console))
102 {
103 print_to_console(printk_console, loglevel, message, len);
104 }
105 else
106 {
107 list_foreach(console_t, con, consoles)
108 {
109 print_to_console(con, loglevel, message, len);
110 }
111 }
112}
113
114void lprintk(int loglevel, const char *format, ...)
115{
116 va_list args;
117 va_start(args, format);
118 lvprintk(loglevel, format, args);
119 va_end(args);
120}
121
122void printk(const char *format, ...)
123{
124 va_list args;
125 va_start(args, format);
126 lvprintk(MOS_LOG_INFO, format, args);
127 va_end(args);
128}
