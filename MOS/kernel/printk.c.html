12static void deduce_level_color(int loglevel, standard_color_t *fg, standard_color_t *bg)
13{
14 *bg = Black;
15 switch (loglevel)
16 {
17 case MOS_LOG_INFO2: *fg = DarkGray; break;
18 case MOS_LOG_INFO: *fg = Gray; break;
19 case MOS_LOG_EMPH: *fg = Cyan; break;
20 case MOS_LOG_WARN: *fg = Brown; break;
21 case MOS_LOG_EMERG: *fg = Red; break;
22 case MOS_LOG_FATAL: *fg = White, *bg = Red; break;
23 default: break; // do not change the color
24 }
25}

27void printk(const char *fmt, ...)
28{
29 char message[PRINTK_BUFFER_SIZE] = { 0 };
30 va_list args;
31 va_start(args, fmt);
32 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
33 va_end(args);
34 const size_t len = strlen(message);
35
36 list_foreach(console_t, console, consoles)
37 {
38 console->write(console, message, len);
39 }
40}

42void lprintk(int loglevel, const char *fmt, ...)
43{
44 standard_color_t fg = White, bg = Black;
45 deduce_level_color(loglevel, &fg, &bg);
46
47 char message[PRINTK_BUFFER_SIZE] = { 0 };
48 va_list args;
49 va_start(args, fmt);
50 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
51 va_end(args);
52 const size_t len = strlen(message);
53
54 standard_color_t prev_fg, prev_bg;
55 list_foreach(console_t, console, consoles)
56 {
57 if (console->caps & CONSOLE_CAP_COLOR)
58 {
59 console->get_color(console, &prev_fg, &prev_bg);
60 console->set_color(console, fg, bg);
61 }
62
63 console->write(console, message, len);
64
65 if (console->caps & CONSOLE_CAP_COLOR)
66 {
67 console->set_color(console, prev_fg, prev_bg);
68 }
69 }
70}
