15vmblock_t mm_make_process_map_cow(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
16{
17 // Note that the block returned by this function contains it's ACTRUAL flags, which may be different from the flags of the original block.
18 // (consider the case where we CoW-map a page that is already CoW-mapped)
19 vmblock_t block = mm_copy_maps(from, fvaddr, to, tvaddr, npages);
20 platform_mm_flag_pages(from, fvaddr, npages, block.flags & ~VM_WRITE);
21 platform_mm_flag_pages(to, tvaddr, npages, block.flags & ~VM_WRITE);
22 return block;
23}

25static void copy_cow_pages_inplace(uintptr_t vaddr, size_t npages)
26{
27 paging_handle_t pg_handle = current_process->pagetable;
28 const vm_flags flags = platform_mm_get_flags(pg_handle, vaddr); // this flags are the same for all pages (in the same proc_vmblock)
29 void *pagetmp = kmalloc(MOS_PAGE_SIZE);
30 for (size_t j = 0; j < npages; j++)
31 {
32 const uintptr_t current_page = vaddr + j * MOS_PAGE_SIZE;
33 memcpy(pagetmp, (void *) current_page, MOS_PAGE_SIZE);
34 mm_unmap_pages(pg_handle, current_page, 1);
35 mm_alloc_pages_at(pg_handle, current_page, 1, flags | VM_WRITE);
36 memcpy((void *) current_page, pagetmp, MOS_PAGE_SIZE);
37 }
38 kfree(pagetmp);
39}

41bool cow_handle_page_fault(uintptr_t fault_addr, bool present, bool is_write, bool is_user, bool is_exec)
42{
43 MOS_UNUSED(is_user);
44 MOS_UNUSED(present);
45
46 if (!current_thread)
47 return false;
48
49 if (!present)
50 return false;
51
52 if (!is_write)
53 return false;
54
55 if (is_write && is_exec)
56 mos_panic("Cannot write and execute at the same time");
57
58 process_t *current_proc = current_process;
59 MOS_ASSERT_X(current_proc->pagetable.pgd == current_cpu->pagetable.pgd, "Page fault in a process that is not the current process?!");
60#if MOS_DEBUG_FEATURE(cow)
61 process_dump_mmaps(current_proc);
62#endif
63
64 for (ssize_t i = 0; i < current_proc->mmaps_count; i++)
65 {
66 const vmblock_t vm = current_proc->mmaps[i].vm;
67 uintptr_t block_start = vm.vaddr;
68 uintptr_t block_end = vm.vaddr + vm.npages * MOS_PAGE_SIZE;
69
70 if (fault_addr < block_start || fault_addr >= block_end)
71 continue;
72
73 proc_vmblock_t *mmap = &current_proc->mmaps[i];
74
75 const bool is_cow = mmap->map_flags & MMAP_COW;
76 const bool is_zero_on_demand = mmap->map_flags & MMAP_ZERO_ON_DEMAND;
77 mos_debug(cow, "fault_addr=" PTR_FMT ", vmblock=" PTR_FMT "-" PTR_FMT, fault_addr, vm.vaddr, vm.vaddr + vm.npages * MOS_PAGE_SIZE);
78
79 if (is_zero_on_demand)
80 {
81 pr_emph("Zero-on-demand page fault in block %ld", i);
82 // TODO: only allocate the page where the fault happened
83 mm_unmap_pages(current_proc->pagetable, vm.vaddr, vm.npages);
84 mm_alloc_pages_at(current_proc->pagetable, vm.vaddr, vm.npages, vm.flags);
85 memzero((void *) vm.vaddr, MOS_PAGE_SIZE * vm.npages);
86 current_proc->mmaps[i].map_flags &= ~MMAP_ZERO_ON_DEMAND;
87 mos_debug(cow, "ZoD resolved");
88 return true;
89 }
90 else if (is_cow)
91 {
92 pr_emph("CoW page fault in block %ld", i);
93 // TODO: only copy the page where the fault happened
94 copy_cow_pages_inplace(vm.vaddr, vm.npages);
95 platform_mm_flag_pages(current_proc->pagetable, vm.vaddr, vm.npages, vm.flags);
96 current_proc->mmaps[i].map_flags &= ~MMAP_COW;
97 mos_debug(cow, "CoW resolved");
98 return true;
99 }
100 else
101 {
102 pr_emerg("%s page fault (%s) in non-cow mapped region", is_user ? "User" : "Kernel", is_write ? "write" : "read");
103 }
104 }
105
106 pr_emph("page fault in unmapped region: " PTR_FMT, fault_addr);
107 return false;
108}
