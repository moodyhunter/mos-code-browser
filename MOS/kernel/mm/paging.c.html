12void mos_kernel_mm_init()
13{
14 liballoc_init(MOS_PAGE_SIZE);
15#if MOS_MM_LIBALLOC_DEBUG
16 mos_install_kpanic_hook(liballoc_dump);
17#endif
18}
19
20void *kpage_alloc(size_t npages, pagealloc_flags type)
21{
22 if (unlikely(npages <= 0))
23 {
24 mos_warn("allocating negative or zero pages");
25 return NULL;
26 }
27
28 if (unlikely(mos_platform->mm_alloc_pages == NULL))
29 mos_panic("platform configuration error: alloc_page is NULL");
30
31 void *ptr = mos_platform->mm_alloc_pages(mos_platform->kernel_pg, npages, type);
32 if (unlikely(ptr == NULL))
33 mos_warn("failed to allocate %zu pages", npages);
34
35 return ptr;
36}
37
38bool kpage_free(void *vptr, size_t npages)
39{
40 if (unlikely(vptr == NULL))
41 {
42 mos_warn("freeing NULL pointer");
43 return false;
44 }
45 if (unlikely(npages <= 0))
46 {
47 mos_warn("freeing negative or zero pages");
48 return false;
49 }
50 if (unlikely(mos_platform->mm_free_pages == NULL))
51 mos_panic("platform configuration error: free_page is NULL");
52 return mos_platform->mm_free_pages(mos_platform->kernel_pg, (uintptr_t) vptr, npages);
53}
