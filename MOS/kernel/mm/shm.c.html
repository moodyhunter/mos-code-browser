15shm_block_t shm_allocate(process_t *owner, size_t npages, mmap_flags flags, vm_flags vmflags)
16{
17 pr_info2("allocating %zu SHM pages in address space " PTR_FMT, npages, owner->pagetable.ptr);
18 vmblock_t block = platform_mm_alloc_pages(owner->pagetable, npages, PGALLOC_HINT_MMAP, vmflags);
19 process_attach_mmap(owner, block, VMTYPE_SHM, flags);
20 return (shm_block_t){ .block = block, .address_space = owner->pagetable };
21}
22
23vmblock_t shm_map_shared_block(shm_block_t source, process_t *owner)
24{
25 pr_info2("sharing %zu pages from address space " PTR_FMT " to address space " PTR_FMT, source.block.npages, source.address_space.ptr, owner->pagetable.ptr);
26 vmblock_t block = platform_mm_get_free_pages(owner->pagetable, source.block.npages, PGALLOC_HINT_MMAP);
27 block = platform_mm_copy_maps(source.address_space, source.block.vaddr, owner->pagetable, block.vaddr, source.block.npages);
28 process_attach_mmap(owner, block, VMTYPE_SHM, MMAP_PRIVATE);
29 return block;
30}
