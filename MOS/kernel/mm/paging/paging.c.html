14static bitmap_line_t kernel_page_map[MOS_PAGEMAP_KERNEL_LINES] = { 0 };
15
16void pagemap_use(page_map_t *map, uintptr_t vaddr, size_t n_pages)
17{
18 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
19 const uintptr_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
20 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
21 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
22
23 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : map->ummap;
24
25 if (unlikely(pagemap_index + n_pages > pagemap_size_lines * BITMAP_LINE_BITS))
26 {
27 mos_panic("pagemap_use: out of bounds");
28 }
29
30 for (size_t i = 0; i < n_pages; i++)
31 {
32 bool is_set = bitmap_get(pagemap, pagemap_size_lines, pagemap_index + i);
33 MOS_ASSERT_X(!is_set, "page " PTR_FMT " is already allocated", (pagemap_index + i) * MOS_PAGE_SIZE + pagemap_base);
34 bitmap_set(pagemap, pagemap_size_lines, pagemap_index + i);
35 is_set = bitmap_get(pagemap, pagemap_size_lines, pagemap_index + i);
36 MOS_ASSERT_X(is_set, "page " PTR_FMT " is not set", (pagemap_index + i) * MOS_PAGE_SIZE + pagemap_base);
37 }
38}

40void pagemap_free(page_map_t *map, uintptr_t vaddr, size_t n_pages)
41{
42 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
43 const uintptr_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
44 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
45 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
46
47 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : map->ummap;
48
49 if (unlikely(pagemap_index + n_pages > pagemap_size_lines * BITMAP_LINE_BITS))
50 {
51 mos_panic("pagemap_free: out of bounds");
52 }
53
54 for (size_t i = 0; i < n_pages; i++)
55 {
56 bool is_set = bitmap_get(pagemap, pagemap_size_lines, pagemap_index + i);
57 MOS_ASSERT_X(is_set, "page " PTR_FMT " is already free", (pagemap_index + i) * MOS_PAGE_SIZE + pagemap_base);
58 bitmap_clear(pagemap, pagemap_size_lines, pagemap_index + i);
59 }
60}

62bool mm_get_is_mapped(paging_handle_t table, uintptr_t vaddr)
63{
64 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
65 const size_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
66 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
67 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
68
69 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : table.um_page_map->ummap;
70 return bitmap_get(pagemap, pagemap_size_lines, pagemap_index);
71}

73vmblock_t mm_get_free_pages(paging_handle_t table, size_t n_pages, pgalloc_hints hints)
74{
75 static const uintptr_t limits[] = {
76 [PGALLOC_HINT_KHEAP] = MOS_ADDR_KERNEL_HEAP,
77 [PGALLOC_HINT_UHEAP] = MOS_ADDR_USER_HEAP,
78 [PGALLOC_HINT_STACK] = MOS_ADDR_USER_STACK,
79 [PGALLOC_HINT_MMAP] = MOS_ADDR_USER_MMAP,
80 };
81
82 uintptr_t vaddr_begin = limits[hints];
83
84 const bool is_kernel = hints == PGALLOC_HINT_KHEAP;
85 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : table.um_page_map->ummap;
86 const uintptr_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
87 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
88 const size_t pagemap_begin_index = (vaddr_begin - pagemap_base) / MOS_PAGE_SIZE;
89
90 size_t page_i = bitmap_find_first_free_n(pagemap, pagemap_size_lines, pagemap_begin_index, n_pages);
91 MOS_ASSERT_X(!bitmap_get(pagemap, pagemap_size_lines, page_i), "page %zu is already allocated", page_i);
92
93 const uintptr_t vaddr = pagemap_base + page_i * MOS_PAGE_SIZE;
94 const bool begins_in_kernel = vaddr >= MOS_KERNEL_START_VADDR;
95 const bool ends_in_kernel = (vaddr + n_pages * MOS_PAGE_SIZE) >= MOS_KERNEL_START_VADDR;
96 if (is_kernel)
97 {
98 MOS_ASSERT_X(begins_in_kernel && ends_in_kernel, "incorrect allocation, expected in kernel space");
99 }
100 else
101 {
102 MOS_ASSERT_X(!begins_in_kernel && !ends_in_kernel, "incorrect allocation, expected in user space");
103 }
104
105 return (vmblock_t){ .vaddr = vaddr, .npages = n_pages };
106}

108vmblock_t mm_alloc_pages(paging_handle_t table, size_t npages, pgalloc_hints hints, vm_flags flags)
109{
110 vmblock_t block = mm_get_free_pages(table, npages, hints);
111 return mm_alloc_pages_at(table, block.vaddr, block.npages, flags);
112}
113
114vmblock_t mm_alloc_pages_at(paging_handle_t table, uintptr_t vaddr, size_t n_pages, vm_flags flags)
115{
116 uintptr_t paddr = pmem_freelist_allocate_free(n_pages);
117 MOS_ASSERT_X(paddr != 0, "could not find free physical memory");
118
119 vmblock_t block = { .vaddr = vaddr, .npages = n_pages, .paddr = paddr, .flags = flags };
120 mm_map_allocated_pages(table, block);
121 return block;
122}
123
124void mm_map_pages(paging_handle_t table, vmblock_t block)
125{
126 pmem_freelist_remove_region(block.paddr, block.npages * MOS_PAGE_SIZE);
127 mm_map_allocated_pages(table, block);
128}
129
130void mm_map_allocated_pages(paging_handle_t table, vmblock_t block)
131{
132 pagemap_use(table.um_page_map, block.vaddr, block.npages);
133 platform_mm_map_pages(PGD_FOR_VADDR(block.vaddr, table), block);
134}
135
136void mm_unmap_pages(paging_handle_t table, uintptr_t vaddr, size_t npages)
137{
138 pagemap_free(table.um_page_map, vaddr, npages);
139 platform_mm_unmap_pages(PGD_FOR_VADDR(vaddr, table), vaddr, npages);
140}
141
142void mm_free_pages(paging_handle_t table, vmblock_t block)
143{
144 mm_unmap_pages(table, block.vaddr, block.npages);
145 pmem_freelist_add_region(block.paddr, block.npages * MOS_PAGE_SIZE);
146}

148vmblock_t mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
149{
150 pagemap_use(to.um_page_map, tvaddr, npages);
151 vmblock_t block = platform_mm_copy_maps(PGD_FOR_VADDR(fvaddr, from), fvaddr, PGD_FOR_VADDR(tvaddr, to), tvaddr, npages);
152 return block;
153}

155paging_handle_t mm_create_user_pgd()
156{
157 paging_handle_t table = platform_mm_create_user_pgd();
158 table.um_page_map = (page_map_t *) kzalloc(sizeof(page_map_t));
159 return table;
160}
161
162void mm_destroy_user_pgd(paging_handle_t table)
163{
164 platform_mm_destroy_user_pgd(table);
165 kfree(table.um_page_map);
166}
