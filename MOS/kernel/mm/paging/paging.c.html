14static bitmap_line_t kernel_page_map[MOS_PAGEMAP_KERNEL_LINES] = { 0 };
15
16static void pagemap_mark_used(page_map_t *map, uintptr_t vaddr, size_t n_pages)
17{
18 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
19 const uintptr_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
20 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
21 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
22
23 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : map->ummap;
24
25 if (unlikely(pagemap_index + n_pages > pagemap_size_lines * BITMAP_LINE_BITS))
26 {
27 mos_panic("pagemap_use: out of bounds");
28 }
29
30 for (size_t i = 0; i < n_pages; i++)
31 {
32 bool is_set = bitmap_get(pagemap, pagemap_size_lines, pagemap_index + i);
33 MOS_ASSERT_X(!is_set, "page " PTR_FMT " is already allocated", (pagemap_index + i) * MOS_PAGE_SIZE + pagemap_base);
34 bitmap_set(pagemap, pagemap_size_lines, pagemap_index + i);
35 is_set = bitmap_get(pagemap, pagemap_size_lines, pagemap_index + i);
36 MOS_ASSERT_X(is_set, "page " PTR_FMT " is not set", (pagemap_index + i) * MOS_PAGE_SIZE + pagemap_base);
37 }
38}

40static void pagemap_mark_free(page_map_t *map, uintptr_t vaddr, size_t n_pages)
41{
42 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
43 const uintptr_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
44 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
45 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
46
47 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : map->ummap;
48
49 if (unlikely(pagemap_index + n_pages > pagemap_size_lines * BITMAP_LINE_BITS))
50 {
51 mos_panic("pagemap_free: out of bounds");
52 }
53
54 for (size_t i = 0; i < n_pages; i++)
55 {
56 bool is_set = bitmap_get(pagemap, pagemap_size_lines, pagemap_index + i);
57 MOS_ASSERT_X(is_set, "page " PTR_FMT " is already free", (pagemap_index + i) * MOS_PAGE_SIZE + pagemap_base);
58 bitmap_clear(pagemap, pagemap_size_lines, pagemap_index + i);
59 }
60}

62vmblock_t mm_get_free_pages(paging_handle_t table, size_t n_pages, pgalloc_hints hints)
63{
64 static const uintptr_t limits[] = {
65 [PGALLOC_HINT_KHEAP] = MOS_ADDR_KERNEL_HEAP,
66 [PGALLOC_HINT_UHEAP] = MOS_ADDR_USER_HEAP,
67 [PGALLOC_HINT_STACK] = MOS_ADDR_USER_STACK,
68 [PGALLOC_HINT_MMAP] = MOS_ADDR_USER_MMAP,
69 };
70
71 uintptr_t vaddr_begin = limits[hints];
72
73 const bool is_kernel = hints == PGALLOC_HINT_KHEAP;
74 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : table.um_page_map->ummap;
75 const uintptr_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
76 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
77 const size_t pagemap_begin_index = (vaddr_begin - pagemap_base) / MOS_PAGE_SIZE;
78
79 size_t page_i = bitmap_find_first_free_n(pagemap, pagemap_size_lines, pagemap_begin_index, n_pages);
80 MOS_ASSERT_X(!bitmap_get(pagemap, pagemap_size_lines, page_i), "page %zu is already allocated", page_i);
81
82 const uintptr_t vaddr = pagemap_base + page_i * MOS_PAGE_SIZE;
83 const bool begins_in_kernel = vaddr >= MOS_KERNEL_START_VADDR;
84 const bool ends_in_kernel = (vaddr + n_pages * MOS_PAGE_SIZE) >= MOS_KERNEL_START_VADDR;
85 if (is_kernel)
86 {
87 MOS_ASSERT_X(begins_in_kernel && ends_in_kernel, "incorrect allocation, expected in kernel space");
88 }
89 else
90 {
91 MOS_ASSERT_X(!begins_in_kernel && !ends_in_kernel, "incorrect allocation, expected in user space");
92 }
93
94 return (vmblock_t){ .vaddr = vaddr, .npages = n_pages };
95}

97vmblock_t mm_alloc_pages(paging_handle_t table, size_t npages, pgalloc_hints hints, vm_flags flags)
98{
99 vmblock_t block = mm_get_free_pages(table, npages, hints);
100 return mm_alloc_pages_at(table, block.vaddr, block.npages, flags);
101}
102
103vmblock_t mm_alloc_pages_at(paging_handle_t table, uintptr_t vaddr, size_t n_pages, vm_flags flags)
104{
105 uintptr_t paddr = pmalloc_alloc(n_pages);
106 MOS_ASSERT_X(paddr != 0, "could not find free physical memory");
107
108 vmblock_t block = { .vaddr = vaddr, .npages = n_pages, .paddr = paddr, .flags = flags };
109 mm_map_allocated_pages(table, block);
110 return block;
111}
112
113void mm_free_pages(paging_handle_t table, vmblock_t block)
114{
115 mm_unmap_pages(table, block.vaddr, block.npages);
116 pmalloc_release_region(block.paddr, block.npages * MOS_PAGE_SIZE);
117}
118
119void mm_map_pages(paging_handle_t table, vmblock_t block)
120{
121 pmalloc_acquire_region(block.paddr, block.npages * MOS_PAGE_SIZE);
122 mm_map_allocated_pages(table, block);
123}
124
125void mm_map_allocated_pages(paging_handle_t table, vmblock_t block)
126{
127 pagemap_mark_used(table.um_page_map, block.vaddr, block.npages);
128 platform_mm_map_pages(PGD_FOR_VADDR(block.vaddr, table), block);
129}
130
131void mm_unmap_pages(paging_handle_t table, uintptr_t vaddr, size_t npages)
132{
133 pagemap_mark_free(table.um_page_map, vaddr, npages);
134 platform_mm_unmap_pages(PGD_FOR_VADDR(vaddr, table), vaddr, npages);
135}

137vmblock_t mm_copy_maps(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t npages)
138{
139 pagemap_mark_used(to.um_page_map, tvaddr, npages);
140 vmblock_t block = platform_mm_copy_maps(PGD_FOR_VADDR(fvaddr, from), fvaddr, PGD_FOR_VADDR(tvaddr, to), tvaddr, npages);
141 return block;
142}

144bool mm_get_is_mapped(paging_handle_t table, uintptr_t vaddr)
145{
146 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
147 const size_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
148 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
149 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
150
151 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : table.um_page_map->ummap;
152 return bitmap_get(pagemap, pagemap_size_lines, pagemap_index);
153}

155vmblock_t mm_get_block_info(paging_handle_t table, uintptr_t vaddr, size_t n_pages)
156{
157 const bool is_kernel = vaddr >= MOS_KERNEL_START_VADDR;
158 const size_t pagemap_base = is_kernel ? MOS_KERNEL_START_VADDR : 0;
159 const size_t pagemap_size_lines = is_kernel ? MOS_PAGEMAP_KERNEL_LINES : MOS_PAGEMAP_USER_LINES;
160 const size_t pagemap_index = (vaddr - pagemap_base) / MOS_PAGE_SIZE;
161
162 bitmap_line_t *pagemap = is_kernel ? kernel_page_map : table.um_page_map->ummap;
163 MOS_ASSERT_X(bitmap_get(pagemap, pagemap_size_lines, pagemap_index), "page %zu is not allocated", pagemap_index);
164
165 return platform_mm_get_block_info(PGD_FOR_VADDR(vaddr, table), vaddr, n_pages);
166}

168paging_handle_t mm_create_user_pgd()
169{
170 paging_handle_t table = platform_mm_create_user_pgd();
171 table.um_page_map = (page_map_t *) kzalloc(sizeof(page_map_t));
172 return table;
173}

175void mm_destroy_user_pgd(paging_handle_t table)

177 platform_mm_destroy_user_pgd(table);
