10#define RESERVED_LOMEM 1 MB
11#define PMEM_FREELIST_SIZE_FOR(mem_size) (((mem_size) / 2 / MOS_PAGE_SIZE) * sizeof(pmlist_node_t))
12
13typedef struct _pmlist_node_t
14{
15 struct _pmlist_node_t *next;
16 uintptr_t paddr;
17 size_t n_pages;
18} pmlist_node_t;
19
20// Do not mark as const, or the compiler will put it in .rodata
21static pmlist_node_t pmlist_storage[PMEM_FREELIST_SIZE_FOR(MOS_MAX_VADDR)] __aligned(MOS_PAGE_SIZE) = { 0 };
22static pmlist_node_t *pmlist_head = (pmlist_node_t *) pmlist_storage;
23static size_t pmlist_count = 0;
24
25static size_t freelist_size()
26{
27 return ALIGN_UP_TO_PAGE(PMEM_FREELIST_SIZE_FOR(platform_info->mem.available));
28}
29
30void mos_pmm_setup(void)
31{
32 pr_info("Initializing Physical Memory Allocator");
33 const size_t list_size = freelist_size();
34 pr_info2("%zu bytes (aligned) required for physical memory freelist", list_size);
35
36 memzero(pmlist_head, list_size);
37
38 // add current physical memory region to the freelist
39 for (size_t i = 0; i < platform_info->mem_regions.count; i++)
40 {
41 memregion_t *r = &platform_info->mem_regions.regions[i];
42 if (!r->available)
43 continue;
44
45 if (r->address + r->size_bytes < RESERVED_LOMEM)
46 {
47 pr_emph("ignored low memory: " PTR_FMT "-" PTR_FMT " (%zu bytes)", r->address, r->address + r->size_bytes, r->size_bytes);
48 continue;
49 }
50
51 size_t alignment_loss = pmalloc_release_region(r->address, r->size_bytes);
52 if (alignment_loss)
53 pr_emph("%zu bytes of memory loss due to alignment", alignment_loss);
54 }
55}

57void pmalloc_dump()
58{
59 pr_info("pmem freelist has %zu entries", pmlist_count);
60 pmlist_node_t *this = pmlist_head;
61 while (this)
62 {
63 const uintptr_t pstart = this->paddr;
64 const uintptr_t pend = pstart + this->n_pages * MOS_PAGE_SIZE;
65 char sbuf[32];
66 format_size(sbuf, sizeof(sbuf), this->n_pages * MOS_PAGE_SIZE);
67 pr_info(" [" PTR_FMT "] entry: " PTR_FMT "-" PTR_FMT " (%zu page(s), %s)", (uintptr_t) this, pstart, pend, this->n_pages, sbuf);
68 this = this->next;
69 }
70}

72uintptr_t pmalloc_alloc(size_t pages)
73{
74 pmlist_node_t *this = pmlist_head;
75 while (this)
76 {
77 if (this->n_pages >= pages)
78 {
79 const uintptr_t addr = this->paddr; // ! pmalloc_acquire_region changes `this->paddr`, make a copy first
80 mos_debug(pmm, "allocated %zu pages from freelist, starting at " PTR_FMT, pages, addr);
81 pmalloc_acquire_region(addr, pages * MOS_PAGE_SIZE);
82 return addr;
83 }
84 this = this->next;
85 }
86
87 mos_panic("no free physical memory found in freelist, %zu pages requested", pages);
88}

90#define this_start (this->paddr)
91#define this_end (this_start + this->n_pages * MOS_PAGE_SIZE)
92
93void pmalloc_acquire_region(uintptr_t start_addr, size_t size_bytes)
94{
95 const uintptr_t end_addr = start_addr + size_bytes;
96
97 // the addresses must be aligned to page size
98 MOS_ASSERT(start_addr % MOS_PAGE_SIZE == 0);
99 MOS_ASSERT(end_addr % MOS_PAGE_SIZE == 0);
100
101 const size_t pages_in_region = (end_addr - start_addr) / MOS_PAGE_SIZE;
102
103 mos_debug(pmm, "removing physical memory region " PTR_FMT "-" PTR_FMT " from freelist.", start_addr, end_addr);
104
105 bool needs_cleanup = false;
106 bool freed = false;
107 void *cleanup_target_memptr = NULL;
108
109 pmlist_node_t *this = pmlist_head;
110 pmlist_node_t *prev = NULL;
111 while (this)
112 {
113 pmlist_node_t *next = this->next;
114 if (this_start <= start_addr && this_end >= end_addr)
115 {
116 // starts before this region
117 if (start_addr < this_start)
118 continue;
119
120 // or ends after this region
121 if (end_addr > this_end)
122 continue;
123
124 const size_t part_1_size = start_addr - this_start;
125 const size_t part_2_size = this_end - end_addr;
126
127 if (part_1_size == 0 && part_2_size != 0)
128 {
129 // part 1 is empty, which means we are removing from the front of the region
130 this->paddr = end_addr;
131 this->n_pages = part_2_size / MOS_PAGE_SIZE;
132 mos_debug(pmm, "pmem block now starts at " PTR_FMT ", with %zu pages", this_start, this->n_pages);
133 }
134 else if (part_1_size != 0 && part_2_size == 0)
135 {
136 // part 2 is empty, which means we are removing the tail of part 1
137 mos_debug(pmm, "shrunk " PTR_FMT "-" PTR_FMT ": ends at " PTR_FMT, this_start, this_end, this_start + (this->n_pages - pages_in_region) * MOS_PAGE_SIZE);
138 this->n_pages -= pages_in_region;
139 }
140 else if (part_1_size == 0 && part_2_size == 0)
141 {
142 // part 1 and part 2 are empty, so remove this entry
143 if (prev)
144 prev->next = next;
145 else
146 pmlist_head = next;
147
148 // have to do some cleanup
149 needs_cleanup = true;
150 cleanup_target_memptr = this;
151 mos_debug(pmm, "removed " PTR_FMT "-" PTR_FMT " from freelist.", this_start, this_end);
152 }
153 else
154 {
155 // neither part 1 nor part 2 is empty, so we have to allocate a new entry for part 2
156 mos_debug(pmm, "split " PTR_FMT "-" PTR_FMT " into " PTR_FMT "-" PTR_FMT " and " PTR_FMT "-" PTR_FMT, this_start, this_end, this_start, start_addr,
157 end_addr, end_addr + part_2_size);
158 pmlist_node_t *new = (pmlist_node_t *) pmlist_storage + pmlist_count++;
159 memset(new, 0, sizeof(pmlist_node_t));
160 new->next = next;
161 this->next = new;
162 this->n_pages = part_1_size / MOS_PAGE_SIZE;
163 new->paddr = end_addr;
164 new->n_pages = part_2_size / MOS_PAGE_SIZE;
165 }
166 freed = true;
167 break;
168 }
169 prev = this;
170 this = next;
171 }
172
173 if (!freed)
174 mos_panic(PTR_FMT "-" PTR_FMT " is not in the freelist.", start_addr, end_addr);
175
176 if (needs_cleanup)
177 {
178 // firstly find who is at 'pmem_freelist_count' that should be copied to 'cleanup_target_memptr'
179 // minus 1 because it's a count, make it an index
180 pmlist_node_t *copy_source = &pmlist_storage[pmlist_count - 1];
181 pmlist_count--;
182
183 pmlist_node_t *r_prev = pmlist_head;
184 for (pmlist_node_t *r_this = r_prev; r_this != copy_source && r_this->next; r_this = r_this->next)
185 r_prev = r_this;
186
187 if (cleanup_target_memptr != copy_source)
188 {
189 memmove(cleanup_target_memptr, copy_source, sizeof(pmlist_node_t));
190 memzero(copy_source, sizeof(pmlist_node_t)); // prevent from being used again
191
192 if (copy_source != r_prev->next)
193 {
194 pmlist_head = cleanup_target_memptr;
195 }
196 else
197 {
198 r_prev->next = cleanup_target_memptr;
199 }
200 }
201 }
202}

204size_t pmalloc_release_region(uintptr_t start_addr, size_t size_bytes)
205{
206 const uintptr_t aligned_start = ALIGN_UP_TO_PAGE(start_addr);
207 const uintptr_t aligned_end = ALIGN_DOWN_TO_PAGE(start_addr + size_bytes);
208
209 const size_t pages_in_region = (aligned_end - aligned_start) / MOS_PAGE_SIZE;
210
211 mos_debug(pmm, "adding physical memory region " PTR_FMT "-" PTR_FMT " to freelist.", aligned_start, aligned_end);
212
213 pmlist_node_t *this = pmlist_head;
214 pmlist_node_t *prev = NULL;
215
216 while (this)
217 {
218 // * if the aligned start/end address is within the current region, it overlaps
219 if ((this_start <= aligned_start && aligned_start < this_end) || (this_start < aligned_end && aligned_end <= this_end))
220 {
221 mos_panic("new pmem " PTR_FMT "-" PTR_FMT " overlaps with " PTR_FMT "-" PTR_FMT, aligned_start, aligned_end, this_start, this_end);
222 return 0;
223 }
224
225 // prepend to 'this' region
226 if (aligned_end == this_start)
227 {
228 uintptr_t new_end = this_start + pages_in_region * MOS_PAGE_SIZE;
229 mos_debug(pmm, "enlarge range [" PTR_FMT "-" PTR_FMT "]: starts at " PTR_FMT, this_start, this_end, new_end);
230 this->paddr = aligned_start;
231 this->n_pages += pages_in_region;
232 goto end;
233 }
234
235 // append to the previous region (if possible)
236 if (prev)
237 {
238 // * the new region "may" overlap with next region
239 // * if we 'append' to the current region, we will not be able to check for such overlap
240 // * (because we would have 'goto end'-ed)
241 const uintptr_t prev_start = prev->paddr;
242 const uintptr_t prev_end = prev_start + prev->n_pages * MOS_PAGE_SIZE;
243
244 if (aligned_start == prev_end)
245 {
246 prev->n_pages += pages_in_region;
247 const uintptr_t new_ends = prev_end + pages_in_region * MOS_PAGE_SIZE;
248 mos_debug(pmm, "enlarged " PTR_FMT "-" PTR_FMT ": ends at " PTR_FMT, prev_start, prev_end, new_ends);
249 goto end;
250 }
251 }
252
253 // same as above, we only 'prepend' the region before 'this'
254 if (aligned_end < this_start)
255 {
256 pmlist_node_t *new_range = &pmlist_storage[pmlist_count++];
257 new_range->paddr = aligned_start;
258 new_range->n_pages = pages_in_region;
259 new_range->next = this;
260 if (prev)
261 prev->next = new_range;
262 else
263 pmlist_head = new_range;
264 goto end;
265 }
266
267 prev = this;
268 this = this->next;
269 }
270
271 MOS_ASSERT(this == NULL && prev != NULL);
272 pmlist_node_t *new_range = &pmlist_storage[pmlist_count++];
273 prev->next = new_range;
274
275 // the freelist is in the correct order, so we can insert the new range here
276 new_range->paddr = aligned_start;
277 new_range->n_pages = pages_in_region;
278 new_range->next = NULL;
279
280end:
281 // return the amount of memory that was lost due to alignment
282 return (aligned_start - start_addr) + ((start_addr + size_bytes) - aligned_end);
283}
