11// This is the byte alignment that memory must be allocated on. IMPORTANT for GTK and other stuff.
12#define ALIGNMENT 16 // or 4
13
14// unsigned char[16], or unsigned short
15#define ALIGN_TYPE char
16
17// Alignment information is stored right before the pointer. This is the number of bytes of information stored there.
18#define ALIGN_INFO sizeof(ALIGN_TYPE) * 16

26/** This macro will conveniently align our pointer upwards */
27#define LIBALLOC_ALIGN_PTR(ptr) \
28 if (ALIGNMENT > 1) \
29 { \
30 uintptr_t diff; \
31 ptr = (void *) ((uintptr_t) ptr + ALIGN_INFO); \
32 diff = (uintptr_t) ptr & (ALIGNMENT - 1); \
33 if (diff != 0) \
34 { \
35 diff = ALIGNMENT - diff; \
36 ptr = (void *) ((uintptr_t) ptr + diff); \
37 } \
38 *((ALIGN_TYPE *) ((uintptr_t) ptr - ALIGN_INFO)) = diff + ALIGN_INFO; \
39 }
40
41#define LIBALLOC_UNALIGN_PTR(ptr) \
42 if (ALIGNMENT > 1) \
43 { \
44 uintptr_t diff = *((ALIGN_TYPE *) ((uintptr_t) ptr - ALIGN_INFO)); \
45 if (diff < (ALIGNMENT + ALIGN_INFO)) \
46 ptr = (void *) ((uintptr_t) ptr - diff); \
47 }

60void *kheap_alloc_page(size_t npages, vm_flags vmflags)
61{
62 if (unlikely(npages <= 0))
63 {
64 mos_warn("allocating negative or zero pages");
65 return NULL;
66 }
67
68 if (unlikely(mos_platform->mm_alloc_pages == NULL))
69 mos_panic("platform configuration error: alloc_page is NULL");
70
71 vmblock_t block = mos_platform->mm_alloc_pages(mos_platform->kernel_pg, npages, PGALLOC_HINT_KHEAP, vmflags);
72 MOS_ASSERT_X(block.vaddr >= MOS_X86_HEAP_BASE_VADDR, "only use this function to free kernel heap pages");
73
74 if (unlikely(block.npages < npages))
75 mos_warn("failed to allocate %zu pages", npages);
76
77 return (void *) block.vaddr;
78}

80bool kheap_free_page(void *vptr, size_t npages)
81{
82 MOS_ASSERT_X(vptr >= (void *) MOS_X86_HEAP_BASE_VADDR, "only use this function to free kernel heap pages");
83
84 if (unlikely(vptr == NULL))
85 {
86 mos_warn("freeing NULL pointer");
87 return false;
88 }
89 if (unlikely(npages <= 0))
90 {
91 mos_warn("freeing negative or zero pages");
92 return false;
93 }
94 if (unlikely(mos_platform->mm_free_pages == NULL))
95 mos_panic("platform configuration error: free_page is NULL");
96 mos_platform->mm_free_pages(mos_platform->kernel_pg, (uintptr_t) vptr, npages);
97 return true;
98}

100// A structure found at the top of all system allocated memory blocks. It details the usage of the memory block.
101typedef struct liballoc_block
102{
103 struct liballoc_block *prev; // Linked list information.
104 struct liballoc_block *next; // Linked list information.
105 size_t pages; // The number of pages in the block.
106 size_t size; // bytes in the block.
107 size_t usage; // bytes used in the block.
108 struct liballoc_part *first; // A pointer to the first allocated memory in the block.
109} liballoc_block_t;
110
111// This is a structure found at the beginning of all sections in a major block which were allocated by a malloc, calloc, realloc call.
112typedef struct liballoc_part
113{
114 struct liballoc_part *prev; // Linked list information.
115 struct liballoc_part *next; // Linked list information.
116 struct liballoc_block *block; // The owning block. A pointer to the major structure.
117 u32 magic; // A magic number to idenfity correctness.
118 size_t size; // The size of the memory allocated. Could be 1 byte or more.
119 size_t req_size; // The size of memory requested.
120} liballoc_part_t;
121
122static liballoc_block_t *l_memroot = NULL; // The root memory block acquired from the system.
123static liballoc_block_t *l_bestbet = NULL; // The major with the most free memory.
124
125static size_t l_page_size = 0; // The size of an individual page. Set up in liballoc_init.
126static size_t l_alloc_n_page_once = 0; // The number of pages to request per chunk. Set up in liballoc_init.
127
128static size_t l_mem_allocated = 0; // Running total of allocated memory.
129static size_t l_mem_inuse = 0; // Running total of used memory.
130static size_t l_warnings = 0; // Number of warnings encountered
131static size_t l_errors = 0; // Number of actual errors
132static size_t l_possible_overruns = 0; // Number of possible overruns

137 pr_info("--------------- Memory data ---------------");

142

144 liballoc_part_t *min = NULL;
145
146 pr_info("Memory Blocks:");
147 while (maj != NULL)

150
151 min = maj->first;
152 while (min != NULL)
153 {
154 pr_info(" %p: %zu bytes", (void *) min, min->size);
155 min = min->next;
156 }
157
158 maj = maj->next;
159 }

163static liballoc_block_t *allocate_new_pages_for(unsigned int size)
164{
165 // This is how much space is required.
166 u32 st = size + sizeof(liballoc_block_t) + sizeof(liballoc_part_t);
167
168 // Perfect amount of space?
169 if ((st % l_page_size) == 0)
170 st = st / (l_page_size);
171 else
172 st = st / (l_page_size) + 1;
173 // No, add the buffer.
174
175 // Make sure it's >= the minimum size.
176 if (st < l_alloc_n_page_once)
177 st = l_alloc_n_page_once;
178
179 liballoc_block_t *maj = (liballoc_block_t *) kheap_alloc_page(st, VM_READ | VM_WRITE);
180 if (maj == NULL)
181 {
182 l_warnings += 1;
183 mos_warn("liballoc: WARNING: liballoc_alloc(%i) returns NULL", st);
184 return NULL;
185 }
186
187 maj->prev = NULL;
188 maj->next = NULL;
189 maj->pages = st;
190 maj->size = st * l_page_size;
191 maj->usage = sizeof(liballoc_block_t);
192 maj->first = NULL;
193
194 l_mem_allocated += maj->size;
195
196#if MOS_MM_LIBALLOC_DEBUG
197 pr_info("liballoc: Allocated %u pages (%zu bytes) at %p for %u bytes to be used.", st, maj->size, (void *) maj, size);
198 pr_info("liballoc: Total memory usage = %i KB", (int) ((l_mem_allocated / (1024))));
199#endif
200
201 return maj;
202}

204void liballoc_init(size_t page_size)
205{
206 l_page_size = page_size;
207 l_alloc_n_page_once = 16;
208
209 MOS_ASSERT_X(l_memroot == NULL, "liballoc_init() called twice");
210
211#if MOS_MM_LIBALLOC_DEBUG
212 pr_info("liballoc: initialization of liballoc " VERSION "");
213 // atexit(liballoc_dump);
214#endif
215
216 // This is the first time we are being used.
217 l_memroot = allocate_new_pages_for(sizeof(liballoc_block_t));
218 if (l_memroot == NULL)
219 {
220#if MOS_MM_LIBALLOC_LOCKS
221 liballoc_unlock();
222#endif
223#if MOS_MM_LIBALLOC_DEBUG
224 pr_info("liballoc: initial l_memRoot initialization failed");
225#endif
226 }
227
228#if MOS_MM_LIBALLOC_DEBUG
229 pr_info("liballoc: set up first memory major %p", (void *) l_memroot);
230#endif
231}

233void *liballoc_malloc(size_t req_size)
234{
235 unsigned long size = req_size;
236
237 // For alignment, we adjust size so there's enough space to align.
238 if (ALIGNMENT > 1)
239 {
240 size += ALIGNMENT + ALIGN_INFO;
241 }
242 // So, ideally, we really want an alignment of 0 or 1 in order to save space.
243
244#if MOS_MM_LIBALLOC_LOCKS
245 liballoc_lock();
246#endif
247
248 if (size == 0)
249 {
250 mos_warn("liballoc: liballoc_malloc(0) called.");
251 l_warnings += 1;
252#if MOS_MM_LIBALLOC_LOCKS
253 liballoc_unlock();
254#endif
255 return liballoc_malloc(1);
256 }
257
258 MOS_ASSERT_X(l_memroot, "liballoc: liballoc_malloc() called before liballoc_init().");
259
260 // Now we need to bounce through every major and find enough space....
261
262 int startedBet = 0;
263 size_t bestSize = 0;
264 liballoc_block_t *block = l_memroot;
265
266 // Start at the best bet....
267 if (l_bestbet != NULL)
268 {
269 bestSize = l_bestbet->size - l_bestbet->usage;
270
271 if (bestSize > (size + sizeof(liballoc_part_t)))
272 {
273 block = l_bestbet;
274 startedBet = 1;
275 }
276 }
277
278 while (block != NULL)
279 {
280 s64 diff = block->size - block->usage;
281 // free memory in the block
282
283 if (bestSize < diff)
284 {
285 // Hmm.. this one has more memory then our bestBet. Remember!
286 l_bestbet = block;
287 bestSize = diff;
288 }
289
290#ifdef USE_CASE1
291 // CASE 1: There is not enough space in this major block.
292 if (diff < (size + sizeof(liballoc_part_t)))
293 {
294#if MOS_MM_LIBALLOC_DEBUG
295 pr_info("CASE 1: Insufficient space in block %p", (void *) block);
296#endif
297
298 // Another major block next to this one?
299 if (block->next != NULL)
300 {
301 block = block->next; // Hop to that one.
302 continue;
303 }
304
305 if (startedBet == 1) // If we started at the best bet,
306 { // let's start all over again.
307 block = l_memroot;
308 startedBet = 0;
309 continue;
310 }
311
312 // Create a new major block next to this one and...
313 block->next = allocate_new_pages_for(size); // next one will be okay.
314 if (block->next == NULL)
315 break; // no more memory.
316 block->next->prev = block;
317 block = block->next;
318
319 // .. fall through to CASE 2 ..
320 }
321#endif
322
323#ifdef USE_CASE2
324 // CASE 2: It's a brand new block.
325 if (block->first == NULL)
326 {
327 block->first = (liballoc_part_t *) ((uintptr_t) block + sizeof(liballoc_block_t));
328
329 block->first->magic = LIBALLOC_MAGIC;
330 block->first->prev = NULL;
331 block->first->next = NULL;
332 block->first->block = block;
333 block->first->size = size;
334 block->first->req_size = req_size;
335 block->usage += size + sizeof(liballoc_part_t);
336
337 l_mem_inuse += size;
338
339 void *p = (void *) ((uintptr_t) (block->first) + sizeof(liballoc_part_t));
340
341 LIBALLOC_ALIGN_PTR(p);
342
343#if MOS_MM_LIBALLOC_LOCKS
344 liballoc_unlock(); // release the lock
345#endif
346
347#if MOS_MM_LIBALLOC_DEBUG
348 pr_info("liballoc: allocating %lu bytes at %p", size, p);
349#endif
350 return p;
351 }
352
353#endif
354
355 // CASE 3: Block in use and enough space at the start of the block.
356#ifdef USE_CASE3
357 {
358 s64 diff = (uintptr_t) block->first - (uintptr_t) block;
359 diff -= sizeof(liballoc_block_t);
360
361 if (diff >= (size + sizeof(liballoc_part_t)))
362 {
363 // Yes, space in front. Squeeze in.
364 block->first->prev = (liballoc_part_t *) ((uintptr_t) block + sizeof(liballoc_block_t));
365 block->first->prev->next = block->first;
366 block->first = block->first->prev;
367
368 block->first->magic = LIBALLOC_MAGIC;
369 block->first->prev = NULL;
370 block->first->block = block;
371 block->first->size = size;
372 block->first->req_size = req_size;
373 block->usage += size + sizeof(liballoc_part_t);
374
375 l_mem_inuse += size;
376
377 void *p = (void *) ((uintptr_t) (block->first) + sizeof(liballoc_part_t));
378 LIBALLOC_ALIGN_PTR(p);
379
380#if MOS_MM_LIBALLOC_LOCKS
381 liballoc_unlock(); // release the lock
382#endif
383#if MOS_MM_LIBALLOC_DEBUG
384 pr_info("liballoc: allocating %lu bytes at %p", size, p);
385#endif
386 return p;
387 }
388 }
389#endif
390
391#ifdef USE_CASE4
392 // CASE 4: There is enough space in this block. But is it contiguous?
393 liballoc_part_t *section = block->first;
394
395 // Looping within the block now...
396 while (section != NULL)
397 {
398 // CASE 4.1: End of a section in a block. check for enough space from last and end?
399 if (section->next == NULL)
400 {
401 // the rest of this block is free... is it big enough?
402 size_t size_left = block->size; // size of the block
403 size_left -= (uintptr_t) section - (uintptr_t) block; // minus the area before this section
404 size_left -= sizeof(liballoc_part_t) + section->size; // minus this section
405
406 // if there's still enough space
407 if (size_left >= sizeof(liballoc_part_t) + size)
408 {
409 // yay....
410 section->next = (liballoc_part_t *) ((uintptr_t) section + sizeof(liballoc_part_t) + section->size);
411 section->next->prev = section;
412
413 // go to the next section
414 section = section->next;
415 section->next = NULL;
416 section->magic = LIBALLOC_MAGIC;
417 section->block = block;
418 section->size = size;
419 section->req_size = req_size;
420
421 block->usage += size + sizeof(liballoc_part_t);
422
423 l_mem_inuse += size;
424
425 void *p = (void *) ((uintptr_t) section + sizeof(liballoc_part_t));
426 LIBALLOC_ALIGN_PTR(p);
427
428#if MOS_MM_LIBALLOC_LOCKS
429 liballoc_unlock(); // release the lock
430#endif
431#if MOS_MM_LIBALLOC_DEBUG
432 pr_info("liballoc: allocating %lu bytes at %p", size, p);
433#endif
434 return p;
435 }
436 }
437
438 // CASE 4.2: Is there space between two minors?
439 if (section->next != NULL)
440 {
441 // is the difference between here and next big enough?
442 s64 diff = (uintptr_t) (section->next);
443 diff -= (uintptr_t) section;
444 diff -= sizeof(liballoc_part_t);
445 diff -= section->size;
446 // minus our existing usage.
447
448 if (diff >= (size + sizeof(liballoc_part_t)))
449 {
450 // yay......
451 liballoc_part_t *new_min = (liballoc_part_t *) ((uintptr_t) section + sizeof(liballoc_part_t) + section->size);
452
453 new_min->magic = LIBALLOC_MAGIC;
454 new_min->next = section->next;
455 new_min->prev = section;
456 new_min->size = size;
457 new_min->req_size = req_size;
458 new_min->block = block;
459 section->next->prev = new_min;
460 section->next = new_min;
461 block->usage += size + sizeof(liballoc_part_t);
462
463 l_mem_inuse += size;
464
465 void *p = (void *) ((uintptr_t) new_min + sizeof(liballoc_part_t));
466 LIBALLOC_ALIGN_PTR(p);
467
468#if MOS_MM_LIBALLOC_LOCKS
469 liballoc_unlock(); // release the lock
470#endif
471#if MOS_MM_LIBALLOC_DEBUG
472 pr_info("liballoc: allocating %lu bytes at %p", size, p);
473#endif
474 return p;
475 }
476 } // min->next != NULL
477
478 section = section->next;
479 } // while min != NULL ...
480
481#endif
482
483#ifdef USE_CASE5
484 // CASE 5: Block full! Ensure next block and loop.
485 if (block->next == NULL)
486 {
487#if MOS_MM_LIBALLOC_DEBUG
488 pr_info("CASE 5: block full");
489#endif
490
491 if (startedBet == 1)
492 {
493 block = l_memroot;
494 startedBet = 0;
495 continue;
496 }
497
498 // we've run out. we need more...
499 block->next = allocate_new_pages_for(size); // next one guaranteed to be okay
500 if (block->next == NULL)
501 break; // uh oh, no more memory.....
502 block->next->prev = block;
503 }
504
505#endif
506
507 block = block->next;
508 } // while (maj != NULL)
509
510#if MOS_MM_LIBALLOC_LOCKS
511 liballoc_unlock(); // release the lock
512#endif
513
514 mos_warn("No memory available.");
515
516#if MOS_MM_LIBALLOC_DEBUG
517 liballoc_dump();
518#endif
519
520 return NULL;
521}

523void liballoc_free(const void *ptr)
524{
525 liballoc_part_t *min;
526 liballoc_block_t *maj;
527
528 if (ptr == NULL)
529 {
530 l_warnings += 1;
531 mos_warn("liballoc: free(NULL) called");
532 return;
533 }
534
535 LIBALLOC_UNALIGN_PTR(ptr);
536
537#if MOS_MM_LIBALLOC_LOCKS
538 liballoc_lock(); // lockit
539#endif
540
541 min = (liballoc_part_t *) ((uintptr_t) ptr - sizeof(liballoc_part_t));
542
543 if (min->magic != LIBALLOC_MAGIC)
544 {
545 l_errors += 1;
546
547 // Check for overrun errors. For all bytes of LIBALLOC_MAGIC
548 if (((min->magic & 0xFFFFFF) == (LIBALLOC_MAGIC & 0xFFFFFF)) || ((min->magic & 0xFFFF) == (LIBALLOC_MAGIC & 0xFFFF)) ||
549 ((min->magic & 0xFF) == (LIBALLOC_MAGIC & 0xFF)))
550 {
551 l_possible_overruns += 1;
552 mos_warn("liballoc: ERROR: Possible 1-3 byte overrun for magic %x != %x", min->magic, LIBALLOC_MAGIC);
553 }
554
555 if (min->magic == LIBALLOC_DEAD)
556 {
557 mos_warn("liballoc: multiple free() attempt on %p", ptr);
558 }
559 else
560 {
561 mos_warn("liballoc: bad free(%p) called.", ptr);
562 }
563#if MOS_MM_LIBALLOC_LOCKS
564 // being lied to...
565 liballoc_unlock(); // release the lock
566#endif
567 return;
568 }
569
570 maj = min->block;
571 l_mem_inuse -= min->size;
572 maj->usage -= (min->size + sizeof(liballoc_part_t));
573 min->magic = LIBALLOC_DEAD; // No mojo.
574
575 if (min->next != NULL)
576 min->next->prev = min->prev;
577 if (min->prev != NULL)
578 min->prev->next = min->next;
579
580 if (min->prev == NULL)
581 maj->first = min->next;
582 // Might empty the block. This was the first
583 // minor.
584
585 // We need to clean up after the majors now....
586
587 if (maj->first == NULL) // Block completely unused.
588 {
589 if (l_memroot == maj)
590 l_memroot = maj->next;
591 if (l_bestbet == maj)
592 l_bestbet = NULL;
593 if (maj->prev != NULL)
594 maj->prev->next = maj->next;
595 if (maj->next != NULL)
596 maj->next->prev = maj->prev;
597 l_mem_allocated -= maj->size;
598
599 kheap_free_page(maj, maj->pages);
600 }
601 else
602 {
603 if (l_bestbet != NULL)
604 {
605 int bestSize = l_bestbet->size - l_bestbet->usage;
606 int majSize = maj->size - maj->usage;
607
608 if (majSize > bestSize)
609 l_bestbet = maj;
610 }
611 }
612
613#if MOS_MM_LIBALLOC_LOCKS
614 liballoc_unlock(); // release the lock
615#endif
616}

618void *liballoc_calloc(size_t nobj, size_t size)
619{
620 MOS_ASSERT_X(nobj > 0, "You Fool! You can't allocate 0 objects!");
621 size_t real_size = nobj * size;
622 void *p = liballoc_malloc(real_size);
623 memset(p, 0, real_size);
624 return p;
625}

627void *liballoc_realloc(void *p, size_t size)
628{
629
630 // Honour the case of size == 0 => free old and return NULL
631 if (size == 0)
632 {
633 liballoc_free(p);
634 return NULL;
635 }
636
637 // In the case of a NULL pointer, return a simple malloc.
638 if (p == NULL)
639 return liballoc_malloc(size);
640
641 // Unalign the pointer if required.
642
643 void *ptr = p;
644 LIBALLOC_UNALIGN_PTR(ptr);
645
646#if MOS_MM_LIBALLOC_LOCKS
647 liballoc_lock(); // lockit
648#endif
649
650 liballoc_part_t *min = (liballoc_part_t *) ((uintptr_t) ptr - sizeof(liballoc_part_t));
651
652 // Ensure it is a valid structure.
653 if (min->magic != LIBALLOC_MAGIC)
654 {
655 l_errors += 1;
656
657 // Check for overrun errors. For all bytes of LIBALLOC_MAGIC
658 if (((min->magic & 0xFFFFFF) == (LIBALLOC_MAGIC & 0xFFFFFF)) || ((min->magic & 0xFFFF) == (LIBALLOC_MAGIC & 0xFFFF)) ||
659 ((min->magic & 0xFF) == (LIBALLOC_MAGIC & 0xFF)))
660 {
661 l_possible_overruns += 1;
662 mos_warn("liballoc: Possible 1-3 byte overrun for magic %x != %x", min->magic, LIBALLOC_MAGIC);
663 }
664
665 if (min->magic == LIBALLOC_DEAD)
666 {
667 mos_warn("liballoc: multiple free() attempt on %p from %p.", ptr, __builtin_return_address(0));
668 }
669 else
670 {
671 mos_warn("liballoc: Bad free(%p) called", ptr);
672 }
673#if MOS_MM_LIBALLOC_LOCKS
674 // being lied to...
675 liballoc_unlock(); // release the lock
676#endif
677 return NULL;
678 }
679
680 // Definitely a memory block.
681
682 unsigned int real_size = min->req_size;
683
684 if (real_size >= size)
685 {
686 min->req_size = size;
687#if MOS_MM_LIBALLOC_LOCKS
688 liballoc_unlock();
689#endif
690 return p;
691 }
692#if MOS_MM_LIBALLOC_LOCKS
693 liballoc_unlock();
694#endif
695
696 // If we got here then we're reallocating to a block bigger than us.
697 ptr = liballoc_malloc(size); // We need to allocate new memory
698 memcpy(ptr, p, real_size);
699 liballoc_free(p);
700
701 return ptr;
702}
