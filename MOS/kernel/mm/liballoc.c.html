12// This is the byte alignment that memory must be allocated on. IMPORTANT for GTK and other stuff.
13#define ALIGNMENT 16 // or 4
14
15// unsigned char[16], or unsigned short
16#define ALIGN_TYPE char
17
18// Alignment information is stored right before the pointer. This is the number of bytes of information stored there.
19#define ALIGN_INFO sizeof(ALIGN_TYPE) * 16
20

27/** This macro will conveniently align our pointer upwards */
28#define LIBALLOC_ALIGN_PTR(ptr) \
29 if (ALIGNMENT > 1) \
30 { \
31 uintptr_t diff; \
32 ptr = (void *) ((uintptr_t) ptr + ALIGN_INFO); \
33 diff = (uintptr_t) ptr & (ALIGNMENT - 1); \
34 if (diff != 0) \
35 { \
36 diff = ALIGNMENT - diff; \
37 ptr = (void *) ((uintptr_t) ptr + diff); \
38 } \
39 *((ALIGN_TYPE *) ((uintptr_t) ptr - ALIGN_INFO)) = diff + ALIGN_INFO; \
40 }
41
42#define LIBALLOC_UNALIGN_PTR(ptr) \
43 if (ALIGNMENT > 1) \
44 { \
45 uintptr_t diff = *((ALIGN_TYPE *) ((uintptr_t) ptr - ALIGN_INFO)); \
46 if (diff < (ALIGNMENT + ALIGN_INFO)) \
47 ptr = (void *) ((uintptr_t) ptr - diff); \
48 }

50#define LIBALLOC_MAGIC 0xaabbccdd
51#define LIBALLOC_DEAD 0xdeaddead

53// A structure found at the top of all system allocated memory blocks. It details the usage of the memory block.
54typedef struct liballoc_block
55{
56 struct liballoc_block *prev; // Linked list information.
57 struct liballoc_block *next; // Linked list information.
58 size_t pages; // The number of pages in the block.
59 size_t size; // bytes in the block.
60 size_t usage; // bytes used in the block.
61 struct liballoc_part *first; // A pointer to the first allocated memory in the block.
62} liballoc_block_t;
63
64// This is a structure found at the beginning of all sections in a major block which were allocated by a malloc, calloc, realloc call.
65typedef struct liballoc_part
66{
67 struct liballoc_part *prev; // Linked list information.
68 struct liballoc_part *next; // Linked list information.
69 struct liballoc_block *block; // The owning block. A pointer to the major structure.
70 u32 magic; // A magic number to idenfity correctness.
71 size_t size; // The size of the memory allocated. Could be 1 byte or more.
72 size_t req_size; // The size of memory requested.
73} liballoc_part_t;
74
75static liballoc_block_t *l_memroot = NULL; // The root memory block acquired from the system.
76static liballoc_block_t *l_bestbet = NULL; // The major with the most free memory.
77
78static size_t l_page_size = 0; // The size of an individual page. Set up in liballoc_init.
79static size_t l_alloc_n_page_once = 0; // The number of pages to request per chunk. Set up in liballoc_init.
80
81static size_t l_mem_allocated = 0; // Running total of allocated memory.
82static size_t l_mem_inuse = 0; // Running total of used memory.
83static size_t l_warnings = 0; // Number of warnings encountered
84static size_t l_errors = 0; // Number of actual errors
85static size_t l_possible_overruns = 0; // Number of possible overruns

90 pr_info("--------------- Memory data ---------------");
91 pr_info("Total Memory Allocated: %zu bytes", l_mem_allocated);
92 pr_info("Memory Used (malloc'ed): %zu bytes", l_mem_inuse);
93 pr_info("Possible Overruns: %zu", l_possible_overruns);
94 pr_info("emitted %zu warning(s) and %zu error(s)", l_warnings, l_errors);
95
96 liballoc_block_t *maj = l_memroot;
97 liballoc_part_t *min = NULL;
98
99 pr_info("Memory Blocks:");
100 while (maj != NULL)
101 {
102 pr_info(" %p: total = %zu, used = %zu", (void *) maj, maj->size, maj->usage);
103
104 min = maj->first;
105 while (min != NULL)
106 {
107 pr_info(" %p: %zu bytes", (void *) min, min->size);
108 min = min->next;
109 }
110
111 maj = maj->next;
112 }
113}

116static liballoc_block_t *allocate_new_pages_for(unsigned int size)
117{
118 // This is how much space is required.
119 u32 st = size + sizeof(liballoc_block_t) + sizeof(liballoc_part_t);
120
121 // Perfect amount of space?
122 if ((st % l_page_size) == 0)
123 st = st / (l_page_size);
124 else
125 st = st / (l_page_size) + 1;
126 // No, add the buffer.
127
128 // Make sure it's >= the minimum size.
129 if (st < l_alloc_n_page_once)
130 st = l_alloc_n_page_once;
131
132 liballoc_block_t *maj = (liballoc_block_t *) kpage_alloc(st, PGALLOC_KHEAP);
133 if (maj == NULL)
134 {
135 l_warnings += 1;
136 mos_warn("liballoc: WARNING: liballoc_alloc(%i) returns NULL", st);
137 return NULL;
138 }
139
140 maj->prev = NULL;
141 maj->next = NULL;
142 maj->pages = st;
143 maj->size = st * l_page_size;
144 maj->usage = sizeof(liballoc_block_t);
145 maj->first = NULL;
146
147 l_mem_allocated += maj->size;
148
149#if MOS_MM_LIBALLOC_DEBUG
150 pr_info("liballoc: Allocated %u pages (%zu bytes) at %p for %u bytes to be used.", st, maj->size, (void *) maj, size);
151 pr_info("liballoc: Total memory usage = %i KB", (int) ((l_mem_allocated / (1024))));
152#endif
153
154 return maj;
155}

157void liballoc_init(size_t page_size)
158{
159 l_page_size = page_size;
160 l_alloc_n_page_once = 16;
161
162 MOS_ASSERT_X(l_memroot == NULL, "liballoc_init() called twice");
163
164#if MOS_MM_LIBALLOC_DEBUG
165 pr_info("liballoc: initialization of liballoc " VERSION "");
166 // atexit(liballoc_dump);
167#endif
168
169 // This is the first time we are being used.
170 l_memroot = allocate_new_pages_for(sizeof(liballoc_block_t));
171 if (l_memroot == NULL)
172 {
173#if MOS_MM_LIBALLOC_LOCKS
174 liballoc_unlock();
175#endif
176#if MOS_MM_LIBALLOC_DEBUG
177 pr_info("liballoc: initial l_memRoot initialization failed");
178#endif
179 }
180
181#if MOS_MM_LIBALLOC_DEBUG
182 pr_info("liballoc: set up first memory major %p", (void *) l_memroot);
183#endif
184}

186void *liballoc_malloc(size_t req_size)
187{
188 unsigned long size = req_size;
189
190 // For alignment, we adjust size so there's enough space to align.
191 if (ALIGNMENT > 1)
192 {
193 size += ALIGNMENT + ALIGN_INFO;
194 }
195 // So, ideally, we really want an alignment of 0 or 1 in order to save space.
196
197#if MOS_MM_LIBALLOC_LOCKS
198 liballoc_lock();
199#endif
200
201 if (size == 0)
202 {
203 mos_warn("liballoc: liballoc_malloc(0) called.");
204 l_warnings += 1;
205#if MOS_MM_LIBALLOC_LOCKS
206 liballoc_unlock();
207#endif
208 return liballoc_malloc(1);
209 }
210
211 MOS_ASSERT_X(l_memroot, "liballoc: liballoc_malloc() called before liballoc_init().");
212
213 // Now we need to bounce through every major and find enough space....
214
215 int startedBet = 0;
216 size_t bestSize = 0;
217 liballoc_block_t *block = l_memroot;
218
219 // Start at the best bet....
220 if (l_bestbet != NULL)
221 {
222 bestSize = l_bestbet->size - l_bestbet->usage;
223
224 if (bestSize > (size + sizeof(liballoc_part_t)))
225 {
226 block = l_bestbet;
227 startedBet = 1;
228 }
229 }
230
231 while (block != NULL)
232 {
233 s64 diff = block->size - block->usage;
234 // free memory in the block
235
236 if (bestSize < diff)
237 {
238 // Hmm.. this one has more memory then our bestBet. Remember!
239 l_bestbet = block;
240 bestSize = diff;
241 }
242
243#ifdef USE_CASE1
244 // CASE 1: There is not enough space in this major block.
245 if (diff < (size + sizeof(liballoc_part_t)))
246 {
247#if MOS_MM_LIBALLOC_DEBUG
248 pr_info("CASE 1: Insufficient space in block %p", (void *) block);
249#endif
250
251 // Another major block next to this one?
252 if (block->next != NULL)
253 {
254 block = block->next; // Hop to that one.
255 continue;
256 }
257
258 if (startedBet == 1) // If we started at the best bet,
259 { // let's start all over again.
260 block = l_memroot;
261 startedBet = 0;
262 continue;
263 }
264
265 // Create a new major block next to this one and...
266 block->next = allocate_new_pages_for(size); // next one will be okay.
267 if (block->next == NULL)
268 break; // no more memory.
269 block->next->prev = block;
270 block = block->next;
271
272 // .. fall through to CASE 2 ..
273 }
274#endif
275
276#ifdef USE_CASE2
277 // CASE 2: It's a brand new block.
278 if (block->first == NULL)
279 {
280 block->first = (liballoc_part_t *) ((uintptr_t) block + sizeof(liballoc_block_t));
281
282 block->first->magic = LIBALLOC_MAGIC;
283 block->first->prev = NULL;
284 block->first->next = NULL;
285 block->first->block = block;
286 block->first->size = size;
287 block->first->req_size = req_size;
288 block->usage += size + sizeof(liballoc_part_t);
289
290 l_mem_inuse += size;
291
292 void *p = (void *) ((uintptr_t) (block->first) + sizeof(liballoc_part_t));
293
294 LIBALLOC_ALIGN_PTR(p);
295
296#if MOS_MM_LIBALLOC_LOCKS
297 liballoc_unlock(); // release the lock
298#endif
299
300#if MOS_MM_LIBALLOC_DEBUG
301 pr_info("liballoc: allocating %lu bytes at %p", size, p);
302#endif
303 return p;
304 }
305
306#endif
307
308 // CASE 3: Block in use and enough space at the start of the block.
309#ifdef USE_CASE3
310 {
311 s64 diff = (uintptr_t) block->first - (uintptr_t) block;
312 diff -= sizeof(liballoc_block_t);
313
314 if (diff >= (size + sizeof(liballoc_part_t)))
315 {
316 // Yes, space in front. Squeeze in.
317 block->first->prev = (liballoc_part_t *) ((uintptr_t) block + sizeof(liballoc_block_t));
318 block->first->prev->next = block->first;
319 block->first = block->first->prev;
320
321 block->first->magic = LIBALLOC_MAGIC;
322 block->first->prev = NULL;
323 block->first->block = block;
324 block->first->size = size;
325 block->first->req_size = req_size;
326 block->usage += size + sizeof(liballoc_part_t);
327
328 l_mem_inuse += size;
329
330 void *p = (void *) ((uintptr_t) (block->first) + sizeof(liballoc_part_t));
331 LIBALLOC_ALIGN_PTR(p);
332
333#if MOS_MM_LIBALLOC_LOCKS
334 liballoc_unlock(); // release the lock
335#endif
336#if MOS_MM_LIBALLOC_DEBUG
337 pr_info("liballoc: allocating %lu bytes at %p", size, p);
338#endif
339 return p;
340 }
341 }
342#endif
343
344#ifdef USE_CASE4
345 // CASE 4: There is enough space in this block. But is it contiguous?
346 liballoc_part_t *section = block->first;
347
348 // Looping within the block now...
349 while (section != NULL)
350 {
351 // CASE 4.1: End of a section in a block. check for enough space from last and end?
352 if (section->next == NULL)
353 {
354 // the rest of this block is free... is it big enough?
355 size_t size_left = block->size; // size of the block
356 size_left -= (uintptr_t) section - (uintptr_t) block; // minus the area before this section
357 size_left -= sizeof(liballoc_part_t) + section->size; // minus this section
358
359 // if there's still enough space
360 if (size_left >= sizeof(liballoc_part_t) + size)
361 {
362 // yay....
363 section->next = (liballoc_part_t *) ((uintptr_t) section + sizeof(liballoc_part_t) + section->size);
364 section->next->prev = section;
365
366 // go to the next section
367 section = section->next;
368 section->next = NULL;
369 section->magic = LIBALLOC_MAGIC;
370 section->block = block;
371 section->size = size;
372 section->req_size = req_size;
373
374 block->usage += size + sizeof(liballoc_part_t);
375
376 l_mem_inuse += size;
377
378 void *p = (void *) ((uintptr_t) section + sizeof(liballoc_part_t));
379 LIBALLOC_ALIGN_PTR(p);
380
381#if MOS_MM_LIBALLOC_LOCKS
382 liballoc_unlock(); // release the lock
383#endif
384#if MOS_MM_LIBALLOC_DEBUG
385 pr_info("liballoc: allocating %lu bytes at %p", size, p);
386#endif
387 return p;
388 }
389 }
390
391 // CASE 4.2: Is there space between two minors?
392 if (section->next != NULL)
393 {
394 // is the difference between here and next big enough?
395 s64 diff = (uintptr_t) (section->next);
396 diff -= (uintptr_t) section;
397 diff -= sizeof(liballoc_part_t);
398 diff -= section->size;
399 // minus our existing usage.
400
401 if (diff >= (size + sizeof(liballoc_part_t)))
402 {
403 // yay......
404 liballoc_part_t *new_min = (liballoc_part_t *) ((uintptr_t) section + sizeof(liballoc_part_t) + section->size);
405
406 new_min->magic = LIBALLOC_MAGIC;
407 new_min->next = section->next;
408 new_min->prev = section;
409 new_min->size = size;
410 new_min->req_size = req_size;
411 new_min->block = block;
412 section->next->prev = new_min;
413 section->next = new_min;
414 block->usage += size + sizeof(liballoc_part_t);
415
416 l_mem_inuse += size;
417
418 void *p = (void *) ((uintptr_t) new_min + sizeof(liballoc_part_t));
419 LIBALLOC_ALIGN_PTR(p);
420
421#if MOS_MM_LIBALLOC_LOCKS
422 liballoc_unlock(); // release the lock
423#endif
424#if MOS_MM_LIBALLOC_DEBUG
425 pr_info("liballoc: allocating %lu bytes at %p", size, p);
426#endif
427 return p;
428 }
429 } // min->next != NULL
430
431 section = section->next;
432 } // while min != NULL ...
433
434#endif
435
436#ifdef USE_CASE5
437 // CASE 5: Block full! Ensure next block and loop.
438 if (block->next == NULL)
439 {
440#if MOS_MM_LIBALLOC_DEBUG
441 pr_info("CASE 5: block full");
442#endif
443
444 if (startedBet == 1)
445 {
446 block = l_memroot;
447 startedBet = 0;
448 continue;
449 }
450
451 // we've run out. we need more...
452 block->next = allocate_new_pages_for(size); // next one guaranteed to be okay
453 if (block->next == NULL)
454 break; // uh oh, no more memory.....
455 block->next->prev = block;
456 }
457
458#endif
459
460 block = block->next;
461 } // while (maj != NULL)
462
463#if MOS_MM_LIBALLOC_LOCKS
464 liballoc_unlock(); // release the lock
465#endif
466
467 mos_warn("No memory available.");
468
469#if MOS_MM_LIBALLOC_DEBUG
470 liballoc_dump();
471#endif
472
473 return NULL;
474}

476void liballoc_free(const void *ptr)
477{
478 liballoc_part_t *min;
479 liballoc_block_t *maj;
480
481 if (ptr == NULL)
482 {
483 l_warnings += 1;
484 mos_warn("liballoc: free(NULL) called");
485 return;
486 }
487
488 LIBALLOC_UNALIGN_PTR(ptr);
489
490#if MOS_MM_LIBALLOC_LOCKS
491 liballoc_lock(); // lockit
492#endif
493
494 min = (liballoc_part_t *) ((uintptr_t) ptr - sizeof(liballoc_part_t));
495
496 if (min->magic != LIBALLOC_MAGIC)
497 {
498 l_errors += 1;
499
500 // Check for overrun errors. For all bytes of LIBALLOC_MAGIC
501 if (((min->magic & 0xFFFFFF) == (LIBALLOC_MAGIC & 0xFFFFFF)) || ((min->magic & 0xFFFF) == (LIBALLOC_MAGIC & 0xFFFF)) ||
502 ((min->magic & 0xFF) == (LIBALLOC_MAGIC & 0xFF)))
503 {
504 l_possible_overruns += 1;
505 mos_warn("liballoc: ERROR: Possible 1-3 byte overrun for magic %x != %x", min->magic, LIBALLOC_MAGIC);
506 }
507
508 if (min->magic == LIBALLOC_DEAD)
509 {
510 mos_warn("liballoc: multiple free() attempt on %p", ptr);
511 }
512 else
513 {
514 mos_warn("liballoc: bad free(%p) called.", ptr);
515 }
516#if MOS_MM_LIBALLOC_LOCKS
517 // being lied to...
518 liballoc_unlock(); // release the lock
519#endif
520 return;
521 }
522
523 maj = min->block;
524 l_mem_inuse -= min->size;
525 maj->usage -= (min->size + sizeof(liballoc_part_t));
526 min->magic = LIBALLOC_DEAD; // No mojo.
527
528 if (min->next != NULL)
529 min->next->prev = min->prev;
530 if (min->prev != NULL)
531 min->prev->next = min->next;
532
533 if (min->prev == NULL)
534 maj->first = min->next;
535 // Might empty the block. This was the first
536 // minor.
537
538 // We need to clean up after the majors now....
539
540 if (maj->first == NULL) // Block completely unused.
541 {
542 if (l_memroot == maj)
543 l_memroot = maj->next;
544 if (l_bestbet == maj)
545 l_bestbet = NULL;
546 if (maj->prev != NULL)
547 maj->prev->next = maj->next;
548 if (maj->next != NULL)
549 maj->next->prev = maj->prev;
550 l_mem_allocated -= maj->size;
551
552 kpage_free(maj, maj->pages);
553 }
554 else
555 {
556 if (l_bestbet != NULL)
557 {
558 int bestSize = l_bestbet->size - l_bestbet->usage;
559 int majSize = maj->size - maj->usage;
560
561 if (majSize > bestSize)
562 l_bestbet = maj;
563 }
564 }
565
566#if MOS_MM_LIBALLOC_LOCKS
567 liballoc_unlock(); // release the lock
568#endif
569}

571void *liballoc_calloc(size_t nobj, size_t size)
572{
573 MOS_ASSERT_X(nobj > 0, "You Fool! You can't allocate 0 objects!");
574 size_t real_size = nobj * size;
575 void *p = liballoc_malloc(real_size);
576 memset(p, 0, real_size);
577 return p;
578}

580void *liballoc_realloc(void *p, size_t size)
581{
582
583 // Honour the case of size == 0 => free old and return NULL
584 if (size == 0)
585 {
586 liballoc_free(p);
587 return NULL;
588 }
589
590 // In the case of a NULL pointer, return a simple malloc.
591 if (p == NULL)
592 return liballoc_malloc(size);
593
594 // Unalign the pointer if required.
595
596 void *ptr = p;
597 LIBALLOC_UNALIGN_PTR(ptr);
598
599#if MOS_MM_LIBALLOC_LOCKS
600 liballoc_lock(); // lockit
601#endif
602
603 liballoc_part_t *min = (liballoc_part_t *) ((uintptr_t) ptr - sizeof(liballoc_part_t));
604
605 // Ensure it is a valid structure.
606 if (min->magic != LIBALLOC_MAGIC)
607 {
608 l_errors += 1;
609
610 // Check for overrun errors. For all bytes of LIBALLOC_MAGIC
611 if (((min->magic & 0xFFFFFF) == (LIBALLOC_MAGIC & 0xFFFFFF)) || ((min->magic & 0xFFFF) == (LIBALLOC_MAGIC & 0xFFFF)) ||
612 ((min->magic & 0xFF) == (LIBALLOC_MAGIC & 0xFF)))
613 {
614 l_possible_overruns += 1;
615 mos_warn("liballoc: Possible 1-3 byte overrun for magic %x != %x", min->magic, LIBALLOC_MAGIC);
616 }
617
618 if (min->magic == LIBALLOC_DEAD)
619 {
620 mos_warn("liballoc: multiple free() attempt on %p from %p.", ptr, __builtin_return_address(0));
621 }
622 else
623 {
624 mos_warn("liballoc: Bad free(%p) called", ptr);
625 }
626#if MOS_MM_LIBALLOC_LOCKS
627 // being lied to...
628 liballoc_unlock(); // release the lock
629#endif
630 return NULL;
631 }
632
633 // Definitely a memory block.
634
635 unsigned int real_size = min->req_size;
636
637 if (real_size >= size)
638 {
639 min->req_size = size;
640#if MOS_MM_LIBALLOC_LOCKS
641 liballoc_unlock();
642#endif
643 return p;
644 }
645#if MOS_MM_LIBALLOC_LOCKS
646 liballoc_unlock();
647#endif
648
649 // If we got here then we're reallocating to a block bigger than us.
650 ptr = liballoc_malloc(size); // We need to allocate new memory
651 memcpy(ptr, p, real_size);
652 liballoc_free(p);
653
654 return ptr;
655}
