14static vmblock_t zero_block;
15
16void mos_kernel_mm_init()
17{
18 // zero fill on demand (read-only)
19 zero_block = mm_alloc_pages(current_cpu->pagetable, 1, PGALLOC_HINT_KHEAP, VM_RW);
20 memzero((void *) zero_block.vaddr, MOS_PAGE_SIZE);
21
22 liballoc_init();
23#if MOS_MM_LIBALLOC_DEBUG
24 mos_install_kpanic_hook(liballoc_dump);
25#endif
26}

28// !! This function is called by liballoc, not intended to be called by anyone else !!
29void *liballoc_alloc_page(size_t npages)
30{
31 if (unlikely(npages <= 0))
32 {
33 mos_warn("allocating negative or zero pages");
34 return NULL;
35 }
36
37 vmblock_t block = mm_alloc_pages(current_cpu->pagetable, npages, PGALLOC_HINT_KHEAP, VM_RW);
38 MOS_ASSERT_X(block.vaddr >= MOS_ADDR_KERNEL_HEAP, "only use this function to free kernel heap pages");
39
40 if (unlikely(block.npages < npages))
41 mos_warn("failed to allocate %zu pages", npages);
42
43 return (void *) block.vaddr;
44}
45
46// !! This function is called by liballoc, not intended to be called by anyone else !!
47bool liballoc_free_page(void *vptr, size_t npages)
48{
49 MOS_ASSERT_X(vptr >= (void *) MOS_ADDR_KERNEL_HEAP, "only use this function to free kernel heap pages");
50
51 if (unlikely(vptr == NULL))
52 {
53 mos_warn("freeing NULL pointer");
54 return false;
55 }
56 if (unlikely(npages <= 0))
57 {
58 mos_warn("freeing negative or zero pages");
59 return false;
60 }
61
62 // TODO: this is a hack
63 vmblock_t block = platform_mm_get_block_info(current_cpu->pagetable, (uintptr_t) vptr, npages);
64 mm_free_pages(current_cpu->pagetable, block);
65 return true;
66}

68vmblock_t mm_alloc_zeroed_pages(paging_handle_t handle, size_t npages, pgalloc_hints hints, vm_flags flags)
69{
70 vmblock_t free_pages = mm_get_free_pages(handle, npages, hints);
71 if (free_pages.npages < npages)
72 {
73 mos_warn("failed to allocate %zu pages", npages);
74 return (vmblock_t){ 0 };
75 }
76
77 // zero fill the pages
78 for (size_t i = 0; i < npages; i++)
79 {
80 // actually, zero_block is always accessible, using [handle] == using [current_cpu->pagetable] as source
81 vmblock_t this_block = {
82 .vaddr = free_pages.vaddr + i * MOS_PAGE_SIZE,
83 .npages = 1,
84 .flags = VM_READ | VM_USER,
85 .paddr = zero_block.paddr,
86 };
87 mm_map_allocated_pages(handle, this_block);
88 }
89
90 // make the pages read-only (because for now, they are mapped to zero_block)
91 platform_mm_flag_pages(handle, free_pages.vaddr, npages, VM_READ | ((flags & VM_USER) ? VM_USER : 0));
92 free_pages.flags = flags; // but set the desired flags correctly
93 return free_pages;
94}

96vmblock_t mm_alloc_zeroed_pages_at(paging_handle_t handle, uintptr_t vaddr, size_t npages, vm_flags flags)
97{
98 if (mm_get_is_mapped(handle, vaddr))
99 {
100 mos_warn("failed to allocate %zu pages at %p: already mapped", npages, (void *) vaddr);
101 return (vmblock_t){ 0 };
102 }
103
104 // zero fill the pages
105 for (size_t i = 0; i < npages; i++)
106 {
107 // actually, zero_block is always accessible, using [handle] == using [current_cpu->pagetable] as source
108 mm_copy_maps(current_cpu->pagetable, zero_block.vaddr, handle, vaddr + i * MOS_PAGE_SIZE, 1);
109 }
110
111 // make the pages read-only (because for now, they are mapped to zero_block)
112 platform_mm_flag_pages(handle, vaddr, npages, VM_READ | ((flags & VM_USER) ? VM_USER : 0));
113 return (vmblock_t){ .vaddr = vaddr, .npages = npages, .flags = flags };
114}

116vmblock_t mm_map_proxy_space(paging_handle_t src, uintptr_t srcvaddr, size_t npages)
117{
118 const vmblock_t proxy_blk = mm_get_free_pages(current_cpu->pagetable, npages, PGALLOC_HINT_KHEAP);
119 mm_copy_maps(src, srcvaddr, current_cpu->pagetable, proxy_blk.vaddr, npages);
120 return proxy_blk;
121}
122
123void mm_unmap_proxy_space(vmblock_t proxy_blk)
124{
125 mm_unmap_pages(current_cpu->pagetable, proxy_blk.vaddr, proxy_blk.npages);
126}

128void mm_copy_pages(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t n_pages)
129{
130 paging_handle_t current_handle = current_cpu->pagetable;
131 if (fvaddr >= MOS_KERNEL_START_VADDR && tvaddr >= MOS_KERNEL_START_VADDR)
132 {
133 // both addresses are in kernel space
134 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
135 }
136 else if (fvaddr >= MOS_KERNEL_START_VADDR)
137 {
138 // from kernel space to user space
139 if (to.pgd == current_handle.pgd)
140 {
141 // if the target is the current pagetable
142 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
143 }
144 else
145 {
146 // otherwise, make a proxy block
147 const vmblock_t proxy_blk = mm_map_proxy_space(to, tvaddr, n_pages);
148 memcpy((void *) proxy_blk.vaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
149 mm_unmap_proxy_space(proxy_blk);
150 }
151 }
152 else if (tvaddr >= MOS_KERNEL_START_VADDR)
153 {
154 // from user space to kernel space
155 if (from.pgd == current_handle.pgd)
156 {
157 // if the source is the current pagetable
158 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
159 }
160 else
161 {
162 // otherwise, make a proxy block
163 const vmblock_t proxy_blk = mm_map_proxy_space(from, fvaddr, n_pages);
164 memcpy((void *) tvaddr, (void *) proxy_blk.vaddr, n_pages * MOS_PAGE_SIZE);
165 mm_unmap_proxy_space(proxy_blk);
166 }
167 }
168 else
169 {
170 // both addresses are in user space
171 if (from.pgd == current_handle.pgd)
172 {
173 if (to.pgd == current_handle.pgd)
174 {
175 // although both addresses are in user space, they are both the current pagetable
176 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
177 }
178 else
179 {
180 // only the source is the current pagetable
181 const vmblock_t proxy_blk = mm_map_proxy_space(to, tvaddr, n_pages);
182 memcpy((void *) proxy_blk.vaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
183 mm_unmap_proxy_space(proxy_blk);
184 }
185 }
186 else if (to.pgd == current_handle.pgd)
187 {
188 // only the target is the current pagetable
189 const vmblock_t proxy_blk = mm_map_proxy_space(from, fvaddr, n_pages);
190 memcpy((void *) tvaddr, (void *) proxy_blk.vaddr, n_pages * MOS_PAGE_SIZE);
191 mm_unmap_proxy_space(proxy_blk);
192 }
193 else
194 {
195 // neither is the current pagetable
196 const vmblock_t proxy_blk_from = mm_map_proxy_space(from, fvaddr, n_pages);
197 const vmblock_t proxy_blk_to = mm_map_proxy_space(to, tvaddr, n_pages);
198 memcpy((void *) proxy_blk_to.vaddr, (void *) proxy_blk_from.vaddr, n_pages * MOS_PAGE_SIZE);
199 mm_unmap_proxy_space(proxy_blk_from);
200 mm_unmap_proxy_space(proxy_blk_to);
201 }
202 }
