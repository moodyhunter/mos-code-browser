11vmblock_t mm_map_proxy_space(paging_handle_t src, uintptr_t srcvaddr, size_t npages)
12{
13 const vmblock_t proxy_blk = platform_mm_get_free_pages(current_cpu->pagetable, npages, PGALLOC_HINT_USERSPACE);
14 platform_mm_copy_maps(src, srcvaddr, current_cpu->pagetable, proxy_blk.vaddr, npages);
15 return proxy_blk;
16}
17
18void mm_unmap_proxy_space(vmblock_t proxy_blk)
19{
20 platform_mm_unmap_pages(current_cpu->pagetable, proxy_blk.vaddr, proxy_blk.npages);
21}

23void mm_copy_pages(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t n_pages)
24{
25 paging_handle_t current_handle = current_cpu->pagetable;
26 if (fvaddr >= MOS_KERNEL_START_VADDR && tvaddr >= MOS_KERNEL_START_VADDR)
27 {
28 // both addresses are in kernel space
29 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
30 }
31 else if (fvaddr >= MOS_KERNEL_START_VADDR)
32 {
33 // from kernel space to user space
34 if (to.ptr == current_handle.ptr)
35 {
36 // if the target is the current pagetable
37 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
38 }
39 else
40 {
41 // otherwise, make a proxy block
42 const vmblock_t proxy_blk = mm_map_proxy_space(to, tvaddr, n_pages);
43 memcpy((void *) proxy_blk.vaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
44 mm_unmap_proxy_space(proxy_blk);
45 }
46 }
47 else if (tvaddr >= MOS_KERNEL_START_VADDR)
48 {
49 // from user space to kernel space
50 if (from.ptr == current_handle.ptr)
51 {
52 // if the source is the current pagetable
53 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
54 }
55 else
56 {
57 // otherwise, make a proxy block
58 const vmblock_t proxy_blk = mm_map_proxy_space(from, fvaddr, n_pages);
59 memcpy((void *) tvaddr, (void *) proxy_blk.vaddr, n_pages * MOS_PAGE_SIZE);
60 mm_unmap_proxy_space(proxy_blk);
61 }
62 }
63 else
64 {
65 // both addresses are in user space
66 if (from.ptr == current_handle.ptr)
67 {
68 if (to.ptr == current_handle.ptr)
69 {
70 // although both addresses are in user space, they are both the current pagetable
71 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
72 }
73 else
74 {
75 // only the source is the current pagetable
76 const vmblock_t proxy_blk = mm_map_proxy_space(to, tvaddr, n_pages);
77 memcpy((void *) proxy_blk.vaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
78 mm_unmap_proxy_space(proxy_blk);
79 }
80 }
81 else if (to.ptr == current_handle.ptr)
82 {
83 // only the target is the current pagetable
84 const vmblock_t proxy_blk = mm_map_proxy_space(from, fvaddr, n_pages);
85 memcpy((void *) tvaddr, (void *) proxy_blk.vaddr, n_pages * MOS_PAGE_SIZE);
86 mm_unmap_proxy_space(proxy_blk);
87 }
88 else
89 {
90 // neither is the current pagetable
91 const vmblock_t proxy_blk_from = mm_map_proxy_space(from, fvaddr, n_pages);
92 const vmblock_t proxy_blk_to = mm_map_proxy_space(to, tvaddr, n_pages);
93 memcpy((void *) proxy_blk_to.vaddr, (void *) proxy_blk_from.vaddr, n_pages * MOS_PAGE_SIZE);
94 mm_unmap_proxy_space(proxy_blk_from);
95 mm_unmap_proxy_space(proxy_blk_to);
96 }
97 }
