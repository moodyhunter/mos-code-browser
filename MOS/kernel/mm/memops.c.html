13static vmblock_t zero_block;
14
15void mos_kernel_mm_init()
16{
17 // zero fill on demand (read-only, user-accessible)
18 zero_block = platform_mm_alloc_pages(current_cpu->pagetable, 1, PGALLOC_HINT_KHEAP, VM_RW);
19 memzero((void *) zero_block.vaddr, MOS_PAGE_SIZE);
20
21 liballoc_init();
22#if MOS_MM_LIBALLOC_DEBUG
23 mos_install_kpanic_hook(liballoc_dump);
24#endif
25}

27// allocate new pages for liballoc
28// !! This function is called by liballoc, not intended to be called by anyone else !!
29void *liballoc_alloc_page(size_t npages)
30{
31 if (unlikely(npages <= 0))
32 {
33 mos_warn("allocating negative or zero pages");
34 return NULL;
35 }
36
37 vmblock_t block = platform_mm_alloc_pages(current_cpu->pagetable, npages, PGALLOC_HINT_KHEAP, VM_RW);
38 MOS_ASSERT_X(block.vaddr >= MOS_ADDR_KERNEL_HEAP, "only use this function to free kernel heap pages");
39
40 if (unlikely(block.npages < npages))
41 mos_warn("failed to allocate %zu pages", npages);
42
43 return (void *) block.vaddr;
44}
45
46// free pages allocated by liballoc
47// !! This function is called by liballoc, not intended to be called by anyone else !!
48bool liballoc_free_page(void *vptr, size_t npages)
49{
50 MOS_ASSERT_X(vptr >= (void *) MOS_ADDR_KERNEL_HEAP, "only use this function to free kernel heap pages");
51
52 if (unlikely(vptr == NULL))
53 {
54 mos_warn("freeing NULL pointer");
55 return false;
56 }
57 if (unlikely(npages <= 0))
58 {
59 mos_warn("freeing negative or zero pages");
60 return false;
61 }
62
63 platform_mm_free_pages(current_cpu->pagetable, (uintptr_t) vptr, npages);
64 return true;
65}

67vmblock_t mm_alloc_zeroed_pages(paging_handle_t handle, size_t npages, pgalloc_hints hints, vm_flags flags)
68{
69 vmblock_t free_pages = platform_mm_get_free_pages(handle, npages, hints);
70 if (free_pages.npages < npages)
71 {
72 mos_warn("failed to allocate %zu pages", npages);
73 return (vmblock_t){ 0 };
74 }
75
76 // zero fill the pages
77 for (size_t i = 0; i < npages; i++)
78 {
79 // actually, zero_block is always accessible, using [handle] == using [current_cpu->pagetable] as source
80 platform_mm_copy_maps(current_cpu->pagetable, zero_block.vaddr, handle, free_pages.vaddr + i * MOS_PAGE_SIZE, 1);
81 }
82
83 // make the pages read-only (because for now, they are mapped to zero_block)
84 platform_mm_flag_pages(handle, free_pages.vaddr, npages, VM_READ | ((flags & VM_USER) ? VM_USER : 0));
85 free_pages.flags = flags; // but set the desired flags correctly
86 return free_pages;
87}

89vmblock_t mm_alloc_zeroed_pages_at(paging_handle_t handle, uintptr_t vaddr, size_t npages, vm_flags flags)
90{
91 if (platform_mm_get_is_mapped(handle, vaddr))
92 {
93 mos_warn("failed to allocate %zu pages at %p: already mapped", npages, (void *) vaddr);
94 return (vmblock_t){ 0 };
95 }
96
97 // zero fill the pages
98 for (size_t i = 0; i < npages; i++)
99 {
100 // actually, zero_block is always accessible, using [handle] == using [current_cpu->pagetable] as source
101 platform_mm_copy_maps(current_cpu->pagetable, zero_block.vaddr, handle, vaddr + i * MOS_PAGE_SIZE, 1);
102 }
103
104 // make the pages read-only (because for now, they are mapped to zero_block)
105 platform_mm_flag_pages(handle, vaddr, npages, VM_READ | ((flags & VM_USER) ? VM_USER : 0));
106 return (vmblock_t){ .vaddr = vaddr, .npages = npages, .flags = flags };
107}

109vmblock_t mm_map_proxy_space(paging_handle_t src, uintptr_t srcvaddr, size_t npages)
110{
111 const vmblock_t proxy_blk = platform_mm_get_free_pages(current_cpu->pagetable, npages, PGALLOC_HINT_KHEAP);
112 platform_mm_copy_maps(src, srcvaddr, current_cpu->pagetable, proxy_blk.vaddr, npages);
113 return proxy_blk;
114}
115
116void mm_unmap_proxy_space(vmblock_t proxy_blk)
117{
118 platform_mm_unmap_pages(current_cpu->pagetable, proxy_blk.vaddr, proxy_blk.npages);
119}

121void mm_copy_pages(paging_handle_t from, uintptr_t fvaddr, paging_handle_t to, uintptr_t tvaddr, size_t n_pages)
122{
123 paging_handle_t current_handle = current_cpu->pagetable;
124 if (fvaddr >= MOS_KERNEL_START_VADDR && tvaddr >= MOS_KERNEL_START_VADDR)
125 {
126 // both addresses are in kernel space
127 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
128 }
129 else if (fvaddr >= MOS_KERNEL_START_VADDR)
130 {
131 // from kernel space to user space
132 if (to.ptr == current_handle.ptr)
133 {
134 // if the target is the current pagetable
135 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
136 }
137 else
138 {
139 // otherwise, make a proxy block
140 const vmblock_t proxy_blk = mm_map_proxy_space(to, tvaddr, n_pages);
141 memcpy((void *) proxy_blk.vaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
142 mm_unmap_proxy_space(proxy_blk);
143 }
144 }
145 else if (tvaddr >= MOS_KERNEL_START_VADDR)
146 {
147 // from user space to kernel space
148 if (from.ptr == current_handle.ptr)
149 {
150 // if the source is the current pagetable
151 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
152 }
153 else
154 {
155 // otherwise, make a proxy block
156 const vmblock_t proxy_blk = mm_map_proxy_space(from, fvaddr, n_pages);
157 memcpy((void *) tvaddr, (void *) proxy_blk.vaddr, n_pages * MOS_PAGE_SIZE);
158 mm_unmap_proxy_space(proxy_blk);
159 }
160 }
161 else
162 {
163 // both addresses are in user space
164 if (from.ptr == current_handle.ptr)
165 {
166 if (to.ptr == current_handle.ptr)
167 {
168 // although both addresses are in user space, they are both the current pagetable
169 memcpy((void *) tvaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
170 }
171 else
172 {
173 // only the source is the current pagetable
174 const vmblock_t proxy_blk = mm_map_proxy_space(to, tvaddr, n_pages);
175 memcpy((void *) proxy_blk.vaddr, (void *) fvaddr, n_pages * MOS_PAGE_SIZE);
176 mm_unmap_proxy_space(proxy_blk);
177 }
178 }
179 else if (to.ptr == current_handle.ptr)
180 {
181 // only the target is the current pagetable
182 const vmblock_t proxy_blk = mm_map_proxy_space(from, fvaddr, n_pages);
183 memcpy((void *) tvaddr, (void *) proxy_blk.vaddr, n_pages * MOS_PAGE_SIZE);
184 mm_unmap_proxy_space(proxy_blk);
185 }
186 else
187 {
188 // neither is the current pagetable
189 const vmblock_t proxy_blk_from = mm_map_proxy_space(from, fvaddr, n_pages);
190 const vmblock_t proxy_blk_to = mm_map_proxy_space(to, tvaddr, n_pages);
191 memcpy((void *) proxy_blk_to.vaddr, (void *) proxy_blk_from.vaddr, n_pages * MOS_PAGE_SIZE);
192 mm_unmap_proxy_space(proxy_blk_from);
193 mm_unmap_proxy_space(proxy_blk_to);
194 }
195 }
