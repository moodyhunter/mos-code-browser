7void io_init(io_t *io, io_flags_t flags, size_t size, const io_op_t *ops)
8{
9 io->flags = flags;
10 io->size = size;
11 io->ops = ops;
12}
13
14void io_ref(io_t *io)
15{
16 mos_debug("io_ref(%p)", (void *) io);
17 io->refcount.atomic++;
18}
19
20void io_unref(io_t *io)
21{
22 mos_debug("io_unref(%p)", (void *) io);
23
24 if (unlikely(io->closed))
25 {
26 mos_warn("io_close: %p is already closed", (void *) io);
27 return;
28 }
29
30 if (io->refcount.atomic > 0)
31 return;
32
33 if (unlikely(!io->ops->close))
34 {
35 mos_warn("io_close: no close function");
36 return;
37 }
38
39 io->closed = true;
40 io->ops->close(io);
41}
42
43size_t io_read(io_t *io, void *buf, size_t count)
44{
45 if (!(io->flags & IO_READABLE))
46 {
47 pr_info2("io_read: %p is not readable\n", (void *) io);
48 return 0;
49 }
50
51 if (unlikely(!io->ops->read))
52 {
53 mos_warn_once("io_read: no read function");
54 return 0;
55 }
56 return io->ops->read(io, buf, count);
57}
58
59size_t io_write(io_t *io, const void *buf, size_t count)
60{
61 if (!(io->flags & IO_WRITABLE))
62 {
63 pr_info2("io_write: %p is not writable\n", (void *) io);
64 return 0;
65 }
66 if (unlikely(!io->ops->write))
67 {
68 mos_warn("io_write: no write function");
69 return 0;
70 }
71 return io->ops->write(io, buf, count);
72}
