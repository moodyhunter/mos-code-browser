8void io_init(io_t *io, io_flags_t flags, const io_op_t *ops)
9{
10 io->flags = flags;
11 io->ops = ops;
12 io->closed = false;
13 refcount_zero(&io->refcount);
14}
15
16io_t *io_ref(io_t *io)
17{
18 mos_debug("io_ref(%p)", (void *) io);
19 if (unlikely(!io))
20 {
21 mos_warn("io is NULL");
22 return NULL;
23 }
24
25 if (unlikely(io->closed))
26 {
27 mos_warn("io_write: %p is already closed", (void *) io);
28 return 0;
29 }
30
31 refcount_inc(&io->refcount);
32 return io;
33}
34
35void io_unref(io_t *io)
36{
37 mos_debug("io_unref(%p)", (void *) io);
38 if (unlikely(!io))
39 {
40 mos_warn("io is NULL");
41 return;
42 }
43
44 if (unlikely(io->closed))
45 {
46 mos_warn("%p is already closed", (void *) io);
47 return;
48 }
49
50 if (refcount_get(&io->refcount) > 0)
51 {
52 refcount_dec(&io->refcount);
53 return;
54 }
55
56 if (unlikely(!io->ops->close))
57 {
58 mos_warn("no close function");
59 return;
60 }
61
62 io->closed = true;
63 io->ops->close(io);
64}
65
66size_t io_read(io_t *io, void *buf, size_t count)
67{
68 if (unlikely(!io))
69 {
70 mos_warn("io is NULL");
71 return 0;
72 }
73 if (unlikely(io->closed))
74 {
75 mos_warn("%p is already closed", (void *) io);
76 return 0;
77 }
78 if (!(io->flags & IO_READABLE))
79 {
80 pr_info2("%p is not readable\n", (void *) io);
81 return 0;
82 }
83 if (unlikely(!io->ops->read))
84 {
85 mos_warn_once("no read function");
86 return 0;
87 }
88 return io->ops->read(io, buf, count);
89}
90
91size_t io_write(io_t *io, const void *buf, size_t count)
92{
93 if (unlikely(!io))
94 {
95 mos_warn("io is NULL");
96 return 0;
97 }
98 if (unlikely(io->closed))
99 {
100 mos_warn("%p is already closed", (void *) io);
101 return 0;
102 }
103 if (!(io->flags & IO_WRITABLE))
104 {
105 pr_info2("%p is not writable\n", (void *) io);
106 return 0;
107 }
108 if (unlikely(!io->ops->write))
109 {
110 mos_warn_once("no write function");
111 return 0;
112 }
113 return io->ops->write(io, buf, count);
114}
