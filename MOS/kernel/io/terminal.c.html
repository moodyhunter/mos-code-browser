10static size_t terminal_io_read(io_t *io, void *buffer, size_t size)
11{
12 terminal_t *terminal = container_of(io, terminal_t, io);
13 switch (terminal->type)
14 {
15 case TERM_TYPE_CONSOLE: return console_read(terminal->console, buffer, size);
16 case TERM_TYPE_PIPE: return io_read(terminal->pipe.read, buffer, size);
17 default: MOS_UNREACHABLE();
18 }
19}
20
21static size_t terminal_io_write(io_t *io, const void *buffer, size_t size)
22{
23 terminal_t *terminal = container_of(io, terminal_t, io);
24 switch (terminal->type)
25 {
26 case TERM_TYPE_CONSOLE: return console_write(terminal->console, buffer, size);
27 case TERM_TYPE_PIPE: return io_write(terminal->pipe.write, buffer, size);
28 default: MOS_UNREACHABLE();
29 }
30}
31
32static void terminal_io_close(io_t *io)
33{
34 terminal_t *terminal = container_of(io, terminal_t, io);
35 switch (terminal->type)
36 {
37 case TERM_TYPE_CONSOLE: break;
38 case TERM_TYPE_PIPE:
39 {
40 io_unref(terminal->pipe.read);
41 io_unref(terminal->pipe.write);
42 }
43 break;
44 }
45 kfree(terminal);
46}
47
48static io_op_t terminal_io_op = {
49 .read = terminal_io_read,
50 .write = terminal_io_write,
51 .close = terminal_io_close,
52};
53
54terminal_t *terminal_create_console(console_t *console)
55{
56 if (console == NULL)
57 {
58 mos_warn("console is NULL");
59 return NULL;
60 }
61 terminal_t *terminal = kcalloc(1, sizeof(terminal_t));
62 terminal->type = TERM_TYPE_CONSOLE;
63 terminal->console = console;
64 io_init(&terminal->io, IO_READABLE | IO_WRITABLE, -1, &terminal_io_op);
65 return terminal;
66}
67
68terminal_t *terminal_create_pipe(io_t *read, io_t *write)
69{
70 terminal_t *terminal = kcalloc(1, sizeof(terminal_t));
71 terminal->type = TERM_TYPE_PIPE;
72 terminal->pipe.read = io_ref(read);
73 terminal->pipe.write = io_ref(write);
74 io_init(&terminal->io, IO_READABLE | IO_WRITABLE, -1, &terminal_io_op);
75 return terminal;
76}
