8static cmdline_option_t *cmdline_get_option(const char *option_name)
9{
10 MOS_ASSERT(mos_cmdline);
11 for (u32 i = 0; i < mos_cmdline->options_count; i++)
12 {
13 if (strcmp(mos_cmdline->options[i]->name, option_name) == 0)
14 return mos_cmdline->options[i];
15 }
16 return NULL;
17}

19void invoke_setup_functions(cmdline_t *cmdline)
20{
21 extern const setup_func_t __MOS_SETUP_START[]; // defined in linker script
22 extern const setup_func_t __MOS_SETUP_END[];
23
24 for (const setup_func_t *func = __MOS_SETUP_START; func < __MOS_SETUP_END; func++)
25 {
26 cmdline_option_t *option = cmdline_get_option(func->param);
27 if (unlikely(!option))
28 {
29 pr_info2("no option given for '%s'", func->param);
30 // mos_debug(init, "no option given for '%s'", opt->param);
31 continue; // no option given for this setup function, skip
32 }
33
34 pr_info2("invoking setup function for '%s'", func->param);
35 const bool result = func->setup_fn(option->argc, option->argv);
36 if (unlikely(!result))
37 {
38 pr_warn("setup function for '%s' failed", func->param);
39 continue;
40 }
41
42 cmdline_remove_option(cmdline, option->name);
43 }
44}
