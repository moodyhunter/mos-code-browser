9void mutex_try_acquire_may_reschedule(bool *mutex)
10{
11 if ((uintptr_t) mutex < MOS_KERNEL_START_VADDR)
12 {
13 pr_emerg("mutex_try_acquire_may_reschedule: tid %d tried to acquire a lock at " PTR_FMT " which is in user space", current_thread->tid, (uintptr_t) mutex);
14 }
15
16 if (__sync_bool_compare_and_swap(mutex, MUTEX_UNLOCKED, MUTEX_LOCKED)) // if the mutex is unlocked, lock it
17 {
18 pr_info2("mutex_acquire: tid %d acquires a free lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
19 return;
20 }
21
22 pr_info2("mutex_acquire: tid %d blocks on a locked lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
23
24 // TODO: this uses a user pointer, which will DEFINITELY cause problems when we have multiple processes
25 wait_condition_t *wc = wc_wait_for_mutex(mutex);
26 reschedule_for_wait_condition(wc);
27 pr_info2("mutex_acquire: tid %d unblocks and acquires a lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
28 __atomic_store_n(mutex, MUTEX_LOCKED, __ATOMIC_SEQ_CST);
29}
30
31bool mutex_release(bool *lock)
32{
33
34 if (__atomic_load_n(lock, __ATOMIC_SEQ_CST) == MUTEX_UNLOCKED)
35 {
36 pr_warn("mutex_release: tid %d tried to release a lock at " PTR_FMT " but it was already unlocked", current_thread->tid, (uintptr_t) lock);
37 return false;
38 }
39
40 if (__sync_bool_compare_and_swap(lock, MUTEX_LOCKED, MUTEX_UNLOCKED)) // if the mutex is locked, unlock it
41 {
42 pr_info2("mutex_release: tid %d releases a lock at " PTR_FMT, current_thread->tid, (uintptr_t) lock);
43 return true;
44 }
45
46 pr_warn("mutex_release: tid %d tried to release a lock at " PTR_FMT " but it was already unlocked", current_thread->tid, (uintptr_t) lock);
47 return false;
48}
