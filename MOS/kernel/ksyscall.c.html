15void define_syscall(panic)(void)
16{
17 if (current_process->effective_uid == 0)
18 mos_panic("Kernel panic called by syscall from process %d (%s), thread %d", current_process->pid, current_process->name, current_thread->tid);
19 else
20 mos_warn("only root can panic");
21}

23fd_t define_syscall(file_open)(const char *path, file_open_flags flags)
24{
25 if (path == NULL)
26 return -1;
27
28 file_t *f = vfs_open(path, flags);
29 if (!f)
30 return -1;
31 return process_attach_fd(current_process, &f->io);
32}
33
34bool define_syscall(file_stat)(const char *path, file_stat_t *stat)
35{
36 if (path == NULL || stat == NULL)
37 return false;
38
39 return vfs_stat(path, stat);
40}

42size_t define_syscall(io_read)(fd_t fd, void *buf, size_t count, size_t offset)
43{
44 if (fd < 0 || buf == NULL)
45 return 0;
46 if (offset)
47 mos_warn("offset is not supported yet");
48 return io_read(current_process->files[fd], buf, count);
49}

51size_t define_syscall(io_write)(fd_t fd, const void *buf, size_t count, size_t offset)
52{
53 if (fd < 0 || buf == NULL)
54 return 0;
55 if (offset)
56 mos_warn("offset is not supported yet");
57 return io_write(current_process->files[fd], buf, count);
58}

60bool define_syscall(io_close)(fd_t fd)
61{
62 if (fd < 0)
63 return false;
64 process_detach_fd(current_process, fd);
65 return true;
66}

68noreturn void define_syscall(exit)(u32 exit_code)
69{
70 int pid = current_process->pid;
71 if (unlikely(pid == 1))
72 mos_panic("init process exited with code %d", exit_code);
73
74 pr_info2("Kernel syscall exit called with code %d from pid %d", exit_code, pid);
75 process_handle_exit(current_process, exit_code);
76 jump_to_scheduler();
77 MOS_UNREACHABLE();
78}

80void define_syscall(yield_cpu)(void)

82 jump_to_scheduler();

85pid_t define_syscall(fork)(void)
86{
87 process_t *parent = current_process;
88 process_t *child = process_handle_fork(parent);
89 if (child == NULL)
90 return 0;
91
92 return current_process == child ? 0 : child->pid; // return 0 for child, pid for parent
93}

103pid_t define_syscall(get_pid)()
104{

109pid_t define_syscall(get_parent_pid)()
110{
111 MOS_ASSERT(current_process && current_process->parent);

115pid_t define_syscall(spawn)(const char *path, int argc, const char *const argv[])
116{
117 MOS_UNUSED(argc);
118 MOS_UNUSED(argv);
119 process_t *process = process_create_from_elf(path, current_process, current_process->effective_uid);
120 if (process == NULL)
121 return -1;
122
123 return process->pid;
124}

126tid_t define_syscall(create_thread)(const char *name, thread_entry_t entry, void *arg)
127{
128 MOS_UNUSED(name);
129 MOS_ASSERT(current_thread);
130 thread_t *thread = thread_new(current_process, THREAD_FLAG_USERMODE, entry, arg);
131 if (thread == NULL)
132 return -1;
133
134 return thread->tid;
135}

137tid_t define_syscall(get_tid)()
138{
139 MOS_ASSERT(current_thread);
140 return current_thread->tid;
141}

143noreturn void define_syscall(thread_exit)(void)
144{
145 MOS_ASSERT(current_thread);
146 pr_info2("Kernel syscall thread_exit called from tid %d", current_thread->tid);
147 thread_handle_exit(current_thread);
148 jump_to_scheduler();
149 MOS_UNREACHABLE();
150}
