36size_t define_ksyscall(io_write)(fd_t fd, const void *buf, size_t count, size_t offset)
37{
38 thread_t *p = current_thread;
39 return io_write(p->owner->file_table[fd], buf, count);
40 MOS_UNUSED(offset);
41 return 0;
42}
