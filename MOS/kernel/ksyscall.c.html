12fd_t define_ksyscall(file_open)(const char *path, file_open_flags flags)
13{
14 file_t *f = vfs_open(path, flags);
15 if (!f)
16 return -1;
17 return process_attach_fd(current_process, &f->io);
18}
19
20bool define_ksyscall(file_stat)(const char *path, file_stat_t *stat)
21{
22 return vfs_stat(path, stat);
23}

25size_t define_ksyscall(io_read)(fd_t fd, void *buf, size_t count, size_t offset)
26{
27 if (offset)
28 mos_warn("offset is not supported yet");
29 return io_read(current_process->files[fd], buf, count);
30}
31
32size_t define_ksyscall(io_write)(fd_t fd, const void *buf, size_t count, size_t offset)
33{
34 if (offset)
35 mos_warn("offset is not supported yet");
36 return io_write(current_process->files[fd], buf, count);
37}

39bool define_ksyscall(io_close)(fd_t fd)
40{
41 process_detach_fd(current_process, fd);
42 return true;
43}

45noreturn void define_ksyscall(panic)(void)
46{
47 mos_panic("Kernel panic called by syscall");
48}

50noreturn void define_ksyscall(exit)(u32 exit_code)
51{
52 int pid = current_process->pid;
53 if (unlikely(pid == 1))
54 mos_panic("init process exited with code %d", exit_code);
55
56 pr_info("Kernel syscall exit called with code %d from pid %d", exit_code, pid);
57 process_handle_exit(current_process, exit_code);
58 jump_to_scheduler();
59 MOS_UNREACHABLE();
60}
61
62void define_ksyscall(yield_cpu)(void)
63{
64 jump_to_scheduler();
65}
66
67pid_t define_ksyscall(fork)(void)
68{
69 process_t *parent = current_process;
70 process_t *child = process_handle_fork(parent);
71 if (child == NULL)
72 return 0;
73
74 return current_process == child ? 0 : child->pid; // return 0 for child, pid for parent
75}
