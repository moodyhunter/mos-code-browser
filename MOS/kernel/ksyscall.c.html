19void define_syscall(poweroff)(bool reboot, u32 magic)
20{
21#define POWEROFF_MAGIC MOS_FOURCC('G', 'B', 'y', 'e')
22 if (current_process->effective_uid == 0)
23 {
24 if (magic != POWEROFF_MAGIC)
25 mos_warn("poweroff syscall called with wrong magic number (0x%x)", magic);
26 if (!reboot)
27 platform_shutdown();
28 else
29 mos_warn("reboot is not implemented yet");
30 }
31 else
32 mos_warn("poweroff syscall called by non-root user");
33}

92noreturn void define_syscall(exit)(u32 exit_code)
93{
94 int pid = current_process->pid;
95 if (unlikely(pid == 1))
96 mos_panic("init process exited with code %d", exit_code);
97
98 pr_info2("Kernel syscall exit called with code %d from pid %d", exit_code, pid);
99 process_handle_exit(current_process, exit_code);
100 reschedule();
101 MOS_UNREACHABLE();
102}

109pid_t define_syscall(fork)(void)
110{
111 process_t *parent = current_process;
112 process_t *child = process_handle_fork(parent);
113 if (child == NULL)
114 return 0;
115
116 return current_process == child ? 0 : child->pid; // return 0 for child, pid for parent
117}

176uintptr_t define_syscall(heap_control)(heap_control_op op, uintptr_t arg)
177{
178 MOS_ASSERT(current_process);
179 process_t *process = current_process;
180
181 proc_vmblock_t *block = NULL;
182 for (int i = 0; i < process->mmaps_count; i++)
183 {
184 if (process->mmaps[i].type == VMTYPE_HEAP)
185 {
186 block = &process->mmaps[i];
187 break;
188 }
189 }
190
191 if (block == NULL)
192 {
193 mos_warn("heap_control called but no heap block found");
194 return 0;
195 }
196
197 switch (op)
198 {
199 case HEAP_GET_BASE: return block->vm.vaddr;
200 case HEAP_GET_TOP: return block->vm.vaddr + block->vm.npages * MOS_PAGE_SIZE;
201 case HEAP_SET_TOP:
202 {
203 if (arg < block->vm.vaddr)
204 {
205 mos_warn("heap_control: new top is below heap base");
206 return 0;
207 }
208
209 if (arg % MOS_PAGE_SIZE)
210 {
211 mos_warn("heap_control: new top is not page-aligned");
212 return 0;
213 }
214
215 if (arg == block->vm.vaddr + block->vm.npages * MOS_PAGE_SIZE)
216 return 0; // no change
217
218 if (arg < block->vm.vaddr + block->vm.npages * MOS_PAGE_SIZE)
219 {
220 mos_warn("heap_control: shrinking heap not supported yet");
221 return 0;
222 }
223
224 return process_grow_heap(process, (arg - block->vm.vaddr) / MOS_PAGE_SIZE);
225 }
226 case HEAP_GET_SIZE: return block->vm.npages * MOS_PAGE_SIZE;
227 case HEAP_GROW_PAGES:
228 {
229 // arg is the number of pages to grow
230 // some bad guy would pass a huge value here :)
231 return process_grow_heap(process, arg);
232 }
233 default: mos_warn("heap_control: unknown op %d", op); return 0;
234 }
235}

237bool define_syscall(wait_for_thread)(tid_t tid)
238{
239 MOS_ASSERT(current_thread);
240 thread_t *target = thread_get(tid);
241 if (target == NULL)
242 return false;
243
244 if (target->owner != current_process)
245 {
246 pr_warn("wait_for_thread(%d) from process %d (%s) but thread belongs to process %d (%s)", tid, current_process->pid, current_process->name, target->owner->pid,
247 target->owner->name);
248 return false;
249 }
250
251 if (target->state == THREAD_STATE_DEAD)
252 {
253 return true; // thread is already dead, no need to wait
254 }
255
256 wait_condition_t *wc = wc_wait_for_thread(target);
257 reschedule_for_wait_condition(wc);
258 return true;
259}

261bool define_syscall(mutex_acquire)(bool *mutex)
262{
263 MOS_ASSERT(current_thread);
264
265 if (__sync_bool_compare_and_swap(mutex, MUTEX_UNLOCKED, MUTEX_LOCKED)) // if the mutex is unlocked, lock it
266 {
267 pr_info2("mutex_acquire: tid %d acquires a free lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
268 return true;
269 }
270
271 pr_info2("mutex_acquire: tid %d blocks on a locked lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
272
273 // TODO: this uses a user pointer, which will DEFINITELY cause problems when we have multiple processes
274 wait_condition_t *wc = wc_wait_for_mutex(mutex);
275 reschedule_for_wait_condition(wc);
276 pr_info2("mutex_acquire: tid %d unblocks and acquires a lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
277 __atomic_store_n(mutex, MUTEX_LOCKED, __ATOMIC_SEQ_CST);
278 return true;
279}
280
281bool define_syscall(mutex_release)(bool *mutex)
282{
283 MOS_ASSERT(current_thread);
284
285 if (__atomic_load_n(mutex, __ATOMIC_SEQ_CST) == MUTEX_UNLOCKED)
286 {
287 pr_warn("mutex_release: tid %d tried to release a lock at " PTR_FMT " but it was already unlocked", current_thread->tid, (uintptr_t) mutex);
288 return false;
289 }
290
291 if (__sync_bool_compare_and_swap(mutex, MUTEX_LOCKED, MUTEX_UNLOCKED)) // if the mutex is locked, unlock it
292 {
293 pr_info2("mutex_release: tid %d releases a lock at " PTR_FMT, current_thread->tid, (uintptr_t) mutex);
294 return true;
295 }
296
297 return false;
298}
