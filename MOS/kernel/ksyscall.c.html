14void define_ksyscall(panic)(void)
15{
16 if (current_process->effective_uid == 0)
17 mos_panic("Kernel panic called by syscall from process %d (%s), thread %d", current_process->pid, current_process->name, current_thread->tid);
18 else
19 mos_warn("only root can panic");
20}

22fd_t define_ksyscall(file_open)(const char *path, file_open_flags flags)
23{
24 if (path == NULL)
25 return -1;
26
27 file_t *f = vfs_open(path, flags);
28 if (!f)
29 return -1;
30 return process_attach_fd(current_process, &f->io);
31}

33bool define_ksyscall(file_stat)(const char *path, file_stat_t *stat)
34{
35 if (path == NULL || stat == NULL)
36 return false;
37
38 return vfs_stat(path, stat);
39}

41size_t define_ksyscall(io_read)(fd_t fd, void *buf, size_t count, size_t offset)
42{
43 if (fd < 0 || buf == NULL)
44 return 0;
45 if (offset)
46 mos_warn("offset is not supported yet");
47 return io_read(current_process->files[fd], buf, count);
48}
49
50size_t define_ksyscall(io_write)(fd_t fd, const void *buf, size_t count, size_t offset)
51{
52 if (fd < 0 || buf == NULL)
53 return 0;
54 if (offset)
55 mos_warn("offset is not supported yet");
56 return io_write(current_process->files[fd], buf, count);
57}

59bool define_ksyscall(io_close)(fd_t fd)
60{
61 if (fd < 0)
62 return false;
63 process_detach_fd(current_process, fd);
64 return true;
65}

67noreturn void define_ksyscall(exit)(u32 exit_code)
68{
69 int pid = current_process->pid;
70 if (unlikely(pid == 1))
71 mos_panic("init process exited with code %d", exit_code);
72
73 pr_info("Kernel syscall exit called with code %d from pid %d", exit_code, pid);
74 process_handle_exit(current_process, exit_code);
75 jump_to_scheduler();
76 MOS_UNREACHABLE();
77}

79void define_ksyscall(yield_cpu)(void)
80{
81 jump_to_scheduler();

84pid_t define_ksyscall(fork)(void)
85{
86 process_t *parent = current_process;
87 process_t *child = process_handle_fork(parent);
88 if (child == NULL)
89 return 0;
90
91 return current_process == child ? 0 : child->pid; // return 0 for child, pid for parent
92}

94pid_t define_ksyscall(exec)(const char *path, const char *const argv[])

96 MOS_UNUSED(path);

102pid_t define_ksyscall(get_pid)()
103{
104 MOS_ASSERT(current_process);
105 return current_process->pid;
106}

108pid_t define_ksyscall(get_parent_pid)()
109{
110 MOS_ASSERT(current_process && current_process->parent);
111 return current_process->parent->pid;
112}

114pid_t define_ksyscall(spawn)(const char *path, int argc, const char *const argv[])
115{
116 MOS_UNUSED(argc);
117 MOS_UNUSED(argv);
118 process_t *process = process_create_from_elf(path, current_process, current_process->effective_uid);
119 if (process == NULL)
120 return -1;
121
122 return process->pid;
123}
