11elf_verify_result elf_verify_header(elf_header_t *header)
12{
13 if (header->identity.magic[0] != '\x7f')
14 return ELF_VERIFY_INVALID_MAGIC;
15
16 if (strncmp(&header->identity.magic[1], "ELF", 3) != 0)
17 return ELF_VERIFY_INVALID_MAGIC_ELF;
18
19 if (header->identity.bits != ELF_BITS_MOS_DEFAULT)
20 return ELF_VERIFY_INVALID_BITS;
21
22 if (header->identity.endianness != ELF_ENDIANNESS_MOS_DEFAULT)
23 return ELF_VERIFY_INVALID_ENDIAN;
24
25 if (header->identity.version != ELF_VERSION_CURRENT)
26 return ELF_VERIFY_INVALID_VERSION;
27
28 if (header->identity.osabi != ELF_OSABI_NONE)
29 return ELF_VERIFY_INVALID_OSABI;
30
31 return ELF_VERIFY_OK;
32}
