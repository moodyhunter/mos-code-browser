12static_assert(sizeof(elf_header_t) == (MOS_32BITS ? 0x34 : 0x40), "elf_header has wrong size");
13static_assert(sizeof(elf_program_hdr_t) == (MOS_32BITS ? 0x20 : 0x38), "elf_program_header has wrong size");
14static_assert(sizeof(elf_section_hdr_t) == (MOS_32BITS ? 0x28 : 0x40), "elf_section_header has wrong size");

16elf_verify_result elf_verify_header(elf_header_t *header)
17{
18 if (header->identity.magic[0] != '\x7f')
19 return ELF_VERIFY_INVALID_MAGIC;
20
21 if (strncmp(&header->identity.magic[1], "ELF", 3) != 0)
22 return ELF_VERIFY_INVALID_MAGIC_ELF;
23
24 if (header->identity.bits != ELF_BITS_MOS_DEFAULT)
25 return ELF_VERIFY_INVALID_BITS;
26
27 if (header->identity.endianness != ELF_ENDIANNESS_MOS_DEFAULT)
28 return ELF_VERIFY_INVALID_ENDIAN;
29
30 if (header->identity.version != ELF_VERSION_CURRENT)
31 return ELF_VERIFY_INVALID_VERSION;
32
33 if (header->identity.osabi != ELF_OSABI_NONE)
34 return ELF_VERIFY_INVALID_OSABI;
35
36 return ELF_VERIFY_OK;
37}

39process_t *create_elf_process(const char *path, uid_t effective_uid)
40{
41 file_t *f = vfs_open(path, FILE_OPEN_READ);
42 if (!f)
43 {
44 mos_warn("failed to open '%s'", path);
45 goto bail_out;
46 }
47
48 size_t npage_required = f->io.size / MOS_PAGE_SIZE + 1;
49 char *const buf = kpage_alloc(npage_required, PGALLOC_NONE);
50
51 size_t size = io_read(&f->io, buf, f->io.size);
52 MOS_ASSERT_X(size == f->io.size, "failed to read init");
53
54 elf_header_t *elf = (elf_header_t *) buf;
55 if (elf->object_type != ELF_OBJTYPE_EXECUTABLE)
56 {
57 pr_emerg("'%s' is not an executable", path);
58 goto bail_out;
59 }
60
61 elf_verify_result verify_result = elf_verify_header(elf);
62 if (verify_result != ELF_VERIFY_OK)
63 {
64 pr_emerg("failed to verify ELF header for '%s', result: %d", path, (int) verify_result);
65 goto bail_out;
66 }
67
68 process_t *proc = allocate_process(NULL, effective_uid, f->fsnode->name, (thread_entry_t) elf->entry_point, NULL);
69
70 for (int i = 0; i < elf->ph.count; i++)
71 {
72 elf_program_hdr_t *ph = (elf_program_hdr_t *) (buf + elf->ph_offset + i * elf->ph.entry_size);
73 mos_debug("program header %d: %c%c%c%s at " PTR_FMT, i,
74 (elf_program_header_flags) ph->p_flags & ELF_PH_F_R ? 'r' : '-', //
75 (elf_program_header_flags) ph->p_flags & ELF_PH_F_W ? 'w' : '-', //
76 (elf_program_header_flags) ph->p_flags & ELF_PH_F_X ? 'x' : '-', //
77 ph->header_type == ELF_PH_T_LOAD ? " (load)" : "", //
78 ph->vaddr //
79 );
80
81 if (!(ph->header_type & ELF_PH_T_LOAD))
82 continue; // skip non-loadable segments
83
84 vm_flags map_flags = ( //
85 (ph->p_flags & ELF_PH_F_R ? VM_READ : 0) | //
86 (ph->p_flags & ELF_PH_F_W ? VM_WRITE : 0) | //
87 (ph->p_flags & ELF_PH_F_X ? VM_EXEC : 0) | //
88 VM_USERMODE //
89 );
90
91 mos_platform->mm_map_kvaddr( //
92 proc->pagetable, //
93 ALIGN_DOWN_TO_PAGE(ph->vaddr), //
94 (uintptr_t) buf + ph->data_offset, //
95 ALIGN_UP_TO_PAGE(ph->segsize_in_mem) / MOS_PAGE_SIZE, //
96 map_flags //
97 );
98 }
99
100#if MOS_DEBUG
101 const char *const strtab = buf + ((elf_section_hdr_t *) (buf + elf->sh_offset + elf->sh_strtab_index * elf->sh.entry_size))->sh_offset;
102 for (int i = 0; i < elf->sh.count; i++)
103 {
104 elf_section_hdr_t *sh = (elf_section_hdr_t *) (buf + elf->sh_offset + i * elf->sh.entry_size);
105 pr_info2("section %2d: %s", i, strtab + sh->name_index);
106 }
107#endif
108
109 process_attach_fd(proc, &f->io);
110
111 return proc;
112
113bail_out:
114 if (f)
115 io_close(&f->io);
116 return NULL;
117}
