19elf_verify_result elf_verify_header(elf_header_t *header)
20{
21 if (header->identity.magic[0] != '\x7f')
22 return ELF_VERIFY_INVALID_MAGIC;
23
24 if (strncmp(&header->identity.magic[1], "ELF", 3) != 0)
25 return ELF_VERIFY_INVALID_MAGIC_ELF;
26
27 if (header->identity.bits != ELF_BITS_MOS_DEFAULT)
28 return ELF_VERIFY_INVALID_BITS;
29
30 if (header->identity.endianness != ELF_ENDIANNESS_MOS_DEFAULT)
31 return ELF_VERIFY_INVALID_ENDIAN;
32
33 if (header->identity.version != ELF_VERSION_CURRENT)
34 return ELF_VERIFY_INVALID_VERSION;
35
36 if (header->identity.osabi != ELF_OSABI_NONE)
37 return ELF_VERIFY_INVALID_OSABI;
38
39 return ELF_VERIFY_OK;
40}

42process_t *elf_create_process(const char *path, process_t *parent, terminal_t *term, uid_t effective_uid)
43{
44 file_t *f = vfs_open(path, FILE_OPEN_READ);
45 if (!f)
46 {
47 mos_warn("failed to open '%s'", path);
48 goto bail_out_1;
49 }
50
51 size_t npage_required = ALIGN_UP_TO_PAGE(f->io.size) / MOS_PAGE_SIZE;
52 const vmblock_t buf_block = platform_mm_alloc_pages(current_cpu->pagetable, npage_required, PGALLOC_HINT_KHEAP, VM_READ | VM_WRITE);
53 char *const buf = (char *) buf_block.vaddr;
54
55 size_t size = io_read(&f->io, buf, f->io.size);
56 MOS_ASSERT_X(size == f->io.size, "failed to read entire file '%s'", path);
57
58 elf_header_t *elf = (elf_header_t *) buf;
59 elf_program_hdr_t **ph_list = kcalloc(sizeof(elf_program_hdr_t *), elf->ph.count);
60
61 if (elf->object_type != ELF_OBJTYPE_EXECUTABLE)
62 {
63 pr_emerg("'%s' is not an executable", path);
64 goto bail_out;
65 }
66
67 elf_verify_result verify_result = elf_verify_header(elf);
68 if (verify_result != ELF_VERIFY_OK)
69 {
70 pr_emerg("failed to verify ELF header for '%s', result: %d", path, (int) verify_result);
71 goto bail_out;
72 }
73
74 process_t *proc = process_new(parent, effective_uid, f->fsnode->name, term, (thread_entry_t) elf->entry_point, NULL);
75
76 for (int i = 0; i < elf->ph.count; i++)
77 {
78 elf_program_hdr_t *ph = (elf_program_hdr_t *) (buf + elf->ph_offset + i * elf->ph.entry_size);
79 mos_debug("program header %d: %c%c%c%s at " PTR_FMT, i,
80 (elf_program_header_flags) ph->p_flags & ELF_PH_F_R ? 'r' : '-', //
81 (elf_program_header_flags) ph->p_flags & ELF_PH_F_W ? 'w' : '-', //
82 (elf_program_header_flags) ph->p_flags & ELF_PH_F_X ? 'x' : '-', //
83 ph->header_type == ELF_PH_T_LOAD ? " (load)" : "", //
84 ph->vaddr //
85 );
86
87 // skip non-loadable segments
88 if (!(ph->header_type & ELF_PH_T_LOAD))
89 continue;
90
91 ph_list[i] = ph;
92 }
93
94 // previous_sh is used to deterimine the ending address of the previous section
95 //! assuming that the sections are sorted by their address (which should be?)
96 elf_section_hdr_t *previouse_sh = NULL;
97
98 const char *const strtab = buf + ((elf_section_hdr_t *) (buf + elf->sh_offset + elf->sh_strtab_index * elf->sh.entry_size))->sh_offset;
99 for (int sh_i = 0; sh_i < elf->sh.count; sh_i++)
100 {
101 elf_section_hdr_t *sh = (elf_section_hdr_t *) (buf + elf->sh_offset + sh_i * elf->sh.entry_size);
102 const char *const name = &strtab[sh->name_index];
103 mos_debug("elf section %2d: %s", sh_i, name);
104
105 if (sh->sh_size == 0)
106 continue; // skip empty sections
107
108 const uintptr_t section_inmem_addr = sh->sh_addr;
109 bool is_loadable = false;
110
111 vm_flags map_flags = VM_USER;
112 for (int ph_i = 0; ph_i < elf->ph.count; ph_i++)
113 {
114 elf_program_hdr_t *ph = ph_list[ph_i];
115 if (!ph)
116 continue;
117
118 if (section_inmem_addr >= ph->vaddr && section_inmem_addr < ph->vaddr + ph->segsize_in_mem)
119 {
120 MOS_ASSERT_X(!is_loadable, "section %d is loadable in multiple program headers", sh_i);
121 mos_debug("section %d (%s), header %d, file offset " PTR_FMT " (%zu bytes) -> mem " PTR_FMT, sh_i, name, ph_i, sh->sh_offset, sh->sh_size,
122 section_inmem_addr);
123 is_loadable = true;
124 map_flags |= ((ph->p_flags & ELF_PH_F_R ? VM_READ : 0) | //
125 (ph->p_flags & ELF_PH_F_W ? VM_WRITE : 0) | //
126 (ph->p_flags & ELF_PH_F_X ? VM_EXEC : 0));
127 }
128 }
129
130 if (!is_loadable)
131 {
132 mos_debug("section %d is not loadable", sh_i);
133 continue;
134 }
135
136 if (unlikely(map_flags & VM_READ && map_flags & VM_WRITE && map_flags & VM_EXEC))
137 mos_warn("section %d (%s) is both readable, writable and executable", sh_i, name);
138
139 const size_t section_inmem_pages = ALIGN_UP_TO_PAGE(sh->sh_size) / MOS_PAGE_SIZE;
140
141 // find if there are any pages that are already mapped because sections may not be page-aligned
142 size_t mapped_pages_n = 0;
143 for (mapped_pages_n = 0; mapped_pages_n < section_inmem_pages; mapped_pages_n++)
144 {
145 uintptr_t page_addr = ALIGN_DOWN_TO_PAGE(section_inmem_addr) + mapped_pages_n * MOS_PAGE_SIZE;
146 if (unlikely(platform_mm_get_is_mapped(proc->pagetable, page_addr)))
147 continue;
148 break;
149 }
150
151 // if mapped_pages_n > 1, then this section must have [at least one page] that is overlapping with another section
152 MOS_ASSERT_X(mapped_pages_n == 0 || mapped_pages_n == 1, "a section is partially mapped in multiple pages, this is impossible");
153
154 const size_t pages_to_map = section_inmem_pages - mapped_pages_n;
155 const uintptr_t first_unmapped_addr = ALIGN_DOWN_TO_PAGE(section_inmem_addr) + mapped_pages_n * MOS_PAGE_SIZE;
156
157 if (mapped_pages_n == 0)
158 {
159 mos_debug("section %d is not previously mapped", sh_i);
160 }
161 else
162 {
163 // this section is previously partially (or fully) mapped, check if it's a NOBITS section and possibly memzero on its
164 // previously mapped pages
165 if (sh->header_type == ELF_SH_T_NOBITS && mapped_pages_n)
166 {
167 mos_debug("section %d is NOBITS, zeroing previously mapped area", sh_i);
168 MOS_ASSERT(previouse_sh); // It's just impossible to have a previously mapped section if this is the first section
169
170 // Memory layout:
171 // 'A'
172 // ...... ->|<- ....... previous page ............ ->|<- this page ...........
173 // .... previous section ->| gap |<- this section ...........................
174 // |<- [zero_size] ->|<- [first_unmapped_addr]
175 // 'B'
176
177 const uintptr_t A = ALIGN_UP_TO_PAGE((uintptr_t) buf + previouse_sh->sh_offset + previouse_sh->sh_size);
178 const uintptr_t zero_size = ALIGN_UP_TO_PAGE(sh->sh_addr) - sh->sh_addr;
179 const uintptr_t B = A - zero_size;
180 memzero((void *) B, zero_size);
181 }
182
183 if (mapped_pages_n == section_inmem_pages)
184 {
185 mos_debug("section %d is already fully mapped", sh_i);
186 continue;
187 }
188 else
189 {
190 mos_debug("section %d is partially mapped (%zu pages), mapping the rest (%zu pages)", sh_i, mapped_pages_n, pages_to_map);
191 }
192 }
193
194 previouse_sh = sh;
195
196 if (mapped_pages_n == section_inmem_pages)
197 continue; // all pages are already mapped
198
199 if (sh->header_type == ELF_SH_T_NOBITS)
200 {
201 vmblock_t zero_block = mm_alloc_zeroed_pages_at(proc->pagetable, first_unmapped_addr, pages_to_map, map_flags);
202 process_attach_mmap(proc, zero_block, VMTYPE_APPDATA_ZERO, MMAP_COW); // zero-fill-on-demand is CoW
203 }
204 else
205 {
206 uintptr_t file_offset = ALIGN_DOWN_TO_PAGE((uintptr_t) buf + sh->sh_offset) + mapped_pages_n * MOS_PAGE_SIZE;
207 vmblock_t block = platform_mm_copy_maps(current_cpu->pagetable, file_offset, proc->pagetable, first_unmapped_addr, pages_to_map);
208 platform_mm_flag_pages(proc->pagetable, block.vaddr, block.npages, map_flags);
209 block.flags = map_flags;
210 process_attach_mmap(proc, block, map_flags & VM_EXEC ? VMTYPE_APPCODE : VMTYPE_APPDATA, MMAP_DEFAULT);
211 }
212 }
213
214 process_attach_ref_fd(proc, &f->io); // do we need this?
215
216 // unmap the buffer from kernel pages
217 platform_mm_unmap_pages(current_cpu->pagetable, buf_block.vaddr, buf_block.npages);
218
219 return proc;
220
221bail_out:
222 if (ph_list)
223 kfree(ph_list);
224 if (buf)
225 platform_mm_free_pages(current_cpu->pagetable, buf_block.vaddr, buf_block.npages);
226bail_out_1:
227 if (f)
228 io_unref(&f->io); // close the file, we should have the file's refcount == 0 here
229
230 return NULL;
231}
