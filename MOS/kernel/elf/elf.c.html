13static_assert(sizeof(elf_header_t) == (MOS_32BITS ? 0x34 : 0x40), "elf_header has wrong size");
14static_assert(sizeof(elf_program_hdr_t) == (MOS_32BITS ? 0x20 : 0x38), "elf_program_header has wrong size");
15static_assert(sizeof(elf_section_hdr_t) == (MOS_32BITS ? 0x28 : 0x40), "elf_section_header has wrong size");

17elf_verify_result elf_verify_header(elf_header_t *header)
18{
19 if (header->identity.magic[0] != '\x7f')
20 return ELF_VERIFY_INVALID_MAGIC;
21
22 if (strncmp(&header->identity.magic[1], "ELF", 3) != 0)
23 return ELF_VERIFY_INVALID_MAGIC_ELF;
24
25 if (header->identity.bits != ELF_BITS_MOS_DEFAULT)
26 return ELF_VERIFY_INVALID_BITS;
27
28 if (header->identity.endianness != ELF_ENDIANNESS_MOS_DEFAULT)
29 return ELF_VERIFY_INVALID_ENDIAN;
30
31 if (header->identity.version != ELF_VERSION_CURRENT)
32 return ELF_VERIFY_INVALID_VERSION;
33
34 if (header->identity.osabi != ELF_OSABI_NONE)
35 return ELF_VERIFY_INVALID_OSABI;
36
37 return ELF_VERIFY_OK;
38}

40process_t *create_elf_process(const char *path, uid_t effective_uid)
41{
42 file_t *f = vfs_open(path, FILE_OPEN_READ);
43 if (!f)
44 {
45 mos_warn("failed to open '%s'", path);
46 goto bail_out_1;
47 }
48
49 size_t npage_required = f->io.size / MOS_PAGE_SIZE + 1;
50 const vmblock_t buf_block = mos_platform->mm_alloc_pages(current_cpu->pagetable, npage_required, PGALLOC_HINT_DEFAULT, VM_READ | VM_WRITE);
51 char *const buf = (char *) buf_block.vaddr;
52
53 size_t size = io_read(&f->io, buf, f->io.size);
54 MOS_ASSERT_X(size == f->io.size, "failed to read init");
55
56 elf_header_t *elf = (elf_header_t *) buf;
57 if (elf->object_type != ELF_OBJTYPE_EXECUTABLE)
58 {
59 pr_emerg("'%s' is not an executable", path);
60 goto bail_out;
61 }
62
63 elf_verify_result verify_result = elf_verify_header(elf);
64 if (verify_result != ELF_VERIFY_OK)
65 {
66 pr_emerg("failed to verify ELF header for '%s', result: %d", path, (int) verify_result);
67 goto bail_out;
68 }
69
70 process_t *proc = process_new(NULL, effective_uid, f->fsnode->name, (thread_entry_t) elf->entry_point, NULL);
71
72 for (int i = 0; i < elf->ph.count; i++)
73 {
74 elf_program_hdr_t *ph = (elf_program_hdr_t *) (buf + elf->ph_offset + i * elf->ph.entry_size);
75 mos_debug("program header %d: %c%c%c%s at " PTR_FMT, i,
76 (elf_program_header_flags) ph->p_flags & ELF_PH_F_R ? 'r' : '-', //
77 (elf_program_header_flags) ph->p_flags & ELF_PH_F_W ? 'w' : '-', //
78 (elf_program_header_flags) ph->p_flags & ELF_PH_F_X ? 'x' : '-', //
79 ph->header_type == ELF_PH_T_LOAD ? " (load)" : "", //
80 ph->vaddr //
81 );
82
83 if (!(ph->header_type & ELF_PH_T_LOAD))
84 continue; // skip non-loadable segments
85
86 vm_flags map_flags = ( //
87 (ph->p_flags & ELF_PH_F_R ? VM_READ : 0) | //
88 (ph->p_flags & ELF_PH_F_W ? VM_WRITE : 0) | //
89 (ph->p_flags & ELF_PH_F_X ? VM_EXEC : 0) | //
90 VM_USER //
91 );
92
93 // rwx
94 if (map_flags & VM_READ && map_flags & VM_WRITE && map_flags & VM_EXEC)
95 mos_warn("segment is writable, readable and executable");
96
97 vmblock_t block = mos_platform->mm_copy_maps( //
98 current_cpu->pagetable, //
99 buf_block.vaddr + ph->data_offset, //
100 proc->pagetable, //
101 ALIGN_DOWN_TO_PAGE(ph->vaddr), //
102 ALIGN_UP_TO_PAGE(ph->segsize_in_mem) / MOS_PAGE_SIZE //
103 );
104
105 mos_platform->mm_flag_pages(proc->pagetable, block.vaddr, block.npages, map_flags);
106 block.flags = map_flags;
107
108 process_attach_mmap(proc, block, (ph->p_flags & ELF_PH_F_X) ? VMTYPE_APPCODE : VMTYPE_APPDATA, false);
109 }
110
111#if MOS_DEBUG
112 process_dump_mmaps(proc);
113 const char *const strtab = buf + ((elf_section_hdr_t *) (buf + elf->sh_offset + elf->sh_strtab_index * elf->sh.entry_size))->sh_offset;
114 for (int i = 0; i < elf->sh.count; i++)
115 {
116 elf_section_hdr_t *sh = (elf_section_hdr_t *) (buf + elf->sh_offset + i * elf->sh.entry_size);
117 pr_info2("ELF section %2d: %s", i, strtab + sh->name_index);
118 }
119#endif
120
121 process_attach_fd(proc, &f->io);
122
123 // unmap the buffer from kernel pages
124 mos_platform->mm_unmap_pages(current_cpu->pagetable, buf_block.vaddr, buf_block.npages);
125
126 return proc;
127
128bail_out:
129 if (buf)
130 mos_platform->mm_free_pages(current_cpu->pagetable, buf_block.vaddr, buf_block.npages);
131bail_out_1:
132 if (f)
133 io_close(&f->io);
134 return NULL;
135}
