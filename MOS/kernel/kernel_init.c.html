25const char *init_path = "/programs/init";
26
27bool setup_init_path(int argc, const char **argv)
28{
29 if (argc != 1)
30 {
31 pr_warn("setup_init_path: expected 1 argument, got %d", argc);
32 for (int i = 0; i < argc; i++)
33 pr_warn(" %d: %s", i, argv[i]);
34 return false;
35 }
36
37 init_path = strdup(argv[0]);
38 pr_emph("init path set to '%s'", init_path);
39 return true;
40}
41
42__setup(init, "init", setup_init_path);

44bool mount_initrd(void)
45{
46 blockdev_t *dev = blockdev_find("initrd");
47 if (!dev)
48 return false;
49 pr_info2("found initrd block device: %s", dev->name);
50
51 mountpoint_t *mount = kmount(&root_path, &fs_cpio, dev);
52 if (!mount)
53 return false;
54
55 pr_info2("mounted initrd as rootfs");
56 return true;
57}

59void mos_start_kernel(const char *cmdline)
60{
61 mos_cmdline = cmdline_create(cmdline);
62
63 pr_info("Welcome to MOS!");
64 pr_emph("MOS %s (%s)", MOS_KERNEL_VERSION, MOS_KERNEL_REVISION_STRING);
65
66 if (mos_cmdline->options_count)
67 pr_emph("MOS Arguments: (total of %zu options)", mos_cmdline->options_count);
68
69 for (u32 i = 0; i < mos_cmdline->options_count; i++)
70 {
71 cmdline_option_t *option = mos_cmdline->options[i];
72 pr_info("%2d: %s", i + 1, option->name);
73 for (u32 j = 0; j < option->argc; j++)
74 pr_info2(" %2d: %s", j + 1, option->argv[j]);
75 }
76
77 invoke_setup_functions(mos_cmdline);
78
79 ipc_init();
80 process_init();
81 thread_init();
82
83 if (unlikely(!mount_initrd()))
84 mos_panic("failed to mount initrd");
85
86 console_t *init_con = console_get("x86");
87 if (!init_con)
88 {
89 init_con = console_get_by_prefix("x86");
90 if (!init_con)
91 mos_panic("failed to find serial console");
92 }
93
94 if (init_con->caps & CONSOLE_CAP_CLEAR)
95 init_con->clear(init_con);
96
97 terminal_t *init_term = terminal_create_console(init_con);
98
99 // construct argv for init process
100 argv_t init_argv = {
101 .argc = mos_cmdline->options_count + 1,
102 .argv = kmalloc(sizeof(char *) * (mos_cmdline->options_count + 2)), // +1 for init path, +1 for NULL
103 };
104
105 init_argv.argv[0] = strdup(init_path);
106
107 for (u32 i = 0; i < mos_cmdline->options_count; i++)
108 {
109 const cmdline_option_t *option = mos_cmdline->options[i];
110
111 // concatenate all arguments into a single string
112 size_t len = strlen(option->name) + 1; // +1 for '='
113 for (u32 j = 0; j < option->argc; j++)
114 len += strlen(option->argv[j]) + 1; // +1 for ',' (or '\0' for last argument)
115
116 char *str = kmalloc(len);
117 size_t offset = 0;
118 strcpy(str, option->name);
119 offset += strlen(option->name);
120
121 if (option->argc)
122 {
123 str[offset++] = '=';
124 for (u32 j = 0; j < option->argc; j++)
125 {
126 strcpy(str + offset, option->argv[j]);
127 offset += strlen(option->argv[j]);
128 if (j != option->argc - 1)
129 str[offset++] = ',';
130 }
131 }
132 str[offset] = '\0'; // null-terminate the string
133 init_argv.argv[i + 1] = str;
134 }
135
136 process_t *init = elf_create_process(init_path, NULL, init_term, (uid_t) 0, init_argv);
137 current_thread = init->threads[0];
138 pr_info("created init process: %s", init->name);
139
140 kthread_init(); // must be called after creating the first init process
141 device_manager_init();
142
143 scheduler();
144 MOS_UNREACHABLE();
145}
