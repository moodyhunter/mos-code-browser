21bool mount_initrd(void)
22{
23 blockdev_t *dev = blockdev_find("initrd");
24 if (!dev)
25 return false;
26 pr_info2("found initrd block device: %s", dev->name);
27
28 mountpoint_t *mount = kmount(&root_path, &fs_cpio, dev);
29 if (!mount)
30 return false;
31
32 pr_info2("mounted initrd as rootfs");
33 return true;
34}

36const char *cmdline_get_init_path(void)
37{
38 cmdline_arg_t *init_arg = mos_cmdline_get_arg("init");
39 if (init_arg && init_arg->params_count > 0)
40 {
41 cmdline_param_t *init_param = init_arg->params[0];
42 if (init_param->param_type == CMDLINE_PARAM_TYPE_STRING)
43 return init_param->val.string;
44
45 pr_warn("init path is not a string, using default '/programs/init'");
46 }
47 return "/programs/init";
48}

50void dump_cmdline(void)
51{
52 pr_emph("MOS %s (%s)", MOS_KERNEL_VERSION, MOS_KERNEL_REVISION);
53 if (mos_cmdline->args_count)
54 pr_emph("MOS Arguments: (total of %zu options)", mos_cmdline->args_count);
55 for (u32 i = 0; i < mos_cmdline->args_count; i++)
56 {
57 cmdline_arg_t *option = mos_cmdline->arguments[i];
58 pr_info("%2d: %s", i, option->arg_name);
59
60 for (u32 j = 0; j < option->params_count; j++)
61 {
62 cmdline_param_t *parameter = option->params[j];
63 switch (parameter->param_type)
64 {
65 case CMDLINE_PARAM_TYPE_STRING: pr_info("%6s%s", "", parameter->val.string); break;
66 case CMDLINE_PARAM_TYPE_BOOL: pr_info("%6s%s", "", parameter->val.boolean ? "true" : "false"); break;
67 default: MOS_UNREACHABLE();
68 }
69 }
70 }
71}

73void mos_start_kernel(const char *cmdline)
74{
75 mos_cmdline = mos_cmdline_create(cmdline);
76 printk_setup_console(); // setup printk console
77
78 pr_info("Welcome to MOS!");
79 dump_cmdline();
80
81#if BUILD_TESTING
82 if (mos_cmdline_get_arg("mos_tests"))
83 mos_test_engine_run_tests();
84#endif
85
86 ipc_init();
87 process_init();
88 thread_init();
89
90 if (unlikely(!mount_initrd()))
91 mos_panic("failed to mount initrd");
92
93 const char *init_path = cmdline_get_init_path();
94
95 console_t *init_con = console_get("x86");
96 if (!init_con)
97 {
98 init_con = console_get_by_prefix("x86");
99 if (!init_con)
100 mos_panic("failed to find serial console");
101 }
102
103 if (init_con->caps & CONSOLE_CAP_CLEAR)
104 init_con->clear(init_con);
105
106 terminal_t *init_term = terminal_create_console(init_con);
107
108 uid_t root_uid = 0;
109 process_t *init = elf_create_process(init_path, NULL, init_term, root_uid);
110 current_thread = init->threads[0];
111 pr_info("created init process: %s", init->name);
112
113 scheduler();
114 MOS_UNREACHABLE();
115}
