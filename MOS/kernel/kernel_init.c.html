19bool mount_initrd(void)
20{
21 blockdev_t *dev = blockdev_find("initrd");
22 if (!dev)
23 return false;
24 pr_info("found initrd block device: %s", dev->name);
25
26 mountpoint_t *mount = kmount(&root_path, &fs_cpio, dev);
27 if (!mount)
28 return false;
29
30 pr_info("mounted initrd as rootfs");
31 return true;
32}

34const char *cmdline_get_init_path(void)
35{
36 cmdline_arg_t *init_arg = mos_cmdline_get_arg("init");
37 if (init_arg && init_arg->params_count > 0)
38 {
39 cmdline_param_t *init_param = init_arg->params[0];
40 if (init_param->param_type == CMDLINE_PARAM_TYPE_STRING)
41 return init_param->val.string;
42
43 pr_warn("init path is not a string, using default '/programs/init'");
44 }
45 return "/programs/init";
46}

48void dump_cmdline(void)
49{
50 pr_emph("MOS %s (%s)", MOS_KERNEL_VERSION, MOS_KERNEL_REVISION);
51 pr_emph("MOS Arguments: (total of %zu options)", mos_cmdline->args_count);
52 for (u32 i = 0; i < mos_cmdline->args_count; i++)
53 {
54 cmdline_arg_t *option = mos_cmdline->arguments[i];
55 pr_info("%2d: %s", i, option->arg_name);
56
57 for (u32 j = 0; j < option->params_count; j++)
58 {
59 cmdline_param_t *parameter = option->params[j];
60 switch (parameter->param_type)
61 {
62 case CMDLINE_PARAM_TYPE_STRING: pr_info("%6s%s", "", parameter->val.string); break;
63 case CMDLINE_PARAM_TYPE_BOOL: pr_info("%6s%s", "", parameter->val.boolean ? "true" : "false"); break;
64 default: MOS_UNREACHABLE();
65 }
66 }
67 }
68}

70void mos_start_kernel(const char *cmdline)
71{
72 current_cpu->pagetable = mos_platform->kernel_pg;
73 mos_cmdline = mos_cmdline_create(cmdline);
74
75 pr_info("Welcome to MOS!");
76 dump_cmdline();
77
78#if MOS_MEME
79 mos_warn("V2Ray 4.45.2 started");
80#endif
81
82#if BUILD_TESTING
83 if (mos_cmdline_get_arg("mos_tests"))
84 mos_test_engine_run_tests();
85#endif
86
87 process_init();
88 thread_init();
89
90 if (unlikely(!mount_initrd()))
91 mos_panic("failed to mount initrd");
92
93 const char *init_path = cmdline_get_init_path();
94
95 uid_t root_uid = 0;
96 process_t *init = process_create_from_elf(init_path, NULL, root_uid);
97 pr_info("created init process: %s", init->name);
98
99 scheduler();
100 MOS_UNREACHABLE();
101}
