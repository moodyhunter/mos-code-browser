21bool mount_initrd(void)
22{
23 blockdev_t *dev = blockdev_find("initrd");
24 if (!dev)
25 return false;
26 pr_info("found initrd block device: %s", dev->name);
27
28 mountpoint_t *mount = kmount(&root_path, &fs_cpio, dev);
29 if (!mount)
30 return false;
31
32 pr_info("mounted initrd to rootfs");
33 return true;
34}

36const char *get_init_path(void)
37{
38 cmdline_arg_t *init_arg = mos_cmdline_get_arg("init");
39 if (init_arg && init_arg->params_count > 0)
40 {
41 cmdline_param_t *init_param = init_arg->params[0];
42 if (init_param->param_type == CMDLINE_PARAM_TYPE_STRING)
43 return init_param->val.string;
44
45 pr_warn("init path is not a string, using default '/init'");
46 }
47 return "/init";
48}

50void dump_cmdline(void)
51{
52 pr_emph("MOS %s (%s)", MOS_KERNEL_VERSION, MOS_KERNEL_REVISION);
53 pr_emph("MOS Arguments: (total of %zu options)", mos_cmdline->args_count);
54 for (u32 i = 0; i < mos_cmdline->args_count; i++)
55 {
56 cmdline_arg_t *option = mos_cmdline->arguments[i];
57 pr_info("%2d: %s", i, option->arg_name);
58
59 for (u32 j = 0; j < option->params_count; j++)
60 {
61 cmdline_param_t *parameter = option->params[j];
62 switch (parameter->param_type)
63 {
64 case CMDLINE_PARAM_TYPE_STRING: pr_info("%6s%s", "", parameter->val.string); break;
65 case CMDLINE_PARAM_TYPE_BOOL: pr_info("%6s%s", "", parameter->val.boolean ? "true" : "false"); break;
66 default: MOS_UNREACHABLE();
67 }
68 }
69 }
70 pr_emph("%-25s'%s'", "Kernel builtin cmdline:", MOS_KERNEL_BUILTIN_CMDLINE);
71}

73void mos_start_kernel(const char *cmdline)
74{
75 mos_cmdline = mos_cmdline_create(cmdline);
76
77 pr_info("Welcome to MOS!");
78 dump_cmdline();
79
80#if MOS_MEME
81 mos_warn("V2Ray 4.45.2 started");
82#endif
83
84 if (mos_cmdline_get_arg("mos_tests"))
85 mos_test_engine_run_tests();
86
87 process_init();
88 thread_init();
89
90 if (unlikely(!mount_initrd()))
91 mos_panic("failed to mount initrd");
92
93 const char *init_path = get_init_path();
94 file_t *init_file = vfs_open(init_path, OPEN_READ);
95 if (!init_file)
96 mos_panic("failed to open init");
97
98 size_t npage_required = init_file->io.size / MOS_PAGE_SIZE + 1;
99 char *initrd_data = kpage_alloc(npage_required, PGALLOC_NONE);
100 size_t r = io_read(&init_file->io, initrd_data, init_file->io.size);
101 MOS_ASSERT_X(r == init_file->io.size, "failed to read init");
102
103 elf_header_t *header = (elf_header_t *) initrd_data;
104 elf_verify_result verify_result = mos_elf_verify_header(header);
105 if (verify_result != ELF_VERIFY_OK)
106 {
107 switch (verify_result)
108 {
109 case ELF_VERIFY_INVALID_ENDIAN: pr_emerg("Invalid ELF endianness"); break;
110 case ELF_VERIFY_INVALID_MAGIC: pr_emerg("Invalid ELF magic"); break;
111 case ELF_VERIFY_INVALID_MAGIC_ELF: pr_emerg("Invalid ELF magic: 'ELF'"); break;
112 case ELF_VERIFY_INVALID_VERSION: pr_emerg("Invalid ELF version"); break;
113 case ELF_VERIFY_INVALID_BITS: pr_emerg("Invalid ELF bits"); break;
114 case ELF_VERIFY_INVALID_OSABI: pr_emerg("Invalid ELF OS ABI"); break;
115 default: MOS_UNREACHABLE();
116 }
117
118 mos_panic("failed to verify 'init' ELF header for '%s'", init_path);
119 }
120
121 if (header->object_type != ELF_OBJECT_EXECUTABLE)
122 mos_panic("'init' is not an executable");
123
124 elf_program_header_t *init_ph = (elf_program_header_t *) (initrd_data + header->program_header_offset);
125 elf_section_header_t *init_sh = (elf_section_header_t *) (initrd_data + header->section_header_offset);
126
127 int i_program = 1, i_section = 1;
128 while (init_ph->header_type != ELF_PH_T_LOAD && i_program < header->program_header.count)
129 init_ph++, i_program++;
130 while (init_sh->header_type != ELF_SH_T_PROGBITS && i_section < header->section_header.count)
131 init_sh++, i_section++;
132
133 MOS_ASSERT_X(init_sh->header_type == ELF_SH_T_PROGBITS, "no code section found");
134 MOS_ASSERT_X(init_ph->header_type == ELF_PH_T_LOAD, "no loadable segment found");
135
136 // the stack memory to be used if we enter the kernelmode by a trap / interrupt
137 // TODO: Per-process stack
138 extern tss32_t tss_entry;
139 tss_entry.esp0 = (u32) kpage_alloc(1, PGALLOC_NONE) + MOS_PAGE_SIZE; // stack grows downwards from the top of the page
140 pr_emph("kernel stack at " PTR_FMT, (uintptr_t) tss_entry.esp0);
141
142 process_t *pinit = create_process((process_id_t){ 1 }, (uid_t){ 0 }, (thread_entry_t) header->entry_point, (void *) 114514);
143
144 mos_platform->mm_map_kvaddr(pinit->pagetable, init_ph->vaddr, (uintptr_t) header + init_ph->data_offset, 1, VM_USERMODE);
145
146 scheduler();
147 MOS_UNREACHABLE();
148}
