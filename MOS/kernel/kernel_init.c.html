21bool mount_initrd(void)
22{
23 blockdev_t *dev = blockdev_find("initrd");
24 if (!dev)
25 return false;
26 pr_info("found initrd block device: %s", dev->name);
27
28 mountpoint_t *mount = kmount(&root_path, &fs_cpio, dev);
29 if (!mount)
30 return false;
31
32 pr_info("mounted initrd as rootfs");
33 return true;
34}

36const char *cmdline_get_init_path(void)
37{
38 cmdline_arg_t *init_arg = mos_cmdline_get_arg("init");
39 if (init_arg && init_arg->params_count > 0)
40 {
41 cmdline_param_t *init_param = init_arg->params[0];
42 if (init_param->param_type == CMDLINE_PARAM_TYPE_STRING)
43 return init_param->val.string;
44
45 pr_warn("init path is not a string, using default '/init'");
46 }
47 return "/init";
48}

50void dump_cmdline(void)
51{
52 pr_emph("MOS %s (%s)", MOS_KERNEL_VERSION, MOS_KERNEL_REVISION);
53 pr_emph("MOS Arguments: (total of %zu options)", mos_cmdline->args_count);
54 for (u32 i = 0; i < mos_cmdline->args_count; i++)
55 {
56 cmdline_arg_t *option = mos_cmdline->arguments[i];
57 pr_info("%2d: %s", i, option->arg_name);
58
59 for (u32 j = 0; j < option->params_count; j++)
60 {
61 cmdline_param_t *parameter = option->params[j];
62 switch (parameter->param_type)
63 {
64 case CMDLINE_PARAM_TYPE_STRING: pr_info("%6s%s", "", parameter->val.string); break;
65 case CMDLINE_PARAM_TYPE_BOOL: pr_info("%6s%s", "", parameter->val.boolean ? "true" : "false"); break;
66 default: MOS_UNREACHABLE();
67 }
68 }
69 }
70 pr_emph("%-25s'%s'", "Kernel builtin cmdline:", MOS_KERNEL_BUILTIN_CMDLINE);
71}

73elf_header_t *open_elf_executable(const char *path)
74{
75 file_t *f = vfs_open(path, FILE_OPEN_READ);
76 if (!f)
77 {
78 pr_emerg("failed to open '%s'", path);
79 goto bail_out;
80 }
81
82 size_t npage_required = f->io.size / MOS_PAGE_SIZE + 1;
83 char *header_buf = kpage_alloc(npage_required, PGALLOC_NONE);
84
85 size_t size = io_read(&f->io, header_buf, f->io.size);
86 MOS_ASSERT_X(size == f->io.size, "failed to read init");
87
88 elf_header_t *elf_header = (elf_header_t *) header_buf;
89 if (elf_header->object_type != ELF_OBJTYPE_EXECUTABLE)
90 {
91 pr_emerg("'%s' is not an executable", path);
92 goto bail_out;
93 }
94
95 elf_verify_result verify_result = elf_verify_header(elf_header);
96 if (verify_result != ELF_VERIFY_OK)
97 {
98 pr_emerg("failed to verify ELF header for '%s', result: %d", path, (int) verify_result);
99 goto bail_out;
100 }
101
102 return elf_header;
103
104bail_out:
105 if (f)
106 io_close(&f->io);
107 return NULL;
108}

110process_t *create_elf_process(elf_header_t *elf)
111{
112 const char *buf = (const char *) elf;
113
114 process_t *proc = allocate_process((process_id_t){ 1 }, (uid_t){ 0 }, (thread_entry_t) elf->entry_point, NULL);
115
116 for (int i = 0; i < elf->ph.count; i++)
117 {
118 elf_program_hdr_t *ph = (elf_program_hdr_t *) (buf + elf->ph_offset + i * elf->ph.entry_size);
119 pr_info("program header %d: %c%c%c%s at " PTR_FMT, i,
120 (elf_program_header_flags) ph->p_flags & ELF_PH_F_R ? 'r' : '-', //
121 (elf_program_header_flags) ph->p_flags & ELF_PH_F_W ? 'w' : '-', //
122 (elf_program_header_flags) ph->p_flags & ELF_PH_F_X ? 'x' : '-', //
123 ph->header_type == ELF_PH_T_LOAD ? " (load)" : "", //
124 ph->vaddr //
125 );
126
127 if (!(ph->header_type & ELF_PH_T_LOAD))
128 continue;
129
130 vm_flags map_flags = ( //
131 (ph->p_flags & ELF_PH_F_R ? VM_READ : 0) | //
132 (ph->p_flags & ELF_PH_F_W ? VM_WRITE : 0) | //
133 (ph->p_flags & ELF_PH_F_X ? VM_EXECUTE : 0) | //
134 VM_USERMODE //
135 );
136
137 mos_platform->mm_map_kvaddr( //
138 proc->pagetable, //
139 ALIGN_DOWN_TO_PAGE(ph->vaddr), //
140 (uintptr_t) buf + ph->data_offset, //
141 ALIGN_UP_TO_PAGE(ph->segsize_in_mem) / MOS_PAGE_SIZE, //
142 map_flags //
143 );
144 }
145
146 const char *const strtab = buf + ((elf_section_hdr_t *) (buf + elf->sh_offset + elf->sh_strtab_index * elf->sh.entry_size))->sh_offset;
147 for (int i = 0; i < elf->sh.count; i++)
148 {
149 elf_section_hdr_t *sh = (elf_section_hdr_t *) (buf + elf->sh_offset + i * elf->sh.entry_size);
150 pr_info2("section %2d: %s", i, strtab + sh->name_index);
151 }
152
153 return proc;
154}

156void mos_start_kernel(const char *cmdline)
157{
158 mos_cmdline = mos_cmdline_create(cmdline);
159
160 pr_info("Welcome to MOS!");
161 dump_cmdline();
162
163#if MOS_MEME
164 mos_warn("V2Ray 4.45.2 started");
165#endif
166
167 if (mos_cmdline_get_arg("mos_tests"))
168 mos_test_engine_run_tests();
169
170 process_init();
171 thread_init();
172
173 if (unlikely(!mount_initrd()))
174 mos_panic("failed to mount initrd");
175
176 const char *init_path = cmdline_get_init_path();
177
178 elf_header_t *init_header = open_elf_executable(init_path);
179 if (!init_header)
180 mos_panic("failed to open init");
181 process_t *init = create_elf_process(init_header);
182 MOS_UNUSED(init);
183
184 // the stack memory to be used if we enter the kernelmode by a trap / interrupt
185 // TODO: Per-process stack
186 extern tss32_t tss_entry;
187 tss_entry.esp0 = (u32) kpage_alloc(1, PGALLOC_NONE) + 1 * MOS_PAGE_SIZE; // stack grows downwards from the top of the page
188 pr_emph("kernel stack at " PTR_FMT, (uintptr_t) tss_entry.esp0);
189
190 scheduler();
191 MOS_UNREACHABLE();
192}
