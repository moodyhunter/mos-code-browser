11typedef struct
12{
13 as_linked_list;
14 kpanic_hook_t *hook;
15} panic_hook_holder_t;
16
17static list_node_t kpanic_hooks = LIST_HEAD_INIT(kpanic_hooks);
18static kmsg_handler_t *kwarn_handler = NULL;
19
20void kwarn_handler_set(kmsg_handler_t *handler)
21{
22 pr_warn("installing a new warning handler...");
23 kwarn_handler = handler;
24}
25
26void kwarn_handler_remove()
27{
28 pr_warn("removing warning handler...");
29 if (!kwarn_handler)
30 mos_warn("no previous warning handler installed");
31 kwarn_handler = NULL;
32}

34noreturn void mos_kpanic(const char *func, u32 line, const char *fmt, ...)
35{
36 static bool in_panic = false;
37 if (in_panic)
38 {
39 pr_fatal("recursive panic detected, aborting...");
40 while (true)
41 ;
42 }
43 in_panic = true;
44 mos_platform->interrupt_disable();
45
46 va_list args;
47 char message[PRINTK_BUFFER_SIZE] = { 0 };
48 va_start(args, fmt);
49 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
50 va_end(args);
51
52 pr_emerg("");
53 pr_fatal("!!!!!!!!!!!!!!!!!!!!!!!!");
54 pr_fatal("!!!!! KERNEL PANIC !!!!!");
55 pr_fatal("!!!!!!!!!!!!!!!!!!!!!!!!");
56 pr_emerg("");
57 pr_emerg("%s", message);
58 pr_emerg(" in function: %s (line %u)", func, line);
59
60 list_foreach(panic_hook_holder_t, holder, kpanic_hooks)
61 {
62 holder->hook();
63 }
64
65 pr_emerg("Halting...");
66 mos_platform->halt_cpu();
67
68 while (1)
69 ;
70}
71
72void mos_kwarn(const char *func, u32 line, const char *fmt, ...)
73{
74 va_list args;
75 if (kwarn_handler)
76 {
77 va_start(args, fmt);
78 kwarn_handler(func, line, fmt, args);
79 va_end(args);
80 return;
81 }
82
83 char message[PRINTK_BUFFER_SIZE] = { 0 };
84 va_start(args, fmt);
85 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
86 va_end(args);
87
88 lprintk(MOS_LOG_WARN, "%s", message);
89 lprintk(MOS_LOG_WARN, " in function: %s (line %u)\n", func, line);
90}
91
92void mos_install_kpanic_hook(kpanic_hook_t *hook)
93{
94 panic_hook_holder_t *holder = kmalloc(sizeof(panic_hook_holder_t));
95 holder->hook = hook;
96 list_node_append(&kpanic_hooks, &holder->list_node);
97}
