9static int dummy_write(console_t *console, const char *message, size_t length)
10{
11 (void) console;
12 (void) message;
13 (void) length;
14 return 0;
15}
16
17// ! This is a compile-time connected linked list.
18static console_t dummy_con = {
19 .name = "dummy",
20 .caps = CONSOLE_CAP_NONE,
21 .write = dummy_write,
22 .list_node = LIST_HEAD_INIT(consoles),
23};
24
25static bool has_console_registered = false;
26list_node_t consoles = LIST_NODE_INIT(dummy_con);
27
28void mos_install_console(console_t *con)
29{
30 if (con->caps & CONSOLE_CAP_SETUP)
31 con->setup(con);
32
33 if (con->caps & CONSOLE_CAP_CLEAR)
34 con->clear(con);
35
36 list_node_append(&consoles, list_node(con));
37 pr_info("console: registered '%s'", con->name);
38
39 if (unlikely(!has_console_registered))
40 {
41 pr_info2("console: removing '%s'", dummy_con.name);
42 list_remove(&dummy_con);
43 has_console_registered = true;
44 }
45}
