10static int dummy_write_to_console(console_t *console, const char *message, size_t length)
11{
12 (void) console;
13 (void) message;
14 (void) length;
15 return 0;
16}

18// ! This is a compile-time connected linked list.
19static console_t dummy_con = {
20 .name = "dummy",
21 .caps = CONSOLE_CAP_NONE,
22 .write_impl = dummy_write_to_console,
23 .list_node = LIST_HEAD_INIT(consoles),
24};
25
26static bool has_console_registered = false;
27list_node_t consoles = LIST_NODE_INIT(dummy_con);
28
29void console_register(console_t *con)
30{
31 if (con->caps & CONSOLE_CAP_SETUP)
32 con->setup(con);
33
34 if (con->caps & CONSOLE_CAP_CLEAR)
35 con->clear(con);
36
37 list_node_append(&consoles, list_node(con));
38 pr_info("console: registered '%s'", con->name);
39
40 if (unlikely(!has_console_registered))
41 {
42 pr_info2("console: removing '%s'", dummy_con.name);
43 list_remove(&dummy_con);
44 has_console_registered = true;
45 }
46}
47
48console_t *console_get(const char *name)
49{
50 list_foreach(console_t, con, consoles)
51 {
52 if (strcmp(con->name, name) == 0)
53 return con;
54 }
55 return NULL;
56}
57
58console_t *console_get_by_prefix(const char *prefix)
59{
60 list_foreach(console_t, con, consoles)
61 {
62 if (strncmp(con->name, prefix, strlen(prefix)) == 0)
63 return con;
64 }
65 return NULL;
66}

68int console_read(console_t *con, char *dest, size_t size)
69{
70 if (con->caps & CONSOLE_CAP_READ)
71 return con->read_impl(con, dest, size);
72 return -1;
73}
74
75int console_write(console_t *con, const char *data, size_t size)
76{
77 return con->write_impl(con, data, size);
78}
