11static io_t *server_io = NULL;
12
13static void device_manager_thread(void *arg)
14{
15 (void) arg;
16 MOS_ASSERT_X(server_io != NULL, "Device manager not initialized");
17
18 while (true)
19 {
20 io_t *io = ipc_accept(server_io);
21 }
22
23 return;
24}
25
26void device_manager_init()
27{
28 MOS_ASSERT_X(server_io == NULL, "Device manager already initialized");
29 server_io = ipc_create(MOS_DEVICE_MANAGER_SERVICE_NAME, 32);
30 kthread_create(device_manager_thread, 0, "device_manager");
31}
