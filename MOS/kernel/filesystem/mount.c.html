10static mountpoint_t *rootmp = NULL;
11
12void kmount_add_mp(mountpoint_t *new_mp)
13{
14 if (rootmp == NULL)
15 {
16 pr_info2("root filesystem mounted");
17 rootmp = new_mp;
18 return;
19 }
20
21 mountpoint_t *parent = kmount_find_submp(rootmp, new_mp->path);
22 if (parent == NULL)
23 {
24 mos_warn("mountpoint %s cannot be placed in tree", new_mp->path->name);
25 MOS_UNREACHABLE();
26 return;
27 }
28
29 new_mp->parent = parent;
30 parent->children = krealloc(parent->children, sizeof(mountpoint_t *) * (parent->children_count + 1));
31 parent->children[parent->children_count] = new_mp;
32}

34mountpoint_t *kmount(fsnode_t *path, filesystem_t *fs, blockdev_t *blockdev)
35{
36 MOS_ASSERT(path != NULL);
37 MOS_ASSERT(fs != NULL);
38 MOS_ASSERT(blockdev != NULL);
39
40 pr_info("mount '%s' on '%s' (blockdev '%s')", fs->name, path->name, blockdev->name);
41 mountpoint_t *new_mp = kcalloc(1, sizeof(mountpoint_t));
42
43 if (!new_mp)
44 return NULL;
45 new_mp->path = path;
46 new_mp->fs = fs;
47 new_mp->dev = blockdev;
48 new_mp->children = NULL;
49
50 if (!fs->op_mount(blockdev, path))
51 {
52 kfree(new_mp);
53 mos_warn("failed to mount filesystem %s at %s", fs->name, path->name);
54 return NULL;
55 }
56
57 kmount_add_mp(new_mp);
58 return new_mp;
59}

61bool kunmount(mountpoint_t *mountpoint)
62{
63 if (mountpoint->refcount.atomic > 0)
64 {
65 mos_warn("mountpoint %s still has %llu references", mountpoint->path->name, mountpoint->refcount.atomic);
66 return false;
67 }
68 if (mountpoint->children_count > 0)
69 {
70 mos_warn("mountpoint %s still has %zu children", mountpoint->path->name, mountpoint->children_count);
71 return false;
72 }
73
74 if (!mountpoint->fs->op_unmount(mountpoint))
75 {
76 mos_warn("failed to unmount filesystem %s at %s", mountpoint->fs->name, mountpoint->path->name);
77 return false;
78 }
79 if (unlikely(mountpoint == rootmp))
80 {
81 pr_warn("unmounting root filesystem");
82 rootmp = NULL;
83 }
84 kfree(mountpoint);
85 return true;
86}
87
88mountpoint_t *kmount_find_mp(fsnode_t *path)
89{
90 if (unlikely(rootmp == NULL))
91 return NULL;
92
93 return kmount_find_submp(rootmp, path);
94}
95
96mountpoint_t *kmount_find_submp(mountpoint_t *mp, fsnode_t *path)
97{
98 if (mp->children == NULL)
99 return mp;
100
101 mountpoint_t *match = NULL;
102 for (size_t i = 0; i < mp->children_count; i++)
103 {
104 mountpoint_t *child = mp->children[i];
105 if (child->path->name[0] == '/')
106 continue;
107 if (strcmp(child->path->name, path->name) == 0)
108 {
109 match = child;
110 break;
111 }
112 }
113 return match;
114}
