13#define CPIO_MODE_FILE_TYPE 0170000 // This masks the file type bits.
14#define CPIO_MODE_SOCKET 0140000 // File type value for sockets.
15#define CPIO_MODE_SYMLINK 0120000 // File type value for symbolic links. For symbolic links, the link body is stored as file data.
16#define CPIO_MODE_FILE 0100000 // File type value for regular files.
17#define CPIO_MODE_BLOCKDEV 0060000 // File type value for block special devices.
18#define CPIO_MODE_DIR 0040000 // File type value for directories.
19#define CPIO_MODE_CHARDEV 0020000 // File type value for character special devices.
20#define CPIO_MODE_FIFO 0010000 // File type value for named pipes or FIFOs.
21#define CPIO_MODE_SUID 0004000 // SUID bit.
22#define CPIO_MODE_SGID 0002000 // SGID bit.
23#define CPIO_MODE_STICKY 0001000 // Sticky bit.

25bool cpio_read_metadata(blockdev_t *dev, const char *target, cpio_metadata_t *metadata)
26{
27 if (unlikely(strcmp(target, "TRAILER!!!") == 0))
28 mos_panic("what the heck are you doing?");
29
30 size_t offset = 0;
31 cpio_newc_header_t header;
32
33 while (true)
34 {
35 dev->read(dev, &header, sizeof(cpio_newc_header_t), offset);
36
37 if (strncmp(header.magic, "07070", 5) != 0 || (header.magic[5] != '1' && header.magic[5] != '2'))
38 {
39 mos_warn("invalid cpio header magic, possibly corrupt archive");
40 return false;
41 }
42
43 offset += sizeof(cpio_newc_header_t);
44
45 size_t filename_len = strntoll(header.namesize, NULL, 16, sizeof(header.namesize) / sizeof(char));
46
47 char filename[filename_len];
48 dev->read(dev, filename, filename_len, offset);
49
50 bool found = strncmp(filename, target, filename_len) == 0;
51
52 if (found)
53 {
54 metadata->header = header;
55 metadata->header_offset = offset - sizeof(cpio_newc_header_t);
56 metadata->name_offset = offset;
57 metadata->name_length = filename_len;
58 }
59
60 if (unlikely(strcmp(filename, "TRAILER!!!") == 0))
61 return false;
62
63 offset += filename_len;
64 offset = ((offset + 3) & ~0x03); // align to 4 bytes
65
66 size_t data_len = strntoll(header.filesize, NULL, 16, sizeof(header.filesize) / sizeof(char));
67 if (found)
68 {
69 metadata->data_offset = offset;
70 metadata->data_length = data_len;
71 }
72
73 offset += data_len;
74 offset = ((offset + 3) & ~0x03); // align to 4 bytes (again)
75
76 if (found)
77 return true;
78 }
79
80 MOS_UNREACHABLE();
81}

83bool cpio_mount(blockdev_t *dev, fsnode_t *path)
84{
85 MOS_UNUSED(path);
86 size_t read = 0;
87 cpio_newc_header_t header;
88 read = dev->read(dev, &header, sizeof(header), 0);
89 if (read != sizeof(header))
90 {
91 mos_warn("failed to read cpio archive");
92 return false;
93 }
94
95 if (strncmp(header.magic, "07070", 5) != 0 || (header.magic[5] != '1' && header.magic[5] != '2'))
96 {
97 mos_warn("invalid cpio header magic, possibly corrupt archive");
98 return false;
99 }
100
101 mos_debug("cpio header: %.6s", header.magic);
102
103 return true;
104}

112bool cpio_open(const mountpoint_t *mp, const fsnode_t *path, file_open_flags flags, file_t *file)
113{
114 const char *strpath = path_to_string_relative(mp->path, path);
115 mos_debug("cpio_open: %s", strpath);
116 if (flags & FILE_OPEN_WRITE)
117 {
118 mos_warn("cpio_open: write not supported");
119 return false;
120 }
121
122 cpio_metadata_t metadata;
123 bool result = cpio_read_metadata(mp->dev, strpath, &metadata);
124
125 if (!result)
126 return false;
127
128 file->pdata = kmalloc(sizeof(cpio_metadata_t));
129 memcpy(file->pdata, &metadata, sizeof(cpio_metadata_t));
130 return true;
131}

133size_t cpio_read(blockdev_t *dev, file_t *file, void *buf, size_t size)
134{
135 cpio_metadata_t *metadata = file->pdata;
136 size_t read = dev->read(dev, buf, MIN(size, metadata->data_length), metadata->data_offset);
137 return read;
138}

146bool cpio_stat(const mountpoint_t *mp, const fsnode_t *path, file_stat_t *stat)
147{
148 const char *strpath = path_to_string_relative(mp->path, path);
149 cpio_metadata_t metadata;
150 bool result = cpio_read_metadata(mp->dev, strpath, &metadata);
151 if (!result)
152 return false;
153
154 stat->size = metadata.data_length;
155
156 stat->uid = strntoll(metadata.header.uid, NULL, 16, sizeof(metadata.header.uid) / sizeof(char));
157 stat->gid = strntoll(metadata.header.gid, NULL, 16, sizeof(metadata.header.gid) / sizeof(char));
158
159 // 0000777 The lower 9 bits specify read/write/execute permissions for world, group, and user following standard POSIX conventions.
160 u32 modebits = strntoll(metadata.header.mode, NULL, 16, sizeof(metadata.header.mode) / sizeof(char));
161 stat->permissions.other = (modebits & 0007) >> 0;
162 stat->permissions.group = (modebits & 0070) >> 3;
163 stat->permissions.owner = (modebits & 0700) >> 6;
164
165 stat->sticky = modebits & CPIO_MODE_STICKY;
166 stat->sgid = modebits & CPIO_MODE_SGID;
167 stat->suid = modebits & CPIO_MODE_SUID;
168
169 switch (modebits & CPIO_MODE_FILE_TYPE)
170 {
171 case CPIO_MODE_FILE: stat->type = FILE_TYPE_REGULAR_FILE; break;
172 case CPIO_MODE_DIR: stat->type = FILE_TYPE_DIRECTORY; break;
173 case CPIO_MODE_SYMLINK: stat->type = FILE_TYPE_SYMLINK; break;
174 case CPIO_MODE_CHARDEV: stat->type = FILE_TYPE_CHAR_DEVICE; break;
175 case CPIO_MODE_BLOCKDEV: stat->type = FILE_TYPE_BLOCK_DEVICE; break;
176 case CPIO_MODE_FIFO: stat->type = FILE_TYPE_NAMED_PIPE; break;
177 case CPIO_MODE_SOCKET: stat->type = FILE_TYPE_SOCKET; break;
178 default: mos_warn("invalid cpio file mode"); return -1;
179 }
180 return true;
181}

183bool cpio_readlink(const mountpoint_t *mp, const fsnode_t *path, char *buf, size_t bufsize)
184{
185 const char *strpath = path_to_string_relative(mp->path, path);
186 cpio_metadata_t metadata;
187 bool result = cpio_read_metadata(mp->dev, strpath, &metadata);
188 if (!result)
189 return false;
190 u32 modebits = strntoll(metadata.header.mode, NULL, 16, sizeof(metadata.header.mode) / sizeof(char));
191
192 if (!(modebits & 0120000))
193 {
194 mos_warn("cpio_readlink: not a symbolic link");
195 return false;
196 }
197
198 size_t read = 0;
199 if (metadata.data_length > bufsize)
200 {
201 mos_warn("cpio_readlink: buffer too small");
202 return false;
203 }
204
205 u32 limit = metadata.data_length < bufsize ? metadata.data_length : bufsize;
206
207 read = mp->dev->read(mp->dev, buf, limit, metadata.data_offset);
208 if (read != limit)
209 {
210 mos_warn("cpio_readlink: failed to read link");
211 return false;
212 }
213 return true;
214}
