12// BEGIN: filesystem's io_t operations
13void vfs_io_ops_close(io_t *io)
14{
15 file_t *file = container_of(io, file_t, io);
16 mountpoint_t *mp = kmount_find_mp(file->fsnode);
17 if (mp == NULL)
18 {
19 mos_warn("no filesystem mounted at %s", file->fsnode->name);
20 return;
21 }
22 mp->fs->op_close(file);
23 tree_trace_to_root(tree_node(file->fsnode), path_treeop_decrement_refcount);
24 kfree(file);
25}

27size_t vfs_io_ops_read(io_t *io, void *buf, size_t count)
28{
29 file_t *file = container_of(io, file_t, io);
30 mountpoint_t *mp = kmount_find_mp(file->fsnode);
31 if (mp == NULL)
32 {
33 mos_warn("no filesystem mounted at %s", file->fsnode->name);
34 return -1;
35 }
36 size_t result = mp->fs->op_read(mp->dev, file, buf, count);

60file_t *vfs_open(const char *path, file_open_flags flags)
61{
62 fsnode_t *node = path_find_fsnode(path);
63 file_t *file = kcalloc(1, sizeof(file_t));
64 bool opened = vfs_path_open(node, flags, file);
65 if (!opened)
66 {
67 kfree(node);
68 kfree(file);
69 return NULL;
70 }
71 file->fsnode = node;
72 MOS_ASSERT_X(refcount_get(&file->io.refcount) == 0, "file has nonzero refcount");
73 return file;
74}

76bool vfs_stat(const char *path, file_stat_t *restrict stat)
77{
78 fsnode_t *p = path_find_fsnode(path);

88fsnode_t *vfs_readlink(const char *path)
89{
90 fsnode_t *p = path_find_fsnode(path);
91 fsnode_t *target = kmalloc(sizeof(fsnode_t));
92 bool opened = vfs_path_readlink(p, &target);
93 if (!opened)
94 {
95 kfree(p);
96 kfree(target);
97 return NULL;
98 }
99 return target;

102bool vfs_path_open(fsnode_t *path, file_open_flags flags, file_t *file)
103{
104 mountpoint_t *mp = kmount_find_mp(path);
105 if (mp == NULL)
106 {
107 mos_warn("no filesystem mounted at %s", path->name);
108 return NULL;
109 }
110
111 mos_debug("opening file %s on fs: %s, blockdev: %s", path->name, mp->fs->name, mp->dev->name);
112
113 file_stat_t stat;
114 if (!mp->fs->op_stat(mp, path, &stat))
115 {
116 mos_warn("stat failed for %s", path->name);
117 return NULL;
118 }
119
120 if (stat.type == FILE_TYPE_SYMLINK)
121 {
122 if (flags & FILE_OPEN_SYMLINK_NO_FOLLOW)
123 goto _continue;
124
125 // TODO: follow symlinks
126 }
127
128_continue:;
129
130 if (!mp->fs->op_open(mp, path, flags, file))
131 {
132 mos_warn("failed to open file %s", path->name);
133 return NULL;
134 }
135
136 tree_trace_to_root(tree_node(path), path_treeop_increment_refcount);
137 io_init(&file->io, (flags & FILE_OPEN_READ) | (flags & FILE_OPEN_WRITE), stat.size, &fs_io_ops);
138 return true;
139}

141bool vfs_path_readlink(fsnode_t *path, fsnode_t **link)
142{
143 mountpoint_t *mp = kmount_find_mp(path);
144 if (mp == NULL)
145 {
146 mos_warn("no filesystem mounted at %s", path->name);
147 return false;
148 }
149
150 file_stat_t stat;
151 if (!mp->fs->op_stat(mp, path, &stat))
152 {
153 mos_warn("stat failed for %s", path->name);
154 return false;
155 }
156
157 if (stat.type != FILE_TYPE_SYMLINK)
158 {
159 mos_warn("%s is not a symlink", path->name);
160 return false;
161 }
162
163 char target[stat.size + 1];
164 if (!mp->fs->op_readlink(mp, path, target, stat.size))
165 {
166 mos_warn("readlink failed for %s", path->name);
167 return false;
168 }
169 target[stat.size] = '\0';
170
171 // ! 'path' is always a symlink (aka, a file), so we have to get its parent
172 bool resolved = path_resolve(path_parent(path), target, link);
173 return resolved;
174}

176bool vfs_path_stat(fsnode_t *path, file_stat_t *restrict stat)
177{
178 mountpoint_t *mp = kmount_find_mp(path);
179 if (mp == NULL)
180 {
181 mos_warn("no filesystem mounted at %s", path->name);
182 return -1;
183 }
184 bool result = mp->fs->op_stat(mp, path, stat);
185 return result;
186}

188void file_io_cleanup(io_t *io)
189{
190 // the file has already been closed, so we can just free the fsnode
191 file_t *file = container_of(io, file_t, io);
192 tree_trace_to_root(tree_node(file->fsnode), path_treeop_decrement_refcount);
193 if (file->fsnode)
194 kfree(file->fsnode);
195 kfree(file);
196}
