11void path_node_get_name(const tree_node_t *node, char **name, size_t *name_len)
12{
13 fsnode_t *path = tree_entry(node, fsnode_t);
14 *name = (char *) path->name;
15 *name_len = strlen(path->name);
16}

32static fsnode_t *impl_path_get_subpath(fsnode_t *cwd, const char *path, size_t path_len)
33{
34 MOS_ASSERT(cwd != NULL);
35 MOS_ASSERT(path != NULL);
36 MOS_ASSERT(path[0] != PATH_SEPARATOR);
37
38 if (path_len == 2 && strncmp(path, "..", 2) == 0)
39 {
40 fsnode_t *parent = path_parent(cwd);
41 if (parent == NULL)
42 {
43 mos_warn("cannot go up from root");
44 return &root_path;
45 }
46 return parent;
47 }
48 else if (path_len == 1 && strncmp(path, ".", 1) == 0)
49 {
50 // do nothing
51 return cwd;
52 }
53 else
54 {
55 const tree_node_t *cnode = tree_find_child_by_name(&path_tree_op, tree_node(cwd), path, path_len);
56 fsnode_t *child;
57 if (cnode == NULL)
58 {
59 child = kzalloc(sizeof(fsnode_t));
60 child->name = duplicate_string(path, path_len);
61 tree_add_child(tree_node(cwd), tree_node(child));
62 }
63 else
64 {
65 child = tree_entry(cnode, fsnode_t);
66 }
67 return child;
68 }
69}

71const char *path_next_segment(const char *path, size_t *segment_len)
72{
73 if (path == NULL || *path == '\0')
74 return NULL;
75 if (*path == PATH_SEPARATOR)
76 path++;
77 const char *next = strchr(path, PATH_SEPARATOR);
78 if (next == NULL)
79 next = path + strlen(path);
80 *segment_len = next - path;
81 return next;
82}

84fsnode_t *path_find_fsnode(const char *path)
85{
86 fsnode_t *target;
87 bool resolved = path_resolve(&root_path, path, &target);
88 if (!resolved)
89 {
90 mos_warn("path_contruct: path not resolved");
91 return NULL;
92 }
93 return target;
94}

96bool path_resolve(fsnode_t *cwd, const char *path, fsnode_t **resolved)
97{
98 MOS_ASSERT(cwd && path && resolved);
99 if (path == NULL || *path == '\0')
100 {
101 *resolved = cwd;
102 return true;
103 }
104
105 fsnode_t *real_cwd = cwd;
106
107 // an absolute path starts with a slash
108 if (path[0] == PATH_SEPARATOR)
109 {
110 path++;
111 real_cwd = &root_path;
112 }
113
114 size_t segment_len = 0;
115 const char *next_seg = path_next_segment(path, &segment_len);
116
117 fsnode_t *current = impl_path_get_subpath(real_cwd, path, segment_len);
118 if (next_seg == NULL || *next_seg == '\0')
119 {
120 *resolved = current;
121 return true;
122 }
123
124 file_stat_t stat = { 0 };
125 vfs_path_stat(current, &stat);
126 if (stat.type == FILE_TYPE_SYMLINK)
127 {
128 fsnode_t *symlink_target; // to be allocated by vfs_path_readlink (aka, **resolved)
129 bool ok = vfs_path_readlink(current, &symlink_target);
130 if (!ok)
131 {
132 *resolved = NULL;
133 return false;
134 }
135 current = symlink_target;
136 }
137
138 return path_resolve(current, next_seg + 1, resolved);
139}

142{
143 return tree_find_common_prefix(tree_node(path), tree_node(prefix)) == tree_node(prefix);

146const char *path_to_string_relative(const fsnode_t *root, const fsnode_t *leaf)
147{
148 if (root == NULL || leaf == NULL)
149 return strdup("");
150
151 if (root == leaf)
152 return strdup(root->name);
153
154 const fsnode_t *current = leaf;
155 char *pool = kmalloc(MOS_PATH_MAX_LENGTH);
156 char *begin = pool + MOS_PATH_MAX_LENGTH;
157
158 do
159 {
160 size_t len = strlen(current->name);
161 begin -= len;
162 memcpy(begin, current->name, len);
163 *(--begin) = '/';
164 current = tree_entry(tree_node(current)->parent, fsnode_t);
165 } while (current && current != root);
166
167 begin++; // remove leading '/'
168
169 const char *result = duplicate_string(begin, pool + MOS_PATH_MAX_LENGTH - begin);
170 kfree(pool);
171 return result;
172}

174void path_treeop_decrement_refcount(const tree_node_t *node)
175{
176 fsnode_t *path = tree_entry(node, fsnode_t);
177 mos_debug(fs, "decreasing refcount of path '%s'", path->name);
178 refcount_dec(&path->refcount);
179}

181void path_treeop_increment_refcount(const tree_node_t *node)
182{
183 fsnode_t *path = tree_entry(node, fsnode_t);
184 mos_debug(fs, "incrementing refcount of path '%s'", path->name);
185 refcount_inc(&path->refcount);
