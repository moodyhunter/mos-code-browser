13process_t *process_handle_fork(process_t *parent)
14{
15 MOS_ASSERT(process_is_valid(parent));
16
17 process_t *child = process_allocate(parent, parent->effective_uid, parent->name);
18 pr_emph("process %d forked to %d", parent->pid, child->pid);
19
20 // copy the parent's memory
21 for (int i = 0; i < parent->mmaps_count; i++)
22 {
23 proc_vmblock_t block = parent->mmaps[i];
24 if (block.map_flags & MMAP_PRIVATE)
25 {
26 mos_debug("private mapping, skipping");
27 continue;
28 }
29
30 vmblock_t child_vmblock;
31 if (block.type == VMTYPE_KSTACK)
32 {
33 // Kernel stacks are special, we need to allocate a new one (not CoW-mapped)
34 MOS_ASSERT_X(block.vm.npages == MOS_STACK_PAGES_KERNEL, "kernel stack size is not %d pages", MOS_STACK_PAGES_KERNEL);
35 child_vmblock = platform_mm_alloc_pages(child->pagetable, block.vm.npages, PGALLOC_HINT_STACK, block.vm.flags);
36 pr_info2("fork %d->%d: kernel stack " PTR_FMT "+%d, flags = [%x]", parent->pid, child->pid, block.vm.vaddr, block.vm.npages, block.vm.flags);
37 process_attach_mmap(child, child_vmblock, VMTYPE_KSTACK, MMAP_DEFAULT);
38 }
39 else
40 {
41 parent->mmaps[i].map_flags |= MMAP_COW;
42 mm_make_process_map_cow(parent->pagetable, block.vm.vaddr, child->pagetable, block.vm.vaddr, block.vm.npages);
43 child_vmblock = parent->mmaps[i].vm; // do not use the return value from mm_make_process_map_cow
44 pr_info2("fork %d->%d: CoW " PTR_FMT "+%d, flags = [%x]", parent->pid, child->pid, block.vm.vaddr, block.vm.npages, block.vm.flags);
45 process_attach_mmap(child, child_vmblock, block.type, MMAP_COW);
46 }
47 }
48
49 // copy the parent's files
50 for (int i = 0; i < parent->files_count; i++)
51 {
52 io_t *file = parent->files[i];
53 process_attach_ref_fd(child, file);
54 }
55
56 // copy the parent's threads
57 for (int i = 0; i < parent->threads_count; i++)
58 {
59 thread_t *parent_thread = parent->threads[i];
60 if (parent_thread != current_thread)
61 continue;
62
63 thread_t *child_thread = thread_allocate(child, parent_thread->flags);
64 child_thread->stack = parent_thread->stack;
65 child_thread->kernel_stack = parent_thread->kernel_stack;
66 child_thread->status = THREAD_STATUS_CREATED;
67 platform_context_copy(parent_thread->context, &child_thread->context);
68
69 process_attach_thread(child, child_thread);
70 hashmap_put(thread_table, &child_thread->tid, child_thread);
71 }
72
73 hashmap_put(process_table, &child->pid, child);
74 return child;
75}
