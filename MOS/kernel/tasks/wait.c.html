7static bool wait_for_thread_is_ready(wait_condition_t *condition)
8{
9 thread_t *thread = condition->arg;
10 return thread->status == THREAD_STATUS_DEAD;
11}
12
13static bool wait_for_mutex_is_ready(wait_condition_t *condition)
14{
15 // return true if the mutex is free (aka false)
16 bool *mutex = condition->arg;
17 return *mutex == MUTEX_UNLOCKED; // TODO: this is unsafe in SMP
18}
19
20wait_condition_t *wc_wait_for_thread(thread_t *target)
21{
22 wait_condition_t *condition = kmalloc(sizeof(wait_condition_t));
23 condition->arg = target;
24 condition->verify = wait_for_thread_is_ready;
25 condition->cleanup = NULL;
26 return condition;
27}
28
29wait_condition_t *wc_wait_for_mutex(bool *mutex)
30{
31 wait_condition_t *condition = kmalloc(sizeof(wait_condition_t));
32 condition->arg = mutex;
33 condition->verify = wait_for_mutex_is_ready;
34 condition->cleanup = NULL;
35 return condition;
36}
37
38void wc_condition_cleanup(wait_condition_t *condition)
39{
40 if (condition->cleanup)
41 condition->cleanup(condition);
42 kfree(condition);
43}
