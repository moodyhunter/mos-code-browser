9static bool thread_is_ready(wait_condition_t *condition)
10{
11 thread_t *thread = condition->arg;
12 return thread->state == THREAD_STATE_DEAD;
13}
14
15static bool mutex_is_ready(wait_condition_t *condition)
16{
17 // return true if the mutex is free (aka false)
18 bool *mutex = condition->arg;
19 return *mutex == MUTEX_UNLOCKED; // TODO: this is unsafe in SMP
20}
21
22wait_condition_t *wc_wait_for_thread(thread_t *target)
23{
24 return wc_wait_for(target, thread_is_ready, NULL);
25}
26
27wait_condition_t *wc_wait_for_mutex(bool *mutex)
28{
29 return wc_wait_for(mutex, mutex_is_ready, NULL);
30}
31
32wait_condition_t *wc_wait_for(void *arg, bool (*verify)(wait_condition_t *condition), void (*cleanup)(wait_condition_t *condition))
33{
34 wait_condition_t *condition = kmalloc(sizeof(wait_condition_t));
35 condition->arg = arg;
36 condition->verify = verify;
37 condition->cleanup = cleanup;
38 return condition;
39}
40
41bool wc_condition_verify(wait_condition_t *condition)
42{
43 MOS_ASSERT_X(condition->verify, "wait condition has no verify function");
44 return condition->verify(condition);
45}
46
47void wc_condition_cleanup(wait_condition_t *condition)
48{
49 if (condition->cleanup)
50 condition->cleanup(condition);
51 kfree(condition);
52}
