5#include "mos/mm/kmalloc.h"

7static bool wait_for_thread_is_ready(wait_condition_t *condition)
8{
9 thread_t *thread = condition->arg;
10 return thread->status == THREAD_STATUS_DEAD;
11}
12
13wait_condition_t *wc_wait_for_thread(thread_t *target)
14{
15 wait_condition_t *condition = kmalloc(sizeof(wait_condition_t));
16 condition->arg = target;
17 condition->verify = wait_for_thread_is_ready;
18 return condition;
19}
20
21void wc_condition_cleanup(wait_condition_t *condition)
22{
23 if (condition->cleanup)
24 condition->cleanup(condition);
25 kfree(condition);
26}
