10static size_t stdin_read(io_t *io, void *buf, size_t count)
11{
12 stdio_t *stdio = container_of(io, stdio_t, io);
13 MOS_ASSERT(stdio->type == STDIO_TYPE_STDIN);
14 MOS_UNUSED(io);
15 MOS_UNUSED(buf);
16 MOS_UNUSED(count);
17 return 0;
18}
19
20static size_t stdout_write(io_t *io, const void *buf, size_t count)
21{
22 if (!buf)
23 return 0;
24 stdio_t *stdio = container_of(io, stdio_t, io);
25 MOS_ASSERT(stdio->type == STDIO_TYPE_STDOUT);
26 lprintk(0, "%.*s", (int) count, (const char *) buf);
27 return 0;
28}
29
30static size_t stderr_write(io_t *io, const void *buf, size_t count)
31{
32 if (!buf)
33 return 0;
34 stdio_t *stdio = container_of(io, stdio_t, io);
35 MOS_ASSERT(stdio->type == STDIO_TYPE_STDERR);
36 lprintk(0, "stderr: %.*s", (int) count, (const char *) buf);
37 return 0;
38}
39
40static const io_op_t task_stdin_op = { .read = stdin_read };
41static const io_op_t task_stdout_op = { .write = stdout_write };
42static const io_op_t task_stderr_op = { .write = stderr_write };
43
44void process_stdio_setup(process_t *process)
45{
46 stdio_t *stdin = kcalloc(1, sizeof(stdio_t));
47 stdin->type = STDIO_TYPE_STDIN;
48 io_init(&stdin->io, IO_READABLE, -1, &task_stdin_op);
49 process_attach_fd(process, &stdin->io);
50
51 stdio_t *stdout = kcalloc(1, sizeof(stdio_t));
52 stdout->type = STDIO_TYPE_STDOUT;
53 io_init(&stdout->io, IO_WRITABLE, -1, &task_stdout_op);
54 process_attach_fd(process, &stdout->io);
55
56 stdio_t *stderr = kcalloc(1, sizeof(stdio_t));
57 stderr->type = STDIO_TYPE_STDERR;
58 io_init(&stderr->io, IO_WRITABLE, -1, &task_stderr_op);
59 process_attach_fd(process, &stderr->io);
60}
