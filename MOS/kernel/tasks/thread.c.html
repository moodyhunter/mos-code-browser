15static u32 STACK_NPAGES = 0;
16hashmap_t *thread_table;
17
18static hash_t hashmap_thread_hash(const void *key)
19{
20 return (hash_t){ .hash = *(tid_t *) key };
21}
22
23static int hashmap_thread_equal(const void *key1, const void *key2)
24{
25 return *(tid_t *) key1 == *(tid_t *) key2;
26}
27
28void thread_init()
29{
30 thread_table = kmalloc(sizeof(hashmap_t));
31 memset(thread_table, 0, sizeof(hashmap_t));
32 hashmap_init(thread_table, MOS_THREAD_STACK_SIZE, hashmap_thread_hash, hashmap_thread_equal);
33 STACK_NPAGES = MOS_THREAD_STACK_SIZE / MOS_PAGE_SIZE;
34}
35
36void thread_deinit()
37{
38 hashmap_deinit(thread_table);
39 kfree(thread_table);
40}

42static tid_t new_thread_id()
43{
44 static tid_t next = 1;
45 return (tid_t){ next++ };
46}
47
48thread_t *create_thread(process_t *owner, thread_flags_t tflags, thread_entry_t entry, void *arg)
49{
50 thread_t *t = kmalloc(sizeof(thread_t));
51 t->magic[0] = 'T';
52 t->magic[1] = 'H';
53 t->magic[2] = 'R';
54 t->magic[3] = 'D';
55 t->tid = new_thread_id();
56 t->owner = owner;
57 t->status = THREAD_STATUS_READY;
58 t->flags = tflags;
59
60 vm_flags sflags = VM_READ | VM_WRITE;
61 pgalloc_hints hints = PGALLOC_HINT_DEFAULT;
62
63 if (tflags & THREAD_FLAG_USERMODE)
64 sflags |= VM_USER, hints = PGALLOC_HINT_USERSPACE;
65
66 // allcate stack for the thread, in the kernel space
67 const vmblock_t stack_block = mos_platform->mm_alloc_pages(current_cpu->pagetable, STACK_NPAGES, hints, sflags);
68 void *const stack = (void *) stack_block.vaddr;
69 stack_init(&t->stack, stack, MOS_THREAD_STACK_SIZE);
70
71 // copy the stack mappping to the process address space
72 vmblock_t blk = mos_platform->mm_copy_maps(current_cpu->pagetable, (uintptr_t) stack, owner->pagetable, (uintptr_t) stack, stack_block.npages);
73 mos_platform->mm_flag_pages(owner->pagetable, blk.vaddr, blk.npages, sflags);
74
75 process_attach_mmap(owner, blk, VMTYPE_STACK, false);
76 mos_platform->context_setup(t, entry, arg);
77
78 hashmap_put(thread_table, &t->tid, t);
79 process_attach_thread(owner, t);
80
81 // unmap the stack from the kernel space
82 mos_platform->mm_unmap_pages(current_cpu->pagetable, stack_block.vaddr, stack_block.npages);
83 return t;
84}
85
86thread_t *get_thread(tid_t tid)
87{
88 return hashmap_get(thread_table, &tid);
89}
