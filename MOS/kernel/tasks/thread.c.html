16#define THREAD_HASHTABLE_SIZE 512
17
18hashmap_t *thread_table;
19
20static hash_t hashmap_thread_hash(const void *key)
21{
22 return (hash_t){ .hash = *(tid_t *) key };
23}
24
25static int hashmap_thread_equal(const void *key1, const void *key2)
26{
27 return *(tid_t *) key1 == *(tid_t *) key2;
28}
29
30static tid_t new_thread_id()
31{
32 static tid_t next = 1;
33 return (tid_t){ next++ };
34}
35
36thread_t *thread_allocate(process_t *owner, thread_flags_t tflags)
37{
38 thread_t *t = kmalloc(sizeof(thread_t));
39 t->magic[0] = 'T';
40 t->magic[1] = 'H';
41 t->magic[2] = 'R';
42 t->magic[3] = 'D';
43 t->tid = new_thread_id();
44 t->owner = owner;
45 t->status = THREAD_STATUS_CREATED;
46 t->flags = tflags;
47
48 return t;
49}
50
51void thread_init()
52{
53 thread_table = kmalloc(sizeof(hashmap_t));
54 memset(thread_table, 0, sizeof(hashmap_t));
55 hashmap_init(thread_table, THREAD_HASHTABLE_SIZE, hashmap_thread_hash, hashmap_thread_equal);
56}
57
58void thread_deinit()
59{
60 hashmap_deinit(thread_table);
61 kfree(thread_table);
62}

64thread_t *thread_new(process_t *owner, thread_flags_t tflags, thread_entry_t entry, void *arg)
65{
66 thread_t *t = thread_allocate(owner, tflags);
67 t->current_instruction = (uintptr_t) entry;
68
69 {
70 // Kernel stack
71 const vmblock_t kstack_blk = mos_platform->mm_alloc_pages(owner->pagetable, MOS_STACK_PAGES_KERNEL, PGALLOC_HINT_USERSPACE, VM_READ | VM_WRITE);
72 stack_init(&t->kernel_stack, (void *) kstack_blk.vaddr, kstack_blk.npages * MOS_PAGE_SIZE);
73 process_attach_mmap(owner, kstack_blk, VMTYPE_KSTACK, false);
74 }
75
76 if (tflags & THREAD_FLAG_USERMODE)
77 {
78 // allcate stack for the thread
79 const vmblock_t ustack_blk = mos_platform->mm_alloc_pages(owner->pagetable, MOS_STACK_PAGES_USER, PGALLOC_HINT_USERSPACE, VM_READ | VM_WRITE | VM_USER);
80 stack_init(&t->stack, (void *) ustack_blk.vaddr, MOS_STACK_PAGES_USER * MOS_PAGE_SIZE);
81 process_attach_mmap(owner, ustack_blk, VMTYPE_STACK, false);
82
83 // map the stack into current kernel's address space, making a proxy stack
84 const vmblock_t ustack_proxy = mm_map_proxy_space(owner->pagetable, ustack_blk.vaddr, ustack_blk.npages);
85 downwards_stack_t proxy_stack;
86 stack_init(&proxy_stack, (void *) ustack_proxy.vaddr, ustack_proxy.npages * MOS_PAGE_SIZE);
87 mos_platform->context_setup(t, &proxy_stack, entry, arg);
88
89 t->stack.head -= proxy_stack.top - proxy_stack.head;
90 mm_unmap_proxy_space(ustack_proxy);
91 }
92
93 hashmap_put(thread_table, &t->tid, t);
94 process_attach_thread(owner, t);
95
96 return t;
97}
98
99thread_t *thread_get(tid_t tid)
100{
101 return hashmap_get(thread_table, &tid);
102}
