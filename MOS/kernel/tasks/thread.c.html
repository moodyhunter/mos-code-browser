15#define THREAD_HASHTABLE_SIZE 4096
16
17// TODO: make this configurable
18#define THREAD_STACK_SIZE 1 MB
19
20static u32 thread_stack_npages = 0;
21hashmap_t *thread_table;
22
23static hash_t hashmap_thread_hash(const void *key)
24{
25 return (hash_t){ .hash = *(tid_t *) key };
26}
27
28static int hashmap_thread_equal(const void *key1, const void *key2)
29{
30 return *(tid_t *) key1 == *(tid_t *) key2;
31}
32
33void thread_init()
34{
35 thread_table = kmalloc(sizeof(hashmap_t));
36 memset(thread_table, 0, sizeof(hashmap_t));
37 hashmap_init(thread_table, THREAD_HASHTABLE_SIZE, hashmap_thread_hash, hashmap_thread_equal);
38 thread_stack_npages = THREAD_STACK_SIZE / MOS_PAGE_SIZE;
39}
40
41void thread_deinit()
42{
43 hashmap_deinit(thread_table);
44 kfree(thread_table);
45}
46
47static tid_t new_thread_id()
48{
49 static tid_t next = 1;
50 return (tid_t){ next++ };
51}
52
53thread_t *create_thread(process_t *owner, thread_flags_t flags, thread_entry_t entry, void *arg)
54{
55 thread_t *thread = kmalloc(sizeof(thread_t));
56 thread->magic[0] = 'T';
57 thread->magic[1] = 'H';
58 thread->magic[2] = 'R';
59 thread->magic[3] = 'D';
60 thread->tid = new_thread_id();
61 thread->owner = owner;
62 thread->status = THREAD_STATUS_READY;
63 thread->flags = flags;
64
65 // allcate stack for the thread
66 void *stack_page = kpage_alloc(thread_stack_npages, PGALLOC_NONE);
67 stack_init(&thread->stack, stack_page, THREAD_STACK_SIZE);
68
69 vm_flags stack_flags = VM_WRITE;
70 if (flags & THREAD_FLAG_USERMODE)
71 stack_flags |= VM_USERMODE;
72
73 // thread stack
74 mos_platform->mm_map_kvaddr(owner->pagetable, (uintptr_t) stack_page, (uintptr_t) stack_page, thread_stack_npages, stack_flags);
75 mos_platform->context_setup(thread, entry, arg);
76
77 hashmap_put(thread_table, &thread->tid, thread);
78 return thread;
79}
80
81thread_t *get_thread(tid_t tid)
82{
83 return hashmap_get(thread_table, &tid);
84}
