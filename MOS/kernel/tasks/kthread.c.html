12static process_t *kthreadd = NULL;
13
14typedef struct kthread_arg
15{
16 thread_entry_t entry;
17 void *arg;
18} kthread_arg_t;
19
20static void kthread_entry(void *arg)
21{
22 kthread_arg_t *kthread_arg = arg;
23 kthread_arg->entry(kthread_arg->arg);
24 kfree(kthread_arg);
25 thread_handle_exit(current_thread);
26}
27
28void kthread_init(void)
29{
30 kthreadd = process_allocate(NULL, 0, "kthreadd");
31 MOS_ASSERT_X(kthreadd->pid == 2, "kthreadd should have pid 2");
32 hashmap_put(process_table, &kthreadd->pid, kthreadd);
33}
34
35thread_t *kthread_create(thread_entry_t entry, void *arg, const char *name)
36{
37 kthread_arg_t *kthread_arg = kzalloc(sizeof(kthread_arg_t));
38 kthread_arg->entry = entry;
39 kthread_arg->arg = arg;
40 thread_t *thread = thread_new(kthreadd, THREAD_MODE_KERNEL, name, kthread_entry, kthread_arg);
41 return thread;
42}
