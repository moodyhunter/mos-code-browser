11static process_t *kthreadd = NULL;
12
13void kthread_init(void)
14{
15 kthreadd = process_allocate(NULL, 0, "kthreadd");
16 MOS_ASSERT_X(kthreadd->pid == 2, "kthreadd should have pid 2");
17 hashmap_put(process_table, &kthreadd->pid, kthreadd);
18}
19
20thread_t *kthread_create(thread_entry_t entry, void *arg, const char *name)
21{
22 thread_t *thread = thread_new(kthreadd, THREAD_MODE_KERNEL, name, entry, arg);
23 return thread;
24}
