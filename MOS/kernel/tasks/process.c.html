14#define PROCESS_HASHTABLE_SIZE 512
15
16hashmap_t *process_table;

18static hash_t process_hash(const void *key)
19{
20 return (hash_t){ .hash = *(pid_t *) key };
21}
22
23static int process_equal(const void *key1, const void *key2)
24{
25 return *(pid_t *) key1 == *(pid_t *) key2;
26}
27
28static pid_t new_process_id(void)
29{
30 static pid_t next = 1;
31 return (pid_t){ next++ };
32}

34process_t *process_allocate(process_t *parent, uid_t euid, const char *name)
35{
36 process_t *proc = kmalloc(sizeof(process_t));
37 memzero(proc, sizeof(process_t));
38
39 proc->magic = PROCESS_MAGIC_PROC;
40
41 proc->pid = new_process_id();
42
43 if (likely(parent))
44 {
45 proc->parent = parent;
46 }
47 else if (unlikely(proc->pid == 1) || unlikely(proc->pid == 2))
48 {
49 proc->parent = proc;
50 pr_emph("special process %d (%s) created", proc->pid, name);
51 }
52 else
53 {
54 pr_emerg("process %d has no parent", proc->pid);
55 kfree(proc);
56 return NULL;
57 }
58
59 if (likely(name))
60 {
61 proc->name = duplicate_string(name, strlen(name));
62 }
63 else
64 {
65 proc->name = "<unknown>";
66 }
67
68 proc->effective_uid = euid;
69
70 if (unlikely(proc->pid == 2))
71 {
72 proc->pagetable.ptr = 0; // ! Special case: PID 2 (kthreadd) uses the kernel page table
73 }
74 else
75 {
76 proc->pagetable = platform_mm_create_user_pgd();
77 }
78
79 return proc;
80}
81
82void process_init(void)
83{
84 process_table = kmalloc(sizeof(hashmap_t));
85 memset(process_table, 0, sizeof(hashmap_t));
86 hashmap_init(process_table, PROCESS_HASHTABLE_SIZE, process_hash, process_equal);
87}
88
89void process_cleanup(void)
90{
91 hashmap_deinit(process_table);
92 kfree(process_table);
93}

95process_t *process_new(process_t *parent, uid_t euid, const char *name, terminal_t *term, thread_entry_t entry, void *arg)
96{
97 process_t *proc = process_allocate(parent, euid, name);
98 if (unlikely(!proc))
99 return NULL;
100
101 if (unlikely(!term))
102 {
103 if (likely(parent))
104 proc->terminal = parent->terminal;
105 else
106 mos_panic("init process has no terminal");
107 }
108
109 proc->terminal = term;
110 process_attach_ref_fd(proc, &term->io);
111 process_attach_ref_fd(proc, &term->io);
112 process_attach_ref_fd(proc, &term->io);
113
114 thread_new(proc, THREAD_MODE_USER, name, entry, arg);
115
116 vmblock_t heap = platform_mm_alloc_pages(proc->pagetable, 0, PGALLOC_HINT_UHEAP, VM_USER_RW);
117 process_attach_mmap(proc, heap, VMTYPE_HEAP, MMAP_DEFAULT);
118
119 void *old_proc = hashmap_put(process_table, &proc->pid, proc);
120 MOS_ASSERT_X(old_proc == NULL, "process already exists, go and buy yourself a lottery :)");
121 return proc;
122}

124process_t *process_get(pid_t pid)
125{
126 process_t *p = hashmap_get(process_table, &pid);
127 if (p == NULL)
128 mos_warn("process %d not found", pid);
129 return p;
130}

132fd_t process_attach_ref_fd(process_t *process, io_t *file)
133{
134 MOS_ASSERT(process_is_valid(process));
135 int fd = process->files_count++;
136 if (fd >= MOS_PROCESS_MAX_OPEN_FILES)
137 {
138 mos_warn("process %d has too many open files", process->pid);
139 return -1;
140 }
141
142 process->files[fd] = io_ref(file);
143 return fd;
144}
145
146io_t *process_get_fd(process_t *process, fd_t fd)
147{
148 MOS_ASSERT(process_is_valid(process));
149 if (fd < 0 || fd >= process->files_count)
150 return NULL;
151 return process->files[fd];
152}
153
154bool process_detach_fd(process_t *process, fd_t fd)
155{
156 MOS_ASSERT(process_is_valid(process));
157 if (fd < 0 || fd >= process->files_count)
158 return false;
159 io_unref(process->files[fd]);
160 process->files[fd] = NULL;
161 process->files_count--;
162 return true;
163}

165void process_attach_thread(process_t *process, thread_t *thread)
166{
167 MOS_ASSERT(process_is_valid(process));
168 MOS_ASSERT(thread_is_valid(thread));
169 MOS_ASSERT(thread->owner == process);
170 mos_debug("process %d attached thread %d", process->pid, thread->tid);
171 process->threads[process->threads_count++] = thread;
172}

174void process_attach_mmap(process_t *process, vmblock_t block, vm_type type, mmap_flags flags)
175{
176 MOS_ASSERT(process_is_valid(process));
177 process->mmaps = krealloc(process->mmaps, sizeof(proc_vmblock_t) * (process->mmaps_count + 1));
178 process->mmaps[process->mmaps_count++] = (proc_vmblock_t){ .vm = block, .type = type, .map_flags = flags };
179}

181void process_handle_exit(process_t *process, int exit_code)
182{
183 MOS_ASSERT(process_is_valid(process));
184 pr_info("process %d exited with code %d", process->pid, exit_code);
185
186 mos_debug("terminating all %lu threads owned by %d", process->threads_count, process->pid);
187 for (int i = 0; i < process->threads_count; i++)
188 {
189 thread_t *thread = process->threads[i];
190 if (thread->state == THREAD_STATE_DEAD)
191 {
192 pr_warn("thread %d is already dead", thread->tid);
193 continue;
194 }
195 thread->state = THREAD_STATE_DEAD; // cleanup will be done by the scheduler
196 }
197
198 mos_debug("closing all %lu files owned by %d", process->files_count, process->pid);
199 for (int i = 0; i < process->files_count; i++)
200 {
201 MOS_ASSERT_X(process->files[i], "file %d is NULL", i);
202 if (!process->files[i]->closed)
203 io_unref(process->files[i]);
204 process->files[i] = NULL;
205 }
206}

208void process_handle_cleanup(process_t *process)
209{
210 MOS_ASSERT(process_is_valid(process));
211 MOS_ASSERT_X(current_process != process, "cannot cleanup current process");
212
213 mos_debug("unmapping all %lu memory regions owned by %d", process->mmaps_count, process->pid);
214 for (int i = 0; i < process->mmaps_count; i++)
215 {
216 const mmap_flags flags = process->mmaps[i].map_flags;
217 const vmblock_t block = process->mmaps[i].vm;
218
219 // they will be unmapped when the last process detaches them
220 // !! TODO: How to track this??
221 if (flags & MMAP_COW)
222 continue;
223
224 platform_mm_free_pages(process->pagetable, block.vaddr, block.npages);
225 }
226}

228uintptr_t process_grow_heap(process_t *process, size_t npages)
229{
230 MOS_ASSERT(process_is_valid(process));
231
232 proc_vmblock_t *heap = NULL;
233 for (int i = 0; i < process->mmaps_count; i++)
234 {
235 if (process->mmaps[i].type == VMTYPE_HEAP)
236 {
237 heap = &process->mmaps[i];
238 break;
239 }
240 }
241
242 MOS_ASSERT(heap != NULL);
243
244 const uintptr_t heap_top = heap->vm.vaddr + heap->vm.npages * MOS_PAGE_SIZE;
245
246 if (heap->map_flags & MMAP_COW)
247 {
248 vmblock_t zeroed = mm_alloc_zeroed_pages_at(process->pagetable, heap_top, npages, VM_USER_RW);
249 MOS_ASSERT(zeroed.npages == npages);
250 }
251 else
252 {
253 vmblock_t new_part = platform_mm_alloc_pages_at(process->pagetable, heap_top, npages, VM_USER_RW);
254 if (new_part.vaddr == 0 || new_part.npages != npages)
255 {
256 mos_warn("failed to grow heap of process %d", process->pid);
257 platform_mm_free_pages(process->pagetable, new_part.vaddr, new_part.npages);
258 return heap_top;
259 }
260 }
261
262 pr_info2("grew heap of process %d by %zu pages", process->pid, npages);
263 heap->vm.npages += npages;
264 return heap_top + npages * MOS_PAGE_SIZE;
265}

267void process_dump_mmaps(process_t *process)
268{
269 pr_info("process %d has %lu memory regions:", process->pid, process->mmaps_count);
270 for (int i = 0; i < process->mmaps_count; i++)
271 {
272 proc_vmblock_t block = process->mmaps[i];
273 const char *type = "unknown";
274 switch (block.type)
275 {
276 case VMTYPE_APPCODE: type = "code"; break;
277 case VMTYPE_APPDATA: type = "data"; break;
278 case VMTYPE_APPDATA_ZERO: type = "data (zeroed)"; break;
279 case VMTYPE_HEAP: type = "heap"; break;
280 case VMTYPE_STACK: type = "stack"; break;
281 case VMTYPE_KSTACK: type = "stack (kernel)"; break;
282 case VMTYPE_SHM: type = "shared memory"; break;
283 case VMTYPE_FILE: type = "file"; break;
284 default: MOS_UNREACHABLE();
285 };
286
287 pr_info(" block %d: " PTR_FMT ", % 3zd page(s), [%c%c%c%c%c%c%c]: %s",
288 i, //
289 block.vm.vaddr, //
290 block.vm.npages, //
291 block.vm.flags & VM_READ ? 'r' : '-', //
292 block.vm.flags & VM_WRITE ? 'w' : '-', //
293 block.vm.flags & VM_EXEC ? 'x' : '-', //
294 block.vm.flags & VM_GLOBAL ? 'g' : '-', //
295 block.vm.flags & VM_USER ? 'u' : '-', //
296 block.map_flags & MMAP_COW ? 'c' : '-', //
297 block.map_flags & MMAP_PRIVATE ? 'p' : '-', //
298 type //
299 );
300 }
