15#define PROCESS_HASHTABLE_SIZE 512
16
17hashmap_t *process_table;
18
19static hash_t process_hash(const void *key)
20{
21 return (hash_t){ .hash = ((process_id_t *) key)->process_id };
22}
23
24static int process_equal(const void *key1, const void *key2)
25{
26 return ((process_id_t *) key1)->process_id == ((process_id_t *) key2)->process_id;
27}
28
29static process_id_t new_process_id(void)
30{
31 static process_id_t next = { 1 };
32 return (process_id_t){ next.process_id++ };
33}
34
35void process_init(void)
36{
37 process_table = kmalloc(sizeof(hashmap_t));
38 memset(process_table, 0, sizeof(hashmap_t));
39 hashmap_init(process_table, PROCESS_HASHTABLE_SIZE, process_hash, process_equal);
40}
41
42void process_deinit(void)
43{
44 hashmap_deinit(process_table);
45 kfree(process_table);
46}
47
48process_t *allocate_process(process_id_t parent_pid, uid_t euid, thread_entry_t entry, void *arg)
49{
50 mos_debug("create_process(parent: %d, euid: %d, arg: %p)", parent_pid.process_id, euid.uid, arg);
51 process_t *process = kmalloc(sizeof(process_t));
52 memset(process, 0, sizeof(process_t));
53
54 process->id = new_process_id();
55 process->effective_uid = euid;
56 process->parent_pid = parent_pid;
57
58 process->pagetable = mos_platform->mm_create_pagetable();
59
60 process_stdio_setup(process);
61 thread_t *main_thread = create_thread(process, THREAD_FLAG_USERMODE, entry, arg);
62 process->main_thread_id = main_thread->id;
63
64 void *old_proc = hashmap_put(process_table, &process->id, process);
65 MOS_ASSERT_X(old_proc == NULL, "process already exists, go and buy yourself a lottery :)");
66 return process;
67}
68
69process_t *get_process(process_id_t pid)
70{
71 process_t *p = hashmap_get(process_table, &pid);
72 if (p == NULL)
73 mos_warn("process %d not found", pid.process_id);
74 return p;
75}
