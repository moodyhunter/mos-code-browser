20#define PROCESS_HASHTABLE_SIZE 512
21
22hashmap_t *process_table;
23
24static hash_t process_hash(const void *key)
25{
26 return (hash_t){ .hash = *(pid_t *) key };
27}
28
29static int process_equal(const void *key1, const void *key2)
30{
31 return *(pid_t *) key1 == *(pid_t *) key2;
32}
33
34static pid_t new_process_id(void)
35{
36 static pid_t next = 1;
37 return (pid_t){ next++ };
38}
39
40process_t *process_allocate(process_t *parent, uid_t euid, const char *name)
41{
42 process_t *proc = kmalloc(sizeof(process_t));
43 memzero(proc, sizeof(process_t));
44
45 proc->magic[0] = 'P';
46 proc->magic[1] = 'R';
47 proc->magic[2] = 'O';
48 proc->magic[3] = 'C';
49 proc->name = "<unknown>";
50
51 proc->pid = new_process_id();
52
53 if (likely(parent))
54 proc->parent = parent;
55 else if (unlikely(proc->pid == 1))
56 proc->parent = proc;
57 else
58 {
59 pr_emerg("process %d has no parent", proc->pid);
60 kfree(proc);
61 return NULL;
62 }
63
64 if (name)
65 {
66 proc->name = kmalloc(strlen(name) + 1);
67 strcpy((char *) proc->name, name);
68 }
69
70 proc->effective_uid = euid;
71 proc->pagetable = mos_platform->mm_create_user_pgd();
72 return proc;
73}
74
75void process_init(void)
76{
77 process_table = kmalloc(sizeof(hashmap_t));
78 memset(process_table, 0, sizeof(hashmap_t));
79 hashmap_init(process_table, PROCESS_HASHTABLE_SIZE, process_hash, process_equal);
80}
81
82void process_deinit(void)
83{
84 hashmap_deinit(process_table);
85 kfree(process_table);
86}
87
88process_t *process_new(process_t *parent, uid_t euid, const char *name, thread_entry_t entry, void *arg)
89{
90 process_t *proc = process_allocate(parent, euid, name);
91 process_stdio_setup(proc);
92 proc->main_thread = thread_new(proc, THREAD_FLAG_USERMODE, entry, arg);
93 void *old_proc = hashmap_put(process_table, &proc->pid, proc);
94 MOS_ASSERT_X(old_proc == NULL, "process already exists, go and buy yourself a lottery :)");
95 return proc;
96}
97
98process_t *process_get(pid_t pid)
99{
100 process_t *p = hashmap_get(process_table, &pid);
101 if (p == NULL)
102 mos_warn("process %d not found", pid);
103 return p;
104}

106fd_t process_attach_fd(process_t *process, io_t *file)
107{
108 MOS_ASSERT(process_is_valid(process));
109 int fd = process->files_count++;
110 if (fd >= MOS_PROCESS_MAX_OPEN_FILES)
111 {
112 mos_warn("process %d has too many open files", process->pid);
113 return -1;
114 }
115
116 process->files[fd] = file;
117 return fd;
118}
119
120bool process_detach_fd(process_t *process, fd_t fd)
121{
122 MOS_ASSERT(process_is_valid(process));
123 if (fd < 0 || fd >= process->files_count)
124 return false;
125 io_close(process->files[fd]);
126 process->files[fd] = NULL;
127 process->files_count--;
128 return true;
129}

131void process_attach_thread(process_t *process, thread_t *thread)
132{
133 MOS_ASSERT(process_is_valid(process));
134 MOS_ASSERT(thread_is_valid(thread));
135 MOS_ASSERT(thread->owner == process);
136 mos_debug("process %d attached thread %d", process->pid, thread->tid);
137 process->threads[process->threads_count++] = thread;
138}

140void process_attach_mmap(process_t *process, vmblock_t block, vm_type type, bool cow)
141{
142 MOS_ASSERT(process_is_valid(process));
143 process->mmaps = krealloc(process->mmaps, sizeof(proc_vmblock_t) * (process->mmaps_count + 1));
144 process->mmaps[process->mmaps_count++] = (proc_vmblock_t){ .vm = block, .type = type, .map_flags = (cow ? MMAP_COW : MMAP_DEFAULT) };
145}
146
147void process_handle_exit(process_t *process, int exit_code)
148{
149 MOS_ASSERT(process_is_valid(process));
150 pr_info("process %d exited with code %d", process->pid, exit_code);
151
152 mos_debug("terminating all %lu threads owned by %d", process->threads_count, process->pid);
153 for (int i = 0; i < process->threads_count; i++)
154 {
155 thread_t *thread = process->threads[i];
156 if (thread->status == THREAD_STATUS_DEAD)
157 {
158 mos_warn("thread %d is already dead", thread->tid);
159 continue;
160 }
161 thread->status = THREAD_STATUS_DEAD;
162 }
163}

165void process_dump_mmaps(process_t *process)
166{
167 for (int i = 0; i < process->mmaps_count; i++)
168 {
169 proc_vmblock_t block = process->mmaps[i];
170 const char *type = "unknown";
171 switch (block.type)
172 {
173 case VMTYPE_APPCODE: type = "code"; break;
174 case VMTYPE_APPDATA: type = "data"; break;
175 case VMTYPE_STACK: type = "stack"; break;
176 case VMTYPE_KSTACK: type = "kernel stack"; break;
177 case VMTYPE_FILE: type = "file"; break;
178 default: MOS_UNREACHABLE();
179 };
180
181 pr_info("block %d: " PTR_FMT ", %zd page(s), %s%sperm: %s%s%s%s%s%s%s -> %s",
182 i, //
183 block.vm.vaddr, //
184 block.vm.npages, //
185 block.map_flags & MMAP_COW ? "cow, " : "", //
186 block.map_flags & MMAP_PRIVATE ? "private, " : "", //
187 block.vm.flags & VM_READ ? "r" : "-", //
188 block.vm.flags & VM_WRITE ? "w" : "-", //
189 block.vm.flags & VM_EXEC ? "x" : "-", //
190 block.vm.flags & VM_WRITE_THROUGH ? "W" : "-", //
191 block.vm.flags & VM_CACHE_DISABLED ? "-" : "c", //
192 block.vm.flags & VM_GLOBAL ? "g" : "-", //
193 block.vm.flags & VM_USER ? "u" : "-", //
194 type //
195 );
196 }
