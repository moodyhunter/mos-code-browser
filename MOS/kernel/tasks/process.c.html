13#define PROCESS_HASHTABLE_SIZE 512
14
15hashmap_t *process_table;
16
17static hash_t process_hash(const void *key)
18{
19 return (hash_t){ .hash = *(pid_t *) key };
20}
21
22static int process_equal(const void *key1, const void *key2)
23{
24 return *(pid_t *) key1 == *(pid_t *) key2;
25}
26
27static pid_t new_process_id(void)
28{
29 static pid_t next = 1;
30 return (pid_t){ next++ };
31}
32
33process_t *process_allocate(process_t *parent, uid_t euid, const char *name)
34{
35 process_t *proc = kmalloc(sizeof(process_t));
36 memzero(proc, sizeof(process_t));
37
38 proc->magic = PROCESS_MAGIC_PROC;
39
40 proc->pid = new_process_id();
41
42 if (likely(parent))
43 {
44 proc->parent = parent;
45 }
46 else if (unlikely(proc->pid == 1))
47 {
48 proc->parent = proc;
49 MOS_ASSERT_ONCE("Only the init process should have a parent of itself");
50 pr_emph("created init process");
51 }
52 else
53 {
54 pr_emerg("process %d has no parent", proc->pid);
55 kfree(proc);
56 return NULL;
57 }
58
59 if (likely(name))
60 {
61 proc->name = kmalloc(strlen(name) + 1);
62 strcpy((char *) proc->name, name);
63 }
64 else
65 {
66 proc->name = "<unknown>";
67 }
68
69 proc->effective_uid = euid;
70 proc->pagetable = platform_mm_create_user_pgd();
71 return proc;
72}
73
74void process_init(void)
75{
76 process_table = kmalloc(sizeof(hashmap_t));
77 memset(process_table, 0, sizeof(hashmap_t));
78 hashmap_init(process_table, PROCESS_HASHTABLE_SIZE, process_hash, process_equal);
79}
80
81void process_cleanup(void)
82{
83 hashmap_deinit(process_table);
84 kfree(process_table);
85}

87process_t *process_new(process_t *parent, uid_t euid, const char *name, terminal_t *term, thread_entry_t entry, void *arg)
88{
89 process_t *proc = process_allocate(parent, euid, name);
90 if (unlikely(!proc))
91 return NULL;
92
93 if (unlikely(!term))
94 {
95 if (likely(parent))
96 proc->terminal = parent->terminal;
97 else
98 mos_panic("init process has no terminal");
99 }
100
101 proc->terminal = term;
102 process_attach_ref_fd(proc, &term->io);
103 process_attach_ref_fd(proc, &term->io);
104 process_attach_ref_fd(proc, &term->io);
105
106 thread_new(proc, THREAD_FLAG_USERMODE, entry, arg);
107 void *old_proc = hashmap_put(process_table, &proc->pid, proc);
108 MOS_ASSERT_X(old_proc == NULL, "process already exists, go and buy yourself a lottery :)");
109 return proc;
110}

112process_t *process_get(pid_t pid)
113{
114 process_t *p = hashmap_get(process_table, &pid);
115 if (p == NULL)
116 mos_warn("process %d not found", pid);
117 return p;
118}

120fd_t process_attach_ref_fd(process_t *process, io_t *file)
121{
122 MOS_ASSERT(process_is_valid(process));
123 int fd = process->files_count++;
124 if (fd >= MOS_PROCESS_MAX_OPEN_FILES)
125 {
126 mos_warn("process %d has too many open files", process->pid);
127 return -1;
128 }
129
130 process->files[fd] = io_ref(file);
131 return fd;
132}
133
134bool process_detach_fd(process_t *process, fd_t fd)
135{
136 MOS_ASSERT(process_is_valid(process));
137 if (fd < 0 || fd >= process->files_count)
138 return false;
139 io_unref(process->files[fd]);
140 process->files[fd] = NULL;
141 process->files_count--;
142 return true;
143}

145void process_attach_thread(process_t *process, thread_t *thread)
146{
147 MOS_ASSERT(process_is_valid(process));
148 MOS_ASSERT(thread_is_valid(thread));
149 MOS_ASSERT(thread->owner == process);
150 mos_debug("process %d attached thread %d", process->pid, thread->tid);
151 process->threads[process->threads_count++] = thread;
152}

154void process_attach_mmap(process_t *process, vmblock_t block, vm_type type, bool cow)
155{
156 MOS_ASSERT(process_is_valid(process));
157 process->mmaps = krealloc(process->mmaps, sizeof(proc_vmblock_t) * (process->mmaps_count + 1));
158 process->mmaps[process->mmaps_count++] = (proc_vmblock_t){ .vm = block, .type = type, .map_flags = (cow ? MMAP_COW : MMAP_DEFAULT) };
159}

161void process_handle_exit(process_t *process, int exit_code)
162{
163 MOS_ASSERT(process_is_valid(process));
164 pr_info("process %d exited with code %d", process->pid, exit_code);
165
166 mos_debug("terminating all %lu threads owned by %d", process->threads_count, process->pid);
167 for (int i = 0; i < process->threads_count; i++)
168 {
169 thread_t *thread = process->threads[i];
170 if (thread->status == THREAD_STATUS_DEAD)
171 {
172 mos_warn("thread %d is already dead", thread->tid);
173 continue;
174 }
175 thread->status = THREAD_STATUS_DEAD; // cleanup will be done by the scheduler
176 }
177
178 mos_debug("closing all %lu files owned by %d", process->files_count, process->pid);
179 for (int i = 0; i < process->files_count; i++)
180 {
181 MOS_ASSERT_X(process->files[i], "file %d is NULL", i);
182 if (!process->files[i]->closed)
183 io_unref(process->files[i]);
184 process->files[i] = NULL;
185 }
186}

188void process_handle_cleanup(process_t *process)
189{
190 MOS_ASSERT(process_is_valid(process));
191 MOS_ASSERT_X(current_process != process, "cannot cleanup current process");
192
193 mos_debug("cleaning up process %d", process->pid);
194 mos_debug("freeing all %lu memory regions owned by %d", process->mmaps_count, process->pid);
195
196 mos_debug("unmapping all %lu memory regions owned by %d", process->mmaps_count, process->pid);
197 for (int i = 0; i < process->mmaps_count; i++)
198 {
199 const mmap_flags flags = process->mmaps[i].map_flags;
200 const vmblock_t block = process->mmaps[i].vm;
201
202 // they will be unmapped when the last process detaches them
203 if (flags & MMAP_COW)
204 continue;
205
206 platform_mm_free_pages(process->pagetable, block.vaddr, block.npages);
207 }
208}

210void process_dump_mmaps(process_t *process)
211{
212 pr_info("process %d has %lu memory regions:", process->pid, process->mmaps_count);
213 for (int i = 0; i < process->mmaps_count; i++)
214 {
215 proc_vmblock_t block = process->mmaps[i];
216 const char *type = "unknown";
217 switch (block.type)
218 {
219 case VMTYPE_APPCODE: type = "code"; break;
220 case VMTYPE_APPDATA: type = "data"; break;
221 case VMTYPE_STACK: type = "stack"; break;
222 case VMTYPE_KSTACK: type = "stack (kernel)"; break;
223 case VMTYPE_FILE: type = "file"; break;
224 default: MOS_UNREACHABLE();
225 };
226
227 pr_info(" block %d: " PTR_FMT ", % 3zd page(s), [%c%c%c%c%c%c%c]: %s",
228 i, //
229 block.vm.vaddr, //
230 block.vm.npages, //
231 block.vm.flags & VM_READ ? 'r' : '-', //
232 block.vm.flags & VM_WRITE ? 'w' : '-', //
233 block.vm.flags & VM_EXEC ? 'x' : '-', //
234 block.vm.flags & VM_GLOBAL ? 'g' : '-', //
235 block.vm.flags & VM_USER ? 'u' : '-', //
236 block.map_flags & MMAP_COW ? 'c' : '-', //
237 block.map_flags & MMAP_PRIVATE ? 'p' : '-', //
238 type //
239 );
240 }
