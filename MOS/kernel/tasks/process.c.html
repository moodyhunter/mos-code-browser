16#define PROCESS_HASHTABLE_SIZE 512
17
18hashmap_t *process_table;
19
20static hash_t process_hash(const void *key)
21{
22 return (hash_t){ .hash = *(pid_t *) key };
23}
24
25static int process_equal(const void *key1, const void *key2)
26{
27 return *(pid_t *) key1 == *(pid_t *) key2;
28}
29
30static pid_t new_process_id(void)
31{
32 static pid_t next = 1;
33 return (pid_t){ next++ };
34}
35
36void process_init(void)
37{
38 process_table = kmalloc(sizeof(hashmap_t));
39 memset(process_table, 0, sizeof(hashmap_t));
40 hashmap_init(process_table, PROCESS_HASHTABLE_SIZE, process_hash, process_equal);
41}
42
43void process_deinit(void)
44{
45 hashmap_deinit(process_table);
46 kfree(process_table);
47}

49process_t *allocate_process(process_t *parent, uid_t euid, const char *name, thread_entry_t entry, void *arg)
50{
51 process_t *proc = kmalloc(sizeof(process_t));
52 memzero(proc, sizeof(process_t));
53
54 proc->magic[0] = 'P';
55 proc->magic[1] = 'R';
56 proc->magic[2] = 'O';
57 proc->magic[3] = 'C';
58
59 proc->pid = new_process_id();
60
61 if (proc->pid == 1)
62 {
63 proc->parent_pid = proc->pid;
64 }
65 else
66 {
67 if (!parent)
68 {
69 pr_emerg("process %d has no parent", proc->pid);
70 kfree(proc);
71 return NULL;
72 }
73 proc->parent_pid = parent->pid;
74 }
75
76 proc->name = "<unknown>";
77 if (name)
78 {
79 proc->name = kmalloc(strlen(name) + 1);
80 strcpy((char *) proc->name, name);
81 }
82
83 proc->effective_uid = euid;
84 proc->pagetable = mos_platform->mm_create_pagetable();
85
86 process_stdio_setup(proc);
87 thread_t *main_thread = create_thread(proc, THREAD_FLAG_USERMODE, entry, arg);
88 proc->main_thread_id = main_thread->tid;
89
90 void *old_proc = hashmap_put(process_table, &proc->pid, proc);
91 MOS_ASSERT_X(old_proc == NULL, "process already exists, go and buy yourself a lottery :)");
92 return proc;
93}

95process_t *get_process(pid_t pid)
96{
97 process_t *p = hashmap_get(process_table, &pid);
98 if (p == NULL)
99 mos_warn("process %d not found", pid);
100 return p;
101}

103fd_t process_attach_fd(process_t *process, io_t *file)
104{
105 MOS_ASSERT(process_is_valid(process));
106 int fd = process->files_count++;
107 if (fd >= MOS_PROCESS_MAX_OPEN_FILES)
108 {
109 mos_warn("process %d has too many open files", process->pid);
110 return -1;
111 }
112
113 process->files[fd] = file;
114 return fd;
115}
116
117bool process_detach_fd(process_t *process, fd_t fd)
118{
119 MOS_ASSERT(process_is_valid(process));
120 if (fd < 0 || fd >= process->files_count)
121 return false;
122 io_close(process->files[fd]);
123 process->files[fd] = NULL;
124 return true;
125}
