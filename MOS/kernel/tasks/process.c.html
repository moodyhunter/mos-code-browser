18#define PROCESS_HASHTABLE_SIZE 512
19
20hashmap_t *process_table;
21
22static hash_t process_hash(const void *key)
23{
24 return (hash_t){ .hash = *(pid_t *) key };
25}
26
27static int process_equal(const void *key1, const void *key2)
28{
29 return *(pid_t *) key1 == *(pid_t *) key2;
30}
31
32static pid_t new_process_id(void)
33{
34 static pid_t next = 1;
35 return (pid_t){ next++ };
36}
37
38void process_init(void)
39{
40 process_table = kmalloc(sizeof(hashmap_t));
41 memset(process_table, 0, sizeof(hashmap_t));
42 hashmap_init(process_table, PROCESS_HASHTABLE_SIZE, process_hash, process_equal);
43}
44
45void process_deinit(void)
46{
47 hashmap_deinit(process_table);
48 kfree(process_table);
49}

51process_t *process_new(process_t *parent, uid_t euid, const char *name, thread_entry_t entry, void *arg)
52{
53 process_t *proc = kmalloc(sizeof(process_t));
54 memzero(proc, sizeof(process_t));
55
56 proc->magic[0] = 'P';
57 proc->magic[1] = 'R';
58 proc->magic[2] = 'O';
59 proc->magic[3] = 'C';
60
61 proc->pid = new_process_id();
62
63 if (proc->pid == 1)
64 proc->parent_pid = proc->pid;
65 else if (!parent)
66 {
67 proc->parent_pid = parent->pid;
68 pr_emerg("process %d has no parent", proc->pid);
69 kfree(proc);
70 return NULL;
71 }
72
73 proc->name = "<unknown>";
74 if (name)
75 {
76 proc->name = kmalloc(strlen(name) + 1);
77 strcpy((char *) proc->name, name);
78 }
79
80 proc->effective_uid = euid;
81 proc->pagetable = mos_platform->mm_create_pagetable();
82
83 process_stdio_setup(proc);
84 proc->main_thread = create_thread(proc, THREAD_FLAG_USERMODE, entry, arg);
85
86 void *old_proc = hashmap_put(process_table, &proc->pid, proc);
87 MOS_ASSERT_X(old_proc == NULL, "process already exists, go and buy yourself a lottery :)");
88 return proc;
89}

91process_t *process_get(pid_t pid)
92{
93 process_t *p = hashmap_get(process_table, &pid);
94 if (p == NULL)
95 mos_warn("process %d not found", pid);
96 return p;
97}

99fd_t process_attach_fd(process_t *process, io_t *file)
100{
101 MOS_ASSERT(process_is_valid(process));
102 int fd = process->files_count++;
103 if (fd >= MOS_PROCESS_MAX_OPEN_FILES)
104 {
105 mos_warn("process %d has too many open files", process->pid);
106 return -1;
107 }
108
109 process->files[fd] = file;
110 return fd;
111}
112
113bool process_detach_fd(process_t *process, fd_t fd)
114{
115 MOS_ASSERT(process_is_valid(process));
116 if (fd < 0 || fd >= process->files_count)
117 return false;
118 io_close(process->files[fd]);
119 process->files[fd] = NULL;
120 return true;
121}

123void process_attach_thread(process_t *process, thread_t *thread)
124{
125 MOS_ASSERT(process_is_valid(process));
126 MOS_ASSERT(thread_is_valid(thread));
127 pr_info("process %d attached thread %d", process->pid, thread->tid);
128 process->threads[process->threads_count++] = thread;
129}

131void process_attach_mmap(process_t *process, vmblock_t block, vm_type type)
132{
133 MOS_ASSERT(process_is_valid(process));
134 process->mmaps = krealloc(process->mmaps, sizeof(vmblock_t) * (process->mmaps_count + 1));
135 process->mmaps[process->mmaps_count++] = (proc_vmblock_t){ .vm = block, .type = type };
136}

138void process_handle_exit(process_t *process, int exit_code)
139{
140 MOS_ASSERT(process_is_valid(process));
141 pr_info("process %d exited with code %d", process->pid, exit_code);
142
143 mos_debug("terminating all %lu threads owned by %d", process->threads_count, process->pid);
144 for (int i = 0; i < process->threads_count; i++)
145 {
146 thread_t *thread = process->threads[i];
147 if (thread->status == THREAD_STATUS_DEAD)
148 {
149 mos_warn("thread %d is already dead", thread->tid);
150 continue;
151 }
152 thread->status = THREAD_STATUS_DEAD;
153 }
154}

156process_t *process_handle_fork(process_t *process)
157{
158 MOS_ASSERT(process_is_valid(process));
159 pr_info("process %d forked", process->pid);
160
161 // TODO: get the returned address (replace NULL)
162 process_t *child = process_new(process, process->effective_uid, process->name, NULL, NULL);
163
164 child->main_thread->status = THREAD_STATUS_WAITING;
165
166 // copy the parent's memory
167 for (int i = 0; i < process->mmaps_count; i++)
168 {
169 proc_vmblock_t block = process->mmaps[i];
170 }
171
172 return NULL;
173}

175void process_dump_mmaps(process_t *process)
176{
177 for (int i = 0; i < process->mmaps_count; i++)
178 {
179 proc_vmblock_t block = process->mmaps[i];
180 const char *type = "unknown";
181 switch (block.type)
182 {
183 case VMTYPE_APPCODE: type = "code"; break;
184 case VMTYPE_APPDATA: type = "data"; break;
185 case VMTYPE_STACK: type = "stack"; break;
186 case VMTYPE_FILE: type = "file"; break;
187 default: MOS_UNREACHABLE();
188 };
189
190 pr_info("block %d: " VPTR_FMT " -> " PPTR_FMT ", %zd bytes, perm: %s%s%s%s%s%s%s -> %s",
191 i, //
192 block.vm.mem.vaddr, //
193 block.vm.mem.paddr, //
194 block.vm.mem.size_bytes, //
195 block.vm.flags & VM_READ ? "r" : "-", //
196 block.vm.flags & VM_WRITE ? "w" : "-", //
197 block.vm.flags & VM_EXEC ? "x" : "-", //
198 block.vm.flags & VM_WRITE_THROUGH ? "W" : "-", //
199 block.vm.flags & VM_CACHE_DISABLED ? "-" : "c", //
200 block.vm.flags & VM_GLOBAL ? "g" : "-", //
201 block.vm.flags & VM_USERMODE ? "u" : "-", //
202 type //
203 );
204 }
