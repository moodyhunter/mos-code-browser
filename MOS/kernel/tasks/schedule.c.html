13static bool should_schedule_to_thread(thread_t *thread)
14{
15 bool state_ok = thread->status == THREAD_STATUS_READY || thread->status == THREAD_STATUS_CREATED;
16 if (!state_ok)
17 return false;
18
19 if (thread->waiting_condition)
20 {
21 wait_condition_t *condition = thread->waiting_condition;
22 MOS_ASSERT_X(condition->verify, "wait condition has no verify function");
23
24 if (!condition->verify(condition))
25 return false;
26
27 wc_condition_cleanup(condition);
28 thread->waiting_condition = NULL;
29 }
30
31 // A thread is
32 // - in READY or CREATED state, and
33 // - its wait condition is satisfied
34 return true;
35}

37void mos_update_current(thread_t *new_current)
38{
39 thread_t *old_current = current_thread;
40 // In the very first switch, current_thread is NULL
41 if (likely(old_current))
42 {
43 // TODO: Add more checks
44 if (old_current->status == THREAD_STATUS_RUNNING || current_thread->status != THREAD_STATUS_DEAD)
45 old_current->status = THREAD_STATUS_READY;
46 }
47 current_thread = new_current;
48 new_current->status = THREAD_STATUS_RUNNING;
49 current_cpu->pagetable = new_current->owner->pagetable;
50}

52bool schedule_to_thread(const void *key, void *value)
53{
54 tid_t *tid = (tid_t *) key;
55 thread_t *thread = (thread_t *) value;
56
57 MOS_ASSERT_X(thread->tid == *tid, "something is wrong with the thread table");
58 if (should_schedule_to_thread(thread))
59 {
60 mos_debug("switching to thread %d", thread->tid);
61 platform_switch_to_thread(&current_cpu->scheduler_stack, thread);
62 }
63 return true;
64}
65
66noreturn void scheduler(void)
67{
68 while (1)
69 hashmap_foreach(thread_table, schedule_to_thread);
70}
71
72void jump_to_scheduler(void)
73{
74 cpu_t *cpu = current_cpu;
75 if (cpu->thread->status != THREAD_STATUS_DEAD)
76 cpu->thread->status = THREAD_STATUS_READY;
77 platform_switch_to_scheduler(&cpu->thread->stack.head, cpu->scheduler_stack);
78}
