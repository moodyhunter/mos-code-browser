11bool schedule_to_thread(const void *key, void *value)
12{
13 tid_t *tid = (tid_t *) key;
14 thread_t *thread = (thread_t *) value;
15 MOS_ASSERT_X(thread->tid == *tid, "something is wrong with the thread table");
16 if (thread->status == THREAD_STATUS_READY)
17 {
18 current_thread = thread;
19 mos_platform->switch_to_thread(&current_cpu->scheduler_stack, thread);
20 }
21 return true;
22}
23
24noreturn void scheduler(void)
25{
26 while (1)
27 hashmap_foreach(thread_table, schedule_to_thread);
28}
29
30void jump_to_scheduler(void)
31{
32 mos_platform->switch_to_scheduler(&current_thread->stack.head, current_cpu->scheduler_stack);
33}
