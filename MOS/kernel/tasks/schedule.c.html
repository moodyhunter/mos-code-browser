12void mos_update_current(thread_t *new_current)
13{
14 thread_t *old_current = current_thread;
15 // In the very first switch, current_thread is NULL
16 if (likely(old_current))
17 {
18 // TODO: Add more checks
19 if (old_current->status == THREAD_STATUS_RUNNING || current_thread->status != THREAD_STATUS_DEAD)
20 old_current->status = THREAD_STATUS_READY;
21 }
22 current_thread = new_current;
23 new_current->status = THREAD_STATUS_RUNNING;
24 current_cpu->pagetable = new_current->owner->pagetable;
25}

27bool schedule_to_thread(const void *key, void *value)
28{
29 tid_t *tid = (tid_t *) key;
30 thread_t *thread = (thread_t *) value;
31
32 MOS_ASSERT_X(thread->tid == *tid, "something is wrong with the thread table");
33 if (thread->status == THREAD_STATUS_READY || thread->status == THREAD_STATUS_FORKED || thread->status == THREAD_STATUS_CREATED)
34 {
35#if MOS_DEBUG
36 process_dump_mmaps(thread->owner);
37#endif
38 platform_switch_to_thread(&current_cpu->scheduler_stack, thread);
39 }
40 return true;
41}
42
43noreturn void scheduler(void)
44{
45 while (1)
46 hashmap_foreach(thread_table, schedule_to_thread);
47}
48
49void jump_to_scheduler(void)
50{
51 cpu_t *cpu = current_cpu;
52 if (cpu->thread->status != THREAD_STATUS_DEAD)
53 cpu->thread->status = THREAD_STATUS_READY;
54 platform_switch_to_scheduler(&cpu->thread->stack.head, cpu->scheduler_stack);
55}
