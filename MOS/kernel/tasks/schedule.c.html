12void mos_update_current(thread_t *thread)
13{
14 // In the very first switch, current_thread is NULL
15 if (likely(current_thread))
16 {
17 // TODO: Add more checks
18 if (current_thread->status == THREAD_STATUS_RUNNING || current_thread->status != THREAD_STATUS_DEAD)
19 current_thread->status = THREAD_STATUS_READY;
20 }
21 current_thread = thread;
22 current_thread->status = THREAD_STATUS_RUNNING;
23 current_cpu->pagetable = thread->owner->pagetable;
24}

26bool schedule_to_thread(const void *key, void *value)
27{
28 tid_t *tid = (tid_t *) key;
29 thread_t *thread = (thread_t *) value;
30
31 MOS_ASSERT_X(thread->tid == *tid, "something is wrong with the thread table");
32 if (thread->status == THREAD_STATUS_READY || thread->status == THREAD_STATUS_FORKED || thread->status == THREAD_STATUS_CREATED)
33 {
34#if MOS_DEBUG
35 process_dump_mmaps(thread->owner);
36#endif
37 mos_platform->switch_to_thread(&current_cpu->scheduler_stack, thread);
38 }
39 return true;
40}
41
42noreturn void scheduler(void)
43{
44 while (1)
45 hashmap_foreach(thread_table, schedule_to_thread);
46}
47
48void jump_to_scheduler(void)
49{
50 cpu_t *cpu = current_cpu;
51 mos_platform->switch_to_scheduler(&cpu->thread->stack.head, cpu->scheduler_stack);
52}
