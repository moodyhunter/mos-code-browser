28s32 atoi(const char *nptr)
29{
30 s32 c;
31 s32 neg = 0;
32 s32 val = 0;
33
34 while (isspace(*nptr))
35 nptr++;
36
37 if (*nptr == '-')
38 {
39 neg = 1;
40 nptr++;
41 }
42 else if (*nptr == '+')
43 {
44 nptr++;
45 }
46
47 while ((c = *nptr++) >= '0' && c <= '9')
48 {
49 val = val * 10 + (c - '0');
50 }
51
52 return neg ? -val : val;
53}

55void format_size(char *buf, size_t buf_size, u64 size)
56{
57#define ARRAY_SIZE(a) (sizeof(a) / sizeof((a)[0]))
58 static const char *units[] = { "B", "KiB", "MiB", "GiB", "TiB", "PiB", "EiB", "ZiB", "YiB" };
59 size_t i = 0;
60 size_t diff = 0;
61 while (size >= 1024 && i < ARRAY_SIZE(units) - 1)
62 {
63 diff = size % 1024;
64 size /= 1024;
65 i++;
66 }
67 if (unlikely(diff == 0 || i == 0))
68 {
69 snprintf(buf, buf_size, "%llu %s", size, units[i]);
70 }
71 else
72 {
73 snprintf(buf, buf_size, "%llu %s + %zu %s", size, units[i], diff, units[i - 1]);
74 }
75}
