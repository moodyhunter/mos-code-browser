17#define MOS_LIB_UNIMPLEMENTED(content) MOS_UNIMPLEMENTED(content)
18#define MOS_LIB_UNREACHABLE() MOS_UNREACHABLE()
19#else // ! Userspace
20#include "libuserspace.h"
21#define MOS_LIB_ASSERT(cond) \
22 if (!(cond)) \
23 fatal_abort("Assertion failed: %s", #cond)
24#define MOS_LIB_ASSERT_X(cond, msg, ...) \
25 do \
26 { \
27 if (!(cond)) \
28 fatal_abort("Assertion failed: %s: " msg, #cond, ##__VA_ARGS__); \
29 } while (0)
30#define MOS_LIB_UNIMPLEMENTED(content) fatal_abort("Unimplemented: " content)
31#define MOS_LIB_UNREACHABLE() fatal_abort("Unreachable code reached")
32#define mos_warn(...) dprintf(stderr, __VA_ARGS__)
33#define mos_panic(...) fatal_abort(__VA_ARGS__)
34#endif
