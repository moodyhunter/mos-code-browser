8#define _ANSI_COLOR_black "0"
9#define _ANSI_COLOR_red "1"
10#define _ANSI_COLOR_green "2"
11#define _ANSI_COLOR_yellow "3"
12#define _ANSI_COLOR_blue "4"
13#define _ANSI_COLOR_magenta "5"

17#define _ANSI_STYLE_regular "0"
18#define _ANSI_STYLE_bright "1"
19#define _ANSI_STYLE_faint "2"
20#define _ANSI_STYLE_italic "3"

27#define _ANSI_COLOR_1(fg) _ANSI_FG _ANSI_COLOR_##fg
28#define _ANSI_COLOR_2(fg, style) _ANSI_STYLE_##style ";" _ANSI_FG _ANSI_COLOR_##fg
29#define _ANSI_COLOR_3(fg, style, bg) _ANSI_STYLE_##style ";" _ANSI_FG _ANSI_COLOR_##fg ";" _ANSI_BG _ANSI_COLOR_##bg
30#define _ANSI_COLOR_N(_1, _2, _3, N, ...) N
31
32// USAGE:
33//
34// ANSI_COLOR(foreground) // a foreground color
35// ANSI_COLOR(foreground, style) // a foreground color with style
36// ANSI_COLOR(foreground, style, background) // a foreground color with style and a background color
37//
38// "_" is to prevent the "must specify at least one argument for '...' parameter of variadic macro" warning
39#define ANSI_COLOR(...) "\033[" _ANSI_COLOR_N(__VA_ARGS__, _ANSI_COLOR_3, _ANSI_COLOR_2, _ANSI_COLOR_1, _)(__VA_ARGS__) "m"
40#define ANSI_COLOR_RESET "\033[0m"
