8always_inline hash_t __pure string_hash(const char *s, const int n)
9{
10 const int p = 31, m = 1e9 + 7;
11 hash_t h = { 0 };
12 long p_pow = 1;
13 for (int i = 0; i < n; i++)
14 {
15 h.hash = (h.hash + (s[i] - 'a' + 1) * p_pow) % m;
16 p_pow = (p_pow * p) % m;
17 }
18 return h;
19}
