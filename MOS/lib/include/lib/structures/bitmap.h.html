19should_inline size_t bitmap_find_first_free_n_from(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t n_bits, size_t start_line)
20{
21 if (start_line >= bitmap_nlines)
22 return 0;
23 return bitmap_find_first_free_n(bitmap + start_line, bitmap_nlines - start_line, n_bits) + start_line * BITMAP_LINE_BITS;
24}
