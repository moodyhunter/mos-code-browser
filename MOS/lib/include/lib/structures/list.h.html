7/**
8 * @defgroup linked_list libs.LinkedList
9 * @ingroup libs
10 * @brief A circular, doubly-linked list.
11 * @{
12 */
13
14typedef struct list_node list_node_t;
15
16/** @brief A node in a linked list. */
17struct list_node
18{
19 list_node_t *prev;
20 list_node_t *next;
21};
22
23/**
24 * @brief Embed a list node into a struct
25 */
26#define as_linked_list list_node_t list_node
27
28// clang-format off
29#define LIST_HEAD_INIT(container) { .prev = &(container), .next = &(container) }
30// clang-format on
31
32#define LIST_NODE_INIT(container) LIST_HEAD_INIT(container.list_node)
33
34/**
35 * @brief Get the container struct of a list node
36 */
37#define list_entry(node, type) container_of((node), type, list_node)
38
39/**
40 * @brief Get the `list_node' of a list element.
41 * This is exactly the reverse of `list_entry' above.
42 */
43#define list_node(element) (&((element)->list_node))
44
45#define list_prepend(element, item) list_node_prepend(list_node(element), list_node(item))
46#define list_append(element, item) list_node_append(list_node(element), list_node(item))
47#define list_insert_before(element, item) list_node_insert_before(list_node(element), list_node(item))
48#define list_insert_after(element, item) list_node_insert_after(list_node(element), list_node(item))
49#define list_remove(element) list_node_remove(list_node(element))
50
51/**
52 * @brief Iterate over a list
53 *
54 * @param t Type of the list elements (e.g. `struct foo')
55 * @param v Name of the variable to use for the current element (e.g. `item')
56 * @param h List Head (e.g. `consoles')
57 */
58#define list_foreach(t, v, h) for (t *v = list_entry((h).next, t); list_node(v) != &(h); v = list_entry(list_node(v)->next, t))
59
60void linked_list_init(list_node_t *head_node);
61bool list_is_empty(list_node_t *head);
62void list_node_remove(list_node_t *link);
63
64void list_node_prepend(list_node_t *head, list_node_t *item);
65void list_node_append(list_node_t *head, list_node_t *item);
66void list_node_insert_before(list_node_t *element, list_node_t *item);
67void list_node_insert_after(list_node_t *element, list_node_t *item);
