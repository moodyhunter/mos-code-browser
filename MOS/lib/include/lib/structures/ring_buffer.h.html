15// A position-only ring buffer specifier.
16typedef struct _ring_buffer_pos
17{
18 size_t size;
19 size_t capacity;
20 size_t head; // index of the first element
21 size_t next_pos; // index of the next element to be inserted
22} ring_buffer_pos_t;
23
24// A managed ring buffer.
25typedef struct _ring_buffer
26{
27 u8 *data;
28 ring_buffer_pos_t pos;
29} ring_buffer_t;
30
31ring_buffer_t *ring_buffer_create(size_t capacity);
32ring_buffer_t *ring_buffer_create_at(void *data, size_t capacity);
33void ring_buffer_pos_init(ring_buffer_pos_t *pos, size_t capacity);
34void ring_buffer_destroy(ring_buffer_t *buffer);
35
36bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity);
37
38size_t ring_buffer_pos_push_back(u8 *buffer, ring_buffer_pos_t *pos, const u8 *data, size_t size);
39size_t ring_buffer_pos_pop_back(u8 *buffer, ring_buffer_pos_t *pos, u8 *buf, size_t size);
40
41size_t ring_buffer_pos_push_front(u8 *buffer, ring_buffer_pos_t *pos, const u8 *data, size_t size);
42size_t ring_buffer_pos_pop_front(u8 *buffer, ring_buffer_pos_t *pos, u8 *buf, size_t size);
43
44should_inline u8 ring_buffer_pos_pop_back_byte(u8 *buffer, ring_buffer_pos_t *pos)
45{
46 u8 data = '\0';
47 ring_buffer_pos_pop_back(buffer, pos, &data, 1);
48 return data;
49}
50
51should_inline u8 ring_buffer_pos_pop_front_byte(u8 *buffer, ring_buffer_pos_t *pos)
52{
53 u8 data = '\0';
54 ring_buffer_pos_pop_front(buffer, pos, &data, 1);
55 return data;
56}
57
58// clang-format off
59should_inline bool ring_buffer_pos_is_full(ring_buffer_pos_t *pos) { return pos->size == pos->capacity; }
60should_inline bool ring_buffer_pos_is_empty(ring_buffer_pos_t *pos) { return pos->size == 0; }
61
62should_inline size_t ring_buffer_pos_push_back_byte(ring_buffer_t *buffer, u8 byte) { return ring_buffer_pos_push_back(buffer->data, &buffer->pos, &byte, 1); }
63should_inline size_t ring_buffer_pos_push_front_byte(ring_buffer_t *buffer, u8 byte) { return ring_buffer_pos_push_front(buffer->data, &buffer->pos, &byte, 1); }

67should_inline bool ring_buffer_is_full(ring_buffer_t *buffer) { return ring_buffer_pos_is_full(&buffer->pos); }
68should_inline bool ring_buffer_is_empty(ring_buffer_t *buffer) { return ring_buffer_pos_is_empty(&buffer->pos); }
