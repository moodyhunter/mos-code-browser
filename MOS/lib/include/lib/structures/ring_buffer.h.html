8/**
9 * \defgroup ringbuffer libs.RingBuffer
10 * \ingroup libs
11 * \brief A ring buffer.
12 * @{
13 */
14typedef struct _ring_buffer
15{
16 u8 *data;
17 size_t size;
18 size_t capacity;
19 size_t head; // index of the first element
20 size_t next_pos; // index of the next element to be inserted
21} ring_buffer_t;
22
23ring_buffer_t *ring_buffer_create(size_t capacity);
24ring_buffer_t *ring_buffer_create_at(void *data, size_t capacity);
25void ring_buffer_destroy(ring_buffer_t *buffer);
26
27bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity);
28
29size_t ring_buffer_push_back(ring_buffer_t *buffer, const u8 *data, size_t size);
30size_t ring_buffer_pop_back(ring_buffer_t *buffer, u8 *data, size_t size);
31
32size_t ring_buffer_push_front(ring_buffer_t *buffer, const u8 *data, size_t size);
33size_t ring_buffer_pop_front(ring_buffer_t *buffer, u8 *data, size_t size);
34
35should_inline bool ring_buffer_is_empty(ring_buffer_t *buffer)
36{
37 return buffer->size == 0;
38}
39
40should_inline bool ring_buffer_is_full(ring_buffer_t *buffer)
41{
42 return buffer->size == buffer->capacity;
43}
44
45should_inline u8 ring_buffer_pop_back_byte(ring_buffer_t *buffer)
46{
47 u8 data = '\0';
48 ring_buffer_pop_back(buffer, &data, 1);
49 return data;
50}
51
52should_inline size_t ring_buffer_push_back_byte(ring_buffer_t *buffer, u8 byte)
53{
54 return ring_buffer_push_back(buffer, &byte, 1);
55}
56
57should_inline u8 ring_buffer_pop_front_byte(ring_buffer_t *buffer)
58{
59 u8 data = '\0';
60 ring_buffer_pop_front(buffer, &data, 1);
61 return data;
62}
63
64should_inline size_t ring_buffer_push_front_byte(ring_buffer_t *buffer, u8 byte)
65{
66 return ring_buffer_push_front(buffer, &byte, 1);
67}
68
69#define ring_buffer_push_back_t(buffer, type, data) ring_buffer_push_back(buffer, (u8 *) &data, sizeof(type))
70#define ring_buffer_pop_back_t(buffer, type, data) ring_buffer_pop_back(buffer, (u8 *) &data, sizeof(type))
71
72#define ring_buffer_push_front_t(buffer, type, data) ring_buffer_push_front(buffer, (u8 *) &data, sizeof(type))
73#define ring_buffer_pop_front_t(buffer, type, data) ring_buffer_pop_front(buffer, (u8 *) &data, sizeof(type))
74
75#define ring_buffer_enqueue_t(buffer, type, data) ring_buffer_push_back_t(buffer, type, data)
76#define ring_buffer_dequeue_t(buffer, type, data) ring_buffer_pop_front_t(buffer, type, data)
77
