3#pragma once
4
5#include "mos/mos_global.h"
6#include "mos/types.h"
7
8typedef struct _ring_buffer
9{
10 u8 *data;
11 size_t size;
12 size_t capacity;
13 size_t head; // index of the first element
14 size_t next_pos; // index of the next element to be inserted
15} ring_buffer_t;
16
17ring_buffer_t *ring_buffer_create(size_t capacity);
18void ring_buffer_destroy(ring_buffer_t *buffer);
19
20bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity);
21
22size_t ring_buffer_push_back(ring_buffer_t *buffer, const u8 *data, size_t size);
23size_t ring_buffer_pop_back(ring_buffer_t *buffer, u8 *data, size_t size);
24
25size_t ring_buffer_push_front(ring_buffer_t *buffer, const u8 *data, size_t size);
26size_t ring_buffer_pop_front(ring_buffer_t *buffer, u8 *data, size_t size);
27
28should_inline bool ring_buffer_is_empty(ring_buffer_t *buffer)
29{
30 return buffer->size == 0;
31}
32
33should_inline bool ring_buffer_is_full(ring_buffer_t *buffer)
34{
35 return buffer->size == buffer->capacity;
36}
37
38should_inline u8 ring_buffer_pop_back_byte(ring_buffer_t *buffer)
39{
40 u8 data = '\0';
41 ring_buffer_pop_back(buffer, &data, 1);
42 return data;
43}
44
45should_inline size_t ring_buffer_push_back_byte(ring_buffer_t *buffer, u8 byte)
46{
47 return ring_buffer_push_back(buffer, &byte, 1);
48}
49
50should_inline u8 ring_buffer_pop_front_byte(ring_buffer_t *buffer)
51{
52 u8 data = '\0';
53 ring_buffer_pop_front(buffer, &data, 1);
54 return data;
55}
56
57should_inline size_t ring_buffer_push_front_byte(ring_buffer_t *buffer, u8 byte)
58{
59 return ring_buffer_push_front(buffer, &byte, 1);
60}
61
62#define ring_buffer_push_back_t(buffer, type, data) ring_buffer_push_back(buffer, (u8 *) &data, sizeof(type))
63#define ring_buffer_pop_back_t(buffer, type, data) ring_buffer_pop_back(buffer, (u8 *) &data, sizeof(type))
64
65#define ring_buffer_push_front_t(buffer, type, data) ring_buffer_push_front(buffer, (u8 *) &data, sizeof(type))
66#define ring_buffer_pop_front_t(buffer, type, data) ring_buffer_pop_front(buffer, (u8 *) &data, sizeof(type))
67
68#define ring_buffer_enqueue_t(buffer, type, data) ring_buffer_push_back_t(buffer, type, data)
69#define ring_buffer_dequeue_t(buffer, type, data) ring_buffer_pop_front_t(buffer, type, data)
