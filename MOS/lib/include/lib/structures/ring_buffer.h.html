5#include "mos/mos_global.h"

8typedef struct _ring_buffer
9{
10 u8 *data;
11 size_t size;
12 size_t capacity;
13 size_t head; // index of the first element
14 size_t tail; // index of the last element
15} ring_buffer_t;
16
17ring_buffer_t *ring_buffer_create(size_t capacity);
18void ring_buffer_destroy(ring_buffer_t *buffer);
19
20bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity);
21
22size_t ring_buffer_write_back(ring_buffer_t *buffer, const u8 *data, size_t size);
23size_t ring_buffer_read_back(ring_buffer_t *buffer, u8 *data, size_t size);
24
25should_inline bool ring_buffer_is_empty(ring_buffer_t *buffer)
26{
27 return buffer->size == 0;
28}
29
30should_inline bool ring_buffer_is_full(ring_buffer_t *buffer)
31{
32 return buffer->size == buffer->capacity;
33}
34
35should_inline u8 ring_buffer_pop_back_byte(ring_buffer_t *buffer)
36{
37 u8 data = '\0';
38 ring_buffer_read_back(buffer, &data, 1);
39 return data;
40}
41
42should_inline size_t ring_buffer_push_back_byte(ring_buffer_t *buffer, u8 byte)
43{
44 return ring_buffer_write_back(buffer, &byte, 1);
45}
