5#include "mos/mos_global.h"
6#include "mos/types.h"
7
8typedef volatile size_t mos_refcount_t;
9
10should_inline size_t refcount_inc(mos_refcount_t *atomic)
11{
12 return __atomic_add_fetch(atomic, 1, __ATOMIC_RELAXED);
13}
14
15should_inline size_t refcount_dec(mos_refcount_t *atomic)
16{
17 return __atomic_sub_fetch(atomic, 1, __ATOMIC_RELAXED);
18}
19
20should_inline void refcount_zero(mos_refcount_t *atomic)
21{
22 __atomic_store_n(atomic, 0, __ATOMIC_RELAXED);
23}
24
25should_inline size_t refcount_get(mos_refcount_t *atomic)
26{
27 return __atomic_load_n(atomic, __ATOMIC_RELAXED);
28}
