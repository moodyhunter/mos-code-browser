9typedef enum
10{
11 LM_none,
12 // char
13 LM_hh,
14 // short
15 LM__h,
16 // long
17 LM__l,
18 // long long
19 LM_ll,
20 // long double
21 LM__L,
22 // intmax_t
23 LM__j,
24 // size_t
25 LM__z,
26 // ptrdiff_t
27 LM__t,
28} length_modifier_t;
29
30typedef struct
31{
32 bool left_aligned : 1; // -
33 bool show_sign : 1; // +
34 bool space_if_positive : 1; // prepend space if positive
35 bool pad_with_zero : 1; // padding with zero
36
37 // For g and G types, trailing zeros are not removed.
38 // For f, F, e, E, g, G types, the output always contains a decimal point.
39 // For o, x, X types, the text 0, 0x, 0X, respectively, is prepended to non-zero numbers.
40 bool hash : 1;
41
42 // POSIX extension '\'' not implemented
43 // glibc 2.2 extension 'I' not implemented
44
45 // If the converted value has fewer characters than the field width, it will be padded with spaces on the left
46 // (or right, if the left-adjustment flag has been given).
47 // A negative field width is taken as a '-' flag followed by a positive field width.
48 // In no case does a nonexistent or small field width cause truncation of a field;
49 // if the result of a conversion is wider than the field width, the field is expanded to contain the conversion result.
50 s32 minimum_width : 24;
51
52 bool has_explicit_precision : 1;
53 // `d, i, o, u, x, X` -> The minimum number of digits to appear;
54 // `a, A, e, E, f, F` -> The number of digits to appear after the radix character;
55 // `g, G` -> The maximum number of significant digits;
56 // `s, S` -> The maximum number of characters to be printed from a string;
57 s32 precision : 16;
58
59 length_modifier_t length : 8;
60} __packed printf_flags_t;
61
62typedef enum
63{
64 BASE_8 = 8,
65 BASE_10 = 10,
66 BASE_16 = 16,
67} base_t;

69static size_t parse_printf_flags(const char *format, printf_flags_t *pflags, va_list *args)
70{
71 const char *start = format;
72#define goto_next_char() (format++)
73#define current (*format)
74
75 while (1)
76 {
77 switch (current)
78 {
79 case '-': pflags->left_aligned = true; break;
80 case '+': pflags->show_sign = true; break;
81 case ' ': pflags->space_if_positive = true; break;
82 case '#': pflags->hash = true; break;
83 // We can read multiple 0s and leave the 'width' field empty, 0-width is meaningless.
84 case '0': pflags->pad_with_zero = true; break;
85 default: goto flag_parse_done;
86 }
87 goto_next_char();
88 };
89
90flag_parse_done:
91 // width
92 pflags->minimum_width = 0;
93 if (current == '*')
94 {
95 pflags->minimum_width = va_arg(*args, s32), goto_next_char();
96 if (pflags->minimum_width < 0)
97 {
98 // A negative field width is taken as a '-' flag followed by a positive field width
99 pflags->left_aligned = true;
100 pflags->minimum_width = -pflags->minimum_width;
101 }
102 }
103 else
104 {
105 while ('0' <= current && current <= '9')
106 pflags->minimum_width = (pflags->minimum_width * 10 + (current - '0')), goto_next_char();
107 }
108
109 // precision
110 pflags->precision = 0;
111 if (current == '.')
112 {
113 pflags->has_explicit_precision = true;
114 goto_next_char();
115 if (current == '*')
116 {
117 pflags->precision = va_arg(*args, s32), goto_next_char();
118 // A negative precision argument is taken as if the precision were omitted.
119 if (pflags->precision < 0)
120 pflags->precision = 0, pflags->has_explicit_precision = false;
121 }
122 else
123 {
124 while ('0' <= current && current <= '9')
125 pflags->precision = pflags->precision * 10 + current - '0', goto_next_char();
126 }
127 }
128
129 // length enums
130 if (current == 'h')
131 {
132 goto_next_char();
133 if (current == 'h')
134 pflags->length = LM_hh, goto_next_char();
135 else
136 pflags->length = LM__h;
137 }
138 else if (current == 'l')
139 {
140 goto_next_char();
141 if (current == 'l')
142 pflags->length = LM_ll, goto_next_char();
143 else
144 pflags->length = LM__l;
145 }
146 else if (current == 'L')
147 {
148 goto_next_char();
149 pflags->length = LM__L;
150 }
151 else if (current == 'j')
152 {
153 goto_next_char();
154 pflags->length = LM__j;
155 }
156 else if (current == 'z')
157 {
158 goto_next_char();
159 pflags->length = LM__z;
160 }
161 else if (current == 't')
162 {
163 goto_next_char();
164 pflags->length = LM__t;
165 }
166
167#undef goto_next_char
168#undef current
169
170 if (unlikely(pflags->left_aligned && pflags->pad_with_zero))
171 {
172 pflags->pad_with_zero = false;
173 mos_warn("printf: '0' flag is ignored by the '-' flag");
174 }
175
176 if (unlikely(pflags->show_sign && pflags->space_if_positive))
177 {
178 pflags->space_if_positive = false;
179 mos_warn("printf: ' ' flag is ignored by the '+' flag");
180 }
181
182 return format - start;
183}

185// writes a character into the buffer, and increses the buffer pointer
186should_inline void buf_putchar(char **pbuf, char c)
187{
188 **pbuf = c;
189 (*pbuf)++;
190}

192// ! prints d, i, o, u, x, and X
193static int printf_diouxX(char *buf, u64 number, printf_flags_t *pflags, char conv)
194{
195 MOS_LIB_ASSERT(conv == 'd' || conv == 'i' || conv == 'o' || conv == 'u' || conv == 'x' || conv == 'X');
196 MOS_LIB_ASSERT(pflags->precision >= 0);
197 MOS_LIB_ASSERT(pflags->minimum_width >= 0);
198
199 if (conv == 'd' || conv == 'i' || conv == 'u')
200 {
201 if (unlikely(pflags->hash))
202 {
203 mos_warn("'#' flag is ignored in d, i mode");
204 pflags->hash = false;
205 }
206 }
207
208 static char lower_hex_digits[] = "0123456789abcdef";
209 static char upper_hex_digits[] = "0123456789ABCDEF";
210
211 base_t base = BASE_10;
212 bool upper_case = false;
213 char *hex_digits = NULL;
214
215 bool is_unsigned_ouxX = conv == 'o' || conv == 'u' || conv == 'x' || conv == 'X';
216 if (is_unsigned_ouxX)
217 {
218 if (unlikely(pflags->show_sign))
219 {
220 mos_warn("'+' flag is ignored in unsigned mode");
221 pflags->show_sign = false;
222 }
223 if (unlikely(pflags->space_if_positive))
224 {
225 mos_warn("' ' flag is ignored in unsigned mode");
226 pflags->space_if_positive = false;
227 }
228
229 base = (conv == 'o' ? BASE_8 : (conv == 'u' ? BASE_10 : BASE_16));
230 upper_case = conv == 'X';
231 hex_digits = upper_case ? upper_hex_digits : lower_hex_digits;
232
233 switch (pflags->length)
234 {
235 case LM_hh: number = (u8) number; break;
236 case LM__h: number = (u16) number; break;
237 case LM_none:
238 case LM__l: number = (u32) number; break;
239 case LM_ll: number = (u64) number; break;
240 case LM__z: number = (size_t) number; break;
241 case LM__t: number = (ptrdiff_t) number; break;
242 case LM__L: number = (u64) number; break;
243 case LM__j:
244 default: MOS_LIB_UNREACHABLE();
245 }
246 }
247
248 // If a precision is given with a numeric conversion (d, i, o, u, x, and X), the 0 flag is ignored.
249 if (pflags->has_explicit_precision && pflags->pad_with_zero)
250 pflags->pad_with_zero = false;
251
252 // The default precision is 1 for d, i, o, u, x, and X
253 if (!pflags->has_explicit_precision)
254 pflags->precision = 1;
255
256 char *start = buf;
257
258 {
259 char num_prefix_buf[5] = { 0 };
260 char num_content_buf[32] = { 0 };
261
262 // Setup prefixes.
263 if (base == BASE_10 && !is_unsigned_ouxX)
264 {
265 bool is_negative = ((s64) number) < 0;
266 if (is_negative)
267 number = -(s64) number, num_prefix_buf[0] = '-';
268 else if (pflags->show_sign)
269 num_prefix_buf[0] = '+'; // 0 is positive too !!!!!
270 else if (pflags->space_if_positive)
271 num_prefix_buf[0] = ' ';
272 }
273 else if (base == BASE_16 && pflags->hash)
274 {
275 // '#' flag turns on prefix '0x' or '0X' for hexadecimal conversions.
276 // For x and X conversions, a nonzero result has the string "0x" (or "0X" for X conversions) prepended to it.
277 if (number != 0)
278 {
279 num_prefix_buf[0] = '0';
280 num_prefix_buf[1] = upper_case ? 'X' : 'x';
281 }
282 }
283 else if (base == BASE_8 && pflags->hash)
284 {
285 // ! BEGIN SPECIAL CASE for base 8 + '#' flag
286 // to be handled later, when we are printing the number
287 // ! END SPECIAL CASE
288 }
289
290 // Print the number.
291 if (number == 0)
292 {
293 // When 0 is printed with an explicit precision 0, the output is empty.
294 // so only print a '0' if the precision is **not** 0.
295 if (pflags->precision != 0)
296 num_content_buf[0] = '0';
297 }
298 else
299 {
300 char *pnumberbuf = num_content_buf;
301 switch (base)
302 {
303 case BASE_8:
304 case BASE_10:
305 while (number > 0)
306 buf_putchar(&pnumberbuf, '0' + (char) (number % base)), number /= base;
307 break;
308 case BASE_16:
309 while (number > 0)
310 buf_putchar(&pnumberbuf, hex_digits[number % 16]), number /= 16;
311 break;
312 default: MOS_LIB_UNREACHABLE();
313 }
314 }
315
316 s32 n_digits = strlen(num_content_buf);
317
318 // ! BEGIN SPECIAL CASE for base 8 + '#' flag
319 if (base == BASE_8 && pflags->hash)
320 {
321 // if there's no padding needed, we'll have to add the '0' prefix
322 if (pflags->precision - n_digits <= 0 && num_content_buf[0] != '0')
323 num_prefix_buf[0] = '0';
324 }
325 // ! END SPECIAL CASE
326
327 s32 precision_padding = MAX(pflags->precision - n_digits, 0);
328 s32 width_to_pad = MAX(pflags->minimum_width - strlen(num_prefix_buf) - precision_padding - n_digits, 0);
329
330 char *pnum_prefix = num_prefix_buf;
331 char *pnum_content = num_content_buf + n_digits;
332 if (pflags->left_aligned)
333 {
334 while (*pnum_prefix)
335 buf_putchar(&buf, *pnum_prefix++);
336 while (precision_padding-- > 0)
337 buf_putchar(&buf, '0');
338 while (pnum_content > num_content_buf)
339 buf_putchar(&buf, *--pnum_content);
340 while (width_to_pad-- > 0)
341 buf_putchar(&buf, ' ');
342 }
343 else
344 {
345 if (pflags->pad_with_zero)
346 {
347 // zero should be after the sign
348 while (*pnum_prefix)
349 buf_putchar(&buf, *pnum_prefix++);
350 while (width_to_pad-- > 0)
351 buf_putchar(&buf, '0');
352 }
353 else
354 {
355 // space should be before the sign
356 while (width_to_pad-- > 0)
357 buf_putchar(&buf, ' ');
358 while (*pnum_prefix)
359 buf_putchar(&buf, *pnum_prefix++);
360 }
361 while (precision_padding-- > 0)
362 buf_putchar(&buf, '0');
363 while (pnum_content > num_content_buf)
364 buf_putchar(&buf, *--pnum_content);
365 }
366 }
367 return buf - start;
368}

370static int printf_cs(char *buf, char *data, printf_flags_t *pflags, char conv)
371{
372 MOS_LIB_ASSERT(conv == 'c' || conv == 's');
373 MOS_LIB_ASSERT(pflags->precision >= 0);
374 MOS_LIB_ASSERT(pflags->minimum_width >= 0);
375
376 if (unlikely(pflags->hash))
377 {
378 mos_warn("printf: '#' flag is ignored in 'c' and 's' mode.");
379 pflags->hash = false;
380 }
381
382 if (unlikely(pflags->pad_with_zero))
383 {
384 mos_warn("printf: '0' flag is ignored in 'c' and 's' mode.");
385 pflags->pad_with_zero = false;
386 }
387
388 if (unlikely(pflags->show_sign))
389 {
390 mos_warn("printf: '+' flag is ignored in 'c' and 's' mode.");
391 pflags->show_sign = false;
392 }
393
394 if (unlikely(pflags->space_if_positive))
395 {
396 mos_warn("printf: ' ' flag is ignored in 'c' and 's' mode.");
397 pflags->space_if_positive = false;
398 }
399
400 if (unlikely(conv == 'c' && pflags->has_explicit_precision))
401 {
402 mos_warn("printf: precision is ignored in 'c' mode.");
403 pflags->has_explicit_precision = false;
404 pflags->precision = 0;
405 }
406
407 char *start = buf;
408
409 s32 printed_len = conv == 'c' ? 1 : strlen(data);
410
411 // If presition is given, the string is truncated to this length.
412 if (pflags->has_explicit_precision)
413 printed_len = MIN(printed_len, pflags->precision);
414
415 s32 width_to_pad = MAX(pflags->minimum_width - printed_len, 0);
416
417 if (pflags->left_aligned)
418 {
419 while (printed_len-- > 0)
420 buf_putchar(&buf, *data++);
421 while (width_to_pad-- > 0)
422 buf_putchar(&buf, ' ');
423 }
424 else
425 {
426 while (width_to_pad-- > 0)
427 buf_putchar(&buf, ' ');
428 while (printed_len-- > 0)
429 buf_putchar(&buf, *data++);
430 }
431
432 return buf - start;
433}

435int vsnprintf(char *buf, size_t size, const char *format, va_list args)
436{
437#pragma message "TODO: Check for buffer overflow."
438 MOS_UNUSED(size);
439 char *start = buf;
440
441 for (; *format; format++)
442 {
443 if (*format == '%')
444 {
445 // skip '%'
446 format++;
447
448 // parse flags
449 printf_flags_t flags = { 0 };
450 size_t c = parse_printf_flags(format, &flags, &args);
451 format += c;
452
453 switch (*format)
454 {
455 case 'd':
456 case 'i':
457 case 'o':
458 case 'u':
459 case 'x':
460 case 'X':
461 {
462 // print a signed integer
463 u64 value;
464 switch (flags.length)
465 {
466 case LM_hh: value = (s8) va_arg(args, s32); break;
467 case LM__h: value = (s16) va_arg(args, s32); break;
468 case LM_none:
469 case LM__l: value = va_arg(args, s32); break;
470 case LM_ll: value = va_arg(args, s64); break;
471 case LM__z: value = va_arg(args, size_t); break;
472 case LM__t: value = va_arg(args, ptrdiff_t); break;
473 case LM__L: value = va_arg(args, s64); break;
474 case LM__j:
475 default: MOS_LIB_UNREACHABLE();
476 }
477 int c = printf_diouxX(buf, value, &flags, *format);
478 buf += c;
479 break;
480 }
481
482 // default precision ; for e, E, f, g, and G it is 0.
483 case 'f':
484 case 'F':
485 {
486 // print a floating point number
487 MOS_LIB_UNIMPLEMENTED("printf: %f / %F");
488 break;
489 }
490 case 'e':
491 case 'E':
492 {
493 // print a floating point number in scientific notation
494 MOS_LIB_UNIMPLEMENTED("printf: %e / %E");
495 break;
496 }
497 case 'g':
498 case 'G':
499 {
500 // print a floating point number in scientific notation
501 MOS_LIB_UNIMPLEMENTED("printf: %g / %G");
502 break;
503 }
504 case 's':
505 {
506 // print string
507 char *string = va_arg(args, char *);
508 int c = printf_cs(buf, string, &flags, *format);
509 buf += c;
510 break;
511 }
512 case 'c':
513 {
514 // print a character
515 char value = (char) va_arg(args, s32);
516 int c = printf_cs(buf, &value, &flags, *format);
517 buf += c;
518 break;
519 }
520 case 'p':
521 {
522 // print a pointer
523 u64 value = (size_t) va_arg(args, void *);
524 buf_putchar(&buf, '0');
525 buf_putchar(&buf, 'x');
526 int c = printf_diouxX(buf, value, &flags, 'x');
527 buf += c;
528 break;
529 }
530 case 'a':
531 case 'A':
532 {
533 // print a hexadecimal number in ASCII
534 MOS_LIB_UNIMPLEMENTED("printf: %a / %A");
535 break;
536 }
537 case 'n':
538 {
539 // print the number of characters printed
540 MOS_LIB_UNIMPLEMENTED("printf: %n");
541 break;
542 }
543 case '%':
544 {
545 // print a '%'
546 buf_putchar(&buf, '%');
547 break;
548 }
549 case '\0':
550 {
551 // end of format string
552 mos_warn("printf: incomplete format specifier");
553 goto end;
554 }
555 default:
556 {
557 mos_warn("printf: unknown format specifier");
558 buf_putchar(&buf, '%');
559 buf_putchar(&buf, *format);
560 break;
561 }
562 }
563 }
564 else
565 {
566 buf_putchar(&buf, *format);
567 }
568 }
569end:
570 buf_putchar(&buf, 0);
571
572 va_end(args);
573 return buf - start;
574}
