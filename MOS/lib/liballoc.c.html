12// This is the byte alignment that memory must be allocated on. IMPORTANT for GTK and other stuff.
13static const int ALIGNMENT = 16; // or 4
14
15// unsigned char[16], or unsigned short
16typedef char liballoc_align_t;
17
18// Alignment information is stored right before the pointer. This is the number of bytes of information stored there.
19#define ALIGN_INFO (sizeof(liballoc_align_t) * 16)
20
21#define USE_CASE1 1
22#define USE_CASE2 1
23#define USE_CASE3 1
24#define USE_CASE4 1
25#define USE_CASE5 1

27/** This macro will conveniently align our pointer upwards */
28#define LIBALLOC_ALIGN_PTR(ptr) \
29 if (ALIGNMENT > 1) \
30 { \
31 uintptr_t diff; \
32 ptr = (void *) ((uintptr_t) ptr + ALIGN_INFO); \
33 diff = (uintptr_t) ptr & (ALIGNMENT - 1); \
34 if (diff != 0) \
35 { \
36 diff = ALIGNMENT - diff; \
37 ptr = (void *) ((uintptr_t) ptr + diff); \
38 } \
39 *((liballoc_align_t *) ((uintptr_t) ptr - ALIGN_INFO)) = diff + ALIGN_INFO; \
40 }
41
42#define LIBALLOC_UNALIGN_PTR(ptr) \
43 if (ALIGNMENT > 1) \
44 { \
45 uintptr_t diff = *((liballoc_align_t *) ((uintptr_t) ptr - ALIGN_INFO)); \
46 if (diff < (ALIGNMENT + ALIGN_INFO)) \
47 ptr = (void *) ((uintptr_t) ptr - diff); \
48 }

69// A structure found at the top of all system allocated memory blocks. It details the usage of the memory block.
70typedef struct liballoc_block
71{
72 struct liballoc_block *prev; // Linked list information.
73 struct liballoc_block *next; // Linked list information.
74 size_t pages; // The number of pages in the block.
75 size_t size; // bytes in the block.
76 size_t usage; // bytes used in the block.
77 struct liballoc_part *first; // A pointer to the first allocated memory in the block.
78} liballoc_block_t;
79
80// This is a structure found at the beginning of all sections in a major block which were allocated by a malloc, calloc, realloc call.
81typedef struct liballoc_part
82{
83 struct liballoc_part *prev; // Linked list information.
84 struct liballoc_part *next; // Linked list information.
85 struct liballoc_block *block; // The owning block. A pointer to the major structure.
86 u32 magic; // A magic number to idenfity correctness.
87 size_t size; // The size of the memory allocated. Could be 1 byte or more.
88 size_t req_size; // The size of memory requested.
89} liballoc_part_t;
90
91static liballoc_block_t *l_memroot = NULL; // The root memory block acquired from the system.
92static liballoc_block_t *l_bestbet = NULL; // The major with the most free memory.
93
94static size_t l_alloc_n_page_once = 0; // The number of pages to request per chunk. Set up in liballoc_init.
95
96static size_t l_mem_allocated = 0; // Running total of allocated memory.
97static size_t l_mem_inuse = 0; // Running total of used memory.
98static size_t l_warnings = 0; // Number of warnings encountered
99static size_t l_errors = 0; // Number of actual errors
100static size_t l_possible_overruns = 0; // Number of possible overruns

102#if LIBALLOC_PRINT_DEBUG_MESSAGES

105 pr_info("--------------- Memory data ---------------");
106 pr_info("Total Memory Allocated: %zu bytes", l_mem_allocated);
107 pr_info("Memory Used (malloc'ed): %zu bytes", l_mem_inuse);
108 pr_info("Possible Overruns: %zu", l_possible_overruns);
109 pr_info("emitted %zu warning(s) and %zu error(s)", l_warnings, l_errors);
110
111 liballoc_block_t *maj = l_memroot;
112 liballoc_part_t *min = NULL;
113
114 pr_info("Memory Blocks:");
115 while (maj != NULL)
116 {
117 pr_info(" %p: total = %zu, used = %zu", (void *) maj, maj->size, maj->usage);
118
119 min = maj->first;
120 while (min != NULL)
121 {
122 pr_info(" %p: %zu bytes", (void *) min, min->size);
123 min = min->next;
124 }
125
126 maj = maj->next;
127 }
128}

131static liballoc_block_t *allocate_new_pages_for(unsigned int size)
132{
133 // This is how much space is required.
134 u32 st = size + sizeof(liballoc_block_t) + sizeof(liballoc_part_t);
135
136 // Perfect amount of space?
137 if ((st % MOS_PAGE_SIZE) == 0)
138 st = st / (MOS_PAGE_SIZE);
139 else
140 st = st / (MOS_PAGE_SIZE) + 1;
141 // No, add the buffer.
142
143 // Make sure it's >= the minimum size.
144 if (st < l_alloc_n_page_once)
145 st = l_alloc_n_page_once;
146
147 liballoc_block_t *maj = (liballoc_block_t *) liballoc_alloc_page(st);
148 if (maj == NULL)
149 {
150 l_warnings += 1;
151 mos_warn("liballoc: WARNING: liballoc_alloc(%i) returns NULL", st);
152 return NULL;
153 }
154
155 maj->prev = NULL;
156 maj->next = NULL;
157 maj->pages = st;
158 maj->size = st * MOS_PAGE_SIZE;
159 maj->usage = sizeof(liballoc_block_t);
160 maj->first = NULL;
161
162 l_mem_allocated += maj->size;
163
164#if LIBALLOC_PRINT_DEBUG_MESSAGES
165 pr_info("liballoc: Allocated %u pages (%zu bytes) at %p for %u bytes to be used.", st, maj->size, (void *) maj, size);
166 pr_info("liballoc: Total memory usage = %i KB", (int) ((l_mem_allocated / (1024))));
167#endif
168
169 return maj;
170}

172static void liballoc_first_alloc()
173{
174 // This is the first time we are being used.
175 l_memroot = allocate_new_pages_for(sizeof(liballoc_block_t));
176 if (l_memroot == NULL)
177 {
178#if MOS_MM_LIBALLOC_LOCKS
179 liballoc_unlock();
180#endif
181#if LIBALLOC_PRINT_DEBUG_MESSAGES
182 pr_info("liballoc: initial l_memRoot initialization failed");
183#endif
184 }
185
186#if LIBALLOC_PRINT_DEBUG_MESSAGES
187 pr_info("liballoc: set up first memory major %p", (void *) l_memroot);
188#endif
189}

191void liballoc_init(void)
192{
193 l_memroot = NULL;
194 l_bestbet = NULL;
195 MOS_LIB_ASSERT_X(l_memroot == NULL, "liballoc_init() called twice");
196 l_alloc_n_page_once = 8;
197#if LIBALLOC_PRINT_DEBUG_MESSAGES
198 pr_info("liballoc: initialization of liballoc " VERSION "");
199#endif
200
201 liballoc_first_alloc();
202}

204void *liballoc_malloc(size_t req_size)
205{
206 unsigned long size = req_size;
207
208 // For alignment, we adjust size so there's enough space to align.
209 if (ALIGNMENT > 1)
210 {
211 size += ALIGNMENT + ALIGN_INFO;
212 }
213 // So, ideally, we really want an alignment of 0 or 1 in order to save space.
214
215#if MOS_MM_LIBALLOC_LOCKS
216 liballoc_lock();
217#endif
218
219 if (size == 0)
220 {
221 mos_warn("liballoc: liballoc_malloc(0) called.");
222 l_warnings += 1;
223#if MOS_MM_LIBALLOC_LOCKS
224 liballoc_unlock();
225#endif
226 return liballoc_malloc(1);
227 }
228
229 if (l_memroot == NULL)
230 {
231 liballoc_first_alloc();
232 }
233
234 // Now we need to bounce through every major and find enough space....
235
236 int startedBet = 0;
237 size_t bestSize = 0;
238 liballoc_block_t *block = l_memroot;
239
240 // Start at the best bet....
241 if (l_bestbet != NULL)
242 {
243 bestSize = l_bestbet->size - l_bestbet->usage;
244
245 if (bestSize > (size + sizeof(liballoc_part_t)))
246 {
247 block = l_bestbet;
248 startedBet = 1;
249 }
250 }
251
252 while (block != NULL)
253 {
254 s64 diff = block->size - block->usage;
255 // free memory in the block
256
257 if (bestSize < diff)
258 {
259 // Hmm.. this one has more memory then our bestBet. Remember!
260 l_bestbet = block;
261 bestSize = diff;
262 }
263
264#if USE_CASE1
265 // CASE 1: There is not enough space in this major block.
266 if (diff < (size + sizeof(liballoc_part_t)))
267 {
268#if LIBALLOC_PRINT_DEBUG_MESSAGES
269 pr_info("CASE 1: Insufficient space in block %p", (void *) block);
270#endif
271
272 // Another major block next to this one?
273 if (block->next != NULL)
274 {
275 block = block->next; // Hop to that one.
276 continue;
277 }
278
279 if (startedBet == 1) // If we started at the best bet,
280 { // let's start all over again.
281 block = l_memroot;
282 startedBet = 0;
283 continue;
284 }
285
286 // Create a new major block next to this one and...
287 block->next = allocate_new_pages_for(size); // next one will be okay.
288 if (block->next == NULL)
289 break; // no more memory.
290 block->next->prev = block;
291 block = block->next;
292
293 // .. fall through to CASE 2 ..
294 }
295#endif
296
297#if USE_CASE2
298 // CASE 2: It's a brand new block.
299 if (block->first == NULL)
300 {
301 block->first = (liballoc_part_t *) ((uintptr_t) block + sizeof(liballoc_block_t));
302
303 block->first->magic = LIBALLOC_MAGIC;
304 block->first->prev = NULL;
305 block->first->next = NULL;
306 block->first->block = block;
307 block->first->size = size;
308 block->first->req_size = req_size;
309 block->usage += size + sizeof(liballoc_part_t);
310
311 l_mem_inuse += size;
312
313 void *p = (void *) ((uintptr_t) (block->first) + sizeof(liballoc_part_t));
314
315 LIBALLOC_ALIGN_PTR(p);
316
317#if MOS_MM_LIBALLOC_LOCKS
318 liballoc_unlock(); // release the lock
319#endif
320
321#if LIBALLOC_PRINT_DEBUG_MESSAGES
322 pr_info("liballoc: allocating %lu bytes at %p", size, p);
323#endif
324 return p;
325 }
326
327#endif
328
329 // CASE 3: Block in use and enough space at the start of the block.
330#if USE_CASE3
331 {
332 s64 diff = (uintptr_t) block->first - (uintptr_t) block;
333 diff -= sizeof(liballoc_block_t);
334
335 if (diff >= (size + sizeof(liballoc_part_t)))
336 {
337 // Yes, space in front. Squeeze in.
338 block->first->prev = (liballoc_part_t *) ((uintptr_t) block + sizeof(liballoc_block_t));
339 block->first->prev->next = block->first;
340 block->first = block->first->prev;
341
342 block->first->magic = LIBALLOC_MAGIC;
343 block->first->prev = NULL;
344 block->first->block = block;
345 block->first->size = size;
346 block->first->req_size = req_size;
347 block->usage += size + sizeof(liballoc_part_t);
348
349 l_mem_inuse += size;
350
351 void *p = (void *) ((uintptr_t) (block->first) + sizeof(liballoc_part_t));
352 LIBALLOC_ALIGN_PTR(p);
353
354#if MOS_MM_LIBALLOC_LOCKS
355 liballoc_unlock(); // release the lock
356#endif
357#if LIBALLOC_PRINT_DEBUG_MESSAGES
358 pr_info("liballoc: allocating %lu bytes at %p", size, p);
359#endif
360 return p;
361 }
362 }
363#endif
364
365#if USE_CASE4
366 // CASE 4: There is enough space in this block. But is it contiguous?
367 liballoc_part_t *section = block->first;
368
369 // Looping within the block now...
370 while (section != NULL)
371 {
372 // CASE 4.1: End of a section in a block. check for enough space from last and end?
373 if (section->next == NULL)
374 {
375 // the rest of this block is free... is it big enough?
376 size_t size_left = block->size; // size of the block
377 size_left -= (uintptr_t) section - (uintptr_t) block; // minus the area before this section
378 size_left -= sizeof(liballoc_part_t) + section->size; // minus this section
379
380 // if there's still enough space
381 if (size_left >= sizeof(liballoc_part_t) + size)
382 {
383 // yay....
384 section->next = (liballoc_part_t *) ((uintptr_t) section + sizeof(liballoc_part_t) + section->size);
385 section->next->prev = section;
386
387 // go to the next section
388 section = section->next;
389 section->next = NULL;
390 section->magic = LIBALLOC_MAGIC;
391 section->block = block;
392 section->size = size;
393 section->req_size = req_size;
394
395 block->usage += size + sizeof(liballoc_part_t);
396
397 l_mem_inuse += size;
398
399 void *p = (void *) ((uintptr_t) section + sizeof(liballoc_part_t));
400 LIBALLOC_ALIGN_PTR(p);
401
402#if MOS_MM_LIBALLOC_LOCKS
403 liballoc_unlock(); // release the lock
404#endif
405#if LIBALLOC_PRINT_DEBUG_MESSAGES
406 pr_info("liballoc: allocating %lu bytes at %p", size, p);
407#endif
408 return p;
409 }
410 }
411
412 // CASE 4.2: Is there space between two minors?
413 if (section->next != NULL)
414 {
415 // is the difference between here and next big enough?
416 s64 diff = (uintptr_t) (section->next);
417 diff -= (uintptr_t) section;
418 diff -= sizeof(liballoc_part_t);
419 diff -= section->size;
420 // minus our existing usage.
421
422 if (diff >= (size + sizeof(liballoc_part_t)))
423 {
424 // yay......
425 liballoc_part_t *new_min = (liballoc_part_t *) ((uintptr_t) section + sizeof(liballoc_part_t) + section->size);
426
427 new_min->magic = LIBALLOC_MAGIC;
428 new_min->next = section->next;
429 new_min->prev = section;
430 new_min->size = size;
431 new_min->req_size = req_size;
432 new_min->block = block;
433 section->next->prev = new_min;
434 section->next = new_min;
435 block->usage += size + sizeof(liballoc_part_t);
436
437 l_mem_inuse += size;
438
439 void *p = (void *) ((uintptr_t) new_min + sizeof(liballoc_part_t));
440 LIBALLOC_ALIGN_PTR(p);
441
442#if MOS_MM_LIBALLOC_LOCKS
443 liballoc_unlock(); // release the lock
444#endif
445#if LIBALLOC_PRINT_DEBUG_MESSAGES
446 pr_info("liballoc: allocating %lu bytes at %p", size, p);
447#endif
448 return p;
449 }
450 } // min->next != NULL
451
452 section = section->next;
453 } // while min != NULL ...
454
455#endif
456
457#if USE_CASE5
458 // CASE 5: Block full! Ensure next block and loop.
459 if (block->next == NULL)
460 {
461#if LIBALLOC_PRINT_DEBUG_MESSAGES
462 pr_info("CASE 5: block full");
463#endif
464
465 if (startedBet == 1)
466 {
467 block = l_memroot;
468 startedBet = 0;
469 continue;
470 }
471
472 // we've run out. we need more...
473 block->next = allocate_new_pages_for(size); // next one guaranteed to be okay
474 if (block->next == NULL)
475 break; // uh oh, no more memory.....
476 block->next->prev = block;
477 }
478
479#endif
480
481 block = block->next;
482 } // while (maj != NULL)
483
484#if MOS_MM_LIBALLOC_LOCKS
485 liballoc_unlock(); // release the lock
486#endif
487
488 mos_warn("No memory available.");
489
490#if LIBALLOC_PRINT_DEBUG_MESSAGES
491 liballoc_dump();
492#endif
493
494 return NULL;
495}

497void liballoc_free(const void *ptr)
498{
499 liballoc_part_t *min;
500 liballoc_block_t *maj;
501
502 if (ptr == NULL)
503 {
504 l_warnings += 1;
505 mos_warn("liballoc: free(NULL) called");
506 return;
507 }
508
509 LIBALLOC_UNALIGN_PTR(ptr);
510
511#if MOS_MM_LIBALLOC_LOCKS
512 liballoc_lock(); // lockit
513#endif
514
515 min = (liballoc_part_t *) ((uintptr_t) ptr - sizeof(liballoc_part_t));
516
517 if (min->magic != LIBALLOC_MAGIC)
518 {
519 l_errors += 1;
520
521 // Check for overrun errors. For all bytes of LIBALLOC_MAGIC
522 if (((min->magic & 0xFFFFFF) == (LIBALLOC_MAGIC & 0xFFFFFF)) || ((min->magic & 0xFFFF) == (LIBALLOC_MAGIC & 0xFFFF)) ||
523 ((min->magic & 0xFF) == (LIBALLOC_MAGIC & 0xFF)))
524 {
525 l_possible_overruns += 1;
526 mos_warn("liballoc: ERROR: Possible 1-3 byte overrun for magic %x != %x", min->magic, LIBALLOC_MAGIC);
527 }
528
529 if (min->magic == LIBALLOC_DEAD)
530 {
531 mos_warn("liballoc: multiple free() attempt on %p", ptr);
532 }
533 else
534 {
535 mos_warn("liballoc: bad free(%p) called.", ptr);
536 }
537#if MOS_MM_LIBALLOC_LOCKS
538 // being lied to...
539 liballoc_unlock(); // release the lock
540#endif
541 return;
542 }
543
544 maj = min->block;
545 l_mem_inuse -= min->size;
546 maj->usage -= (min->size + sizeof(liballoc_part_t));
547 min->magic = LIBALLOC_DEAD; // No mojo.
548
549 if (min->next != NULL)
550 min->next->prev = min->prev;
551 if (min->prev != NULL)
552 min->prev->next = min->next;
553
554 if (min->prev == NULL)
555 maj->first = min->next;
556 // Might empty the block. This was the first
557 // minor.
558
559 // We need to clean up after the majors now....
560
561 if (maj->first == NULL) // Block completely unused.
562 {
563 if (l_memroot == maj)
564 l_memroot = maj->next;
565 if (l_bestbet == maj)
566 l_bestbet = NULL;
567 if (maj->prev != NULL)
568 maj->prev->next = maj->next;
569 if (maj->next != NULL)
570 maj->next->prev = maj->prev;
571 l_mem_allocated -= maj->size;
572
573 liballoc_free_page(maj, maj->pages);
574 }
575 else
576 {
577 if (l_bestbet != NULL)
578 {
579 int bestSize = l_bestbet->size - l_bestbet->usage;
580 int majSize = maj->size - maj->usage;
581
582 if (majSize > bestSize)
583 l_bestbet = maj;
584 }
585 }
586
587#if MOS_MM_LIBALLOC_LOCKS
588 liballoc_unlock(); // release the lock
589#endif
590}

592void *liballoc_calloc(size_t nobj, size_t size)
593{
594 MOS_LIB_ASSERT_X(nobj > 0, "You Fool! You can't allocate 0 objects!");
595 size_t real_size = nobj * size;
596 void *p = liballoc_malloc(real_size);
597 memzero(p, real_size);
598 return p;
599}

601void *liballoc_realloc(void *p, size_t size)
602{
603
604 // Honour the case of size == 0 => free old and return NULL
605 if (size == 0)
606 {
607 liballoc_free(p);
608 return NULL;
609 }
610
611 // In the case of a NULL pointer, return a simple malloc.
612 if (p == NULL)
613 return liballoc_malloc(size);
614
615 // Unalign the pointer if required.
616
617 void *ptr = p;
618 LIBALLOC_UNALIGN_PTR(ptr);
619
620#if MOS_MM_LIBALLOC_LOCKS
621 liballoc_lock(); // lockit
622#endif
623
624 liballoc_part_t *min = (liballoc_part_t *) ((uintptr_t) ptr - sizeof(liballoc_part_t));
625
626 // Ensure it is a valid structure.
627 if (min->magic != LIBALLOC_MAGIC)
628 {
629 l_errors += 1;
630
631 // Check for overrun errors. For all bytes of LIBALLOC_MAGIC
632 if (((min->magic & 0xFFFFFF) == (LIBALLOC_MAGIC & 0xFFFFFF)) || ((min->magic & 0xFFFF) == (LIBALLOC_MAGIC & 0xFFFF)) ||
633 ((min->magic & 0xFF) == (LIBALLOC_MAGIC & 0xFF)))
634 {
635 l_possible_overruns += 1;
636 mos_warn("liballoc: Possible 1-3 byte overrun for magic %x != %x", min->magic, LIBALLOC_MAGIC);
637 }
638
639 if (min->magic == LIBALLOC_DEAD)
640 {
641 mos_warn("liballoc: multiple free() attempt on %p from %p.", ptr, __builtin_return_address(0));
642 }
643 else
644 {
645 mos_warn("liballoc: Bad free(%p) called", ptr);
646 }
647#if MOS_MM_LIBALLOC_LOCKS
648 // being lied to...
649 liballoc_unlock(); // release the lock
650#endif
651 return NULL;
652 }
653
654 // Definitely a memory block.
655
656 unsigned int real_size = min->req_size;
657
658 if (real_size >= size)
659 {
660 min->req_size = size;
661#if MOS_MM_LIBALLOC_LOCKS
662 liballoc_unlock();
663#endif
664 return p;
665 }
666#if MOS_MM_LIBALLOC_LOCKS
667 liballoc_unlock();
668#endif
669
670 // If we got here then we're reallocating to a block bigger than us.
671 ptr = liballoc_malloc(size); // We need to allocate new memory
672 memcpy(ptr, p, real_size);
673 liballoc_free(p);
674
675 return ptr;
676}
