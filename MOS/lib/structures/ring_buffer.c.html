10static u8 ring_buffer_get(u8 *data, ring_buffer_pos_t *pos, size_t index)
11{
12 return data[(pos->head + index) % pos->size];
13}

15ring_buffer_t *ring_buffer_create(size_t capacity)
16{
17 if (capacity == 0)
18 return NULL; // forget about it
19
20 ring_buffer_t *rb = liballoc_malloc(sizeof(ring_buffer_t));
21 if (!rb)
22 return NULL;
23 rb->data = liballoc_malloc(capacity);
24 if (!rb->data)
25 {
26 liballoc_free(rb);
27 return NULL;
28 }
29 ring_buffer_pos_init(&rb->pos, capacity);
30 return rb;
31}

33ring_buffer_t *ring_buffer_create_at(void *data, size_t capacity)

46void ring_buffer_pos_init(ring_buffer_pos_t *pos, size_t capacity)

60bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity)
61{
62 if (new_capacity < buffer->pos.size)
63 return false;
64 void *new_data = liballoc_malloc(new_capacity);
65 if (!new_data)
66 return false;
67 size_t i = 0;
68 while (i < buffer->pos.size)
69 {
70 ((char *) new_data)[i] = ring_buffer_get(buffer->data, &buffer->pos, i);
71 i++;
72 }
73 liballoc_free(buffer->data);
74 buffer->data = new_data;
75 buffer->pos.capacity = new_capacity;
76 buffer->pos.head = 0;
77 buffer->pos.next_pos = buffer->pos.size;
78 return true;
79}

81size_t ring_buffer_pos_push_back(u8 *data, ring_buffer_pos_t *pos, const u8 *target, size_t size)
82{
83 if (pos->size + size > pos->capacity)
84 return 0;
85
86 size_t first_part_i = pos->next_pos;
87 size_t first_part_size = MIN(size, pos->capacity - pos->next_pos);
88 size_t second_part = size - first_part_size;
89 memcpy(data + first_part_i, target, first_part_size);
90 memcpy(data, target + first_part_size, second_part);
91 pos->next_pos = (pos->next_pos + size) % pos->capacity;
92 pos->size += size;
93 return size;
94}

96size_t ring_buffer_pos_pop_back(u8 *data, ring_buffer_pos_t *pos, u8 *target, size_t size)
97{
98 if (size > pos->size)
99 size = pos->size;
100
101 size_t first_part_i = (pos->capacity + pos->next_pos - size) % pos->capacity;
102 size_t first_part_size = MIN(size, pos->capacity - first_part_i);
103
104 size_t second_part_i = 0;
105 size_t second_part_size = size - first_part_size;
106
107 memcpy(target, data + first_part_i, first_part_size);
108 memcpy(target + first_part_size, data + second_part_i, second_part_size);
109
110 pos->next_pos = (pos->capacity + pos->next_pos - size) % pos->capacity;
111 pos->size -= size;
112
113 return size;
114}

116size_t ring_buffer_pos_push_front(u8 *data, ring_buffer_pos_t *pos, const u8 *target, size_t size)
117{
118 if (pos->size + size > pos->capacity)
119 return 0;
120
121 size_t first_part_i = (pos->capacity + pos->head - size) % pos->capacity;
122 size_t first_part_size = MIN(size, pos->capacity - first_part_i);
123
124 size_t second_part_i = 0;
125 size_t second_part_size = size - first_part_size;
126
127 memcpy(data + first_part_i, target, first_part_size);
128 memcpy(data + second_part_i, target + first_part_size, second_part_size);
129
130 pos->head = (pos->capacity + pos->head - size) % pos->capacity;
131 pos->size += size;
132 return size;
133}

135size_t ring_buffer_pos_pop_front(u8 *data, ring_buffer_pos_t *pos, u8 *target, size_t size)
136{
137 if (size > pos->size)
138 size = pos->size;
139
140 size_t first_part_i = pos->head;
141 size_t first_part_size = MIN(size, pos->capacity - first_part_i);
142
143 size_t second_part_i = 0;
144 size_t second_part_size = size - first_part_size;
145
146 memcpy(target, data + first_part_i, first_part_size);
147 memcpy(target + first_part_size, data + second_part_i, second_part_size);
148
149 pos->head = (pos->head + size) % pos->capacity;
150 pos->size -= size;
151 return size;
152}
