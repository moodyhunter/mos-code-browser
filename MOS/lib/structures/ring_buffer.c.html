8static u8 ring_buffer_get(ring_buffer_t *buffer, size_t index)
9{
10 return buffer->data[(buffer->head + index) % buffer->size];
11}
12
13ring_buffer_t *ring_buffer_create(size_t capacity)
14{
15 if (capacity == 0)
16 return NULL; // forget about it
17
18 ring_buffer_t *rb = kmalloc(sizeof(ring_buffer_t));
19 if (!rb)
20 return NULL;
21 rb->data = kmalloc(capacity);
22 if (!rb->data)
23 {
24 kfree(rb);
25 return NULL;
26 }
27 rb->capacity = capacity;
28 rb->size = 0;
29 rb->head = 0;
30 rb->tail = 0;
31 return rb;
32}
33
34void ring_buffer_destroy(ring_buffer_t *buffer)
35{
36 kfree(buffer->data);
37 kfree(buffer);
38}
39
40bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity)
41{
42 if (new_capacity < buffer->size)
43 return false;
44 void *new_data = kmalloc(new_capacity);
45 if (!new_data)
46 return false;
47 size_t i = 0;
48 while (i < buffer->size)
49 {
50 ((char *) new_data)[i] = ring_buffer_get(buffer, i);
51 i++;
52 }
53 kfree(buffer->data);
54 buffer->data = new_data;
55 buffer->capacity = new_capacity;
56 buffer->head = 0;
57 buffer->tail = buffer->size;
58 return true;
59}

61size_t ring_buffer_write_back(ring_buffer_t *buffer, const u8 *data, size_t size)
62{
63 size_t written = 0;
64 while (written < size && buffer->size < buffer->capacity)
65 {
66 buffer->data[buffer->head] = ((const u8 *) data)[written];
67 buffer->head = (buffer->head + 1) % buffer->capacity;
68 buffer->size++;
69 written++;
70 }
71 return written;
72}
73
74size_t ring_buffer_read_back(ring_buffer_t *buffer, u8 *data, size_t size)
75{
76 size_t read = 0;
77 while (read < size && buffer->size > 0)
78 {
79 ((u8 *) data)[read] = buffer->data[buffer->tail];
80 buffer->tail = (buffer->tail + 1) % buffer->capacity;
81 buffer->size--;
82 read++;
83 }
84 return read;
85}
