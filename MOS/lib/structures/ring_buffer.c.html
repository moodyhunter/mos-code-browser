9static u8 ring_buffer_get(ring_buffer_t *buffer, size_t index)
10{
11 return buffer->data[(buffer->head + index) % buffer->size];
12}

14ring_buffer_t *ring_buffer_create(size_t capacity)
15{
16 if (capacity == 0)
17 return NULL; // forget about it
18
19 ring_buffer_t *rb = liballoc_malloc(sizeof(ring_buffer_t));
20 if (!rb)
21 return NULL;
22 rb->data = liballoc_malloc(capacity);
23 if (!rb->data)
24 {
25 liballoc_free(rb);
26 return NULL;
27 }
28 rb->capacity = capacity;
29 rb->size = 0;
30 rb->head = 0;
31 rb->next_pos = 0;
32 return rb;
33}

57bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity)
58{
59 if (new_capacity < buffer->size)
60 return false;
61 void *new_data = liballoc_malloc(new_capacity);
62 if (!new_data)
63 return false;
64 size_t i = 0;
65 while (i < buffer->size)
66 {
67 ((char *) new_data)[i] = ring_buffer_get(buffer, i);
68 i++;
69 }
70 liballoc_free(buffer->data);
71 buffer->data = new_data;
72 buffer->capacity = new_capacity;
73 buffer->head = 0;
74 buffer->next_pos = buffer->size;
75 return true;
76}

78size_t ring_buffer_push_back(ring_buffer_t *buffer, const u8 *data, size_t size)
79{
80 size_t written = 0;
81 while (written < size && buffer->size < buffer->capacity)
82 {
83 buffer->data[buffer->next_pos] = ((const u8 *) data)[written];
84 buffer->next_pos = (buffer->next_pos + 1) % buffer->capacity;
85 buffer->size++;
86 written++;
87 }
88 return written;
89}

91size_t ring_buffer_pop_back(ring_buffer_t *buffer, u8 *data, size_t size)
92{
93 size_t read = 0;
94 while (read < size && buffer->size > 0)
95 {
96 buffer->next_pos = (buffer->capacity + buffer->next_pos - 1) % buffer->capacity;
97 ((u8 *) data)[read] = buffer->data[buffer->next_pos];
98 buffer->data[buffer->next_pos] = 0;
99 buffer->size--;
100 read++;
101 }
102 return read;
103}
