8static u8 ring_buffer_get(ring_buffer_t *buffer, size_t index)
9{
10 return buffer->data[(buffer->head + index) % buffer->size];
11}
12
13ring_buffer_t *ring_buffer_create(size_t capacity)
14{
15 if (capacity == 0)
16 return NULL; // forget about it
17
18 ring_buffer_t *rb = mos_lib_malloc(sizeof(ring_buffer_t));
19 if (!rb)
20 return NULL;
21 rb->data = mos_lib_malloc(capacity);
22 if (!rb->data)
23 {
24 mos_lib_free(rb);
25 return NULL;
26 }
27 rb->capacity = capacity;
28 rb->size = 0;
29 rb->head = 0;
30 rb->next_pos = 0;
31 return rb;
32}

34void ring_buffer_destroy(ring_buffer_t *buffer)
35{
36 mos_lib_free(buffer->data);
37 mos_lib_free(buffer);
38}

40bool ring_buffer_resize(ring_buffer_t *buffer, size_t new_capacity)
41{
42 if (new_capacity < buffer->size)
43 return false;
44 void *new_data = mos_lib_malloc(new_capacity);
45 if (!new_data)
46 return false;
47 size_t i = 0;
48 while (i < buffer->size)
49 {
50 ((char *) new_data)[i] = ring_buffer_get(buffer, i);
51 i++;
52 }
53 mos_lib_free(buffer->data);
54 buffer->data = new_data;
55 buffer->capacity = new_capacity;
56 buffer->head = 0;
57 buffer->next_pos = buffer->size;
58 return true;
59}

61size_t ring_buffer_push_back(ring_buffer_t *buffer, const u8 *data, size_t size)
62{
63 size_t written = 0;
64 while (written < size && buffer->size < buffer->capacity)
65 {
66 buffer->data[buffer->next_pos] = ((const u8 *) data)[written];
67 buffer->next_pos = (buffer->next_pos + 1) % buffer->capacity;
68 buffer->size++;
69 written++;
70 }
71 return written;
72}

74size_t ring_buffer_pop_back(ring_buffer_t *buffer, u8 *data, size_t size)
75{
76 size_t read = 0;
77 while (read < size && buffer->size > 0)
78 {
79 buffer->next_pos = (buffer->capacity + buffer->next_pos - 1) % buffer->capacity;
80 ((u8 *) data)[read] = buffer->data[buffer->next_pos];
81 buffer->data[buffer->next_pos] = 0;
82 buffer->size--;
83 read++;
84 }
85 return read;
86}
