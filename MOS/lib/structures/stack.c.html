9void stack_init(downwards_stack_t *stack, void *mem_region_bottom, size_t size)
10{
11 // the stack grows downwards
12 uintptr_t stack_top = (uintptr_t) mem_region_bottom + size;
13 stack->top = stack_top;
14 stack->head = stack_top;
15 stack->capacity = size;
16}
17
18void stack_deinit(downwards_stack_t *stack)
19{
20 memset(stack, 0, sizeof(downwards_stack_t));
21}
22
23void *stack_grow(downwards_stack_t *stack, size_t size)
24{
25 // high memory | top -----> head -----> top - capacity | low memory
26 if (unlikely(stack->head - (stack->top - stack->capacity) < size))
27 {
28 mos_warn("stack overflow on stack %p, attempted to push %zu bytes", (void *) stack, size);
29 return NULL;
30 }
31
32 stack->head = stack->head - size;
33 return (void *) stack->head;
34}
35
36void stack_push(downwards_stack_t *stack, void *data, size_t size)
37{
38 // high memory | top -----> head -----> top - capacity | low memory
39 if (unlikely(stack->head - (stack->top - stack->capacity) < size))
40 {
41 mos_warn("stack overflow on stack %p, attempted to push %zu bytes", (void *) stack, size);
42 return;
43 }
44
45 stack->head = stack->head - size;
46 memcpy((void *) stack->head, data, size);
47}
48
49void stack_pop(downwards_stack_t *stack, size_t size, void *data)
50{
51 // high memory | top -----> head -----> top - capacity | low memory
52 if (unlikely(stack->head - stack->top < size))
53 {
54 mos_warn("stack underflow on stack %p, attempted to pop %zu bytes", (void *) stack, size);
55 return;
56 }
57
58 if (data != NULL)
59 memcpy(data, (void *) stack->head, size);
60 stack->head = stack->head + size;
61}
