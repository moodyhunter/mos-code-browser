8bitmap_line_t *bitmap_create(size_t size)
9{
10 size_t nlines = BITMAP_LINE_COUNT(size);
11 bitmap_line_t *bitmap = liballoc_malloc(nlines * sizeof(bitmap_line_t));
12 bitmap_zero(bitmap, nlines);
13 return bitmap;
14}

16void bitmap_zero(bitmap_line_t *bitmap, size_t bitmap_nlines)
17{
18 memzero(bitmap, bitmap_nlines * sizeof(bitmap_line_t));
19}

21void bitmap_set(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t index)
22{
23 size_t line = index / BITMAP_LINE_BITS;
24 size_t bit = index % BITMAP_LINE_BITS;
25 if (line >= bitmap_nlines)
26 return;
27 bitmap[line] |= ((bitmap_line_t) 1 << bit);
28}

30void bitmap_clear(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t index)

33 size_t bit = index % BITMAP_LINE_BITS;

39bool bitmap_get(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t index)
40{
41 size_t line = index / BITMAP_LINE_BITS;
42 size_t bit = index % BITMAP_LINE_BITS;
43 if (line >= bitmap_nlines)
44 return false;
45 return bitmap[line] & ((bitmap_line_t) 1 << bit);
46}

48size_t bitmap_find_first_free_n(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t begin_bit, size_t n_bits)
49{
50 size_t target_starting_line = begin_bit / BITMAP_LINE_BITS;
51 size_t target_starting_bit = begin_bit % BITMAP_LINE_BITS;
52 size_t free_bits = 0;
53
54 for (size_t line_i = target_starting_line; free_bits < n_bits; line_i++)
55 {
56 if (line_i >= bitmap_nlines)
57 return 0;
58
59 bitmap_line_t line = bitmap[line_i];
60
61 if (line == 0)
62 {
63 free_bits += BITMAP_LINE_BITS;
64 continue;
65 }
66 else if (line == (bitmap_line_t) ~0)
67 {
68 // a full line of occupied bits
69 target_starting_line = line_i + 1;
70 free_bits = 0;
71 target_starting_bit = 0;
72 continue;
73 }
74
75 for (size_t bit = 0; bit < BITMAP_LINE_BITS; bit++)
76 {
77 if (free_bits >= n_bits)
78 break;
79
80 if (!bitmap_get(&line, bitmap_nlines, bit))
81 {
82 // we have found a free line
83 free_bits++;
84 }
85 else
86 {
87 // this bit is occupied
88 free_bits = 0;
89 target_starting_bit = bit + 1;
90 target_starting_line = line_i;
91 }
92 }
93 }
94
95 return target_starting_line * BITMAP_LINE_BITS + target_starting_bit;
96}

98void bitmap_set_range(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t start_bit, size_t end_bit, bool value)
99{
100 size_t start_line = start_bit / BITMAP_LINE_BITS;
101 size_t start_bit_in_line = start_bit % BITMAP_LINE_BITS;
102 size_t end_line = end_bit / BITMAP_LINE_BITS;
103 size_t end_bit_in_line = end_bit % BITMAP_LINE_BITS;
104
105 if (start_line >= bitmap_nlines || end_line >= bitmap_nlines)
106 return;
107
108 if (start_line == end_line)
109 {
110 bitmap_line_t line = bitmap[start_line];
111 bitmap_line_t mask = ((bitmap_line_t) 1 << (end_bit_in_line - start_bit_in_line + 1)) - 1;
112 mask <<= start_bit_in_line;
113 if (value)
114 line |= mask;
115 else
116 line &= ~mask;
117 bitmap[start_line] = line;
118 return;
119 }
120
121 bitmap_line_t line = bitmap[start_line];
122 bitmap_line_t mask = ((bitmap_line_t) 1 << (BITMAP_LINE_BITS - start_bit_in_line)) - 1;
123 mask <<= start_bit_in_line;
124 if (value)
125 line |= mask;
126 else
127 line &= ~mask;
128 bitmap[start_line] = line;
129
130 for (size_t i = start_line + 1; i < end_line; i++)
131 {
132 if (value)
133 bitmap[i] = (bitmap_line_t) ~0;
134 else
135 bitmap[i] = 0;
136 }
137
138 line = bitmap[end_line];
139 mask = ((bitmap_line_t) 1 << (end_bit_in_line + 1)) - 1;
140 if (value)
141 line |= mask;
142 else
143 line &= ~mask;
144 bitmap[end_line] = line;
145}
