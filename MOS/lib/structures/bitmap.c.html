8bitmap_line_t *bitmap_create(size_t size)
9{
10 size_t nlines = bitmap_nlines(size);
11 bitmap_line_t *bitmap = liballoc_malloc(nlines * sizeof(bitmap_line_t));
12 bitmap_zero(bitmap, nlines);
13 return bitmap;
14}
15
16size_t bitmap_nlines(size_t size)
17{
18 return ALIGN_UP(size, BITMAP_LINE_BITS) / BITMAP_LINE_BITS;
19}
20
21void bitmap_zero(bitmap_line_t *bitmap, size_t bitmap_nlines)
22{
23 memzero(bitmap, bitmap_nlines * sizeof(bitmap_line_t));
24}
25
26void bitmap_set(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t index)
27{
28 size_t line = index / BITMAP_LINE_BITS;
29 size_t bit = index % BITMAP_LINE_BITS;
30 if (line >= bitmap_nlines)
31 return;
32 bitmap[line] |= ((bitmap_line_t) 1 << bit);
33}
34
35void bitmap_clear(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t index)
36{
37 size_t line = index / BITMAP_LINE_BITS;
38 size_t bit = index % BITMAP_LINE_BITS;
39 if (line >= bitmap_nlines)
40 return;
41 bitmap[line] &= ~((bitmap_line_t) 1 << bit);
42}

44bool bitmap_get(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t index)
45{
46 size_t line = index / BITMAP_LINE_BITS;
47 size_t bit = index % BITMAP_LINE_BITS;
48 if (line >= bitmap_nlines)
49 return false;
50 return bitmap[line] & ((bitmap_line_t) 1 << bit);
51}

53size_t bitmap_find_first_free_n(bitmap_line_t *bitmap, size_t bitmap_nlines, size_t n_bits)
54{
55 size_t n_free_pages = 0;
56 size_t target_starting_bit = 0;
57 size_t map_line_i = 0;
58 for (size_t i = 0; n_free_pages < n_bits; i++)
59 {
60 if (i >= bitmap_nlines)
61 return 0;
62
63 bitmap_line_t line = bitmap[i];
64
65 if (line == 0)
66 {
67 n_free_pages += BITMAP_LINE_BITS;
68 continue;
69 }
70 else if (line == (bitmap_line_t) ~0)
71 {
72 // a full line of used pages
73 map_line_i = i + 1;
74 n_free_pages = 0;
75 target_starting_bit = 0;
76 continue;
77 }
78
79 for (size_t bit = 0; bit < BITMAP_LINE_BITS; bit++)
80 {
81 if (!bitmap_get(&line, bitmap_nlines, bit))
82 {
83 // we have found a free page
84 n_free_pages++;
85 }
86 else
87 {
88 // this page is used
89 if (n_free_pages >= n_bits)
90 // but we have enough free pages
91 break;
92 n_free_pages = 0;
93 target_starting_bit = bit + 1;
94 map_line_i = i;
95 }
96 }
97 }
98
99 return map_line_i * BITMAP_LINE_BITS + target_starting_bit;
100}
