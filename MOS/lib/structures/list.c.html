5void linked_list_init(list_node_t *node)
6{
7 node->prev = node;
8 node->next = node;
9}

11bool list_is_empty(list_node_t *head)
12{
13 return head->next == head;
14}
15
16void list_node_remove(list_node_t *node)
17{
18 node->prev->next = node->next;
19 node->next->prev = node->prev;
20
21 // detach the node from the list
22 // ! Is this really necessary?
23 node->next = node;
24 node->prev = node;
25}
26
27// ! Internal API
28static void list_node_insert(list_node_t *prev, list_node_t *new_node, list_node_t *next)
29{
30 new_node->prev = prev;
31 new_node->next = next;
32 prev->next = new_node;
33 next->prev = new_node;
34}
35
36void list_node_prepend(list_node_t *head, list_node_t *item)
37{
38 list_node_insert(head, item, head->next);
39}
40
41void list_node_append(list_node_t *head, list_node_t *item)
42{
43 // The list is circular, so accessing the list tail is like the prev of its head.
44 list_node_insert(head->prev, item, head);
45}
46
47void list_node_insert_before(list_node_t *element, list_node_t *item)
48{
49 list_node_insert(element->prev, item, element);
50}
51
52void list_node_insert_after(list_node_t *element, list_node_t *item)
53{
54 list_node_insert(element, item, element->next);
55}
