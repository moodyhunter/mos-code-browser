8void tree_add_child(tree_node_t *parent, tree_node_t *child)
9{
10 MOS_LIB_ASSERT(parent != NULL);
11 MOS_LIB_ASSERT(child != NULL);
12 MOS_LIB_ASSERT(child->parent == NULL);
13 MOS_LIB_ASSERT(child->n_children == 0);
14 MOS_LIB_ASSERT(child->children == NULL);
15 child->parent = parent;
16 parent->n_children++;
17 parent->children = mos_lib_realloc(parent->children, parent->n_children * sizeof(tree_node_t *));
18 parent->children[parent->n_children - 1] = child;
19}

29const tree_node_t *tree_find_child_by_name(tree_op_t *op, const tree_node_t *node, const char *name, size_t name_len)
30{
31 if (node == NULL)
32 return NULL;
33 if (name_len == 0)
34 return node;
35 for (size_t i = 0; i < node->n_children; i++)
36 {
37 const tree_node_t *child = node->children[i];
38
39 char *child_name = NULL;
40 size_t child_name_len = 0;
41 op->get_node_name(child, &child_name, &child_name_len);
42 if (name_len == child_name_len && strncmp(name, child_name, name_len) == 0)
43 return child;
44 }
45 return NULL;
46}

48const tree_node_t *tree_find_common_prefix(const tree_node_t *node1, const tree_node_t *node2)
49{
50 if (node1 == NULL || node2 == NULL)
51 return NULL;
52 if (node1 == node2)
53 return node1;
54 if (node1->parent == NULL || node2->parent == NULL)
55 return NULL;
56 if (node1->parent == node2)
57 return node2;
58 if (node2->parent == node1)
59 return node1;
60 if (node1->parent == node2->parent)
61 return node1->parent;
62 return tree_find_common_prefix(node1->parent, node2->parent);
63}

65void tree_trace_to_root(const tree_node_t *node, void (*trace_func)(const tree_node_t *node))
66{
67 if (node == NULL)
68 return;
69 trace_func(node);
70 tree_trace_to_root(node->parent, trace_func);
