9typedef struct hashmap_entry
10{
11 const void *key;
12 void *value;
13 hashmap_entry_t *next;
14} hashmap_entry_t;
15
16void hashmap_init(hashmap_t *map, size_t capacity, hashmap_hash_t hash_func, hashmap_key_compare_t compare_func)
17{
18 MOS_ASSERT(map);
19 if (map->magic == HASHMAP_MAGIC)
20 {
21 mos_panic("hashmap_init: hashmap %p is already initialized", (void *) map);
22 return;
23 }
24 map->magic = HASHMAP_MAGIC;
25 map->entries = kmalloc(sizeof(hashmap_entry_t *) * capacity);
26 memset(map->entries, 0, sizeof(hashmap_entry_t *) * capacity);
27 map->capacity = capacity;
28 map->size = 0;
29 map->hash_func = hash_func;
30 map->key_compare_func = compare_func;
31}
32
33void hashmap_deinit(hashmap_t *map)
34{
35 MOS_ASSERT_X(map && map->magic == HASHMAP_MAGIC, "hashmap_put: hashmap %p is not initialized", (void *) map);
36 for (size_t i = 0; i < map->capacity; i++)
37 {
38 hashmap_entry_t *entry = map->entries[i];
39 while (entry != NULL)
40 {
41 hashmap_entry_t *next = entry->next;
42 kfree(entry);
43 entry = next;
44 }
45 }
46 kfree(map->entries);
47}
48
49void *hashmap_put(hashmap_t *map, const void *key, void *value)
50{
51 MOS_ASSERT_X(map && map->magic == HASHMAP_MAGIC, "hashmap_put: hashmap %p is not initialized", (void *) map);
52 size_t index = map->hash_func(key).hash % map->capacity;
53 hashmap_entry_t *entry = map->entries[index];
54 while (entry != NULL)
55 {
56 if (map->key_compare_func(entry->key, key))
57 {
58 // key already exists, replace value
59 void *old_value = entry->value;
60 entry->value = value;
61 return old_value;
62 }
63 entry = entry->next;
64 }
65 entry = kmalloc(sizeof(hashmap_entry_t));
66 entry->key = key;
67 entry->value = value;
68 entry->next = map->entries[index];
69 map->entries[index] = entry;
70 map->size++;
71 return NULL;
72}
73
74void *hashmap_get(const hashmap_t *map, const void *key)
75{
76 MOS_ASSERT_X(map && map->magic == HASHMAP_MAGIC, "hashmap_put: hashmap %p is not initialized", (void *) map);
77 size_t index = map->hash_func(key).hash % map->capacity;
78 hashmap_entry_t *entry = map->entries[index];
79 while (entry != NULL)
80 {
81 if (map->key_compare_func(entry->key, key))
82 {
83 return entry->value;
84 }
85 entry = entry->next;
86 }
87 return NULL;
88}
89
90void *hashmap_remove(hashmap_t *map, const void *key)
91{
92 MOS_ASSERT_X(map && map->magic == HASHMAP_MAGIC, "hashmap_put: hashmap %p is not initialized", (void *) map);
93 size_t index = map->hash_func(key).hash % map->capacity;
94 hashmap_entry_t *entry = map->entries[index];
95 hashmap_entry_t *prev = NULL;
96 while (entry != NULL)
97 {
98 if (map->key_compare_func(entry->key, key))
99 {
100 if (prev == NULL)
101 {
102 map->entries[index] = entry->next;
103 }
104 else
105 {
106 prev->next = entry->next;
107 }
108 void *value = entry->value;
109 kfree(entry);
110 map->size--;
111 return value;
112 }
113 prev = entry;
114 entry = entry->next;
115 }
116 return NULL;
117}
118
119void hashmap_foreach(hashmap_t *map, hashmap_foreach_func_t func)
120{
121 MOS_ASSERT_X(map && map->magic == HASHMAP_MAGIC, "hashmap_put: hashmap %p is not initialized", (void *) map);
122 for (size_t i = 0; i < map->capacity; i++)
123 {
124 hashmap_entry_t *entry = map->entries[i];
125 while (entry != NULL)
126 {
127 if (!func(entry->key, entry->value))
128 return;
129 entry = entry->next;
130 }
131 }
132}
