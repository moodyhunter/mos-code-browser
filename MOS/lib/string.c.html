8size_t strlen(const char *str)
9{
10 size_t len = 0;
11 while (str[len])
12 len++;
13 return len;
14}

16s32 strcmp(const char *s1, const char *s2)
17{
18 size_t i = 0;
19 while (s1[i] && s2[i] && s1[i] == s2[i])
20 i++;
21 return s1[i] - s2[i];
22}

24s32 strncmp(const char *str1, const char *str2, size_t n)
25{
26 u8 c1, c2;
27 while (n-- > 0)
28 {
29 c1 = (u8) *str1++;
30 c2 = (u8) *str2++;
31 if (c1 != c2)
32 return c1 - c2;
33 if (c1 == '\0')
34 return 0;
35 }
36 return 0;
37}

39void *memcpy(void *restrict dest, const void *restrict src, size_t n)
40{
41 typedef s64 largeint_t;
42 largeint_t *ldest = (largeint_t *) dest;
43 largeint_t *lsrc = (largeint_t *) src;
44
45 while (n >= sizeof(largeint_t))
46 {
47 *ldest++ = *lsrc++;
48 n -= sizeof(largeint_t);
49 }
50
51 char *cdest = (char *) ldest;
52 char *csrc = (char *) lsrc;
53
54 while (n > 0)
55 {
56 *cdest++ = *csrc++;
57 n--;
58 }
59
60 return dest;
61}

63void *memmove(void *dest, const void *source, size_t length)
64{
65 // https://github.com/eblot/newlib/blob/master/newlib/libc/string/memmove.c
66 char *dst = dest;
67 const char *src = source;
68
69 if (src < dst && dst < src + length)
70 {
71 /* Have to copy backwards */
72 src += length;
73 dst += length;
74 while (length--)
75 *--dst = *--src;
76 }
77 else
78 {
79 while (length--)
80 *dst++ = *src++;
81 }
82
83 return dest;
84}

94void memzero(void *s, size_t n)
95{
96 typedef u64 largeint_t;
97 largeint_t *lsrc = (largeint_t *) s;
98
99 while (n >= sizeof(largeint_t))
100 {
101 *lsrc++ = 0;
102 n -= sizeof(largeint_t);
103 }
104
105 char *csrc = (char *) lsrc;
106
107 while (n > 0)

109 *csrc++ = 0;

131{
132 while (n > 0 && *src)

139 *dest++ = 0;

144const char *duplicate_string_n(const char *src, size_t len)
145{

157s64 strntoll(const char *str, char **endptr, int base, size_t n)
158{
159 s64 result = 0;
160 bool negative = false;
161 size_t i = 0;
162
163 if (*str == '-')
164 negative = true, str++, i++;
165 else if (*str == '+')
166 str++, i++;
167
168 while (i < n && *str)
169 {
170 char c = *str;
171 if (c >= '0' && c <= '9')
172 result *= base, result += c - '0';
173 else if (c >= 'a' && c <= 'z')
174 result *= base, result += c - 'a' + 10;
175 else if (c >= 'A' && c <= 'Z')
176 result *= base, result += c - 'A' + 10;
177 else
178 break;
179 str++;
180 i++;
181 }
182 if (endptr)
183 *endptr = (char *) str;
184 return negative ? -result : result;
185}
