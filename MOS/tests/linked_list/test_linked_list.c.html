101MOS_TEST_CASE(test_list_node_insert)
102{
103 list_node_t list_head = LIST_HEAD_INIT(list_head);
104 test_structure s1 = { 1, LIST_NODE_INIT(s1), 2 };
105 test_structure s2 = { 3, LIST_NODE_INIT(s2), 4 };
106 test_structure s3 = { 5, LIST_NODE_INIT(s3), 6 };
107 test_structure s4 = { 7, LIST_NODE_INIT(s4), 8 };
108 test_structure s5 = { 9, LIST_NODE_INIT(s5), 10 };
109
110 test_structure new_s = { 11, LIST_NODE_INIT(new_s), 12 };
111 test_structure new_s2 = { 13, LIST_NODE_INIT(new_s2), 14 };
112
113 // Connect the list.
114 list_node_append(&list_head, &s1.list_node); // list_head -> s1
115 list_node_append(&list_head, &s2.list_node); // list_head -> s1 -> s2
116 list_node_append(&list_head, &s3.list_node); // list_head -> s1 -> s2 -> s3
117 list_node_append(&list_head, &s4.list_node); // list_head -> s1 -> s2 -> s3 -> s4
118 list_node_append(&list_head, &s5.list_node); // list_head -> s1 -> s2 -> s3 -> s4 -> s5
119
120 // Insert a new node before s3.
121 list_node_insert_before(&s3.list_node, &new_s.list_node); // s1 -> s2 -> new_s -> s3 -> s4 -> s5 -> list_head
122 MOS_TEST_CHECK(new_s.list_node.next, &s3.list_node);
123 MOS_TEST_CHECK(new_s.list_node.prev, &s2.list_node);
124 MOS_TEST_CHECK(s2.list_node.next, &new_s.list_node);
125 MOS_TEST_CHECK(s3.list_node.prev, &new_s.list_node);
126
127 // original parts of the list should be unchanged.
128 MOS_TEST_CHECK(s2.list_node.prev, &s1.list_node);
129 MOS_TEST_CHECK(s3.list_node.next, &s4.list_node);
130
131 // Insert a new node after s4
132 list_node_insert_after(&s4.list_node, &new_s2.list_node); // s1 -> s2 -> new_s -> s3 -> s4 -> new_s2 -> s5 -> list_head
133 MOS_TEST_CHECK(new_s2.list_node.next, &s5.list_node);
134 MOS_TEST_CHECK(new_s2.list_node.prev, &s4.list_node);
135 MOS_TEST_CHECK(s4.list_node.next, &new_s2.list_node);
136 MOS_TEST_CHECK(s5.list_node.prev, &new_s2.list_node);
137
138 // original parts of the list should be unchanged.
139 MOS_TEST_CHECK(s4.list_node.prev, &s3.list_node);
140 MOS_TEST_CHECK(s5.list_node.next, &list_head);
141}

146 test_structure s1 = { 1, LIST_NODE_INIT(s1), 2 };
147 test_structure s2 = { 3, LIST_NODE_INIT(s2), 4 };
148 test_structure s3 = { 5, LIST_NODE_INIT(s3), 6 };
149 test_structure s4 = { 7, LIST_NODE_INIT(s4), 8 };
150 test_structure s5 = { 9, LIST_NODE_INIT(s5), 10 };
151 list_node_append(&list_head, &s1.list_node); // list_head -> s1
152 list_node_append(&list_head, &s2.list_node); // list_head -> s1 -> s2
153 list_node_append(&list_head, &s3.list_node); // list_head -> s1 -> s2 -> s3
154 list_node_append(&list_head, &s4.list_node); // list_head -> s1 -> s2 -> s3 -> s4
155 list_node_append(&list_head, &s5.list_node); // list_head -> s1 -> s2 -> s3 -> s4 -> s5
156
157 // Remove s3.
158 list_node_remove(&s3.list_node); // list_head -> s1 -> s2 -> s4 -> s5
159 MOS_TEST_CHECK(s1.list_node.next, &s2.list_node);
160 MOS_TEST_CHECK(s2.list_node.next, &s4.list_node);
161 MOS_TEST_CHECK(s4.list_node.next, &s5.list_node);
162 MOS_TEST_CHECK(s5.list_node.next, &list_head);
163 MOS_TEST_CHECK(list_head.prev, &s5.list_node);
164
165 MOS_TEST_CHECK(s1.list_node.prev, &list_head);
166 MOS_TEST_CHECK(s2.list_node.prev, &s1.list_node);
167 MOS_TEST_CHECK(s4.list_node.prev, &s2.list_node);
168 MOS_TEST_CHECK(s5.list_node.prev, &s4.list_node);
169 MOS_TEST_CHECK(list_head.next, &s1.list_node);
170}

185MOS_TEST_CASE(test_list_foreach)
186{
187 list_node_t list_head = LIST_HEAD_INIT(list_head);
188 test_structure s1 = { 1, LIST_NODE_INIT(s1), 2 };
189 test_structure s2 = { 3, LIST_NODE_INIT(s2), 4 };
190 test_structure s3 = { 5, LIST_NODE_INIT(s3), 6 };
191 test_structure s4 = { 7, LIST_NODE_INIT(s4), 8 };
192 test_structure s5 = { 9, LIST_NODE_INIT(s5), 10 };
193 list_node_append(&list_head, &s1.list_node); // list_head -> s1
194 list_node_append(&list_head, &s2.list_node); // list_head -> s1 -> s2
195 list_node_append(&list_head, &s3.list_node); // list_head -> s1 -> s2 -> s3
196 list_node_append(&list_head, &s4.list_node); // list_head -> s1 -> s2 -> s3 -> s4
197 list_node_append(&list_head, &s5.list_node); // list_head -> s1 -> s2 -> s3 -> s4 -> s5
198 int i = 0;
199 list_node_t *node = list_head.next;
200 while (node != &list_head)
201 {
202 i++;
203 node = node->next;
204 }
205 MOS_TEST_CHECK(i, 5);
206
207 // sum the list.
208 int sum_before = 0;
209 int sum_after = 0;
210 list_foreach(test_structure, node, list_head)
211 {
212 sum_before += node->value_before;
213 sum_after += node->value_after;
214 }
215 MOS_TEST_CHECK(sum_before, 25);
216 MOS_TEST_CHECK(sum_after, 30);
217}
