17void for_each_console_print_with_color(standard_color_t fg, standard_color_t bg, const char *message, size_t length)
18{
19 list_foreach(console_t, console, consoles)
20 {
21 standard_color_t prev_fg, prev_bg;
22 if (console->caps & CONSOLE_CAP_COLOR)
23 {
24 console->get_color(console, &prev_fg, &prev_bg);
25 console->set_color(console, fg, bg);
26 }
27
28 console_write(console, message, length);
29
30 if (console->caps & CONSOLE_CAP_COLOR)
31 {
32 console->set_color(console, prev_fg, prev_bg);
33 }
34 }
35}

37void mos_test_engine_log(standard_color_t color, char symbol, char *format, ...)
38{
39 char prefix[5];
40
41 if (symbol)
42 snprintf(prefix, 5, "[%c] ", symbol);
43 else
44 snprintf(prefix, 5, " ");
45
46 for_each_console_print_with_color(Gray, Black, prefix, 4);
47
48 char message[512];
49 va_list args;
50 va_start(args, format);
51 vsprintf(message, format, args);
52 va_end(args);
53
54 for_each_console_print_with_color(color, Black, message, strlen(message));
55}

57static void test_engine_warning_handler(const char *func, u32 line, const char *fmt, va_list args)
58{
59 char message[PRINTK_BUFFER_SIZE] = { 0 };
60 if (test_engine_n_warning_expected > 0)
61 {
62 // MOS_TEST_LOG(MOS_TEST_BLUE, '\0', "expected warning: %s", msg);
63 }
64 else
65 {
66 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
67 lprintk(MOS_LOG_WARN, "warning: %s", message);
68 lprintk(MOS_LOG_WARN, " in function: %s (line %u)\n", func, line);
69 mos_panic("unexpected warning");
70 }
71
72 test_engine_n_warning_expected--;
73}

75void mos_test_engine_run_tests()
76{
77 kwarn_handler_set(test_engine_warning_handler);
78
79 mos_test_result_t result = { 0 };
80
81 cmdline_arg_t *skip_tests_option = mos_cmdline_get_arg("mos_tests_skip");
82 cmdline_arg_t *mos_tests_skip_prefix_option = mos_cmdline_get_arg("mos_tests_skip_prefix");
83
84 MOS_TEST_FOREACH_TEST_CASE(test_case)
85 {
86 bool should_skip = false;
87 for (u32 i = 0; skip_tests_option && i < skip_tests_option->params_count; i++)
88 {
89 cmdline_param_t *parameter = skip_tests_option->params[i];
90 if (strcmp(parameter->val.string, test_case->test_name) == 0)
91 {
92 MOS_TEST_LOG(MOS_TEST_YELLOW, 'S', "Test %s skipped by kernel cmdline", test_case->test_name);
93 should_skip = true;
94 break;
95 }
96 }
97
98 for (u32 i = 0; mos_tests_skip_prefix_option && i < mos_tests_skip_prefix_option->params_count; i++)
99 {
100 cmdline_param_t *parameter = mos_tests_skip_prefix_option->params[i];
101 if (strncmp(parameter->val.string, test_case->test_name, strlen(parameter->val.string)) == 0)
102 {
103 MOS_TEST_LOG(MOS_TEST_YELLOW, 'S', "Test %s skipped by kernel cmdline (prefix %s)", test_case->test_name, parameter->val.string);
104 should_skip = true;
105 break;
106 }
107 }
108
109 if (should_skip)
110 continue;
111
112 mos_test_result_t r = { 0 };
113 test_case->test_func(&r);
114
115 result.n_total += r.n_total;
116 result.n_failed += r.n_failed;
117 result.n_skipped += r.n_skipped;
118
119 if (result.n_failed > 0)
120 mos_panic("TEST FAILED.");
121 }
122
123 kwarn_handler_remove();
124
125 u32 passed = result.n_total - result.n_failed - result.n_skipped;
126 pr_emph("ALL %u TESTS PASSED: (%u succeed, %u failed, %u skipped)", result.n_total, passed, result.n_failed, result.n_skipped);
127
128 if (mos_cmdline_get_arg("mos_tests_halt_on_success"))
129 platform_halt_cpu();
130 else
131 platform_shutdown();
132}
