19void for_each_console_print_with_color(standard_color_t fg, standard_color_t bg, const char *message, size_t length)
20{
21 list_foreach(console_t, console, consoles)
22 {
23 standard_color_t prev_fg, prev_bg;
24 if (console->caps & CONSOLE_CAP_COLOR)
25 {
26 console->get_color(console, &prev_fg, &prev_bg);
27 console->set_color(console, fg, bg);
28 }
29
30 console_write(console, message, length);
31
32 if (console->caps & CONSOLE_CAP_COLOR)
33 {
34 console->set_color(console, prev_fg, prev_bg);
35 }
36 }
37}

39void mos_test_engine_log(standard_color_t color, char symbol, char *format, ...)
40{
41 char prefix[5];
42
43 if (symbol)
44 snprintf(prefix, 5, "[%c] ", symbol);
45 else
46 snprintf(prefix, 5, " ");
47
48 for_each_console_print_with_color(Gray, Black, prefix, 4);
49
50 char message[512];
51 va_list args;
52 va_start(args, format);
53 vsprintf(message, format, args);
54 va_end(args);
55
56 for_each_console_print_with_color(color, Black, message, strlen(message));
57}

59static void test_engine_warning_handler(const char *func, u32 line, const char *fmt, va_list args)
60{
61 char message[PRINTK_BUFFER_SIZE] = { 0 };
62 if (test_engine_n_warning_expected > 0)
63 {
64 // MOS_TEST_LOG(MOS_TEST_BLUE, '\0', "expected warning: %s", msg);
65 }
66 else
67 {
68 vsnprintf(message, PRINTK_BUFFER_SIZE, fmt, args);
69 lprintk(MOS_LOG_WARN, "warning: %s", message);
70 lprintk(MOS_LOG_WARN, " in function: %s (line %u)\n", func, line);
71 mos_panic("unexpected warning");
72 }
73
74 test_engine_n_warning_expected--;
75}

80static bool mos_test_engine_setup_skip_prefix_list(int argc, const char **argv)
81{
82 test_engine_skip_prefix_list = kmalloc(sizeof(char *) * argc);
83 for (int i = 0; i < argc; i++)
84 test_engine_skip_prefix_list[i] = strdup(argv[i]);
85 return true;
86}

88__setup(tests_skip, "mos_tests_skip_prefix", mos_test_engine_setup_skip_prefix_list);

90static bool mos_test_engine_setup_do_not_shutdown(int argc, const char **argv)
91{
92 MOS_UNUSED(argc);
93 MOS_UNUSED(argv);
94 test_engine_do_not_shutdown = true;
95 return true;
96}

100static bool mos_test_engine_should_skip(const char *test_name)
101{
102 if (!test_engine_skip_prefix_list)
103 return false;
104
105 for (int i = 0; test_engine_skip_prefix_list[i]; i++)
106 {
107 if (strncmp(test_name, test_engine_skip_prefix_list[i], strlen(test_engine_skip_prefix_list[i])) == 0)
108 return true;
109 }
110
111 return false;
112}

114static bool mos_test_engine_run_tests(int argc, const char **argv)
115{
116 MOS_UNUSED(argc);
117 MOS_UNUSED(argv);
118 kwarn_handler_set(test_engine_warning_handler);
119
120 mos_test_result_t result = { 0 };
121
122 MOS_TEST_FOREACH_TEST_CASE(test_case)
123 {
124 bool should_skip = mos_test_engine_should_skip(test_case->test_name);
125
126 if (should_skip)
127 continue;
128
129 mos_test_result_t r = { 0 };
130 test_case->test_func(&r);
131
132 result.n_total += r.n_total;
133 result.n_failed += r.n_failed;
134 result.n_skipped += r.n_skipped;
135
136 if (result.n_failed > 0)
137 mos_panic("TEST FAILED.");
138 }
139
140 kwarn_handler_remove();
141
142 u32 passed = result.n_total - result.n_failed - result.n_skipped;
143 pr_emph("ALL %u TESTS PASSED: (%u succeed, %u failed, %u skipped)", result.n_total, passed, result.n_failed, result.n_skipped);
144
145 if (test_engine_do_not_shutdown)
146 platform_halt_cpu();
147 else
148 platform_shutdown();
149
150 MOS_UNREACHABLE();
151}
152
153__setup(tests_run, "mos_tests", mos_test_engine_run_tests);
