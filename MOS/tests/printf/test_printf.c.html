13int tst_printf(char *buffer, const char *format, ...) __attribute__((format(printf, 2, 3)));
14
15#define PRINTF_TEST(expected, format, ...) \
16 do \
17 { \
18 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
19 { \
20 MOS_TEST_SKIP(); \
21 break; \
22 } \
23 tst_printf(buffer, format __VA_OPT__(, ) __VA_ARGS__); \
24 MOS_TEST_CHECK_STRING(buffer, expected); \
25 } while (0)
26
27int tst_printf(char *buffer, const char *format, ...)
28{
29 va_list args;
30 va_start(args, format);
31 int ret = vsnprintf(buffer, 0, format, args);
32 va_end(args);
33 return ret;
34}
