6MOS_TEST_CASE(simple_cmdline)
7{
8 const char *single = "single";
9 cmdline_t *cmd = mos_cmdline_create(single);
10 MOS_TEST_CHECK(cmd != NULL, true);
11 MOS_TEST_CHECK(cmd->args_count, 1);
12 MOS_TEST_CHECK_STRING(cmd->arguments[0]->arg_name, "single");
13 MOS_TEST_CHECK(cmd->arguments[0]->params_count, 0);
14 mos_cmdline_destroy(cmd);
15}

72MOS_TEST_CASE(one_arg_with_multiple_options)
73{
74 const char *one_arg_with_multiple_options = "one=nana1,nana2,nana3,nana4,false";
75 cmdline_t *cmd = mos_cmdline_create(one_arg_with_multiple_options);

83 MOS_TEST_CHECK_STRING(cmd->arguments[0]->params[1]->val.string, "nana2");
84 MOS_TEST_CHECK(cmd->arguments[0]->params[2]->param_type, CMDLINE_PARAM_TYPE_STRING);
85 MOS_TEST_CHECK_STRING(cmd->arguments[0]->params[2]->val.string, "nana3");
86 MOS_TEST_CHECK(cmd->arguments[0]->params[3]->param_type, CMDLINE_PARAM_TYPE_STRING);
