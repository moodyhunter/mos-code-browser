10{

15 MOS_TEST_CHECK(map.size, 0);

20}

22MOS_TEST_CASE(hashmap_put_single)
23{

27
28 MOS_TEST_CHECK(map.capacity, 135);

33 MOS_TEST_CHECK(map.size, 1);
34

134MOS_TEST_CASE(hashmap_remove_function)
135{
136 void *old;
137 hashmap_t map = { 0 };
138 hashmap_common_type_init(&map, 10, string);
139 MOS_TEST_CHECK(map.magic, HASHMAP_MAGIC);
140 MOS_TEST_CHECK(map.capacity, 10);
141 MOS_TEST_CHECK(map.size, 0);
142
143 old = hashmap_put(&map, "foo", "foo1");
144 MOS_TEST_CHECK(map.capacity, 10);
145 MOS_TEST_CHECK(map.size, 1);
146 MOS_TEST_CHECK(old, NULL);
147 MOS_TEST_CHECK_STRING(hashmap_get(&map, "foo"), "foo1");
148
149 old = hashmap_remove(&map, "foo");
150 MOS_TEST_CHECK(map.capacity, 10);
151 MOS_TEST_CHECK(map.size, 0);
152 MOS_TEST_CHECK_STRING(old, "foo1");
153 const char *nothing = hashmap_get(&map, "foo");
154 MOS_TEST_CHECK(nothing, NULL);
155
156 old = hashmap_remove(&map, "foo");
157 MOS_TEST_CHECK(old, NULL);
158 MOS_TEST_CHECK(map.capacity, 10);
159 MOS_TEST_CHECK(map.size, 0);
160
161 MOS_TEST_CHECK(hashmap_get(&map, "foo"), NULL);
162 hashmap_deinit(&map);
163}

165static size_t test_hashmap_foreach_count = 0;
166
167bool test_foreach_function(const void *key, void *value)
168{
169 MOS_UNUSED(key);
170 MOS_UNUSED(value);
171 test_hashmap_foreach_count++;
172 return true;
173}
174
175bool test_foreach_stop_at_quux(const void *key, void *value)
176{
177 MOS_UNUSED(value);
178 test_hashmap_foreach_count++;
179 if (strcmp(key, "quux") == 0)
180 return false;
181 return true;
182}
183
184MOS_TEST_CASE(hashmap_foreach_function)
185{
186 hashmap_t map = { 0 };
187 hashmap_common_type_init(&map, 10, string);
188 MOS_TEST_CHECK(map.magic, HASHMAP_MAGIC);
189 MOS_TEST_CHECK(map.capacity, 10);
190 MOS_TEST_CHECK(map.size, 0);
191 hashmap_put(&map, "foo", "foo1");
192 hashmap_put(&map, "bar", "bar1");
193 hashmap_put(&map, "baz", "baz1");
194 hashmap_put(&map, "qux", "qux1");
195 hashmap_put(&map, "quux", "quux1");
196 hashmap_put(&map, "corge", "corge1");
197 hashmap_put(&map, "grault", "grault1");
198 hashmap_put(&map, "garply", "garply1");
199 hashmap_put(&map, "waldo", "waldo1");
200 hashmap_put(&map, "fred", "fred1");
201 hashmap_put(&map, "plugh", "plugh1");
202 hashmap_put(&map, "xyzzy", "xyzzy1");
203
204 test_hashmap_foreach_count = 0;
205 hashmap_foreach(&map, test_foreach_function);
206 MOS_TEST_CHECK(test_hashmap_foreach_count, map.size);
207
208 test_hashmap_foreach_count = 0;
209 hashmap_foreach(&map, test_foreach_stop_at_quux);
210 MOS_TEST_CHECK(test_hashmap_foreach_count, 4);
211 hashmap_deinit(&map);
212}
