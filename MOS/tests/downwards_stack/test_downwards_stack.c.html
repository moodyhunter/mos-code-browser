3#include "lib/structures/stack.h"
4#include "mos/mm/kmalloc.h"
5#include "test_engine.h"

7MOS_TEST_CASE(stack_init_deinit)
8{
9 downwards_stack_t stack;
10 stack_init(&stack, (void *) 12345, 6789);
11 MOS_TEST_CHECK(stack.base, 12345 + 6789);
12 MOS_TEST_CHECK(stack.head, 12345 + 6789);
13 MOS_TEST_CHECK(stack.capacity, 6789);
14 stack_deinit(&stack);
15
16 MOS_TEST_CHECK(stack.base, (uintptr_t) NULL);
17 MOS_TEST_CHECK(stack.head, (uintptr_t) NULL);
18 MOS_TEST_CHECK(stack.capacity, 0);
19}

21MOS_TEST_CASE(stack_push_pop_stack)
22{
23 downwards_stack_t stack = { 0 };
24 void *addr = kmalloc(4096);
25 stack_init(&stack, addr, 4096);
26
27 uintptr_t stack_bottom = (uintptr_t) addr + 4096;
28 MOS_TEST_CHECK(stack.capacity, 4096);
29 MOS_TEST_CHECK(stack.head, stack_bottom);
30 MOS_TEST_CHECK(stack.base, stack_bottom);
31
32 int pushed_1[10] = { 12345, 54321, 67890, 98765, 43210, 56789, 1234, 54321, 67890, 98765 };
33
34 stack_push(&stack, pushed_1, sizeof(pushed_1));
35 MOS_TEST_CHECK(stack.base, stack_bottom);
36 MOS_TEST_CHECK(stack.head, stack_bottom - sizeof(pushed_1));
37 MOS_TEST_CHECK(stack.capacity, 4096);
38
39 int pushed_2[10] = { 4444, 5555, 6666, 7777, 8888, 9999, 10101, 11011, 12012, 13013 };
40 stack_push(&stack, pushed_2, sizeof(pushed_2));
41 MOS_TEST_CHECK(stack.base, stack_bottom);
42 MOS_TEST_CHECK(stack.head, stack_bottom - sizeof(pushed_1) - sizeof(pushed_2));
43 MOS_TEST_CHECK(stack.capacity, 4096);
44
45 int tmp[10] = { 0 };
46 stack_pop(&stack, sizeof(tmp), tmp);
47 MOS_TEST_CHECK(stack.base, stack_bottom);
48 MOS_TEST_CHECK(stack.head, stack_bottom - sizeof(pushed_1));
49 MOS_TEST_CHECK(stack.capacity, 4096);
50 MOS_TEST_CHECK(tmp[0], pushed_2[0]);
51 MOS_TEST_CHECK(tmp[1], pushed_2[1]);
52 MOS_TEST_CHECK(tmp[2], pushed_2[2]);
53 MOS_TEST_CHECK(tmp[3], pushed_2[3]);
54 MOS_TEST_CHECK(tmp[4], pushed_2[4]);
55 MOS_TEST_CHECK(tmp[5], pushed_2[5]);
56 MOS_TEST_CHECK(tmp[6], pushed_2[6]);
57 MOS_TEST_CHECK(tmp[7], pushed_2[7]);
58 MOS_TEST_CHECK(tmp[8], pushed_2[8]);
59 MOS_TEST_CHECK(tmp[9], pushed_2[9]);
60
61 stack_pop(&stack, sizeof(tmp), tmp);
62 MOS_TEST_CHECK(stack.base, stack_bottom);
63 MOS_TEST_CHECK(stack.head, stack_bottom);
64 MOS_TEST_CHECK(stack.capacity, 4096);
65 MOS_TEST_CHECK(tmp[0], pushed_1[0]);
66 MOS_TEST_CHECK(tmp[1], pushed_1[1]);
67 MOS_TEST_CHECK(tmp[2], pushed_1[2]);
68 MOS_TEST_CHECK(tmp[3], pushed_1[3]);
69 MOS_TEST_CHECK(tmp[4], pushed_1[4]);
70 MOS_TEST_CHECK(tmp[5], pushed_1[5]);
71 MOS_TEST_CHECK(tmp[6], pushed_1[6]);
72 MOS_TEST_CHECK(tmp[7], pushed_1[7]);
73 MOS_TEST_CHECK(tmp[8], pushed_1[8]);
74 MOS_TEST_CHECK(tmp[9], pushed_1[9]);
75
76 stack_deinit(&stack);
77 kfree(addr);
78}
