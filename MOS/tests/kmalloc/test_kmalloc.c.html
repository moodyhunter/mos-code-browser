7MOS_TEST_CASE(kmalloc_single)
8{
9 void *p = kmalloc(1024);
10 MOS_TEST_ASSERT(p != NULL, "kmalloc failed");
11 memset(p, 0, 1024);
12 kfree(p);
13}

59MOS_TEST_CASE(kmalloc_a_lot)
60{
61 void *pointers[100];
62 for (int t = 0; t < 20; t++)
63 {
64 for (int i = 0; i < 50; i++)
65 {
66 pointers[i] = kmalloc(71);
67 MOS_TEST_ASSERT(pointers[i] != NULL, "failed to allocate memory");
68 memset(pointers[i], 0, 71);
69 }
70
71 for (int i = 0; i < 50; i++)
72 {
73 kfree(pointers[i]);
74 }
75 }
76}
