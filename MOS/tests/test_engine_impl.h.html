38#define MOS_TEST_DEFINE_CONDITION(condition, message) \
39 const char *_mt_test_cond_##condition##_message = message; \
40 static bool condition

42#define MOS_TEST_CONDITIONAL(cond) \
43 for (MOS_TEST_CURRENT_TEST_SKIPPED = !(cond), (*_mt_loop_leave) = false, __extension__({ \
44 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
45 MOS_TEST_LOG(MOS_TEST_YELLOW, '\0', "Skipped '%s': condition '%s' not met.", _mt_test_cond_##cond##_message, #cond); \
46 }); \
47 !(*_mt_loop_leave); (*_mt_loop_leave) = true, MOS_TEST_CURRENT_TEST_SKIPPED = false)

49#define MOS_TEST_SKIP() \
50 do \
51 { \
52 ++_MT_result->n_total; \
53 ++_MT_result->n_skipped; \
54 } while (0)

56#define MOS_TEST_CASE(_TestName) \
57 static void _TestName(mos_test_result_t *, bool *, bool *); \
58 static void _MT_WRAP_TEST_NAME(_TestName)(mos_test_result_t * result) \
59 { \
60 MOS_TEST_LOG(MOS_TEST_BLUE, 'T', "Starting test " #_TestName " (line %d)", __LINE__); \
61 _MT_RUN_TEST_AND_PRINT_RESULT(result, _TestName); \
62 } \
63 _MT_REGISTER_TEST_CASE(_TestName, _MT_WRAP_TEST_NAME(_TestName)); \
64 static void _TestName(mos_test_result_t *_MT_result, __maybe_unused bool *_mt_test_skipped, __maybe_unused bool *_mt_loop_leave)

66#define MOS_TEST_DECL_PTEST(_PTestName, ptest_args_printf_format, ...) \

68 static void _PTestName(mos_test_result_t *_MT_result, __maybe_unused bool *_mt_test_skipped, __maybe_unused bool *_mt_loop_leave, __VA_ARGS__)

70#define MOS_TEST_PTEST_INSTANCE(_PTestName, ...) \

83#define MOS_TEST_EXPECT_WARNING_N(N, body, msg) \
84 do \
85 { \
86 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
87 { \
88 MOS_TEST_SKIP(); \
89 break; \
90 } \
91 test_engine_n_warning_expected = N; \
92 body; \
93 if (test_engine_n_warning_expected != 0) \
94 MOS_TEST_FAIL("%d more expected warning(s) not seen: %s, line %d", test_engine_n_warning_expected, msg, __LINE__); \
95 } while (0)

97#define MOS_TEST_EXPECT_WARNING(body, msg) MOS_TEST_EXPECT_WARNING_N(1, body, msg)

120#define MOS_TEST_ASSERT(condition, format, ...) \
121 do \

123 if (MOS_TEST_CURRENT_TEST_SKIPPED) \

133/*
134 * Check if actual value is equal expected value.
135 *
136 * Arguments:
137 * - expected - expected value
138 * - actual - actual value
139 */
140#define MOS_TEST_CHECK(actual, expected) \
141 do \
142 { \
143 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
144 { \
145 MOS_TEST_SKIP(); \
146 break; \
147 } \
148 ++_MT_result->n_total; \
149 if ((expected) != (actual)) \
150 MOS_TEST_FAIL("'%s' is %d, expected %d", #actual, (actual), (expected)); \
151 } while (false)
152
153/*
154 * Check if actual string is equal expected string.
155 *
156 * Arguments:
157 * - expected - expected string
158 * - actual - actual string
159 */
160#define MOS_TEST_CHECK_STRING(actual, expected) \
161 do \
162 { \
163 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
164 { \
165 MOS_TEST_SKIP(); \
166 break; \
167 } \
168 ++_MT_result->n_total; \
169 if (strcmp(expected, actual) != 0) \
170 MOS_TEST_FAIL("values are different (expected = '%s', actual = '%s'), at line %u", (expected), (actual), (__LINE__)); \
171 } while (false)
172
173/*
174 * Check if actual value is not differ from expected value by more then epsilon.
175 *
176 * Arguments:
177 * - expected - expected value
178 * - actual - actual value
179 * - epsilon - maximum acceptable difference
180 */
181#define MOS_TEST_CHECK_EPS(actual, expected, epsilon) \
182 do \
183 { \
184 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
185 { \
186 MOS_TEST_SKIP(); \
187 break; \
188 } \
189 ++_MT_result->n_total; \
190 if (_MT_FLOATABS((expected) - (actual)) > (epsilon)) \
191 MOS_TEST_FAIL("values differ by more then %f (expected = %f, actual = %f)", (epsilon), (expected), (actual)); \
192 } while (false)
193
194/*
195 * Check if every value in the actual array is equal to corresponding value
196 * in the expected array.
197 *
198 * Arguments:
199 * - expected - expected array pointer
200 * - actual - actual array pointer
201 * - elements - number of array elements
202 */
203#define MOS_TEST_CHECK_ARRAY(actual, expected, elements) \
204 do \
205 { \
206 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
207 { \
208 MOS_TEST_SKIP(); \
209 break; \
210 } \
211 ++_MT_result->n_total; \
212 for (unsigned i = 0; i < (unsigned) (elements); ++i) \
213 { \
214 if ((expected)[i] != (actual)[i]) \
215 { \
216 MOS_TEST_FAIL("memories differ at %u-th position (expected = %d, actual = %d)", i, (expected)[i], (actual)[i]); \
217 break; \
218 } \
219 } \
220 } while (false)
221
222/*
223 * Check if every value in the actual array is not differ from corresponding value
224 * in the expected array by more then epsilon.
225 *
226 * Arguments:
227 * - expected - expected array pointer
228 * - actual - actual array pointer
229 * - epsilon - maximum acceptable difference
230 * - elements - number of array elements
231 */
232#define MOS_TEST_CHECK_ARRAY_EPS(actual, expected, elements, epsilon) \
233 do \
234 { \
235 if (MOS_TEST_CURRENT_TEST_SKIPPED) \
236 { \
237 MOS_TEST_SKIP(); \
238 break; \
239 } \
240 ++_MT_result->n_total; \
241 for (unsigned i = 0; i < (unsigned) (elements); ++i) \
242 { \
243 if (_MT_FLOATABS((expected)[i] - (actual)[i]) > (epsilon)) \
244 { \
245 MOS_TEST_FAIL("memories differ at %u by more then %f (expected = %f, actual = %f)", i, (epsilon), (expected)[i], (actual)[i]); \
246 break; \
247 } \
248 } \
249 } while (false)

251#define _MT_RUN_TEST_AND_PRINT_RESULT(_ResultVar, _TestFunc) \
252 do \
253 { \
254 bool _mt_test_skipped = false; \
255 bool _mt_loop_leave = false; \
256 _TestFunc(_ResultVar, &_mt_test_skipped, &_mt_loop_leave); \
257 u32 total = _ResultVar->n_total; \
258 u32 failed = _ResultVar->n_failed; \
259 u32 skipped = _ResultVar->n_skipped; \
260 u32 passed = total - failed - skipped; \
261 if (failed == 0) \
262 if (skipped == 0) \
263 MOS_TEST_LOG(MOS_TEST_GREEN, '\0', "%s: All %u test(s) passed", #_TestFunc, total); \
264 else \
265 MOS_TEST_LOG(MOS_TEST_GREEN, '\0', "%s: All %u test(s) passed (%u skipped)", #_TestFunc, total, skipped); \
266 else \
267 MOS_TEST_LOG(MOS_TEST_RED, 'X', "%s: %u out of %u test(s) failed (%u passed, %u skipped)", #_TestFunc, failed, total, passed, skipped); \
268 } while (0)
