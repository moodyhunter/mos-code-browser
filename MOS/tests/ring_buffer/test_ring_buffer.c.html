8MOS_TEST_CASE(ringbuffer_creation_and_destruction)
9{
10 ring_buffer_t *rb = ring_buffer_create(10);
11
12 MOS_TEST_CHECK(rb != NULL, true);
13 MOS_TEST_CHECK(rb->capacity, 10);
14 MOS_TEST_CHECK(rb->size, 0);
15 MOS_TEST_CHECK(rb->head, 0);
16 MOS_TEST_CHECK(rb->next_pos, 0);
17
18 ring_buffer_destroy(rb);
19
20 rb = ring_buffer_create(0);
21 MOS_TEST_CHECK(rb, NULL);
22}

24MOS_TEST_CASE(ringbuffer_put_and_get)
25{
26 ring_buffer_t *rb = ring_buffer_create(5);
27 size_t written;
28
29 written = ring_buffer_push_back_byte(rb, 'a');
30 MOS_TEST_CHECK(rb->size, 1);
31 MOS_TEST_CHECK(rb->head, 0);
32 MOS_TEST_CHECK(rb->next_pos, 1);
33 MOS_TEST_CHECK(written, 1);
34

36 MOS_TEST_CHECK(rb->size, 2);
37 MOS_TEST_CHECK(rb->head, 0);
38 MOS_TEST_CHECK(rb->next_pos, 2);
39 MOS_TEST_CHECK(written, 1);
40

42 MOS_TEST_CHECK(rb->size, 3);
43 MOS_TEST_CHECK(rb->head, 0);
44 MOS_TEST_CHECK(rb->next_pos, 3);
45 MOS_TEST_CHECK(written, 1);
46
47 written = ring_buffer_push_back_byte(rb, 'd');
48 MOS_TEST_CHECK(rb->size, 4);
49 MOS_TEST_CHECK(rb->head, 0);
50 MOS_TEST_CHECK(rb->next_pos, 4);
51 MOS_TEST_CHECK(written, 1);
52

54 MOS_TEST_CHECK(rb->size, 5);

56 MOS_TEST_CHECK(rb->next_pos, 0);
57 MOS_TEST_CHECK(written, 1);
58

60 MOS_TEST_CHECK(rb->size, 5);
61 MOS_TEST_CHECK(rb->head, 0);
62 MOS_TEST_CHECK(rb->next_pos, 0);
63 MOS_TEST_CHECK(written, 0); // full
64

67 MOS_TEST_CHECK(rb->head, 0);
68 MOS_TEST_CHECK(rb->next_pos, 4);

75 MOS_TEST_CHECK(c, 'd');
76
77 c = ring_buffer_pop_back_byte(rb);
78 MOS_TEST_CHECK(rb->size, 2);
79 MOS_TEST_CHECK(rb->head, 0);
80 MOS_TEST_CHECK(rb->next_pos, 2);
81 MOS_TEST_CHECK(c, 'c');
82

99 MOS_TEST_CHECK(c, 0); // empty

102}

252MOS_TEST_CASE(ringbuffer_full_and_empty)
253{
254 ring_buffer_t *rb = ring_buffer_create(1);
255 size_t written;
256 bool full;
257 bool empty;
258
259 full = ring_buffer_is_full(rb);
260 MOS_TEST_CHECK(full, false);
261
262 empty = ring_buffer_is_empty(rb);
263 MOS_TEST_CHECK(empty, true);
264
265 written = ring_buffer_push_back_byte(rb, 'a');
266 MOS_TEST_CHECK(rb->size, 1);
267 MOS_TEST_CHECK(rb->head, 0);
268 MOS_TEST_CHECK(rb->next_pos, 0);
269 MOS_TEST_CHECK(written, 1);
270
271 full = ring_buffer_is_full(rb);
272 MOS_TEST_CHECK(full, true);
273
274 empty = ring_buffer_is_empty(rb);
275 MOS_TEST_CHECK(empty, false);
276
277 char c = ring_buffer_pop_back_byte(rb);
278 MOS_TEST_CHECK(rb->size, 0);

290MOS_TEST_CASE(ringbuffer_complicated_ops)
291{
292 ring_buffer_t *rb = ring_buffer_create(10);
293
294 bool full;
295 size_t written;
296 char c;
297
298 ring_buffer_push_back_byte(rb, '1');
299 ring_buffer_push_back_byte(rb, '2');
300 ring_buffer_push_back_byte(rb, '3');
301 ring_buffer_push_back_byte(rb, '4');
302 ring_buffer_push_back_byte(rb, '5');
303 MOS_TEST_CHECK(rb->size, 5);
304 MOS_TEST_CHECK(rb->head, 0);
305 MOS_TEST_CHECK(rb->next_pos, 5);
306
307 ring_buffer_push_back_byte(rb, '6');
308 ring_buffer_push_back_byte(rb, '7');
309 ring_buffer_push_back_byte(rb, '8');
310 ring_buffer_push_back_byte(rb, '9');
311 ring_buffer_push_back_byte(rb, '0');
312 MOS_TEST_CHECK(rb->size, 10);
313 MOS_TEST_CHECK(rb->head, 0);
314 MOS_TEST_CHECK(rb->next_pos, 0);
315
316 // |1|2|3|4|5|6|7|8|9|0|
317
318 full = ring_buffer_is_full(rb);
319 MOS_TEST_CHECK(full, true);
320
321 c = ring_buffer_pop_front_byte(rb);
322 MOS_TEST_CHECK(c, '1');
323
324 c = ring_buffer_pop_front_byte(rb);
325 MOS_TEST_CHECK(c, '2');
326
327 c = ring_buffer_pop_front_byte(rb);
328 MOS_TEST_CHECK(c, '3');
329
330 c = ring_buffer_pop_back_byte(rb);
331 MOS_TEST_CHECK(c, '0');
332
333 c = ring_buffer_pop_back_byte(rb);
334 MOS_TEST_CHECK(c, '9');
335
336 c = ring_buffer_pop_back_byte(rb);
337 MOS_TEST_CHECK(c, '8');
338
339 // | | | |4|5|6|7| | | |
340
341 written = ring_buffer_push_front_byte(rb, 'a'); // | | |a|4|5|6|7| | | |
342 MOS_TEST_CHECK(written, 1);
343
344 written = ring_buffer_push_front_byte(rb, 'b'); // | |b|a|4|5|6|7| | | |
345 MOS_TEST_CHECK(rb->head, 1);
346 MOS_TEST_CHECK(rb->next_pos, 7);
347 MOS_TEST_CHECK(written, 1);
348
349 written = ring_buffer_push_front_byte(rb, 'c'); // |c|b|a|4|5|6|7| | | |
350 MOS_TEST_CHECK(rb->head, 0);
351 MOS_TEST_CHECK(rb->next_pos, 7);
352 MOS_TEST_CHECK(written, 1);
353
354 written = ring_buffer_push_front_byte(rb, 'd'); // |c|b|a|4|5|6|7| | |d|
355 MOS_TEST_CHECK(rb->head, 9);
356 MOS_TEST_CHECK(rb->next_pos, 7);
357 MOS_TEST_CHECK(written, 1);
358
359 written = ring_buffer_push_front_byte(rb, 'e'); // |c|b|a|4|5|6|7| |e|d|
360 MOS_TEST_CHECK(rb->head, 8);
361 MOS_TEST_CHECK(rb->next_pos, 7);
362 MOS_TEST_CHECK(written, 1);
363
364 written = ring_buffer_push_front_byte(rb, 'f'); // |c|b|a|4|5|6|7|f|e|d|
365 MOS_TEST_CHECK(rb->head, 7);
366 MOS_TEST_CHECK(rb->next_pos, 7);
367 MOS_TEST_CHECK(written, 1);
368
369 written = ring_buffer_push_front_byte(rb, 'g'); // |c|b|a|4|5|6|7|f|e|d|
370 MOS_TEST_CHECK(rb->head, 7);
371 MOS_TEST_CHECK(rb->next_pos, 7);
372 MOS_TEST_CHECK(written, 0);
373
374 full = ring_buffer_is_full(rb);
375 MOS_TEST_CHECK(full, true);
376
377 c = ring_buffer_pop_back_byte(rb); // |c|b|a|4|5|6| |f|e|d|
