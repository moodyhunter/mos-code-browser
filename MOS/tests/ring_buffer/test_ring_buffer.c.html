8MOS_TEST_CASE(ringbuffer_creation_and_destruction)
9{
10 ring_buffer_t *rb = ring_buffer_create(10);
11
12 MOS_TEST_CHECK(rb != NULL, true);
13 MOS_TEST_CHECK(rb->capacity, 10);
14 MOS_TEST_CHECK(rb->size, 0);
15 MOS_TEST_CHECK(rb->head, 0);
16 MOS_TEST_CHECK(rb->tail, 0);
17
18 ring_buffer_destroy(rb);
19
20 rb = ring_buffer_create(0);
21 MOS_TEST_CHECK(rb, NULL);
22}

24MOS_TEST_CASE(ringbuffer_put_and_get)
25{
26 ring_buffer_t *rb = ring_buffer_create(5);
27 size_t written;
28
29 written = ring_buffer_push_back_byte(rb, 'a');
30 MOS_TEST_CHECK(rb->size, 1);
31 MOS_TEST_CHECK(rb->head, 1);
32 MOS_TEST_CHECK(rb->tail, 0);
33 MOS_TEST_CHECK(written, 1);
34
35 written = ring_buffer_push_back_byte(rb, 'b');
36 MOS_TEST_CHECK(rb->size, 2);
37 MOS_TEST_CHECK(rb->head, 2);
38 MOS_TEST_CHECK(rb->tail, 0);
39 MOS_TEST_CHECK(written, 1);
40
41 written = ring_buffer_push_back_byte(rb, 'c');
42 MOS_TEST_CHECK(rb->size, 3);
43 MOS_TEST_CHECK(rb->head, 3);
44 MOS_TEST_CHECK(rb->tail, 0);
45 MOS_TEST_CHECK(written, 1);
46
47 written = ring_buffer_push_back_byte(rb, 'd');
48 MOS_TEST_CHECK(rb->size, 4);
49 MOS_TEST_CHECK(rb->head, 4);
50 MOS_TEST_CHECK(rb->tail, 0);
51 MOS_TEST_CHECK(written, 1);
52
53 written = ring_buffer_push_back_byte(rb, 'e');
54 MOS_TEST_CHECK(rb->size, 5);
55 MOS_TEST_CHECK(rb->head, 0);
56 MOS_TEST_CHECK(rb->tail, 0);
57 MOS_TEST_CHECK(written, 1);
58
59 written = ring_buffer_push_back_byte(rb, 'f');
60 MOS_TEST_CHECK(rb->size, 5);
61 MOS_TEST_CHECK(rb->head, 0);
62 MOS_TEST_CHECK(rb->tail, 0);
63 MOS_TEST_CHECK(written, 0); // full
64
65 char c = ring_buffer_pop_back_byte(rb);
66 MOS_TEST_CHECK(rb->size, 4);
67 MOS_TEST_CHECK(rb->head, 0);
68 MOS_TEST_CHECK(rb->tail, 1);
69 MOS_TEST_CHECK(c, 'a');
70
71 c = ring_buffer_pop_back_byte(rb);
72 MOS_TEST_CHECK(rb->size, 3);
73 MOS_TEST_CHECK(rb->head, 0);
74 MOS_TEST_CHECK(rb->tail, 2);
75 MOS_TEST_CHECK(c, 'b');
76
77 c = ring_buffer_pop_back_byte(rb);
78 MOS_TEST_CHECK(rb->size, 2);
79 MOS_TEST_CHECK(rb->head, 0);
80 MOS_TEST_CHECK(rb->tail, 3);
81 MOS_TEST_CHECK(c, 'c');
82
83 c = ring_buffer_pop_back_byte(rb);
84 MOS_TEST_CHECK(rb->size, 1);
85 MOS_TEST_CHECK(rb->head, 0);
86 MOS_TEST_CHECK(rb->tail, 4);
87 MOS_TEST_CHECK(c, 'd');
88
89 c = ring_buffer_pop_back_byte(rb);
90 MOS_TEST_CHECK(rb->size, 0);
91 MOS_TEST_CHECK(rb->head, 0);
92 MOS_TEST_CHECK(rb->tail, 0);
93 MOS_TEST_CHECK(c, 'e');
94
95 c = ring_buffer_pop_back_byte(rb);
96 MOS_TEST_CHECK(rb->size, 0);
97 MOS_TEST_CHECK(rb->head, 0);
98 MOS_TEST_CHECK(rb->tail, 0);
99 MOS_TEST_CHECK(c, 0); // empty
100}
