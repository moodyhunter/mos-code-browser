3#include "lib/containers.h"
4#include "lib/structures/tree.h"
5#include "mos/filesystem/filesystem.h"

10MOS_TEST_CASE(test_fs_api_path_is_prefix)
11{
12 fsnode_t root = {
13 .name = "/",
14 .tree_node = {
15 .parent = tree_node(&root),
16 .n_children = 0,
17 .children = NULL,
18 },
19 };
20
21 // /bin
22 fsnode_t bin = {
23 .name = "bin",
24 .tree_node = {
25 .parent = tree_node(&root),
26 .n_children = 0,
27 .children = NULL,
28 },
29 };
30
31 // /bin/tools
32 fsnode_t bin_tools = {
33 .name = "tools",
34 .tree_node = {
35 .parent = tree_node(&bin),
36 .n_children = 0,
37 .children = NULL,
38 },
39 };
40
41 // /temp
42 fsnode_t temp = {
43 .name = "temp",
44 .tree_node = {
45 .parent = tree_node(&root),
46 .n_children = 0,
47 .children = NULL,
48 },
49 };
50
51 bool result = false;
52
53 result = path_verify_prefix(&temp, &bin_tools);
54 MOS_TEST_CHECK(result, false);
55
56 result = path_verify_prefix(&bin_tools, &bin_tools);
57 MOS_TEST_CHECK(result, true);
58
59 result = path_verify_prefix(&bin_tools, &temp);
60 MOS_TEST_CHECK(result, false);
61
62 result = path_verify_prefix(&bin_tools, &root);
63 MOS_TEST_CHECK(result, true);
64
65 result = path_verify_prefix(&root, &root);
66 MOS_TEST_CHECK(result, true);
67
68 result = path_verify_prefix(&root, &bin);
69 MOS_TEST_CHECK(result, false);
70
71 result = path_verify_prefix(&bin, &root);
72 MOS_TEST_CHECK(result, true);
73
74 result = path_verify_prefix(&bin_tools, &bin);
75 MOS_TEST_CHECK(result, true);
76}
