34static void invoke_init(void)
35{
36 typedef void (*func_ptr)(void);
37 extern char __init_array_start, __init_array_end;
38 for (func_ptr *func = (void *) &__init_array_start; func != (void *) &__init_array_end; func++)
39 if (func)
40 (*func)();
41}
42
43static void __attribute__((constructor)) __liballoc_userspace_init(void)
44{
45 liballoc_init();
46}

48void _start(void)
49{
50 extern int main(void);
51 extern void invoke_init(void);
52 invoke_init();
53 int r = main();
54 syscall_exit(r);
55}

57static void thread_start(void *_arg)
58{
59 thread_entry_t entry = ((thread_start_args_t *) _arg)->entry;
60 void *entry_arg = ((thread_start_args_t *) _arg)->arg;
61 free(_arg);
62 entry(entry_arg);
63 syscall_thread_exit();
64}

66void dvprintf(int fd, const char *fmt, va_list ap)
67{
68 char buf[256];
69 vsnprintf(buf, sizeof(buf), fmt, ap);
70 syscall_io_write(fd, buf, strlen(buf), 0);
71}

73void dprintf(int fd, const char *fmt, ...)
74{

100void *liballoc_alloc_page(size_t npages)
101{
102 uintptr_t new_top = syscall_heap_control(HEAP_GROW_PAGES, npages);
103 if (new_top == 0)
104 return NULL;
105
106 return (void *) (new_top - npages * MOS_PAGE_SIZE);
107}

117tid_t start_thread(const char *name, thread_entry_t entry, void *arg)
118{
119 thread_start_args_t *thread_start_args = malloc(sizeof(thread_start_args_t));
120 thread_start_args->entry = entry;
121 thread_start_args->arg = arg;
122 return syscall_create_thread(name, thread_start, thread_start_args);
123}
