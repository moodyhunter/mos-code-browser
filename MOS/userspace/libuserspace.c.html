12typedef struct thread_start_args
13{
14 thread_entry_t entry;
15 void *arg;
16} thread_start_args_t;
17
18u64 __stack_chk_guard = 0;
19
20noreturn void __stack_chk_fail(void)
21{
22 syscall_exit(-1);
23 while (1)
24 ;
25}
26
27void __stack_chk_fail_local(void)
28{
29 __stack_chk_fail();
30}
31
32void _start(void)
33{
34 extern int main(void);
35 int r = main();
36 syscall_exit(r);
37}
38
39void _thread_start(void *arg)
40{
41 thread_start_args_t *args = (thread_start_args_t *) arg;
42 args->entry(args->arg);
43 syscall_thread_exit();
44}
45
46void dvprintf(int fd, const char *fmt, va_list ap)
47{
48 char buf[256];
49 vsnprintf(buf, sizeof(buf), fmt, ap);
50 syscall_io_write(fd, buf, strlen(buf), 0);
51}
52
53void dprintf(int fd, const char *fmt, ...)
54{
55 va_list ap;
56 va_start(ap, fmt);
57 dvprintf(fd, fmt, ap);
58 va_end(ap);
59}
60
61void printf(const char *fmt, ...)
62{
63 va_list ap;
64 va_start(ap, fmt);
65 dvprintf(stdout, fmt, ap);
66 va_end(ap);
67}

70static thread_start_args_t thread_start_args;

72void start_thread(const char *name, thread_entry_t entry, void *arg)
73{
74 thread_start_args = (thread_start_args_t){ .entry = entry, .arg = arg };
75 syscall_create_thread(name, _thread_start, &thread_start_args);
76}
