9typedef struct thread_start_args
10{
11 thread_entry_t entry;
12 void *arg;
13} thread_start_args_t;
14
15extern int main(void);
16u64 __stack_chk_guard = 0;
17
18noreturn void __stack_chk_fail(void)
19{
20 syscall_exit(-1);
21 while (1)
22 ;
23}
24
25void __stack_chk_fail_local(void)
26{
27 __stack_chk_fail();
28}
29
30void _start(void)
31{
32 int r = main();
33 syscall_exit(r);
34}
35
36void _thread_start(void *arg)
37{
38 thread_start_args_t *args = (thread_start_args_t *) arg;
39 args->entry(args->arg);
40 syscall_thread_exit();
41}
42
43void printf(const char *fmt, ...)
44{
45 char buf[4096];
46 va_list args;
47 va_start(args, fmt);
48 vsprintf(buf, fmt, args);
49 va_end(args);
50 syscall_io_write(stdout, buf, strlen(buf), 0);
51}

54static thread_start_args_t thread_start_args;

56void start_thread(const char *name, thread_entry_t entry, void *arg)
57{
58 thread_start_args = (thread_start_args_t){ .entry = entry, .arg = arg };
59 syscall_create_thread(name, _thread_start, &thread_start_args);
60}
