3#include "lib/stdio.h"
4#include "libuserspace.h"
5#include "mos/ksyscall/usermode.h"
6
7void printf(const char *fmt, ...)
8{
9 char buf[4096];
10 va_list args;
11 va_start(args, fmt);
12 vsprintf(buf, fmt, args);
13 va_end(args);
14 invoke_ksyscall_io_write(stdout, buf, strlen(buf), 0);
15}
16
17static char buf[4 KB] = { 0 };
18
19int main(void)
20{
21 volatile int mm = 11;
22 int fd = invoke_ksyscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
23 if (fd < 0)
24 printf("Failed to open /assets/msg.txt\n");
25 else
26 {
27 size_t read = invoke_ksyscall_io_read(fd, buf, 512, 0);
28 printf("\n");
29 invoke_ksyscall_io_write(stdout, buf, read, 0);
30 invoke_ksyscall_io_close(fd);
31 }
32
33 pid_t my_pid = invoke_ksyscall_get_pid();
34 printf("My PID: %d\n", my_pid);
35
36 pid_t ping_pid = invoke_ksyscall_spawn("/programs/kmsg-ping", 0, NULL);
37 pid_t pong_pid = invoke_ksyscall_spawn("/programs/kmsg-pong", 0, NULL);
38 printf("ping pid: %d\n", ping_pid);
39 printf("pong pid: %d\n", pong_pid);
40
41 my_pid = invoke_ksyscall_fork();
42
43 if (mm != 11)
44 invoke_ksyscall_panic();
45
46 if (my_pid == 0)
47 {
48 printf("Child process\n");
49 }
50 else
51 {
52 printf("Parent process\n");
53 }
54
55 pid_t parent = invoke_ksyscall_get_parent_pid();
56 printf("Parent PID: %d\n", parent);
57
58 while (1)
59 invoke_ksyscall_yield_cpu();
60
61 return 0;
62}
