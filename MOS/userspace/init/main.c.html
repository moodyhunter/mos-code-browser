3#include "lib/memory.h"
4#include "lib/string.h"
5#include "libuserspace.h"
6#include "mos/syscall/usermode.h"
7
8static char buf[4 KB] = { 0 };
9
10void thread_work(void *arg)
11{
12 int *value = (int *) arg;
13 pid_t process = syscall_get_pid();
14 printf("Thread started, value = %d, from process %d\n", *value, process);
15}
16
17static int value = 0;
18
19int main(void)
20{
21 printf("\n");
22 int fd = syscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
23 if (fd < 0)
24 printf("Failed to open /assets/msg.txt\n");
25 else
26 {
27 size_t read = syscall_io_read(fd, buf, 512, 0);
28 syscall_io_write(stdout, buf, read, 0);
29 syscall_io_close(fd);
30 }
31
32 char buf[256] = { 0 };
33
34 long read = syscall_io_read(stdin, buf, 256, 0);
35 if (read > 0)
36 {
37 printf("Read %d bytes from stdin", read);
38 syscall_io_write(stdout, buf, read, 0);
39 }
40
41 pid_t my_pid = syscall_get_pid();
42 printf("My PID: %d\n", my_pid);
43
44 value = 3456787;
45
46 pid_t ping_pid = syscall_spawn("/programs/kmsg-ping", 0, NULL);
47 pid_t pong_pid = syscall_spawn("/programs/kmsg-pong", 0, NULL);
48 printf("ping pid: %d\n", ping_pid);
49 printf("pong pid: %d\n", pong_pid);
50
51 start_thread("worker", thread_work, &value);
52 my_pid = syscall_fork();
53
54 if (my_pid == 0)
55 {
56 printf("Child process\n");
57 uintptr_t heap_top = syscall_heap_control(HEAP_GET_TOP, 0);
58 printf("Child top: " PTR_FMT "\n", heap_top);
59 uintptr_t new_top = syscall_heap_control(HEAP_SET_TOP, heap_top + 16 KB);
60 printf("Child new top: " PTR_FMT "\n", new_top);
61
62 char *ptr = (char *) new_top - 1;
63 *ptr = 0x42;
64 u32 pid_new_forked = syscall_fork();
65 if (pid_new_forked == 0)
66 {
67 printf("Child process of child process\n");
68 syscall_exit(0);
69 }
70 else
71 {
72 printf("Parent process of child process\n");
73 }
74 }
75 else
76 {
77 printf("Parent process\n");
78 }
79
80 pid_t parent = syscall_get_parent_pid();
81 printf("Parent PID: %d\n", parent);
82
83 // get heap address
84 uintptr_t heap = syscall_heap_control(HEAP_GET_TOP, 0);
85 printf("Heap base: " PTR_FMT "\n", heap);
86
87 // grow the heap by 16 KB
88 uintptr_t new_heap = syscall_heap_control(HEAP_GROW_PAGES, 4); // in pages
89
90 printf("New heap top: " PTR_FMT "\n", new_heap);
91
92 // write to the heap
93 char *ptr = (char *) new_heap - 1;
94 *ptr = 0x42;
95
96 char *data = malloc(50);
97 strcpy(data, "Hello world!");
98 printf("Data: %s\n", data);
99
100 while (1)
101 ;
102
103 return 0;
104}
