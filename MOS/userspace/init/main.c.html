9#define STRINGIFY(x) #x
10#define TOSTRING(x) STRINGIFY(x)
11
12#define print(msg) print_impl(TOSTRING(__LINE__) ": msg: " msg)
13
14void print_impl(const char *str)
15{
16 invoke_ksyscall_io_write(stdout, str, strlen(str), 0);
17 invoke_ksyscall_io_write(stdout, "\n", 1, 0);
18}
19
20void print_err(const char *str)
21{
22 invoke_ksyscall_io_write(stderr, str, strlen(str), 0);
23 invoke_ksyscall_io_write(stderr, "\n", 1, 0);
24}
25
26void print_int(int i)
27{
28 switch (i)
29 {
30 case 0: print("0"); break;
31 case 1: print("1"); break;
32 case 2: print("2"); break;
33 case 3: print("3"); break;
34 case 4: print("4"); break;
35 case 5: print("5"); break;
36 case 6: print("6"); break;
37 case 7: print("7"); break;
38 case 8: print("8"); break;
39 case 9: print("9"); break;
40 case 10: print("10"); break;
41 }
42}

44static char buf[2 KB] = { 0 };
45
46int main(void)
47{
48 volatile int mm = 11;
49 int fd = invoke_ksyscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
50 if (fd < 0)
51 print_err("Failed to open /assets/msg.txt");
52 else
53 {
54 size_t read = invoke_ksyscall_io_read(fd, buf, 512, 0);
55 print("\n");
56 invoke_ksyscall_io_write(stdout, buf, read, 0);
57 invoke_ksyscall_io_close(fd);
58 }
59
60 pid_t my_pid = invoke_ksyscall_get_pid();
61 print_int(my_pid);
62
63 pid_t ping_pid = invoke_ksyscall_spawn("/programs/kmsg-ping", 0, NULL);
64 pid_t pong_pid = invoke_ksyscall_spawn("/programs/kmsg-pong", 0, NULL);
65 print_int(ping_pid);
66 print_int(pong_pid);
67
68 my_pid = invoke_ksyscall_fork();
69
70 if (mm != 11)
71 invoke_ksyscall_panic();
72
73 if (my_pid == 0)
74 {
75 print("Child process");
76 }
77 else
78 {
79 print("Parent process");
80 }
81
82 pid_t parent = invoke_ksyscall_get_parent_pid();
83 print_int(parent);
84
85 while (1)
86 invoke_ksyscall_yield_cpu();
87
88 return 0;
89}
