3#include "libuserspace.h"
4#include "mos/syscall/usermode.h"

6static char buf[4 KB] = { 0 };
7
8void thread_work(void *arg)
9{
10 int *value = (int *) arg;
11 pid_t process = syscall_get_pid();
12 printf("Thread started, value = %d, from process %d\n", *value, process);
13}
14
15static int value = 0;
16
17int main(void)
18{
19 printf("\n");
20 int fd = syscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
21 if (fd < 0)
22 printf("Failed to open /assets/msg.txt\n");
23 else
24 {
25 size_t read = syscall_io_read(fd, buf, 512, 0);
26 syscall_io_write(stdout, buf, read, 0);
27 syscall_io_close(fd);
28 }
29
30 char buf[256] = { 0 };
31
32 long read = syscall_io_read(stdin, buf, 256, 0);
33 if (read > 0)
34 {
35 printf("Read %d bytes from stdin", read);
36 syscall_io_write(stdout, buf, read, 0);
37 }
38
39 pid_t my_pid = syscall_get_pid();
40 printf("My PID: %d\n", my_pid);
41
42 value = 3456787;
43
44 pid_t ping_pid = syscall_spawn("/programs/kmsg-ping", 0, NULL);
45 pid_t pong_pid = syscall_spawn("/programs/kmsg-pong", 0, NULL);
46 printf("ping pid: %d\n", ping_pid);
47 printf("pong pid: %d\n", pong_pid);
48
49 start_thread("worker", thread_work, &value);
50
51 my_pid = syscall_fork();
52
53 if (my_pid == 0)
54 {
55 printf("Child process\n");
56 uintptr_t heap_top = syscall_heap_control(HEAP_OP_GET_TOP, 0);
57 printf("Child top: " PTR_FMT "\n", heap_top);
58 uintptr_t new_top = syscall_heap_control(HEAP_OP_SET_TOP, heap_top + 16 KB);
59 printf("Child new top: " PTR_FMT "\n", new_top);
60
61 char *ptr = (char *) new_top - 1;
62 *ptr = 0x42;
63 u32 pid_new_forked = syscall_fork();
64 if (pid_new_forked == 0)
65 {
66 printf("Child process of child process\n");
67 syscall_exit(0);
68 }
69 else
70 {
71 printf("Parent process of child process\n");
72 }
73 }
74 else
75 {
76 printf("Parent process\n");
77 }
78
79 pid_t parent = syscall_get_parent_pid();
80 printf("Parent PID: %d\n", parent);
81
82 // get heap address
83 uintptr_t heap = syscall_heap_control(HEAP_OP_GET_TOP, 0);
84 printf("Heap base: " PTR_FMT "\n", heap);
85
86 // grow the heap by 16 KB
87 uintptr_t new_heap = syscall_heap_control(HEAP_OP_GROW, 16 KB);
88
89 printf("New heap top: " PTR_FMT "\n", new_heap);
90
91 // write to the heap
92 char *ptr = (char *) new_heap - 1;
93 *ptr = 0x42;
94
95 while (1)
96 ;
97
98 return 0;
99}
