12u64 __stack_chk_guard = 0;
13
14void noreturn __stack_chk_fail(void)
15{
16 invoke_ksyscall_panic();
17 while (1)
18 ;
19}
20
21void __stack_chk_fail_local(void)
22{
23 __stack_chk_fail();
24}
25
26int strlen(const char *str)
27{
28 int len = 0;
29 while (str[len])
30 len++;
31 return len;
32}
33
34void print(const char *str)
35{
36 invoke_ksyscall_io_write(stdout, str, strlen(str), 0);
37}
38
39void print_err(const char *str)
40{
41 invoke_ksyscall_io_write(stderr, str, strlen(str), 0);
42}
43
44char buf[2 KB] = { 0 };
45
46int main(void)
47{
48 int fd = invoke_ksyscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
49 if (fd < 0)
50 print_err("Failed to open /assets/msg.txt");
51 else
52 {
53 size_t read = invoke_ksyscall_io_read(fd, buf, 512, 0);
54 print("\n");
55 invoke_ksyscall_io_write(stdout, buf, read, 0);
56 invoke_ksyscall_io_close(fd);
57 }
58
59 while (1)
60 invoke_ksyscall_yield_cpu();
61
62 return 0;
63}
