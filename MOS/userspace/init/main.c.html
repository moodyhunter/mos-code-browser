19static void file_api()
20{
21 file_stat_t stat = { 0 };
22 if (syscall_file_stat("/assets/msg.txt", &stat))
23 {
24 printf("File size: %d bytes\n", stat.size);
25 printf("Owner: %d:%d\n", stat.uid, stat.gid);
26 printf("Permissions: %c%c%c%c%c%c%c%c%c",
27 stat.permissions.owner & FILE_PERM_READ ? 'r' : '-', //
28 stat.permissions.owner & FILE_PERM_WRITE ? 'w' : '-', //
29 stat.permissions.owner & FILE_PERM_EXEC ? 'x' : '-', //
30 stat.permissions.group & FILE_PERM_READ ? 'r' : '-', //
31 stat.permissions.group & FILE_PERM_WRITE ? 'w' : '-', //
32 stat.permissions.group & FILE_PERM_EXEC ? 'x' : '-', //
33 stat.permissions.other & FILE_PERM_READ ? 'r' : '-', //
34 stat.permissions.other & FILE_PERM_WRITE ? 'w' : '-', //
35 stat.permissions.other & FILE_PERM_EXEC ? 'x' : '-' //
36 );
37 if (stat.suid)
38 printf("[SUID]");
39 if (stat.sgid)
40 printf("[SGID]");
41 if (stat.sticky)
42 printf("[STICKY]");
43 printf("\n");
44
45 int fd = syscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
46 if (fd >= 0)
47 {
48 size_t read = syscall_io_read(fd, buf, 512, 0);
49 syscall_io_write(stdout, buf, read, 0);
50 syscall_io_close(fd);
51 }
52 else
53 {
54 printf("Failed to open /assets/msg.txt\n");
55 }
56 }
57 else
58 {
59 printf("Failed to stat /assets/msg.txt");
60 }
61}

93 printf("Child top: " PTR_FMT "\n", heap_top);
94 uintptr_t new_top = syscall_heap_control(HEAP_SET_TOP, heap_top + 16 KB);
95 printf("Child new top: " PTR_FMT "\n", new_top);

97 char *ptr = (char *) new_top - 1;
98 *ptr = 0x42;
99 u32 pid_new_forked = syscall_fork();

115 pid_t parent = syscall_get_parent_pid();

118 // get heap address
119 uintptr_t heap = syscall_heap_control(HEAP_GET_TOP, 0);
120 printf("Heap base: " PTR_FMT "\n", heap);

122 // grow the heap by 16 KB
123 uintptr_t new_heap = syscall_heap_control(HEAP_GROW_PAGES, 4); // in pages
124
125 printf("New heap top: " PTR_FMT "\n", new_heap);
126
127 // write to the heap
128 char *ptr = (char *) new_heap - 1;
129 *ptr = 0x42;
130
131 char *data = malloc(50);
132 strcpy(data, "Hello world!");
133 printf("Data: %s\n", data);
134
135 if (syscall_get_pid() != 1)
136 syscall_exit(0);

139 ;

142}
