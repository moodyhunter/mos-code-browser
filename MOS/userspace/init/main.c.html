3#include "libuserspace.h"
4#include "mos/syscall/usermode.h"
5
6static char buf[4 KB] = { 0 };
7
8void thread_work(void *arg)
9{
10 int *value = (int *) arg;
11 pid_t process = syscall_get_pid();
12 printf("Thread started, value = %d, from process %d\n", *value, process);
13}
14
15static int value = 0;
16
17int main(void)
18{
19 printf("\n");
20 int fd = syscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
21 if (fd < 0)
22 printf("Failed to open /assets/msg.txt\n");
23 else
24 {
25 size_t read = syscall_io_read(fd, buf, 512, 0);
26 syscall_io_write(stdout, buf, read, 0);
27 syscall_io_close(fd);
28 }
29
30 char buf[256] = { 0 };
31
32 long read = syscall_io_read(stdin, buf, 256, 0);
33 if (read > 0)
34 {
35 printf("Read %d bytes from stdin", read);
36 syscall_io_write(stdout, buf, read, 0);
37 }
38
39 pid_t my_pid = syscall_get_pid();
40 printf("My PID: %d\n", my_pid);
41
42 value = 3456787;
43
44 pid_t ping_pid = syscall_spawn("/programs/kmsg-ping", 0, NULL);
45 pid_t pong_pid = syscall_spawn("/programs/kmsg-pong", 0, NULL);
46 printf("ping pid: %d\n", ping_pid);
47 printf("pong pid: %d\n", pong_pid);
48
49 start_thread("worker", thread_work, &value);
50
51 my_pid = syscall_fork();
52
53 if (my_pid == 0)
54 {
55 printf("Child process\n");
56 syscall_exit(0);
57 }
58 else
59 {
60 printf("Parent process\n");
61 }
62
63 pid_t parent = syscall_get_parent_pid();
64 printf("Parent PID: %d\n", parent);
65
66 while (1)
67 ;
68
69 return 0;
70}
