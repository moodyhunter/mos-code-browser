3#include "libuserspace.h"
4#include "mos/filesystem/filesystem.h"
5#include "mos/ksyscall/usermode.h"
6#include "mos/mos_global.h"
7#include "mos/types.h"
8
9void print(const char *str)
10{
11 invoke_ksyscall_io_write(stdout, str, strlen(str), 0);
12}
13
14void print_err(const char *str)
15{
16 invoke_ksyscall_io_write(stderr, str, strlen(str), 0);
17}
18
19char buf[2 KB] = { 0 };
20
21int main(void)
22{
23 int fd = invoke_ksyscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
24 if (fd < 0)
25 print_err("Failed to open /assets/msg.txt");
26 else
27 {
28 size_t read = invoke_ksyscall_io_read(fd, buf, 512, 0);
29 print("\n");
30 invoke_ksyscall_io_write(stdout, buf, read, 0);
31 invoke_ksyscall_io_close(fd);
32 }
33
34 while (1)
35 invoke_ksyscall_yield_cpu();
36
37 return 0;
38}
