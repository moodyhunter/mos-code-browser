3#include "lib/memory.h"
4#include "lib/string.h"
5#include "libuserspace.h"
6#include "mos/syscall/usermode.h"
7
8static char buf[4 KB] = { 0 };
9
10static void thread_work(void *arg)
11{
12 int *value = (int *) arg;
13 pid_t process = syscall_get_pid();
14 printf("Thread started, value = %d, from process %d\n", *value, process);
15}
16
17static int value = 0;
18
19int main(void)
20{
21 printf("\n");
22 int fd = syscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
23 if (fd < 0)
24 printf("Failed to open /assets/msg.txt\n");
25 else
26 {
27 size_t read = syscall_io_read(fd, buf, 512, 0);
28 syscall_io_write(stdout, buf, read, 0);
29 syscall_io_close(fd);
30 }
31
32 char buf[256] = { 0 };
33
34 long read = syscall_io_read(stdin, buf, 256, 0);
35 if (read > 0)
36 {
37 printf("Read %d bytes from stdin", read);
38 syscall_io_write(stdout, buf, read, 0);
39 }
40
41 syscall_spawn("/programs/locks", 0, NULL);
42
43 pid_t my_pid = syscall_get_pid();
44 printf("My PID: %d\n", my_pid);
45
46 value = 3456787;
47 pid_t ping_pid = syscall_spawn("/programs/kmsg-ping", 0, NULL);
48 pid_t pong_pid = syscall_spawn("/programs/kmsg-pong", 0, NULL);
49 printf("ping pid: %d\n", ping_pid);
50 printf("pong pid: %d\n", pong_pid);
51
52 start_thread("worker", thread_work, &value);
53 my_pid = syscall_fork();
54
55 if (my_pid == 0)
56 {
57 printf("Child process\n");
58 uintptr_t heap_top = syscall_heap_control(HEAP_GET_TOP, 0);
59 printf("Child top: " PTR_FMT "\n", heap_top);
60 uintptr_t new_top = syscall_heap_control(HEAP_SET_TOP, heap_top + 16 KB);
61 printf("Child new top: " PTR_FMT "\n", new_top);
62
63 char *ptr = (char *) new_top - 1;
64 *ptr = 0x42;
65 u32 pid_new_forked = syscall_fork();
66 if (pid_new_forked == 0)
67 {
68 printf("Child process of child process\n");
69 syscall_exit(0);
70 }
71 else
72 {
73 printf("Parent process of child process\n");
74 }
75 }
76 else
77 {
78 printf("Parent process\n");
79 }
80
81 pid_t parent = syscall_get_parent_pid();
82 printf("Parent PID: %d\n", parent);
83
84 // get heap address
85 uintptr_t heap = syscall_heap_control(HEAP_GET_TOP, 0);
86 printf("Heap base: " PTR_FMT "\n", heap);
87
88 // grow the heap by 16 KB
89 uintptr_t new_heap = syscall_heap_control(HEAP_GROW_PAGES, 4); // in pages
90
91 printf("New heap top: " PTR_FMT "\n", new_heap);
92
93 // write to the heap
94 char *ptr = (char *) new_heap - 1;
95 *ptr = 0x42;
96
97 char *data = malloc(50);
98 strcpy(data, "Hello world!");
99 printf("Data: %s\n", data);
100
101 while (1)
102 ;
103
104 return 0;
105}
