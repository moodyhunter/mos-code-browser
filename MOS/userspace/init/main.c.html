3#include "mos/ksyscall/usermode.h"
4#include "mos/types.h"
5
6#define stdin 0
7#define stdout 1
8#define stderr 2
9
10int strlen(const char *str)
11{
12 int len = 0;
13 while (str[len])
14 len++;
15 return len;
16}
17
18void print(const char *str)
19{
20 invoke_ksyscall_io_write(stdout, str, strlen(str), 0);
21 invoke_ksyscall_io_write(stderr, str, strlen(str), 0);
22}
23
24void _start(void)
25{
26 const char *x = "Hello, world! MOS userspace '/bin/init'";
27 print(x);
28
29 while (true)
30 {
31 print("yielding cpu...");
32 invoke_ksyscall_yield_cpu();
33 print("we are back!");
34 }
35}
