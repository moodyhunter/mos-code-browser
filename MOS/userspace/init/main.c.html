8void printf(const char *fmt, ...)
9{
10 char buf[4096];
11 va_list args;
12 va_start(args, fmt);
13 vsprintf(buf, fmt, args);
14 va_end(args);
15 syscall_io_write(stdout, buf, strlen(buf), 0);
16}

18static char buf[4 KB] = { 0 };
19
20void thread_work(void *arg)
21{
22 int *value = (int *) arg;
23 pid_t process = syscall_get_pid();
24 printf("Thread started, value = %d, from process %d\n", *value, process);
25 // TODO: implement thread termination
26 while (1)
27 ;
28}

30static int value = 0;
31
32int main(void)
33{
34 int fd = syscall_file_open("/assets/msg.txt", FILE_OPEN_READ);
35 if (fd < 0)
36 printf("Failed to open /assets/msg.txt\n");
37 else
38 {
39 size_t read = syscall_io_read(fd, buf, 512, 0);
40 printf("\n");
41 syscall_io_write(stdout, buf, read, 0);
42 syscall_io_close(fd);
43 }
44
45 pid_t my_pid = syscall_get_pid();
46 printf("My PID: %d\n", my_pid);
47
48 value = 3456787;
49
50 pid_t ping_pid = syscall_spawn("/programs/kmsg-ping", 0, NULL);
51 pid_t pong_pid = syscall_spawn("/programs/kmsg-pong", 0, NULL);
52 printf("ping pid: %d\n", ping_pid);
53 printf("pong pid: %d\n", pong_pid);
54
55 syscall_create_thread("worker", thread_work, &value);
56
57 my_pid = syscall_fork();
58
59 if (my_pid == 0)
60 {
61 printf("Child process\n");
62 syscall_exit(0);
63 }
64 else
65 {
66 printf("Parent process\n");
67 }
68
69 pid_t parent = syscall_get_parent_pid();
70 printf("Parent PID: %d\n", parent);
71
72 while (1)
73 ;
74
75 return 0;
76}
