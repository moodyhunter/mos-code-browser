7int main(void)

9 printf("pong\n");
10 fd_t client = syscall_ipc_connect("kmsg-ping-pong", IPC_CONNECT_DEFAULT, MOS_PAGE_SIZE);
11
12 // client read
13 char client_buf[150];
14 size_t read_size = syscall_io_read(client, client_buf, 150, 0);
15 printf("Client: Received '%.*s'\n", (int) read_size, client_buf);
16
17 size_t written = syscall_io_write(client, "Nice Fox!", 10, 0);
18
19 if (written != 10)
20 {
21 printf("Client: failed to write to ipc channel\n");
22 }
23
24 return 0;
25}
