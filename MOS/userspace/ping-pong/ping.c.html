3#include "lib/string.h"
4#include "libuserspace.h"
5#include "mos/syscall/usermode.h"
6
7int main(void)
8{
9 printf("ping\n");
10 fd_t fd = syscall_ipc_open("kmsg-ping-pong", 0, MOS_PAGE_SIZE);
11 if (fd < 0)
12 {
13 printf("failed to open ipc channel");
14 return 1;
15 }
16
17 char *data = "ping";
18 size_t written = syscall_io_write(fd, data, strlen(data), 0);
19 if (written != strlen(data))
20 {
21 printf("failed to write to ipc channel");
22 return 1;
23 }
24 char buf[150];
25 size_t read = syscall_io_read(fd, buf, 150, 0);
26 if (read != strlen(data))
27 {
28 printf("failed to read from ipc channel");
29 return 1;
30 }
31 printf("pong: %s", buf);
32 return 0;
33}
