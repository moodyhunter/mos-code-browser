8const char *data = "The quick brown fox jumps over the lazy dog.";
9const size_t data_size = 43;
10
11static void thread_main(void *arg)
12{
13 fd_t client_fd = (fd_t) arg;
14
15 char buf[150];
16 size_t sz = sprintf(buf, "Welcome to the server, client fd %d!", client_fd);
17
18 size_t written = syscall_io_write(client_fd, buf, sz, 0);
19 if (written != sz)
20 printf("Server: failed to write to ipc channel\n");
21
22 size_t read_size = syscall_io_read(client_fd, buf, 150, 0);
23 printf("Server: Received '%.*s' from client %d\n", (int) read_size, buf, client_fd);
24
25 syscall_io_close(client_fd);
26}
27
28int main(void)
29{
30 for (int i = 0; i < 10; i++)
31 syscall_spawn("/programs/kmsg-pong", 0, NULL);
32 printf("ping\n");
33 fd_t fd = syscall_ipc_create("kmsg-ping-pong", 32);
34 if (fd < 0)
35 {
36 printf("failed to open ipc channel\n");
37 return -1;
38 }
39
40 while (true)
41 {
42 fd_t client_fd = syscall_ipc_accept(fd);
43 if (client_fd < 0)
44 {
45 printf("failed to accept ipc channel\n");
46 return -1;
47 }
48
49 printf("Server: Accepted fd %d\n", client_fd);
50 start_thread("child", thread_main, (void *) client_fd);
51 }
52
53 return 0;
54}
