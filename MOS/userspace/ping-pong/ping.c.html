10const size_t data_size = 43;

12static void thread_main(void *arg)
13{
14 fd_t client_fd = (fd_t) arg;
15
16 char buf[150];
17 size_t sz = sprintf(buf, "Welcome to the server, client fd %d!", client_fd);
18
19 size_t written = syscall_io_write(client_fd, buf, sz, 0);
20 if (written != sz)
21 printf("Server: failed to write to ipc channel\n");
22
23 size_t read_size = syscall_io_read(client_fd, buf, 150, 0);
24 printf("Server: Received '%.*s' from client %d\n", (int) read_size, buf, client_fd);
25
26 syscall_io_close(client_fd);
27}
28
29int main(int argc, char **argv)
30{
31 if (argc != 2)
32 {
33 printf("Usage: %s <ping-pong-ipc-channel>\n", argv[0]);
34 printf("Starts a server that accepts connections on the given ipc name.\n");
35 return -1;
36 }
37
38 const char *ipc_name = argv[1];
39 printf("Server: Starting with ipc name '%s'\n", ipc_name);
40
41 const char *args[3] = {
42 "/programs/kmsg-pong",
43 ipc_name,
44 NULL,
45 };
46
47 for (int i = 0; i < 10; i++)
48 syscall_spawn("/programs/kmsg-pong", 2, args);
49
50 fd_t fd = syscall_ipc_create(ipc_name, 32);
51 if (fd < 0)
52 {
53 printf("failed to open ipc channel\n");
54 return -1;
55 }
56
57 while (true)
58 {
59 fd_t client_fd = syscall_ipc_accept(fd);
60 if (client_fd < 0)
61 {
62 printf("failed to accept ipc channel\n");
63 return -1;
64 }
65
66 printf("Server: Accepted fd %d\n", client_fd);
67 start_thread("child", thread_main, (void *) client_fd);
68 }
69
70 return 0;
71}
