3#include "lib/string.h"
4#include "libuserspace.h"
5#include "mos/syscall/usermode.h"
6
7int main(void)
8{
9 syscall_spawn("/programs/kmsg-pong", 0, NULL);
10 printf("ping\n");
11 fd_t fd = syscall_ipc_create("kmsg-ping-pong", 32);
12 if (fd < 0)
13 {
14 printf("failed to open ipc channel\n");
15 return -1;
16 }
17
18 fd_t client_fd;
19 while ((client_fd = syscall_ipc_accept(fd)) < 0)
20 ;
21
22 char *data = "ping";
23 size_t written = syscall_io_write(client_fd, data, strlen(data), 0);
24 if (written != strlen(data))
25 {
26 printf("server: failed to write to ipc channel\n");
27 }
28
29 while (1)
30 ;
31
32 return 0;
33}
